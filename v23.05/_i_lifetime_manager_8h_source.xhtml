40 /** Interface for managing the lifetime of objects */
41 class ILifetimeManager
42 {
43 public:
44  /** Virtual Destructor */
45  virtual ~ILifetimeManager() = default;
46  /** Registers a group to the lifetime manager and assigns a group id
47  *
48  * @param[in] group The group id of the group
49  */
50  virtual void register_group(IMemoryGroup *group) = 0;
51  /** Unbound and release elements associated with a group
52  *
53  * @param[in] group Group to unbound its elements
54  *
55  * @return True if group was registered and released else false.
56  */
57  virtual bool release_group(IMemoryGroup *group) = 0;
58  /** Registers and starts lifetime of an object
59  *
60  * @param[in] obj Object to register
61  */
62  virtual void start_lifetime(void *obj) = 0;
63  /** Ends lifetime of an object
64  *
65  * @param[in] obj Object
66  * @param[in] obj_memory Object memory
67  * @param[in] size Size of the given object at given time
68  * @param[in] alignment Alignment requirements for the object
69  */
70  virtual void end_lifetime(void *obj, IMemory &obj_memory, size_t size, size_t alignment) = 0;
71  /** Checks if the lifetime of the registered object is complete
72  *
73  * @return True if all object lifetimes are finalized else false.
74  */
75  virtual bool are_all_finalized() const = 0;
76  /** Creates a memory pool depending on the memory requirements

82  virtual std::unique_ptr<IMemoryPool> create_pool(IAllocator *allocator) = 0;
