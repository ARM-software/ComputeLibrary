77 Status NEConvolutionLayerReshapeWeights::validate(const ITensorInfo *weights, const ITensorInfo *biases, const ITensorInfo *output)
78 {
79  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(weights);
80  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(weights, 1,
81  DataType::QASYMM8, DataType::QASYMM8_SIGNED, DataType::QSYMM8_PER_CHANNEL,
82  DataType::BFLOAT16, DataType::F16, DataType::F32);
83  ARM_COMPUTE_RETURN_ERROR_ON(weights->num_dimensions() > 4);
84 
85  if(biases != nullptr)
86  {
87  const int idx_kernels = get_data_layout_dimension_index(weights->data_layout(), DataLayoutDimension::BATCHES);
88  ARM_COMPUTE_RETURN_ERROR_ON(is_data_type_quantized_asymmetric(weights->data_type()));
89  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(weights, biases);
90  ARM_COMPUTE_RETURN_ERROR_ON(biases->dimension(0) != weights->dimension(idx_kernels));
91  ARM_COMPUTE_RETURN_ERROR_ON(biases->num_dimensions() > 1);
92  }
93 
94  if((output != nullptr) && (output->total_size() != 0))
95  {
96  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(weights, output);
97 
98  NEWeightsReshapeKernel::validate(weights, biases, output);
99  }
100 
101  return Status{};
102 }

118 void NEGEMMConvolutionLayer::configure_mm(const ITensor *input, const ITensor *weights, const ITensor *biases, ITensor *output, const ActivationLayerInfo &act_info, int gemm_3d_depth)
119 {
120  ARM_COMPUTE_ERROR_ON_NULLPTR(input, weights);
121  ARM_COMPUTE_ERROR_THROW_ON(validate_mm(input->info(), weights->info(), biases == nullptr ? nullptr : biases->info(), output == nullptr ? nullptr : output->info(),
122  act_info, gemm_3d_depth, _skip_im2col));
123 
124  // Create GEMMInfo structure
125  const GEMMInfo &gemm_info = GEMMInfo(false, false, true /* Reshape weights only for the first run */,
126  gemm_3d_depth, _skip_im2col /* Reinterpret the input as 3D if im2col is skipped */,
127  false, GEMMLowpOutputStageInfo(), false, false, act_info);
128 
129  // Supported activations in GEMM
130  const std::set<ActivationLayerInfo::ActivationFunction> supported_acts = { ActivationLayerInfo::ActivationFunction::RELU,
131  ActivationLayerInfo::ActivationFunction::BOUNDED_RELU,
132  ActivationLayerInfo::ActivationFunction::LU_BOUNDED_RELU
133  };
134 
135  if(_is_quantized)
136  {
137  // Since we need negative offsets for computing convolution, we need to change QuantizationInfo()
138  // Extract and negate input and weights offset
139  const QuantizationInfo iqinfo = input->info()->quantization_info();
140  const QuantizationInfo wqinfo = weights->info()->quantization_info();
141  const QuantizationInfo oqinfo = (output->info()->total_size() == 0) ? iqinfo : output->info()->quantization_info();
142  const UniformQuantizationInfo uiqinfo = iqinfo.uniform();
143  const UniformQuantizationInfo uoqinfo = oqinfo.uniform();
144  const DataType data_type = input->info()->data_type();
145 
146  input->info()->set_quantization_info(QuantizationInfo(uiqinfo.scale, -uiqinfo.offset));
147  if(!is_data_type_quantized_per_channel(weights->info()->data_type()))
148  {
149  const UniformQuantizationInfo uwqinfo = wqinfo.uniform();
150  weights->info()->set_quantization_info(QuantizationInfo(uwqinfo.scale, -uwqinfo.offset));
151  }
152 
153  // Merge activation with output stage
154  PixelValue type_min{};
155  PixelValue type_max{};
156  std::tie(type_min, type_max) = get_min_max(data_type);
157  int32_t min_activation = type_min.get<int32_t>();
158  int32_t max_activation = type_max.get<int32_t>();
159 
160  if(supported_acts.count(act_info.activation()) != 0)
161  {
162  std::tie(min_activation, max_activation) = get_quantized_activation_min_max(act_info, data_type, uoqinfo);
163  }
164 
165  GEMMLowpOutputStageInfo output_info;
166  output_info.type = GEMMLowpOutputStageType::QUANTIZE_DOWN_FIXEDPOINT;
167  output_info.gemmlowp_offset = uoqinfo.offset;
168  output_info.gemmlowp_min_bound = min_activation;
169  output_info.gemmlowp_max_bound = max_activation;
170  output_info.is_quantized_per_channel = (weights->info()->data_type() == DataType::QSYMM8_PER_CHANNEL);
171  quantization::calculate_quantized_multipliers(iqinfo, wqinfo, oqinfo, output_info);
172 
173  _mm_gemmlowp.configure(input, weights, biases, output, GEMMInfo(false, false, true, gemm_3d_depth, _skip_im2col, false, output_info));
174 
175  // Revert back QuantizatioInfo as input and weights could be used in other convolution layers
176  input->info()->set_quantization_info(iqinfo);
177  weights->info()->set_quantization_info(wqinfo);
178  }
179  else
180  {
181  // Configure matrix multiply function
182  _mm_gemm.configure(input, weights, biases, output, 1.0f, 0.0f, gemm_info);
183  }
184 }

186 Status NEGEMMConvolutionLayer::validate_mm(const ITensorInfo *input, const ITensorInfo *weights, const ITensorInfo *biases, const ITensorInfo *output,
187  const ActivationLayerInfo &act_info, int gemm_3d_depth, bool skip_im2col)
188 {
189  const DataType data_type = input->data_type();
190  const bool is_quantized = is_data_type_quantized_asymmetric(data_type);
191  const bool is_activation_enabled = act_info.enabled();
192 
193  // Create GEMMInfo structure
194  const GEMMInfo gemm_info = GEMMInfo(false, false, true /* Reshape weights only for the first run */,
195  gemm_3d_depth, skip_im2col /* Reinterpret the input as 3D if im2col is skipped */,
196  false, GEMMLowpOutputStageInfo(), false, false, act_info);
197 
198  if(is_quantized)
199  {
200  // Since we need negative offsets for computing convolution, we need to change QuantizationInfo()
201  // Extract and negate input and weights offset
202  const QuantizationInfo &iqinfo = input->quantization_info();
203  const QuantizationInfo &wqinfo = weights->quantization_info();
204  const QuantizationInfo &oqinfo = (output->total_size() == 0) ? iqinfo : output->quantization_info();
205  const UniformQuantizationInfo uoqinfo = oqinfo.uniform();
206 
207  // Merge activation with output stage
208  PixelValue type_min{};
209  PixelValue type_max{};
210  std::tie(type_min, type_max) = get_min_max(data_type);
211  int32_t min_activation = type_min.get<int32_t>();
212  int32_t max_activation = type_max.get<int32_t>();
213 
214  const std::set<ActivationLayerInfo::ActivationFunction> supported_acts = { ActivationLayerInfo::ActivationFunction::RELU,
215  ActivationLayerInfo::ActivationFunction::BOUNDED_RELU,
216  ActivationLayerInfo::ActivationFunction::LU_BOUNDED_RELU
217  };
218  if(is_activation_enabled && supported_acts.count(act_info.activation()) != 0)
219  {
220  std::tie(min_activation, max_activation) = get_quantized_activation_min_max(act_info, data_type, uoqinfo);
221  }
222 
223  GEMMLowpOutputStageInfo output_info;
224  output_info.type = GEMMLowpOutputStageType::QUANTIZE_DOWN_FIXEDPOINT;
225  output_info.gemmlowp_offset = uoqinfo.offset;
226  output_info.gemmlowp_min_bound = min_activation;
227  output_info.gemmlowp_max_bound = max_activation;
228  output_info.is_quantized_per_channel = (weights->data_type() == DataType::QSYMM8_PER_CHANNEL);
229  ARM_COMPUTE_RETURN_ON_ERROR(quantization::calculate_quantized_multipliers(iqinfo, wqinfo, oqinfo, output_info));
230 
231  // Perform validation step on GEMMLowp
232  std::unique_ptr<ITensorInfo> input_qa = input->clone();
233  std::unique_ptr<ITensorInfo> weights_qa = weights->clone();
234  input_qa->set_quantization_info(QuantizationInfo(iqinfo.uniform().scale, -iqinfo.uniform().offset));
235  weights_qa->set_quantization_info(QuantizationInfo(wqinfo.uniform().scale, -wqinfo.uniform().offset));
236  return NEGEMMLowpMatrixMultiplyCore::validate(input_qa.get(), weights_qa.get(), biases, output, GEMMInfo(false, false, true, gemm_3d_depth, skip_im2col, false, output_info));
237  }
238  else
239  {
240  // Perform validation step on Matrix multiply function
241  return NEGEMM::validate(input, weights, nullptr, output, 1.0f, 0.0f, gemm_info);
242  }
243 }

245 Status NEGEMMConvolutionLayer::validate_gemm3d(const ITensorInfo *input_info, const ITensorInfo *weights_info, const ActivationLayerInfo &act_info, int gemm_3d_depth, bool skip_im2col)
246 {
247  const DataType data_type = input_info->data_type();
248  const unsigned int mult_y = skip_im2col ? 1U : gemm_3d_depth;
249  const unsigned int mult_z = skip_im2col ? gemm_3d_depth : 1U;
250 
251  // Set dummy tensor shapes for the validation
252  const TensorInfo dummy_input_info(TensorShape(4U, 4U * mult_y, 1U * mult_z), 1, data_type, input_info->quantization_info());
253  const TensorInfo dummy_weights_info(TensorShape(4U, 4U), 1, data_type, weights_info->quantization_info());
254  const TensorInfo dummy_output_info(TensorShape(4U, 4U, gemm_3d_depth), 1, data_type, input_info->quantization_info());
255 
256  return validate_mm(&dummy_input_info, &dummy_weights_info, nullptr, &dummy_output_info, act_info, gemm_3d_depth, skip_im2col);
257 }

259 void NEGEMMConvolutionLayer::configure(const ITensor *input, const ITensor *weights, const ITensor *biases, ITensor *output, const PadStrideInfo &conv_info, const WeightsInfo &weights_info,
260  const Size2D &dilation, const ActivationLayerInfo &act_info, unsigned int num_groups)
261 {
262  ARM_COMPUTE_ERROR_ON_NULLPTR(input, weights, output);
263  ARM_COMPUTE_UNUSED(num_groups, weights_info);
264  ARM_COMPUTE_ERROR_THROW_ON(NEGEMMConvolutionLayer::validate(input->info(),
265  weights->info(),
266  biases != nullptr ? biases->info() : nullptr,
267  output->info(),
268  conv_info,
269  weights_info,
270  dilation,
271  act_info,
272  num_groups));
273 
274  const DataType data_type = input->info()->data_type();
275  const DataLayout data_layout = input->info()->data_layout();
276  const int idx_width = get_data_layout_dimension_index(data_layout, DataLayoutDimension::WIDTH);
277  const int idx_height = get_data_layout_dimension_index(data_layout, DataLayoutDimension::HEIGHT);
278  const int idx_kernels = get_data_layout_dimension_index(data_layout, DataLayoutDimension::BATCHES);
279 
280  const unsigned int kernel_width = weights->info()->dimension(idx_width);
281  const unsigned int kernel_height = weights->info()->dimension(idx_height);
282 
283  _is_prepared = weights_info.retain_internal_weights();
284  _original_weights = weights;
285  _is_quantized = is_data_type_quantized_asymmetric(input->info()->data_type());
286  _data_layout = data_layout;
287  _skip_im2col = (data_layout == DataLayout::NHWC && kernel_width == 1 && kernel_height == 1 && conv_info.stride().first == 1 && conv_info.stride().second == 1);
288 
289  const ITensor *gemm_input_to_use = input;
290  ITensor *gemm_output_to_use = output;
291 
292  // Get convolved dimensions
293  unsigned int conv_w = 0;
294  unsigned int conv_h = 0;
295  std::tie(conv_w, conv_h) = scaled_dimensions(input->info()->dimension(idx_width),
296  input->info()->dimension(idx_height),
297  kernel_width,
298  kernel_height,
299  conv_info,
300  dilation);
301 
302  // Check if GEMM3D is supported
303  if(data_layout == DataLayout::NHWC)
304  {
305  _skip_col2im = bool(validate_gemm3d(input->info(), weights->info(), act_info, conv_h, true));
306  // If not supported, we need to perform im2col and col2im (or reshape layer)
307  if(!_skip_col2im)
308  {
309  _skip_im2col = false;
310  }
311  }
312  else
313  {
314  _skip_col2im = false;
315  }
316 
317  // Get parameters from conv_info
318  unsigned int stride_x = 0;
319  unsigned int stride_y = 0;
320  std::tie(stride_x, stride_y) = conv_info.stride();
321 
322  unsigned int mat_weights_cols = weights->info()->dimension(idx_kernels);
323 
324  // _weights_reshaped will be auto configured in the kernel.
325  // Just append biases and do not transpose 1xW as it will be reshaped in NEGEMM
326  const ITensor *weights_to_use = weights;
327 
328  if(_weights_manager && _weights_manager->are_weights_managed(weights))
329  {
330  _reshape_weights_managed.configure(weights, nullptr);
331  weights_to_use = _weights_manager->acquire(weights, &_reshape_weights_managed);
332  }
333  else
334  {
335  _reshape_weights.configure(weights, nullptr, &_weights_reshaped);
336  weights_to_use = &_weights_reshaped;
337  }
338 
339  // Create tensor to store im2col reshaped inputs
340  if(!_skip_im2col)
341  {
342  _memory_group.manage(&_im2col_output);
343 
344  // Configure
345  _im2col_kernel = arm_compute::support::cpp14::make_unique<NEIm2ColKernel>();
346  _im2col_kernel->configure(input, &_im2col_output, Size2D(kernel_width, kernel_height), conv_info, false, dilation);
347 
348  // Update GEMM input
349  gemm_input_to_use = &_im2col_output;
350  }
351 
352  // Create temporary GEMM output tensor in case we cannot skip col2im
353  const DataType output_data_type = data_type == DataType::BFLOAT16 ? DataType::F32 : data_type;
354  if(!_skip_col2im)
355  {
356  TensorShape shape_gemm;
357 
358  // Calculate GEMM output shape
359  shape_gemm = _im2col_output.info()->tensor_shape();
360  shape_gemm.set(0, mat_weights_cols);
361  shape_gemm.set(1, conv_w * conv_h);
362 
363  // FIXME: input->clone() doesn't work with subtensors for grouped convolutions.
364  TensorInfo info_gemm(shape_gemm, 1, output_data_type);
365  info_gemm.set_quantization_info(output->info()->quantization_info()).set_data_layout(input->info()->data_layout());
366  _gemm_output.allocator()->init(info_gemm);
367  _memory_group.manage(&_gemm_output);
368 
369  // Update GEMM output
370  gemm_output_to_use = &_gemm_output;
371  }
372 
373  // Configure GEMM
374  // In case we need to skip col2im, GEMM3D (gemm_3d_depth != 0) must be called in order to avoid reshaping the output matrix
375  const unsigned int gemm_3d_depth = _skip_col2im ? conv_h : 0;
376  configure_mm(gemm_input_to_use, weights_to_use, biases, gemm_output_to_use, act_info, gemm_3d_depth);
377 
378  if(!_skip_im2col)
379  {
380  _im2col_output.allocator()->allocate();
381  }
382 
383  if(!_skip_col2im)
384  {
385  if(_data_layout == DataLayout::NCHW)
386  {
387  // Configure col2im
388  _col2im_kernel = arm_compute::support::cpp14::make_unique<NECol2ImKernel>();
389  _col2im_kernel->configure(gemm_output_to_use, output, Size2D(conv_w, conv_h));
390  }
391  else
392  {
393  // Configure reshape layer
394  _reshape_layer.configure(gemm_output_to_use, output);
395  }
396  }
397 
398  if(_is_quantized && !_skip_col2im)
399  {
400  _tmp_output.allocator()->allocate();
401  }
402 
403  if(!_skip_col2im || _is_quantized)
404  {
405  _gemm_output.allocator()->allocate();
406  }
407 
408  ARM_COMPUTE_ERROR_ON_MSG((output->info()->dimension(idx_width) != conv_w) || (output->info()->dimension(idx_height) != conv_h),
409  "Output shape does not match the expected one");
410 }

412 Status NEGEMMConvolutionLayer::validate(const ITensorInfo *input, const ITensorInfo *weights, const ITensorInfo *biases, const ITensorInfo *output, const PadStrideInfo &conv_info,
413  const WeightsInfo &weights_info, const Size2D &dilation, const ActivationLayerInfo &act_info, unsigned int num_groups)
414 {
415  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(input, weights, output);
416  ARM_COMPUTE_RETURN_ERROR_ON_MSG(weights_info.are_reshaped(), "Weights already reshaped are not supported!");
417  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(input, 1, DataType::QASYMM8, DataType::QASYMM8_SIGNED, DataType::BFLOAT16, DataType::F16, DataType::F32);
418  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(weights, 1, DataType::QASYMM8, DataType::QASYMM8_SIGNED, DataType::QSYMM8_PER_CHANNEL, DataType::BFLOAT16, DataType::F16, DataType::F32);
419  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_LAYOUT(input, weights);
420  ARM_COMPUTE_RETURN_ERROR_ON_MSG(num_groups > 1, "Grouping (num_groups != 1) is not supported on NEON");
421 
422  const DataLayout data_layout = input->data_layout();
423  const DataType data_type = input->data_type();
424  const int idx_width = get_data_layout_dimension_index(data_layout, DataLayoutDimension::WIDTH);
425  const int idx_height = get_data_layout_dimension_index(data_layout, DataLayoutDimension::HEIGHT);
426  const int idx_channel = get_data_layout_dimension_index(data_layout, DataLayoutDimension::CHANNEL);
427  const int idx_kernels = get_data_layout_dimension_index(data_layout, DataLayoutDimension::BATCHES);
428 
429  const unsigned int kernel_width = weights->dimension(idx_width);
430  const unsigned int kernel_height = weights->dimension(idx_height);
431 
432  TensorInfo im2col_reshaped_info{};
433  TensorInfo info_gemm{};
434  TensorInfo tmp_info{};
435  TensorInfo weights_reshaped_info{};
436  const ITensorInfo *gemm_input_to_use = input;
437  const ITensorInfo *gemm_output_to_use = output;
438  const ITensorInfo *weights_to_use = weights;
439 
440  const bool append_bias = false;
441  const bool is_quantized = is_data_type_quantized_asymmetric(data_type);
442  const bool is_bf16 = data_type == DataType::BFLOAT16;
443  bool skip_im2col = (data_layout == DataLayout::NHWC && kernel_width == 1 && kernel_height == 1 && conv_info.stride().first == 1 && conv_info.stride().second == 1);
444 
445  // Get convolved dimensions
446  unsigned int conv_w = 0;
447  unsigned int conv_h = 0;
448 
449  std::tie(conv_w, conv_h) = scaled_dimensions(input->dimension(idx_width),
450  input->dimension(idx_height),
451  kernel_width,
452  kernel_height,
453  conv_info,
454  dilation);
455 
456  // Check if GEMM3D is supported
457  bool skip_col2im = false;
458  if(data_layout == DataLayout::NHWC)
459  {
460  skip_col2im = bool(validate_gemm3d(input, weights, act_info, conv_h, true));
461  // If not supported, we need to perform im2col and col2im (or reshape layer)
462  if(!skip_col2im)
463  {
464  skip_im2col = false;
465  }
466  }
467 
468  if(skip_col2im)
469  {
470  // If not supported, we need to perform im2col and col2im (or reshape layer)
471  if(!bool(validate_gemm3d(input, weights, act_info, conv_h, skip_im2col)))
472  {
473  skip_im2col = false;
474  skip_col2im = false;
475  }
476  }
477 
478  ARM_COMPUTE_RETURN_ERROR_ON(weights->dimension(idx_channel) != input->dimension(idx_channel));
479  ARM_COMPUTE_RETURN_ERROR_ON(weights->num_dimensions() > 4);
480 
481  // Validate biases
482  if(biases != nullptr)
483  {
484  if(is_quantized)
485  {
486  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(biases, 1, DataType::S32);
487  }
488  else if(is_bf16)
489  {
490  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(biases, 1, DataType::F32);
491  }
492  else
493  {
494  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input, biases);
495  }
496  ARM_COMPUTE_RETURN_ERROR_ON(biases->dimension(0) != weights->dimension(idx_kernels));
497  ARM_COMPUTE_RETURN_ERROR_ON(biases->num_dimensions() > 1);
498  }
499 
500  unsigned int mat_weights_cols = weights->dimension(idx_kernels);
501  unsigned int mat_weights_rows = weights->dimension(idx_width) * weights->dimension(idx_height) * weights->dimension(idx_channel);
502 
503  // Output tensor auto inizialization if not yet initialized
504  ARM_COMPUTE_RETURN_ON_ERROR(NEConvolutionLayerReshapeWeights::validate(weights, nullptr, nullptr));
505  weights_reshaped_info = TensorInfo(compute_weights_reshaped_shape(*weights, append_bias), 1, data_type);
506  weights_reshaped_info.set_quantization_info(weights->quantization_info());
507  weights_to_use = &weights_reshaped_info;
508 
509  if(!skip_im2col)
510  {
511  // Create tensor info for im2col reshaped inputs
512  // For NEON the batch size is on the fourth dimension
513  // TODO (giaiod01): Auto-initialize the output shape of im2col COMPMID-1482
514  TensorShape shape_im2col = input->tensor_shape();
515  shape_im2col.set(0, mat_weights_rows);
516  shape_im2col.set(1, conv_w * conv_h);
517  shape_im2col.set(2, 1);
518 
519  im2col_reshaped_info = TensorInfo(shape_im2col, 1, data_type);
520  im2col_reshaped_info.set_quantization_info(input->quantization_info());
521 
522  ARM_COMPUTE_RETURN_ON_ERROR(NEIm2ColKernel::validate(input, &im2col_reshaped_info, Size2D(kernel_width, kernel_height), conv_info, append_bias, dilation));
523  gemm_input_to_use = &im2col_reshaped_info;
524  }
525 
526  // Create temporary GEMM output tensor in case we cannot skip col2im
527  const DataType output_data_type = data_type == DataType::BFLOAT16 ? DataType::F32 : data_type;
528  if(!skip_col2im)
529  {
530  TensorShape shape_gemm = gemm_input_to_use->tensor_shape();
531  shape_gemm.set(0, mat_weights_cols);
532  shape_gemm.set(1, conv_w * conv_h);
533  info_gemm = TensorInfo(shape_gemm, 1, output_data_type);
534  }
535  else
536  {
537  info_gemm = TensorInfo(output->tensor_shape(), 1, output_data_type);
538  }
539  info_gemm.set_quantization_info(output->quantization_info()).set_data_layout(input->data_layout());
540  gemm_output_to_use = &info_gemm;
541  ARM_COMPUTE_RETURN_ON_ERROR(validate_mm(gemm_input_to_use, weights_to_use, biases, gemm_output_to_use, act_info, skip_col2im ? conv_h : 0, skip_im2col));
542 
543  // Validate Col2Im/ReshapeLayer
544  if(!skip_col2im && (data_layout == DataLayout::NCHW))
545  {
546  ARM_COMPUTE_RETURN_ON_ERROR(NECol2ImKernel::validate(gemm_output_to_use, output, Size2D(conv_w, conv_h)));
547  }
548 
549  return Status{};
550 }

552 void NEGEMMConvolutionLayer::run()
553 {
554  prepare();
555 
556  MemoryGroupResourceScope scope_mg(_memory_group);
557 
558  if(!_skip_im2col)
559  {
560  // Run input reshaping
561  unsigned int y_dim = get_data_layout_dimension_index(_data_layout, DataLayoutDimension::HEIGHT);
562  NEScheduler::get().schedule(_im2col_kernel.get(), y_dim);
563  }
564 
565  // Runs NEGEMM or NEGEMMLowpMatrixMultiplyCore functions
566  if(_is_quantized)
567  {
568  // Run gemmlowp
569  _mm_gemmlowp.run();
570  }
571  else
572  {
573  // Run gemm
574  _mm_gemm.run();
575  }
576 
577  // Reshape output matrix
578  if(!_skip_col2im)
579  {
580  if(_data_layout == DataLayout::NCHW)
581  {
582  NEScheduler::get().schedule(_col2im_kernel.get(), Window::DimY);
583  }
584  else
585  {
586  _reshape_layer.run();
587  }
588  }

591 void NEGEMMConvolutionLayer::prepare()
592 {
593  if(!_is_prepared)
594  {
595  if(_weights_manager && _weights_manager->are_weights_managed(_original_weights))
596  {
597  _weights_manager->run(_original_weights, &_reshape_weights_managed);
598  }
599  else
600  {
601  // Run weights reshaping and mark original weights tensor as unused
602  _weights_reshaped.allocator()->allocate();
603  _reshape_weights.run();
604  _original_weights->mark_as_unused();
605  }
606 
607  // Prepare GEMM
608  _is_quantized ? _mm_gemmlowp.prepare() : _mm_gemm.prepare();
609  if(!_weights_reshaped.is_used())
610  {
611  _weights_reshaped.allocator()->free();
612  }
613 
614  _is_prepared = true;
615  }
