63 void OffsetLifetimeManager::update_blobs_and_mappings()
64 {
65  ARM_COMPUTE_ERROR_ON(!are_all_finalized());
66  ARM_COMPUTE_ERROR_ON(_active_group == nullptr);
67 
68  // Update blob size
69  size_t max_aggregated_size = 0;
70  std::for_each(std::begin(_free_blobs), std::end(_free_blobs), [&](const Blob & b)
71  {
72  max_aggregated_size += b.max_size;
73  _blob.alignment = std::max(_blob.alignment, b.max_alignment);
74  });
75  max_aggregated_size += _free_blobs.size() * _blob.alignment;
76  _blob.owners = std::max(_blob.owners, _free_blobs.size());
77  _blob.size = std::max(_blob.size, max_aggregated_size);
78 
79  // Calculate group mappings
80  auto &group_mappings = _active_group->mappings();
81  size_t offset = 0;
82  for(auto &free_blob : _free_blobs)
83  {
84  for(auto &bound_element_id : free_blob.bound_elements)
85  {
86  ARM_COMPUTE_ERROR_ON(_active_elements.find(bound_element_id) == std::end(_active_elements));
87  Element &bound_element = _active_elements[bound_element_id];
88  group_mappings[bound_element.handle] = offset;
89  }
90  offset += free_blob.max_size;
91  offset = align_offset(offset, _blob.alignment);
92  ARM_COMPUTE_ERROR_ON(offset > _blob.size);
93  }
94 }
