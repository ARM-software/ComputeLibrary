99 std::vector<std::unique_ptr<Printer>> CommonOptions::create_printers()
100 {
101  std::vector<std::unique_ptr<Printer>> printers;
102 
103  if(pretty_console->value() && (log_file->is_set() || log_format->value() != LogFormat::PRETTY))
104  {
105  auto pretty_printer = support::cpp14::make_unique<PrettyPrinter>();
106  pretty_printer->set_color_output(color_output->value());
107  printers.push_back(std::move(pretty_printer));
108  }
109 
110  std::unique_ptr<Printer> printer;
111  switch(log_format->value())
112  {
113  case LogFormat::JSON:
114  printer = support::cpp14::make_unique<JSONPrinter>();
115  break;
116  case LogFormat::NONE:
117  break;
118  case LogFormat::PRETTY:
119  default:
120  auto pretty_printer = support::cpp14::make_unique<PrettyPrinter>();
121  // Don't use colours if we print to a file:
122  pretty_printer->set_color_output((!log_file->is_set()) && color_output->value());
123  printer = std::move(pretty_printer);
124  break;
125  }
126 
127  if(log_file->is_set())
128  {
129  log_streams.push_back(std::make_shared<std::ofstream>(log_file->value()));
130  if(printer != nullptr)
131  {
132  printer->set_stream(*log_streams.back().get());
133  }
134  }
135 
136  if(printer != nullptr)
137  {
138  printers.push_back(std::move(printer));
139  }
140 
141  if(json_file->is_set())
142  {
143  printers.push_back(support::cpp14::make_unique<JSONPrinter>());
144  log_streams.push_back(std::make_shared<std::ofstream>(json_file->value()));
145  printers.back()->set_stream(*log_streams.back().get());
146  }
147 
148  if(pretty_file->is_set())
149  {
150  printers.push_back(support::cpp14::make_unique<PrettyPrinter>());
151  log_streams.push_back(std::make_shared<std::ofstream>(pretty_file->value()));
152  printers.back()->set_stream(*log_streams.back().get());
153  }
154 
155  return printers;
156 }
