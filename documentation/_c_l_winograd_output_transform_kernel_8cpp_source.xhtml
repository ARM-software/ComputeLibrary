49 Status validate_arguments(const ITensorInfo *input, const ITensorInfo *bias, const ITensorInfo *output, const WinogradInfo &winograd_info, const ActivationLayerInfo &act_info)
50 {
51  if(act_info.enabled())
52  {
53  ARM_COMPUTE_RETURN_ERROR_ON_F16_UNSUPPORTED(input);
54  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(input, 1, DataType::U8, DataType::QASYMM8, DataType::F16, DataType::F32);
55  ARM_COMPUTE_RETURN_ERROR_ON_MSG((input->data_type() == DataType::QASYMM8) && (act_info.activation() != ActivationLayerInfo::ActivationFunction::LU_BOUNDED_RELU)
56  && (act_info.activation() != ActivationLayerInfo::ActivationFunction::BOUNDED_RELU)
57  && (act_info.activation() != ActivationLayerInfo::ActivationFunction::RELU)
58  && (act_info.activation() != ActivationLayerInfo::ActivationFunction::LOGISTIC),
59  "For QASYMM8 only logistic, relu, lower bounded relu and lower-upper bounded relu are supported");
60  }
61  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(input, 1, DataType::F32, DataType::F16);
62  ARM_COMPUTE_RETURN_ERROR_ON_F16_UNSUPPORTED(input);
63 
64  ARM_COMPUTE_RETURN_ERROR_ON(output->data_layout() != winograd_info.output_data_layout);
65 
66  const PadStrideInfo conv_info = winograd_info.convolution_info;
67  const Size2D output_tile_size = winograd_info.output_tile_size;
68  const Size2D kernel_size = winograd_info.kernel_size;
69  const Size2D input_dimensions = winograd_info.input_dimensions;
70  const unsigned int num_channels = (winograd_info.kernel_size.width + winograd_info.output_tile_size.width - 1) * (winograd_info.kernel_size.height + winograd_info.output_tile_size.height - 1);
71 
72  ARM_COMPUTE_RETURN_ERROR_ON_MSG(!cl_winograd_convolution_layer_supported(output_tile_size, kernel_size, winograd_info.output_data_layout), "Winograd output transform not supported");
73  ARM_COMPUTE_RETURN_ERROR_ON_MSG(input->dimension(2) != num_channels, "Wrong number of channels");
74 
75  // Compute number of elements to process in the X and Y direction
76  // Compute the number of output tiles along the x and y direction of size "output_tile_size"
77  const Size2D num_tiles = compute_winograd_convolution_tiles(input_dimensions,
78  kernel_size,
79  output_tile_size,
80  conv_info);
81 
82  ARM_COMPUTE_RETURN_ERROR_ON(input->dimension(1) != static_cast<unsigned int>((num_tiles.area())));
83 
84  if(bias != nullptr)
85  {
86  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input, bias);
87  ARM_COMPUTE_RETURN_ERROR_ON(input->dimension(0) != bias->dimension(0));
88  }
89 
90  // Checks performed when output is configured
91  if(output->total_size() != 0)
92  {
93  const TensorInfo tensor_info_output = input->clone()->set_tensor_shape(compute_winograd_output_transform_shape(*input, winograd_info));
94 
95  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_SHAPES(output, &tensor_info_output);
96  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input, output);
97  }
98 
99  return Status{};
100 }

102 std::pair<Status, Window> validate_and_configure_window(ITensorInfo *input, ITensorInfo *bias, ITensorInfo *output, const Size2D &output_tile_size)
103 {
104  ARM_COMPUTE_ERROR_ON_NULLPTR(input, output);
105 
106  constexpr unsigned int num_elems_processed_per_iteration = 1;
107 
108  Window win = calculate_max_window(*input, Steps(num_elems_processed_per_iteration));
109  bool window_changed = false;
110 
111  int output_static_window_end_x = 0;
112  int output_static_window_end_y = 0;
113 
114  if(output->data_layout() == DataLayout::NCHW)
115  {
116  output_static_window_end_x = ceil_to_multiple(output->dimension(0), output_tile_size.width);
117  output_static_window_end_y = ceil_to_multiple(output->dimension(1), output_tile_size.height);
118  }
119  else
120  {
121  output_static_window_end_x = output->dimension(0);
122  output_static_window_end_y = std::max(ceil_to_multiple(output->dimension(1), output_tile_size.width), output->dimension(1) + 1 /* For out of bound reads towards the z axis */);
123  }
124 
125  AccessWindowRectangle input_access(input, 0, 0, num_elems_processed_per_iteration, num_elems_processed_per_iteration);
126  AccessWindowStatic output_access(output, 0, 0, output_static_window_end_x, output_static_window_end_y);
127  window_changed = update_window_and_padding(win, input_access, output_access);
128  output->set_valid_region(ValidRegion(Coordinates(), output->tensor_shape()));
129 
130  if(bias != nullptr)
131  {
132  AccessWindowStatic bias_access(bias, 0, 0, bias->dimension(0), bias->dimension(1));
133  window_changed = window_changed || update_window_and_padding(win, bias_access);
134  }
135 
136  Status err = (window_changed) ? ARM_COMPUTE_CREATE_ERROR(ErrorCode::RUNTIME_ERROR, "Insufficient Padding!") : Status{};
137  return std::make_pair(err, win);
138 }

146 void CLWinogradOutputTransformKernel::configure(const ICLTensor *input, const ICLTensor *bias, ICLTensor *output, const WinogradInfo &winograd_info, const ActivationLayerInfo &act_info)
147 {
148  ARM_COMPUTE_ERROR_ON_NULLPTR(input, output);
149 
150  // Output tensor auto initialization if not yet initialized
151  auto_init_if_empty(*output->info(), input->info()->clone()->set_tensor_shape(compute_winograd_output_transform_shape(*input->info(), winograd_info)));
152 
153  ARM_COMPUTE_ERROR_THROW_ON(validate_arguments(input->info(), (bias != nullptr ? bias->info() : nullptr), output->info(), winograd_info, act_info));
154 
155  _input = input;
156  _bias = bias;
157  _output = output;
158  _is_nhwc = winograd_info.output_data_layout == DataLayout::NHWC;
159 
160  // Compute num_tiles_x
161  const Size2D input_dimensions = winograd_info.input_dimensions;
162  const Size2D kernel_size = winograd_info.kernel_size;
163  const Size2D output_tile_size = winograd_info.output_tile_size;
164  const PadStrideInfo conv_info = winograd_info.convolution_info;
165 
166  // Compute the number of output tiles along the x and y direction of size "output_tile_size"
167  const Size2D num_tiles = compute_winograd_convolution_tiles(input_dimensions,
168  kernel_size,
169  output_tile_size,
170  conv_info);
171  const size_t total_batches = output->info()->tensor_shape().total_size_upper(3);
172 
173  // Set build options
174  CLBuildOptions build_opts;
175  build_opts.add_option("-DACTIVATION_TYPE=" + lower_string(string_from_activation_func(act_info.activation())));
176  build_opts.add_option_if(act_info.enabled(), "-DA_VAL=" + float_to_string_with_full_precision(act_info.a()));
177  build_opts.add_option_if(act_info.enabled(), "-DB_VAL=" + float_to_string_with_full_precision(act_info.b()));
178 
179  if((output_tile_size.x() == 2) || (output_tile_size.x() == 1 && output_tile_size.y() == 2))
180  {
181  build_opts.add_option("-DVEC_SIZE=2");
182  }
183  else if((output_tile_size.x() == 4) || (output_tile_size.x() == 1 && output_tile_size.y() == 4))
184  {
185  build_opts.add_option("-DVEC_SIZE=4");
186  }
187 
188  build_opts.add_option_if(_bias != nullptr, std::string("-DHAS_BIAS"));
189  build_opts.add_option("-DNUM_TILES_X=" + support::cpp11::to_string(num_tiles.width));
190  build_opts.add_option("-DOUTPUT_TILE_W=" + support::cpp11::to_string(output_tile_size.width));
191  build_opts.add_option("-DOUTPUT_TILE_H=" + support::cpp11::to_string(output_tile_size.height));
192  build_opts.add_option("-DDATA_TYPE=" + get_cl_type_from_data_type(input->info()->data_type()));
193  build_opts.add_option_if(total_batches > 1, "-DSRC_DEPTH=" + support::cpp11::to_string(_input->info()->dimension(2)));
194  build_opts.add_option_if(winograd_info.kernel_size.height == 1, "-DWINOGRAD_OUTPUT_TRANSFORM_HORIZONTAL");
195  build_opts.add_option_if(winograd_info.kernel_size.width == 1, "-DWINOGRAD_OUTPUT_TRANSFORM_VERTICAL");
196 
197  // Create kernel
198  std::string kernel_name = "winograd_output_transform_" + output_tile_size.to_string() + "_" + kernel_size.to_string() + "_" + lower_string(string_from_data_layout(winograd_info.output_data_layout));
199  _kernel = static_cast<cl::Kernel>(CLKernelLibrary::get().create_kernel(kernel_name, build_opts.options()));
200 
201  // Configure kernel window
202  auto win_config = validate_and_configure_window(input->info(), (bias != nullptr ? bias->info() : nullptr), output->info(), winograd_info.output_tile_size);
203  ARM_COMPUTE_ERROR_THROW_ON(win_config.first);
204  ICLKernel::configure_internal(win_config.second);
205 
206  // Set config_id for enabling LWS tuning
207  _config_id = kernel_name;
208  _config_id += "_";
209  _config_id += lower_string(string_from_data_type(input->info()->data_type()));
210  _config_id += "_";
211  _config_id += support::cpp11::to_string(input->info()->dimension(0));
212  _config_id += "_";
213  _config_id += support::cpp11::to_string(input->info()->dimension(1));
214  _config_id += "_";
215  _config_id += support::cpp11::to_string(output->info()->dimension(0));
216  _config_id += "_";
217  _config_id += support::cpp11::to_string(output->info()->dimension(1));
218  _config_id += "_";
219  _config_id += lower_string(string_from_data_layout(winograd_info.output_data_layout));
220 }

230 void CLWinogradOutputTransformKernel::run(const Window &window, cl::CommandQueue &queue)
231 {
232  ARM_COMPUTE_ERROR_ON_UNCONFIGURED_KERNEL(this);
233  ARM_COMPUTE_ERROR_ON_INVALID_SUBWINDOW(ICLKernel::window(), window);
234 
235  // Collapse window
236  Window window_collapsed = window.collapse_if_possible(ICLKernel::window(), Window::DimZ);
237 
238  // Get initial windows
239  Window slice = window_collapsed.first_slice_window_4D();
240  slice.set(Window::DimZ, Window::Dimension(0, 1, 1));
241 
242  // Setup output slice
243  Window slice_out(slice);
244  slice_out.set(Window::DimX, Window::Dimension(0, 0, 0));
245  slice_out.set(Window::DimY, Window::Dimension(0, 0, 0));
246 
247  if(_bias != nullptr)
248  {
249  unsigned int idx1 = 2 * num_arguments_per_4D_tensor();
250  Window slice_biases;
251  slice_biases.use_tensor_dimensions(_bias->info()->tensor_shape());
252  add_1D_tensor_argument(idx1, _bias, slice_biases);
253  }
254 
255  if(_is_nhwc)
256  {
257  unsigned int idx2 = 2 * num_arguments_per_4D_tensor() + ((_bias != nullptr) ? num_arguments_per_1D_tensor() : 0);
258  _kernel.setArg(idx2, static_cast<int>(_output->info()->total_size() - _output->info()->strides_in_bytes().y()));
259  }
260 
261  do
262  {
263  unsigned int idx = 0;
264  add_4D_tensor_argument(idx, _input, slice);
265  add_4D_tensor_argument(idx, _output, slice_out);
266  enqueue(queue, *this, slice, lws_hint());
267  }
268  while(window.slide_window_slice_3D(slice) && window.slide_window_slice_3D(slice_out));
269 }
