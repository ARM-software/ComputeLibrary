82 template <>
83 std::unique_ptr<IFunction> create_convolution_layer<NEConvolutionLayerFunctions, NETargetInfo>(ConvolutionLayerNode &node,
84  GraphContext &ctx)
85 {
86  validate_node<NETargetInfo>(node, 3 /* expected inputs */, 1 /* expected outputs */);
87 
88  // Extract IO and info
89  NETargetInfo::TensorType *input = get_backing_tensor<NETargetInfo>(node.input(0));
90  NETargetInfo::TensorType *weights = get_backing_tensor<NETargetInfo>(node.input(1));
91  NETargetInfo::TensorType *biases = get_backing_tensor<NETargetInfo>(node.input(2));
92  NETargetInfo::TensorType *output = get_backing_tensor<NETargetInfo>(node.output(0));
93 
94  const bool is_quantized = is_data_type_quantized_asymmetric(input->info()->data_type());
95 
96  if(is_quantized)
97  {
98  biases->info()->set_data_type(DataType::S32);
99  }
100 
101  const PadStrideInfo conv_info = node.convolution_info();
102  const ConvolutionMethod conv_algorithm = node.convolution_method();
103  const ActivationLayerInfo fused_act = node.fused_activation();
104 
105  // Create and configure function (we assume that functions have been validated before creation)
106  std::shared_ptr<IMemoryManager> mm = get_memory_manager(ctx, Target::NEON);
107  std::unique_ptr<IFunction> func;
108  std::string func_name;
109  if(conv_algorithm == ConvolutionMethod::Direct)
110  {
111  std::tie(func, func_name) = create_named_memory_managed_function<NEDirectConvolutionLayer>(
112  std::string("DirectConvolutionLayer"), mm, input, weights, biases, output, conv_info, fused_act);
113  }
114  else if(conv_algorithm == ConvolutionMethod::GEMM)
115  {
116  std::tie(func, func_name) = create_named_memory_managed_function<NEGEMMConvolutionLayer>(
117  std::string("GEMMConvolutionLayer"), mm, input, weights, biases, output, conv_info, WeightsInfo(), Size2D(1, 1), fused_act);
118  }
119  else if(conv_algorithm == ConvolutionMethod::Winograd)
120  {
121  std::tie(func, func_name) = create_named_memory_managed_function<NEWinogradConvolutionLayer>(
122  std::string("WinogradConvolutionLayer"), mm, input, weights, biases, output, conv_info, fused_act);
123  }
124  else
125  {
126  std::tie(func, func_name) = create_named_memory_managed_function<NEConvolutionLayer>(
127  std::string("ConvolutionLayer"), mm, input, weights, biases, output, conv_info, WeightsInfo(), Size2D(1, 1), fused_act);
128  }
129 
130  // Log info
131  std::ostringstream qss;
132  if(is_quantized)
133  {
134  qss << " Input QuantInfo: " << input->info()->quantization_info()
135  << " Weights QuantInfo: " << weights->info()->quantization_info()
136  << " Output QuantInfo: " << output->info()->quantization_info();
137  }
138  ARM_COMPUTE_LOG_GRAPH_INFO("Instantiated " << func_name
139  << " Target " << NETargetInfo::TargetType
140  << " Data Type: " << input->info()->data_type()
141  << qss.str()
142  << " Input shape: " << input->info()->tensor_shape()
143  << " Weights shape: " << weights->info()->tensor_shape()
144  << " Output shape: " << output->info()->tensor_shape()
145  << (fused_act.enabled() ? " " + to_string(fused_act.activation()) : "")
146  << std::endl);
147  return func;
148 }

150 template <>
151 std::unique_ptr<IFunction> create_normalization_layer<NENormalizationLayer, NETargetInfo>(NormalizationLayerNode &node, GraphContext &ctx)
152 {
153  validate_node<NETargetInfo>(node, 1 /* expected inputs */, 1 /* expected outputs */);
154 
155  // Extract IO and info
156  NETargetInfo::TensorType *input = get_backing_tensor<NETargetInfo>(node.input(0));
157  NETargetInfo::TensorType *output = get_backing_tensor<NETargetInfo>(node.output(0));
158  const NormalizationLayerInfo norm_info = node.normalization_info();
159  ARM_COMPUTE_ERROR_ON(input == nullptr);
160  ARM_COMPUTE_ERROR_ON(output == nullptr);
161 
162  // Create and configure function
163  auto func = support::cpp14::make_unique<NENormalizationLayer>(get_memory_manager(ctx, NETargetInfo::TargetType));
164  func->configure(input, output, norm_info);
165 
166  // Log info
167  ARM_COMPUTE_LOG_GRAPH_INFO("Instantiated "
168  << node.name()
169  << " Type: " << node.type()
170  << " Target: " << NETargetInfo::TargetType
171  << " Data Type: " << input->info()->data_type()
172  << " Input shape: " << input->info()->tensor_shape()
173  << " Output shape: " << output->info()->tensor_shape()
174  << " Normalization info: " << norm_info.type()
175  << std::endl);
176 
177  return std::move(func);
178 }

181 std::unique_ptr<IFunction> NEFunctionFactory::create(INode *node, GraphContext &ctx)
182 {
183  if(node == nullptr)
184  {
185  return nullptr;
186  }
187 
188  NodeType type = node->type();
189  switch(type)
190  {
191  case NodeType::ActivationLayer:
192  return detail::create_activation_layer<NEActivationLayer, NETargetInfo>(*polymorphic_downcast<ActivationLayerNode *>(node));
193  case NodeType::BatchNormalizationLayer:
194  return detail::create_batch_normalization_layer<NEBatchNormalizationLayer, NETargetInfo>(*polymorphic_downcast<BatchNormalizationLayerNode *>(node));
195  case NodeType::ChannelShuffleLayer:
196  return detail::create_channel_shuffle_layer<NEChannelShuffleLayer, NETargetInfo>(*polymorphic_downcast<ChannelShuffleLayerNode *>(node));
197  case NodeType::ConvolutionLayer:
198  return detail::create_convolution_layer<NEConvolutionLayerFunctions, NETargetInfo>(*polymorphic_downcast<ConvolutionLayerNode *>(node), ctx);
199  case NodeType::DeconvolutionLayer:
200  return detail::create_deconvolution_layer<NEDeconvolutionLayer, NETargetInfo>(*polymorphic_downcast<DeconvolutionLayerNode *>(node), ctx);
201  case NodeType::ConcatenateLayer:
202  return detail::create_concatenate_layer<NEConcatenateLayer, NETargetInfo>(*polymorphic_downcast<ConcatenateLayerNode *>(node));
203  case NodeType::DepthwiseConvolutionLayer:
204  return detail::create_depthwise_convolution_layer<NEDepthwiseConvolutionLayerFunctions, NETargetInfo>(*polymorphic_downcast<DepthwiseConvolutionLayerNode *>(node));
205  case NodeType::DetectionOutputLayer:
206  return detail::create_detection_output_layer<CPPDetectionOutputLayer, NETargetInfo>(*polymorphic_downcast<DetectionOutputLayerNode *>(node));
207  case NodeType::EltwiseLayer:
208  return detail::create_eltwise_layer<NEEltwiseFunctions, NETargetInfo>(*polymorphic_downcast<EltwiseLayerNode *>(node));
209  case NodeType::FlattenLayer:
210  return detail::create_flatten_layer<NEFlattenLayer, NETargetInfo>(*polymorphic_downcast<FlattenLayerNode *>(node));
211  case NodeType::FullyConnectedLayer:
212  return detail::create_fully_connected_layer<NEFullyConnectedLayer, NETargetInfo>(*polymorphic_downcast<FullyConnectedLayerNode *>(node), ctx);
213  case NodeType::NormalizationLayer:
214  return detail::create_normalization_layer<NENormalizationLayer, NETargetInfo>(*polymorphic_downcast<NormalizationLayerNode *>(node), ctx);
215  case NodeType::PermuteLayer:
216  return detail::create_permute_layer<NEPermute, NETargetInfo>(*polymorphic_downcast<PermuteLayerNode *>(node));
217  case NodeType::PoolingLayer:
218  return detail::create_pooling_layer<NEPoolingLayer, NETargetInfo>(*polymorphic_downcast<PoolingLayerNode *>(node));
219  case NodeType::PriorBoxLayer:
220  return detail::create_priorbox_layer<NEPriorBoxLayer, NETargetInfo>(*polymorphic_downcast<PriorBoxLayerNode *>(node));
221  case NodeType::ReorgLayer:
222  return detail::create_reorg_layer<NEReorgLayer, NETargetInfo>(*polymorphic_downcast<ReorgLayerNode *>(node));
223  case NodeType::ReshapeLayer:
224  return detail::create_reshape_layer<NEReshapeLayer, NETargetInfo>(*polymorphic_downcast<ReshapeLayerNode *>(node));
225  case NodeType::ResizeLayer:
226  return detail::create_resize_layer<NEScale, NETargetInfo>(*polymorphic_downcast<ResizeLayerNode *>(node));
227  case NodeType::SoftmaxLayer:
228  return detail::create_softmax_layer<NESoftmaxLayer, NETargetInfo>(*polymorphic_downcast<SoftmaxLayerNode *>(node), ctx);
229  case NodeType::UpsampleLayer:
230  return detail::create_upsample_layer<NEUpsampleLayer, NETargetInfo>(*polymorphic_downcast<UpsampleLayerNode *>(node), ctx);
231  case NodeType::YOLOLayer:
232  return detail::create_yolo_layer<NEYOLOLayer, NETargetInfo>(*polymorphic_downcast<YOLOLayerNode *>(node), ctx);
233  default:
234  return nullptr;
235  }
