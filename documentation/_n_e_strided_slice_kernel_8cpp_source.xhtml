41 Status validate_arguments(const ITensorInfo *input, const ITensorInfo *output,
42  const Coordinates &starts, const Coordinates &ends, const BiStrides &strides,
43  int32_t begin_mask, int32_t end_mask, int32_t shrink_axis_mask)
44 {
45  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(input, output);
46  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(input, 1,
47  DataType::U8, DataType::S8, DataType::QASYMM8,
48  DataType::U16, DataType::S16, DataType::QSYMM16,
49  DataType::U32, DataType::S32,
50  DataType::F16, DataType::F32);
51 
52  ARM_COMPUTE_RETURN_ERROR_ON(input->tensor_shape().num_dimensions() > 4);
53  ARM_COMPUTE_RETURN_ERROR_ON(starts.num_dimensions() > input->num_dimensions());
54  ARM_COMPUTE_RETURN_ERROR_ON(ends.num_dimensions() > input->num_dimensions());
55  ARM_COMPUTE_RETURN_ERROR_ON(strides.num_dimensions() > input->num_dimensions());
56  ARM_COMPUTE_RETURN_ERROR_ON(std::any_of(strides.cbegin(), strides.cbegin() + strides.num_dimensions(), [](int i)
57  {
58  return i == 0;
59  }));
60 
61  // Get expected output shape
62  const TensorShape exp_output_shape = arm_compute::misc::shape_calculator::compute_strided_slice_shape(*input,
63  starts, ends, strides,
64  begin_mask, end_mask, shrink_axis_mask);
65  ARM_COMPUTE_RETURN_ERROR_ON(exp_output_shape.total_size() == 0);
66 
67  // Checks output if configured
68  if(output->total_size() != 0)
69  {
70  const TensorInfo exp_output_info = output->clone()->set_tensor_shape(exp_output_shape);
71  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_SHAPES(output, &exp_output_info);
72  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input, output);
73  }
74 
75  return Status{};
76 }

78 std::pair<Status, Window> validate_and_configure_window(ITensorInfo *input, ITensorInfo *output,
79  const Coordinates &starts, const Coordinates &ends, const BiStrides &strides,
80  int32_t begin_mask, int32_t end_mask, int32_t shrink_axis_mask)
81 {

84  starts, ends, strides,

86  auto_init_if_empty(*output, input->clone()->set_tensor_shape(output_shape));

88  // Create window

92  return std::make_pair(Status{}, win);

95 void strided_slice_generic(const ITensor *input, ITensor *output,
96  const Coordinates &starts, const BiStrides &strides, int32_t shrink_axis_mask,
97  const Window &window)
98 {
99  Iterator output_it(output, window);
100  const size_t width_size = input->info()->element_size();
101 
102  const bool is_shrink_w = arm_compute::helpers::bit_ops::is_bit_set(shrink_axis_mask, 0);
103  const bool is_shrink_h = arm_compute::helpers::bit_ops::is_bit_set(shrink_axis_mask, 1);
104  const bool is_shrink_c = arm_compute::helpers::bit_ops::is_bit_set(shrink_axis_mask, 2);
105  const bool is_shrink_n = arm_compute::helpers::bit_ops::is_bit_set(shrink_axis_mask, 3);
106 
107  unsigned int index = 0;
108  const int idx_w = is_shrink_w ? 0 : index++;
109  const int idx_h = is_shrink_h ? 0 : index++;
110  const int idx_c = is_shrink_c ? 0 : index++;
111  const int idx_n = is_shrink_n ? 0 : index;
112 
113  BiStrides shrinked_strides;
114  shrinked_strides.set(0, is_shrink_w ? 0 : strides[0]);
115  shrinked_strides.set(1, is_shrink_h ? 0 : strides[1]);
116  shrinked_strides.set(2, is_shrink_c ? 0 : strides[2]);
117  shrinked_strides.set(3, is_shrink_n ? 0 : strides[3]);
118 
119  execute_window_loop(window, [&](const Coordinates & id)
120  {
121  const int w_coord = starts[0] + (id[idx_w] * shrinked_strides[0]);
122  const int h_coord = starts[1] + (id[idx_h] * shrinked_strides[1]);
123  const int c_coord = starts[2] + (id[idx_c] * shrinked_strides[2]);
124  const int n_coord = starts[3] + (id[idx_n] * shrinked_strides[3]);
125 
126  Coordinates in_coords(w_coord, h_coord, c_coord, n_coord);
127  std::copy_n(input->ptr_to_element(in_coords), width_size, output_it.ptr());
128  },
129  output_it);
130 }

138 void NEStridedSliceKernel::configure(const ITensor *input, ITensor *output,
139  const Coordinates &starts, const Coordinates &ends, const BiStrides &strides,
140  int32_t begin_mask, int32_t end_mask, int32_t shrink_axis_mask)
141 {
142  ARM_COMPUTE_ERROR_ON_NULLPTR(input, output);
143  ARM_COMPUTE_ERROR_THROW_ON(validate_arguments(input->info(), output->info(), starts, ends, strides, begin_mask, end_mask, shrink_axis_mask));
144 
145  _input = input;
146  _output = output;
147  _shrink_mask = shrink_axis_mask;
148 
149  const TensorShape &input_shape = input->info()->tensor_shape();
150 
151  Coordinates ends_abs;
152  std::tie(_starts_abs, ends_abs, _final_strides) = arm_compute::helpers::tensor_transform::calculate_strided_slice_coords(
153  input_shape,
154  starts, ends, strides,
155  begin_mask, end_mask, shrink_axis_mask);
156 
157  // Configure kernel window
158  auto win_config = validate_and_configure_window(input->info(), output->info(), starts, ends, strides, begin_mask, end_mask, shrink_axis_mask);
159  ARM_COMPUTE_ERROR_THROW_ON(win_config.first);
160  INEKernel::configure(win_config.second);
161 }

175 void NEStridedSliceKernel::run(const Window &window, const ThreadInfo &info)
176 {
177  ARM_COMPUTE_UNUSED(info);
178  ARM_COMPUTE_ERROR_ON_UNCONFIGURED_KERNEL(this);
179  ARM_COMPUTE_ERROR_ON_INVALID_SUBWINDOW(INEKernel::window(), window);
