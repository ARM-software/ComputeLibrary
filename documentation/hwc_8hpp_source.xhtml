378 template <typename T>
379 static inline int mali_ioctl(int fd, T &arg)
380 {
381  auto *hdr = &arg.header;
382  const int cmd = _IOC(_IOC_READ | _IOC_WRITE, LINUX_UK_BASE_MAGIC, hdr->id, sizeof(T));
383 
384  if(ioctl(fd, cmd, &arg))
385  return -1;
386  if(hdr->ret)
387  return -1;
388 
389  return 0;
