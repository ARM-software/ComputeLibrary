365 template<typename T>
366 static inline int mali_ioctl(int fd, T &arg)
367 {
368  auto *hdr = &arg.header;
369  const int cmd = _IOC(_IOC_READ | _IOC_WRITE, LINUX_UK_BASE_MAGIC, hdr->id, sizeof(T));
370 
371  if (ioctl(fd, cmd, &arg))
372  return -1;
373  if (hdr->ret)
374  return -1;
375 
376  return 0;
377 }
