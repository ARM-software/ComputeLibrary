39 template <typename T>
40 SimpleTensor<T> scale_core(const SimpleTensor<T> &in, float scale_x, float scale_y, InterpolationPolicy policy, BorderMode border_mode, T constant_border_value,
41  SamplingPolicy sampling_policy, bool ceil_policy_scale)
42 {
43  // Add 1 if ceil_policy_scale is true
44  const size_t round_value = ceil_policy_scale ? 1U : 0U;
45  TensorShape shape_scaled(in.shape());
46  shape_scaled.set(0, (in.shape()[0] + round_value) * scale_x);
47  shape_scaled.set(1, (in.shape()[1] + round_value) * scale_y);
48  SimpleTensor<T> out(shape_scaled, in.data_type());
49 
50  // Compute the ratio between source width/height and destination width/height
51  const auto wr = static_cast<float>(in.shape()[0]) / static_cast<float>(out.shape()[0]);
52  const auto hr = static_cast<float>(in.shape()[1]) / static_cast<float>(out.shape()[1]);
53 
54  const auto width = static_cast<int>(in.shape().x());
55  const auto height = static_cast<int>(in.shape().y());
56 
57  // Determine border size
58  const int border_size = (border_mode == BorderMode::UNDEFINED) ? 0 : 1;
59 
60  // Area interpolation behaves as Nearest Neighbour in case of up-sampling
61  if(policy == InterpolationPolicy::AREA && wr <= 1.f && hr <= 1.f)
62  {
63  policy = InterpolationPolicy::NEAREST_NEIGHBOR;
64  }
65 
66  for(int element_idx = 0, count = 0; element_idx < out.num_elements(); ++element_idx, ++count)
67  {
68  Coordinates id = index2coord(out.shape(), element_idx);
69  int idx = id.x();
70  int idy = id.y();
71  float x_src = 0;
72  float y_src = 0;
73 
74  switch(policy)
75  {
76  case InterpolationPolicy::NEAREST_NEIGHBOR:
77  {
78  switch(sampling_policy)
79  {
80  case SamplingPolicy::TOP_LEFT:
81  x_src = std::floor(idx * wr);
82  y_src = std::floor(idy * hr);
83  break;
84  case SamplingPolicy::CENTER:
85  //Calculate the source coords without -0.5f is equivalent to round the x_scr/y_src coords
86  x_src = (idx + 0.5f) * wr;
87  y_src = (idy + 0.5f) * hr;
88  break;
89  default:
90  ARM_COMPUTE_ERROR("Unsupported sampling policy.");
91  }
92 
93  id.set(0, x_src);
94  id.set(1, y_src);
95 
96  // If coordinates in range of tensor's width or height
97  if(is_valid_pixel_index(x_src, y_src, width, height, border_size))
98  {
99  out[element_idx] = tensor_elem_at(in, id, border_mode, constant_border_value);
100  }
101  break;
102  }
103  case InterpolationPolicy::BILINEAR:
104  {
105  switch(sampling_policy)
106  {
107  case SamplingPolicy::TOP_LEFT:
108  x_src = idx * wr;
109  y_src = idy * hr;
110  break;
111  case SamplingPolicy::CENTER:
112  x_src = (idx + 0.5f) * wr - 0.5f;
113  y_src = (idy + 0.5f) * hr - 0.5f;
114  break;
115  default:
116  ARM_COMPUTE_ERROR("Unsupported sampling policy.");
117  }
118 
119  id.set(0, std::floor(x_src));
120  id.set(1, std::floor(y_src));
121  if(is_valid_pixel_index(x_src, y_src, width, height, border_size))
122  {
123  out[element_idx] = bilinear_policy(in, id, x_src, y_src, border_mode, constant_border_value);
124  }
125  else
126  {
127  if(border_mode == BorderMode::CONSTANT)
128  {
129  out[element_idx] = constant_border_value;
130  }
131  else if(border_mode == BorderMode::REPLICATE)
132  {
133  id.set(0, utility::clamp<int>(x_src, 0, width - 1));
134  id.set(1, utility::clamp<int>(y_src, 0, height - 1));
135  out[element_idx] = in[coord2index(in.shape(), id)];
136  }
137  }
138  break;
139  }
140  case InterpolationPolicy::AREA:
141  {
142  int x_from = std::floor(idx * wr - 0.5f - x_src);
143  int y_from = std::floor(idy * hr - 0.5f - y_src);
144  int x_to = std::ceil((idx + 1) * wr - 0.5f - x_src);
145  int y_to = std::ceil((idy + 1) * hr - 0.5f - y_src);
146  const int xi = std::floor(x_src);
147  const int yi = std::floor(y_src);
148 
149  // Clamp position to borders
150  x_src = std::max(-static_cast<float>(border_size), std::min(x_src, static_cast<float>(width - 1 + border_size)));
151  y_src = std::max(-static_cast<float>(border_size), std::min(y_src, static_cast<float>(height - 1 + border_size)));
152 
153  // Clamp bounding box offsets to borders
154  x_from = ((x_src + x_from) < -border_size) ? -border_size : x_from;
155  y_from = ((y_src + y_from) < -border_size) ? -border_size : y_from;
156  x_to = ((x_src + x_to) >= (width + border_size)) ? (width - 1 + border_size) : x_to;
157  y_to = ((y_src + y_to) >= (height + border_size)) ? (height - 1 + border_size) : y_to;
158  ARM_COMPUTE_ERROR_ON((x_to - x_from + 1) == 0 || (y_to - y_from + 1) == 0);
159 
160  float sum = 0;
161  for(int j = yi + y_from, je = yi + y_to; j <= je; ++j)
162  {
163  for(int i = xi + x_from, ie = xi + x_to; i <= ie; ++i)
164  {
165  id.set(0, static_cast<int>(i));
166  id.set(1, static_cast<int>(j));
167  sum += tensor_elem_at(in, id, border_mode, constant_border_value);
168  }
169  }
170  out[element_idx] = sum / ((x_to - x_from + 1) * (y_to - y_from + 1));
171 
172  break;
173  }
174  default:
175  ARM_COMPUTE_ERROR("Unsupported interpolation mode");
176  }
177  }
178 
179  return out;
180 }

189 template <>
190 SimpleTensor<uint8_t> scale(const SimpleTensor<uint8_t> &src, float scale_x, float scale_y, InterpolationPolicy policy, BorderMode border_mode, uint8_t constant_border_value,
191  SamplingPolicy sampling_policy, bool ceil_policy_scale)
192 {
193  SimpleTensor<uint8_t> dst;
194  if(src.quantization_info().uniform().scale != 0.f)
195  {
196  SimpleTensor<float> src_tmp = convert_from_asymmetric(src);
197  float constant_border_value_f = dequantize_qasymm8(constant_border_value, src.quantization_info());
198  SimpleTensor<float> dst_tmp = scale_core<float>(src_tmp, scale_x, scale_y, policy, border_mode, constant_border_value_f, sampling_policy, ceil_policy_scale);
199  dst = convert_to_asymmetric(dst_tmp, src.quantization_info());
200  }
201  else
202  {
203  dst = scale_core<uint8_t>(src, scale_x, scale_y, policy, border_mode, constant_border_value, sampling_policy, ceil_policy_scale);
204  }
205  return dst;
206 }
