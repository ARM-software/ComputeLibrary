32 inline uint8_t pixel_area_c1u8_clamp(const uint8_t *first_pixel_ptr, size_t stride, size_t width, size_t height, float wr, float hr, int x, int y)
33 {
34  ARM_COMPUTE_ERROR_ON(first_pixel_ptr == nullptr);
35 
36  // Calculate sampling position
37  float in_x = (x + 0.5f) * wr - 0.5f;
38  float in_y = (y + 0.5f) * hr - 0.5f;
39 
40  // Get bounding box offsets
41  int x_from = std::floor(x * wr - 0.5f - in_x);
42  int y_from = std::floor(y * hr - 0.5f - in_y);
43  int x_to = std::ceil((x + 1) * wr - 0.5f - in_x);
44  int y_to = std::ceil((y + 1) * hr - 0.5f - in_y);
45 
46  // Clamp position to borders
47  in_x = std::max(-1.f, std::min(in_x, static_cast<float>(width)));
48  in_y = std::max(-1.f, std::min(in_y, static_cast<float>(height)));
49 
50  // Clamp bounding box offsets to borders
51  x_from = ((in_x + x_from) < -1) ? -1 : x_from;
52  y_from = ((in_y + y_from) < -1) ? -1 : y_from;
53  x_to = ((in_x + x_to) > width) ? (width - in_x) : x_to;
54  y_to = ((in_y + y_to) > height) ? (height - in_y) : y_to;
55 
56  // Get pixel index
57  const int xi = std::floor(in_x);
58  const int yi = std::floor(in_y);
59 
60  // Bounding box elements in each dimension
61  const int x_elements = (x_to - x_from + 1);
62  const int y_elements = (y_to - y_from + 1);
63  ARM_COMPUTE_ERROR_ON(x_elements == 0 || y_elements == 0);
64 
65  // Sum pixels in area
66  int sum = 0;
67  for(int j = yi + y_from, je = yi + y_to; j <= je; ++j)
68  {
69  const uint8_t *ptr = first_pixel_ptr + j * stride + xi + x_from;
70  sum = std::accumulate(ptr, ptr + x_elements, sum);
71  }
72 
73  // Return average
74  return sum / (x_elements * y_elements);
75 }

96 struct ForEachDimension
97 {
98  template <typename L, typename... Ts>
99  static void unroll(const Window &w, Coordinates &id, L &&lambda_function, Ts &&... iterators)
100  {
101  const auto &d = w[dim - 1];
102 
103  for(auto v = d.start(); v < d.end(); v += d.step(), IncrementIterators < dim - 1 >::unroll(iterators...))
104  {
105  id.set(dim - 1, v);

285 inline ValidRegion calculate_valid_region_scale(const ITensorInfo &src_info, const TensorShape &dst_shape, InterpolationPolicy policy, BorderSize border_size, bool border_undefined)
286 {
287  const auto wr = static_cast<float>(dst_shape[0]) / static_cast<float>(src_info.tensor_shape()[0]);
288  const auto hr = static_cast<float>(dst_shape[1]) / static_cast<float>(src_info.tensor_shape()[1]);
289 
290  ValidRegion valid_region{ Coordinates(), dst_shape, src_info.tensor_shape().num_dimensions() };
291 
292  Coordinates &anchor = valid_region.anchor;
293  TensorShape &shape = valid_region.shape;
294 
295  anchor.set(0, (policy == InterpolationPolicy::BILINEAR
296  && border_undefined) ?
297  ((static_cast<int>(src_info.valid_region().anchor[0]) + border_size.left + 0.5f) * wr - 0.5f) :
298  ((static_cast<int>(src_info.valid_region().anchor[0]) + 0.5f) * wr - 0.5f));
299  anchor.set(1, (policy == InterpolationPolicy::BILINEAR
300  && border_undefined) ?
301  ((static_cast<int>(src_info.valid_region().anchor[1]) + border_size.top + 0.5f) * hr - 0.5f) :
302  ((static_cast<int>(src_info.valid_region().anchor[1]) + 0.5f) * hr - 0.5f));
303  float shape_out_x = (policy == InterpolationPolicy::BILINEAR
304  && border_undefined) ?
305  ((static_cast<int>(src_info.valid_region().anchor[0]) + static_cast<int>(src_info.valid_region().shape[0]) - 1) - 1 + 0.5f) * wr - 0.5f :
306  ((static_cast<int>(src_info.valid_region().anchor[0]) + static_cast<int>(src_info.valid_region().shape[0])) + 0.5f) * wr - 0.5f;
307  float shape_out_y = (policy == InterpolationPolicy::BILINEAR
308  && border_undefined) ?
309  ((static_cast<int>(src_info.valid_region().anchor[1]) + static_cast<int>(src_info.valid_region().shape[1]) - 1) - 1 + 0.5f) * hr - 0.5f :
310  ((static_cast<int>(src_info.valid_region().anchor[1]) + static_cast<int>(src_info.valid_region().shape[1])) + 0.5f) * hr - 0.5f;
311 
312  shape.set(0, shape_out_x - anchor[0]);
313  shape.set(1, shape_out_y - anchor[1]);
314 
315  return valid_region;
316 }

318 inline Coordinates index2coords(const TensorShape &shape, int index)
319 {
320  int num_elements = shape.total_size();
321 
322  ARM_COMPUTE_ERROR_ON_MSG(index < 0 || index >= num_elements, "Index has to be in [0, num_elements]!");
323  ARM_COMPUTE_ERROR_ON_MSG(num_elements == 0, "Cannot create coordinate from empty shape!");
324 
325  Coordinates coord{ 0 };
326 
327  for(int d = shape.num_dimensions() - 1; d >= 0; --d)
328  {
329  num_elements /= shape[d];
330  coord.set(d, index / num_elements);
331  index %= num_elements;
332  }
333 
334  return coord;
335 }

337 inline int coords2index(const TensorShape &shape, const Coordinates &coord)
338 {
339  int num_elements = shape.total_size();
340  ARM_COMPUTE_UNUSED(num_elements);
341  ARM_COMPUTE_ERROR_ON_MSG(num_elements == 0, "Cannot create linear index from empty shape!");
342 
343  int index = 0;
344  int stride = 1;
345 
346  for(unsigned int d = 0; d < coord.num_dimensions(); ++d)
347  {
348  index += coord[d] * stride;
349  stride *= shape[d];
350  }
351 
352  return index;
