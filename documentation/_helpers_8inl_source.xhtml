32 inline uint8_t pixel_area_c1u8_clamp(const uint8_t *first_pixel_ptr, size_t stride, size_t width, size_t height, float wr, float hr, int x, int y)
33 {
34  ARM_COMPUTE_ERROR_ON(first_pixel_ptr == nullptr);
35 
36  // Calculate sampling position
37  float in_x = (x + 0.5f) * wr - 0.5f;
38  float in_y = (y + 0.5f) * hr - 0.5f;
39 
40  // Get bounding box offsets
41  int x_from = std::floor(x * wr - 0.5f - in_x);
42  int y_from = std::floor(y * hr - 0.5f - in_y);
43  int x_to = std::ceil((x + 1) * wr - 0.5f - in_x);
44  int y_to = std::ceil((y + 1) * hr - 0.5f - in_y);
45 
46  // Clamp position to borders
47  in_x = std::max(-1.f, std::min(in_x, static_cast<float>(width)));
48  in_y = std::max(-1.f, std::min(in_y, static_cast<float>(height)));
49 
50  // Clamp bounding box offsets to borders
51  x_from = ((in_x + x_from) < -1) ? -1 : x_from;
52  y_from = ((in_y + y_from) < -1) ? -1 : y_from;
53  x_to = ((in_x + x_to) > width) ? (width - in_x) : x_to;
54  y_to = ((in_y + y_to) > height) ? (height - in_y) : y_to;
55 
56  // Get pixel index
57  const int xi = std::floor(in_x);
58  const int yi = std::floor(in_y);
59 
60  // Bounding box elements in each dimension
61  const int x_elements = (x_to - x_from + 1);
62  const int y_elements = (y_to - y_from + 1);
63  ARM_COMPUTE_ERROR_ON(x_elements == 0 || y_elements == 0);
64 
65  // Sum pixels in area
66  int sum = 0;
67  for(int j = yi + y_from, je = yi + y_to; j <= je; ++j)
68  {
69  const uint8_t *ptr = first_pixel_ptr + j * stride + xi + x_from;
70  sum = std::accumulate(ptr, ptr + x_elements, sum);
71  }
72 
73  // Return average
74  return sum / (x_elements * y_elements);
75 }

101 struct ForEachDimension
102 {
103  template <typename L, typename... Ts>
104  static void unroll(const Window &w, Coordinates &id, L &&lambda_function, Ts &&... iterators)
105  {
106  const auto &d = w[dim - 1];
107 
108  for(auto v = d.start(); v < d.end(); v += d.step(), IncrementIterators < dim - 1 >::unroll(iterators...))
109  {
110  id.set(dim - 1, v);
111  ForEachDimension < dim - 1 >::unroll(w, id, lambda_function, iterators...);
112  }
113  }

290 inline ValidRegion calculate_valid_region_scale(const ITensorInfo &src_info, const TensorShape &dst_shape, InterpolationPolicy policy, BorderSize border_size, bool border_undefined)
291 {
292  const auto wr = static_cast<float>(dst_shape[0]) / static_cast<float>(src_info.tensor_shape()[0]);
293  const auto hr = static_cast<float>(dst_shape[1]) / static_cast<float>(src_info.tensor_shape()[1]);
294  Coordinates anchor;
295  anchor.set_num_dimensions(src_info.tensor_shape().num_dimensions());
296  TensorShape new_dst_shape(dst_shape);
297  anchor.set(0, (policy == InterpolationPolicy::BILINEAR
298  && border_undefined) ?
299  ((static_cast<int>(src_info.valid_region().anchor[0]) + border_size.left + 0.5f) * wr - 0.5f) :
300  ((static_cast<int>(src_info.valid_region().anchor[0]) + 0.5f) * wr - 0.5f));
301  anchor.set(1, (policy == InterpolationPolicy::BILINEAR
302  && border_undefined) ?
303  ((static_cast<int>(src_info.valid_region().anchor[1]) + border_size.top + 0.5f) * hr - 0.5f) :
304  ((static_cast<int>(src_info.valid_region().anchor[1]) + 0.5f) * hr - 0.5f));
305  float shape_out_x = (policy == InterpolationPolicy::BILINEAR
306  && border_undefined) ?
307  ((static_cast<int>(src_info.valid_region().anchor[0]) + static_cast<int>(src_info.valid_region().shape[0]) - 1) - 1 + 0.5f) * wr - 0.5f :
308  ((static_cast<int>(src_info.valid_region().anchor[0]) + static_cast<int>(src_info.valid_region().shape[0])) + 0.5f) * wr - 0.5f;
309  float shape_out_y = (policy == InterpolationPolicy::BILINEAR
310  && border_undefined) ?
311  ((static_cast<int>(src_info.valid_region().anchor[1]) + static_cast<int>(src_info.valid_region().shape[1]) - 1) - 1 + 0.5f) * hr - 0.5f :
312  ((static_cast<int>(src_info.valid_region().anchor[1]) + static_cast<int>(src_info.valid_region().shape[1])) + 0.5f) * hr - 0.5f;
313 
314  new_dst_shape.set(0, shape_out_x - anchor[0]);
315  new_dst_shape.set(1, shape_out_y - anchor[1]);
316 
317  return ValidRegion(std::move(anchor), std::move(new_dst_shape));
318 }

320 inline Coordinates index2coords(const TensorShape &shape, int index)
321 {
322  int num_elements = shape.total_size();
323 
324  ARM_COMPUTE_ERROR_ON_MSG(index < 0 || index >= num_elements, "Index has to be in [0, num_elements]!");
325  ARM_COMPUTE_ERROR_ON_MSG(num_elements == 0, "Cannot create coordinate from empty shape!");
326 
327  Coordinates coord{ 0 };
328 
329  for(int d = shape.num_dimensions() - 1; d >= 0; --d)
330  {
331  num_elements /= shape[d];
332  coord.set(d, index / num_elements);
333  index %= num_elements;
334  }
335 
336  return coord;
337 }

339 inline int coords2index(const TensorShape &shape, const Coordinates &coord)
340 {
341  int num_elements = shape.total_size();
342  ARM_COMPUTE_UNUSED(num_elements);
343  ARM_COMPUTE_ERROR_ON_MSG(num_elements == 0, "Cannot create linear index from empty shape!");
344 
345  int index = 0;
346  int stride = 1;
347 
348  for(unsigned int d = 0; d < coord.num_dimensions(); ++d)
349  {
350  index += coord[d] * stride;
351  stride *= shape[d];
352  }
353 
354  return index;
