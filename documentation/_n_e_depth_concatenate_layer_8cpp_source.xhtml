46 void NEDepthConcatenateLayer::configure(const std::vector<ITensor *> &inputs_vector, ITensor *output) // NOLINT
47 {
48  _num_inputs = inputs_vector.size();
49 
50  std::vector<ITensorInfo *> inputs_vector_info;
51  for(unsigned int i = 0; i < _num_inputs; i++)
52  {
53  inputs_vector_info.emplace_back(inputs_vector.at(i)->info());
54  }
55  TensorShape output_shape = arm_compute::misc::shape_calculator::calculate_concatenate_shape(inputs_vector_info, Window::DimZ);
56 
57  // Output auto inizialitation if not yet initialized
58  auto_init_if_empty(*output->info(), output_shape, 1, inputs_vector[0]->info()->data_type());
59  ARM_COMPUTE_ERROR_THROW_ON(NEDepthConcatenateLayer::validate(inputs_vector_info, output->info()));
60 
61  unsigned int depth_offset = 0;
62  _concat_kernels_vector.reserve(_num_inputs);
63  _border_handlers_vector.reserve(_num_inputs);
64  for(unsigned int i = 0; i < _num_inputs; ++i)
65  {
66  auto concat_kernel = support::cpp14::make_unique<NEDepthConcatenateLayerKernel>();
67  auto border_kernel = support::cpp14::make_unique<NEFillBorderKernel>();
68  concat_kernel->configure(inputs_vector.at(i), depth_offset, output);
69  border_kernel->configure(inputs_vector.at(i), concat_kernel->border_size(), BorderMode::CONSTANT, PixelValue(static_cast<float>(0.f)));
70  _border_handlers_vector.emplace_back(std::move(border_kernel));
71  _concat_kernels_vector.emplace_back(std::move(concat_kernel));
72 
73  depth_offset += inputs_vector.at(i)->info()->dimension(2);
74  }
75 
76  // Set valid region from shape
77  output->info()->set_valid_region(ValidRegion(Coordinates(), output_shape));
78 }

80 Status NEDepthConcatenateLayer::validate(const std::vector<ITensorInfo *> &inputs_vector, const ITensorInfo *output)
81 {
82  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(output);
83  ARM_COMPUTE_RETURN_ERROR_ON(inputs_vector.size() < 2);
84 
85  // Output auto inizialitation if not yet initialized
86  TensorInfo tmp_output_info = *output->clone();
87  TensorShape output_shape = arm_compute::misc::shape_calculator::calculate_concatenate_shape(inputs_vector, Window::DimZ);
88  auto_init_if_empty(tmp_output_info, output_shape, 1, inputs_vector[0]->data_type());
89 
90  unsigned int depth_offset = 0;
91  for(const auto &input : inputs_vector)
92  {
93  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(input);
94  ARM_COMPUTE_RETURN_ON_ERROR(NEDepthConcatenateLayerKernel::validate(input, depth_offset, &tmp_output_info));
95  depth_offset += input->dimension(2);
96  }
97 
98  return Status{};
99 }
