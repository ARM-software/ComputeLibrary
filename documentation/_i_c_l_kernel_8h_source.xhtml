296 template <typename T, unsigned int dimension_size>
297 void ICLKernel::add_array_argument(unsigned &idx, const ICLArray<T> *array, const Strides &strides, unsigned int num_dimensions, const Window &window)
298 {
299  ARM_COMPUTE_ERROR_ON(array == nullptr);
300 
301  // Calculate offset to the start of the window
302  unsigned int offset_first_element = 0;
303 
304  for(unsigned int n = 0; n < num_dimensions; ++n)
305  {
306  offset_first_element += window[n].start() * strides[n];
307  }
308 
309  unsigned int idx_start = idx;
310  _kernel.setArg(idx++, array->cl_buffer());
311 
312  for(unsigned int dimension = 0; dimension < dimension_size; dimension++)
313  {
314  _kernel.setArg<cl_uint>(idx++, strides[dimension]);
315  _kernel.setArg<cl_uint>(idx++, strides[dimension] * window[dimension].step());
316  }
317 
318  _kernel.setArg<cl_uint>(idx++, offset_first_element);
319 
320  ARM_COMPUTE_ERROR_ON_MSG(idx_start + num_arguments_per_array<dimension_size>() != idx,
321  "add_%dD_array_argument() is supposed to add exactly %d arguments to the kernel", dimension_size, num_arguments_per_array<dimension_size>());
322  ARM_COMPUTE_UNUSED(idx_start);
323 }
324 }
