293 template <typename T>
294 int SimpleTensor<T>::num_channels() const
295 {
296  switch(_format)
297  {
298  case Format::U8:
299  case Format::S16:
300  case Format::U16:
301  case Format::S32:
302  case Format::U32:
303  case Format::F32:
304  return 1;
305  case Format::RGB888:
306  return 3;
307  case Format::UNKNOWN:
308  return _num_channels;
309  default:
310  ARM_COMPUTE_ERROR("NOT SUPPORTED!");
311  }
