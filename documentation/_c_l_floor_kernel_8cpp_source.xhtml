41 Status validate_arguments(const ITensorInfo *input, const ITensorInfo *output)
42 {
43  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(input, output);
44  ARM_COMPUTE_RETURN_ERROR_ON_F16_UNSUPPORTED(input);
45  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(input, 1, DataType::F16, DataType::F32);
46 
47  // Validate in case of configured output

50  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input, output);
51  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_SHAPES(input, output);
52  }
53 
54  return Status{};
55 }

57 std::pair<Status, Window> validate_and_configure_window(ITensorInfo *input, ITensorInfo *output)
58 {
59  auto_init_if_empty(*output, *input);
60 
61  const unsigned int num_elems_processed_per_iteration = 16 / input->element_size();
62 
63  Window win = calculate_max_window(*input, Steps(num_elems_processed_per_iteration));
64  AccessWindowHorizontal input_access(input, 0, num_elems_processed_per_iteration);
65  AccessWindowHorizontal output_access(output, 0, num_elems_processed_per_iteration);
66  bool window_changed = update_window_and_padding(win, input_access, output_access);
67  output_access.set_valid_region(win, input->valid_region());
68 
69  Status err = (window_changed) ? ARM_COMPUTE_CREATE_ERROR(ErrorCode::RUNTIME_ERROR, "Insufficient Padding!") : Status{};
70  return std::make_pair(err, win);
71 }

79 void CLFloorKernel::configure(const ICLTensor *input, ICLTensor *output)
80 {
81  ARM_COMPUTE_ERROR_ON_NULLPTR(input, output);
82 
83  // Auto initialize output
84  auto_init_if_empty(*output->info(), input->info()->tensor_shape(), 1, input->info()->data_type());
85 
86  // Validate
87  ARM_COMPUTE_ERROR_THROW_ON(validate_arguments(input->info(), output->info()));
88 
89  _input = input;
90  _output = output;
91 
92  const unsigned int num_elems_processed_per_iteration = 16 / input->info()->element_size();
93 
94  // Create kernel
95  std::set<std::string> build_opts;
96  build_opts.emplace(("-DDATA_TYPE=" + get_cl_type_from_data_type(input->info()->data_type())));
97  build_opts.emplace(("-DVEC_SIZE=" + support::cpp11::to_string(num_elems_processed_per_iteration)));
98  _kernel = static_cast<cl::Kernel>(CLKernelLibrary::get().create_kernel("floor_layer", build_opts));
99 
100  // Configure kernel window
101  auto win_config = validate_and_configure_window(input->info(), output->info());
102  ARM_COMPUTE_ERROR_THROW_ON(win_config.first);
103  ICLKernel::configure_internal(win_config.second);
104 }

114 void CLFloorKernel::run(const Window &window, cl::CommandQueue &queue)
115 {
116  ARM_COMPUTE_ERROR_ON_UNCONFIGURED_KERNEL(this);
117  ARM_COMPUTE_ERROR_ON_INVALID_SUBWINDOW(ICLKernel::window(), window);
118 
119  Window collapsed = window.collapse_if_possible(ICLKernel::window(), Window::DimZ);
120  Window slice = collapsed.first_slice_window_3D();
121 
122  do
123  {
124  unsigned int idx = 0;
125  add_3D_tensor_argument(idx, _input, slice);
126  add_3D_tensor_argument(idx, _output, slice);
127  enqueue(queue, *this, slice);
128  }
129  while(collapsed.slide_window_slice_3D(slice));
130 }
