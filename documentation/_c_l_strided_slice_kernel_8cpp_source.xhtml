43 Status validate_arguments(const ITensorInfo *input, const ITensorInfo *output,
44  const Coordinates &starts, const Coordinates &ends, const BiStrides &strides,
45  int32_t begin_mask, int32_t end_mask, int32_t shrink_axis_mask)
46 {
47  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(input, output);
48  ARM_COMPUTE_RETURN_ERROR_ON_F16_UNSUPPORTED(input);
49  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(input, 1,
50  DataType::U8, DataType::S8, DataType::QASYMM8,
51  DataType::U16, DataType::S16,
52  DataType::U32, DataType::S32,
53  DataType::F16, DataType::F32);
54 
55  ARM_COMPUTE_RETURN_ERROR_ON(input->tensor_shape().num_dimensions() > 4);
56  ARM_COMPUTE_RETURN_ERROR_ON(starts.num_dimensions() > input->num_dimensions());
57  ARM_COMPUTE_RETURN_ERROR_ON(ends.num_dimensions() > input->num_dimensions());
58  ARM_COMPUTE_RETURN_ERROR_ON(strides.num_dimensions() > input->num_dimensions());
59  ARM_COMPUTE_RETURN_ERROR_ON(std::any_of(strides.cbegin(), strides.cbegin() + strides.num_dimensions(), [](int i)
60  {
61  return i == 0;
62  }));
63 
64  // Get expected output shape
65  const TensorShape exp_output_shape = arm_compute::misc::shape_calculator::compute_strided_slice_shape(*input,
66  starts, ends, strides,
67  begin_mask, end_mask, shrink_axis_mask);
68  ARM_COMPUTE_RETURN_ERROR_ON(exp_output_shape.total_size() == 0);
69 
70  // Checks output if configured
71  if(output->total_size() != 0)
72  {
73  const TensorInfo exp_output_info = output->clone()->set_tensor_shape(exp_output_shape);
74  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_SHAPES(output, &exp_output_info);
75  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input, output);
76  }
77 
78  return Status{};
79 }

81 std::pair<Status, Window> validate_and_configure_window(ITensorInfo *input, ITensorInfo *output,
82  const Coordinates &starts, const Coordinates &ends, const BiStrides &strides,
83  int32_t begin_mask, int32_t end_mask, int32_t shrink_axis_mask)
84 {
85  // Output tensor auto initialization if not yet initialized
86  const TensorShape output_shape = arm_compute::misc::shape_calculator::compute_strided_slice_shape(*input,
87  starts, ends, strides,
88  begin_mask, end_mask, shrink_axis_mask);
89  auto_init_if_empty(*output, input->clone()->set_tensor_shape(output_shape));
90 
91  // Create window
92  const unsigned int num_elems_processed_per_iteration = 1;
93 
94  Window win = calculate_max_window(*output, Steps(num_elems_processed_per_iteration));
95  output->set_valid_region(ValidRegion(Coordinates(), output->tensor_shape()));
96 
97  return std::make_pair(Status{}, win);
98 }

106 void CLStridedSliceKernel::configure(const ICLTensor *input, ICLTensor *output,
107  const Coordinates &starts, const Coordinates &ends, const BiStrides &strides,
108  int32_t begin_mask, int32_t end_mask, int32_t shrink_axis_mask)
109 {
110  ARM_COMPUTE_ERROR_ON_NULLPTR(input, output);
111  ARM_COMPUTE_ERROR_THROW_ON(validate_arguments(input->info(), output->info(), starts, ends, strides, begin_mask, end_mask, shrink_axis_mask));
112 
113  _input = input;
114  _output = output;
115 
116  const TensorShape &input_shape = input->info()->tensor_shape();
117 
118  Coordinates starts_abs;
119  Coordinates ends_abs;
120  Coordinates final_strides;
121  std::tie(starts_abs, ends_abs, final_strides) = arm_compute::helpers::tensor_transform::calculate_strided_slice_coords(
122  input_shape,
123  starts, ends, strides,
124  begin_mask, end_mask, shrink_axis_mask);
125 
126  // Configure kernel window
127  auto win_config = validate_and_configure_window(input->info(), output->info(), starts, ends, strides, begin_mask, end_mask, shrink_axis_mask);
128  ARM_COMPUTE_ERROR_THROW_ON(win_config.first);
129 
130  // Enable multiple elements processing along x if stride_x is 1 and output width greater than the access vector size
131  const int vec_size_x = 16 / input->info()->element_size();
132  const int output_width_x = output->info()->tensor_shape().x();
133  const bool is_shrink_on_x = arm_compute::helpers::bit_ops::is_bit_set(shrink_axis_mask, 0);
134  const bool multi_access_x = !is_shrink_on_x && (final_strides.x() == 1) && (output_width_x / vec_size_x > 0);
135 
136  // Update window if needed
137  if(multi_access_x)
138  {
139  Window &updated_window = std::get<1>(win_config);
140  updated_window.set(Window::DimX,
141  Window::Dimension(updated_window.x().start(), ceil_to_multiple(updated_window.x().end(), vec_size_x), vec_size_x));
142  }
143  ICLKernel::configure_internal(win_config.second);
144 
145  // Create build options
146  CLBuildOptions build_opts;
147  build_opts.add_option("-DDATA_TYPE=" + get_cl_type_from_data_type(input->info()->data_type()));
148  for(unsigned int i = 0; i < input_shape.num_dimensions(); ++i)
149  {
150  const bool is_shrink = arm_compute::helpers::bit_ops::is_bit_set(shrink_axis_mask, i);
151  build_opts.add_option("-DSTART_" + support::cpp11::to_string(i) + "=" + support::cpp11::to_string(starts_abs[i]));
152  build_opts.add_option("-DSTRIDE_" + support::cpp11::to_string(i) + "=" + support::cpp11::to_string(final_strides[i]));
153  build_opts.add_option_if(is_shrink, "-DSHRINK_" + support::cpp11::to_string(i));
154  }
155  build_opts.add_option_if(multi_access_x, "-DLAST_ACCESSED_X=" + support::cpp11::to_string(std::max<int>(output_width_x - vec_size_x, 0)));
156  build_opts.add_option_if(multi_access_x, "-DVEC_SIZE=" + support::cpp11::to_string(vec_size_x));
157  build_opts.add_option_if_else(input_shape.num_dimensions() > 2,
158  "-DSRC_DEPTH=" + support::cpp11::to_string(input_shape.z()),
159  "-DSRC_DEPTH=1");
160  build_opts.add_option_if_else(_output->info()->num_dimensions() > 2,
161  "-DDST_DEPTH=" + support::cpp11::to_string(_output->info()->tensor_shape().z()),
162  "-DDST_DEPTH=1");
163 
164  // Create kernel
165  _kernel = static_cast<cl::Kernel>(CLKernelLibrary::get().create_kernel("strided_slice", build_opts.options()));
166 
167  // Set config_id for enabling LWS tuning
168  _config_id = "strided_slice";
169  _config_id += "_";
170  _config_id += lower_string(string_from_data_type(input->info()->data_type()));
171  for(unsigned int i = 0; i < input_shape.num_dimensions(); ++i)
172  {
173  _config_id += "_";
174  _config_id += support::cpp11::to_string(input->info()->dimension(i));
175  _config_id += "_";
176  _config_id += support::cpp11::to_string(starts_abs[i]);
177  _config_id += "_";
178  _config_id += support::cpp11::to_string(ends_abs[i]);
179  _config_id += "_";
180  _config_id += support::cpp11::to_string(final_strides[i]);
181  }
182 }

196 void CLStridedSliceKernel::run(const Window &window, cl::CommandQueue &queue)
197 {
198  ARM_COMPUTE_ERROR_ON_UNCONFIGURED_KERNEL(this);
199  ARM_COMPUTE_ERROR_ON_INVALID_SUBWINDOW(ICLKernel::window(), window);
200 
201  Window window_collapsed = window.collapse_if_possible(ICLKernel::window(), Window::DimZ);
202  Window slice = window_collapsed.first_slice_window_4D();
203 
204  do
205  {
206  unsigned int idx = 0;
207  add_4D_tensor_argument(idx, _input, slice);
208  add_4D_tensor_argument(idx, _output, slice);
209  enqueue(queue, *this, slice, lws_hint());
210  }
211  while(window_collapsed.slide_window_slice_4D(slice));
212 }
