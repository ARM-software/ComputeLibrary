90 int main(int argc, char **argv)
91 {
92 #ifdef ARM_COMPUTE_CL
93  CLTuner cl_tuner(false);
94  if(opencl_is_available())
95  {
96  auto ctx_dev_err = create_opencl_context_and_device();
97  ARM_COMPUTE_ERROR_ON_MSG(std::get<2>(ctx_dev_err) != CL_SUCCESS, "Failed to create OpenCL context");
98  CLScheduler::get()
99  .default_init_with_context(std::get<1>(ctx_dev_err), std::get<0>(ctx_dev_err), &cl_tuner);
100  }
101 #endif /* ARM_COMPUTE_CL */
102 
103 #ifdef ARM_COMPUTE_GC
104  GCScheduler::get().default_init();
105 #endif /* ARM_COMPUTE_GC */
106 
107  framework::Framework &framework = framework::Framework::get();
108 
109  utils::CommandLineParser parser;
110 
111  std::set<framework::DatasetMode> allowed_modes
112  {
113  framework::DatasetMode::DISABLED,
114  framework::DatasetMode::PRECOMMIT,
115  framework::DatasetMode::NIGHTLY,
116  framework::DatasetMode::ALL
117  };
118 
119  framework::CommonOptions options(parser);
120 
121  auto dataset_mode = parser.add_option<utils::EnumOption<framework::DatasetMode>>("mode", allowed_modes, framework::DatasetMode::PRECOMMIT);
122  dataset_mode->set_help("For managed datasets select which group to use");
123  auto filter = parser.add_option<utils::SimpleOption<std::string>>("filter", ".*");
124  filter->set_help("Regular expression to select test cases");
125  auto filter_id = parser.add_option<utils::SimpleOption<std::string>>("filter-id");
126  filter_id->set_help("List of test ids. ... can be used to define a range.");
127  auto stop_on_error = parser.add_option<utils::ToggleOption>("stop-on-error");
128  stop_on_error->set_help("Abort execution after the first failed test (useful for debugging)");
129  auto seed = parser.add_option<utils::SimpleOption<std::random_device::result_type>>("seed", std::random_device()());
130  seed->set_help("Global seed for random number generation");
131  auto list_tests = parser.add_option<utils::ToggleOption>("list-tests", false);
132  list_tests->set_help("List all test names");
133  auto test_instruments = parser.add_option<utils::ToggleOption>("test-instruments", false);
134  test_instruments->set_help("Test if the instruments work on the platform");
135  auto error_on_missing_assets = parser.add_option<utils::ToggleOption>("error-on-missing-assets", false);
136  error_on_missing_assets->set_help("Mark a test as failed instead of skipping it when assets are missing");
137  auto assets = parser.add_positional_option<utils::SimpleOption<std::string>>("assets");
138  assets->set_help("Path to the assets directory");
139 #ifdef ARM_COMPUTE_CL
140  auto enable_tuner = parser.add_option<utils::ToggleOption>("enable-tuner");
141  enable_tuner->set_help("Enable OpenCL dynamic tuner");
142 
143  const std::set<CLTunerMode> supported_tuner_modes
144  {
145  CLTunerMode::EXHAUSTIVE,
146  CLTunerMode::NORMAL,
147  CLTunerMode::RAPID
148  };
149  auto tuner_mode = parser.add_option<utils::EnumOption<CLTunerMode>>("tuner-mode", supported_tuner_modes, CLTunerMode::NORMAL);
150  tuner_mode->set_help("Configures the time taken by the tuner to tune. Slow tuner produces the most performant LWS configuration");
151 
152  auto tuner_file = parser.add_option<utils::SimpleOption<std::string>>("tuner-file", "");
153  tuner_file->set_help("File to load/save CLTuner values");
154 #endif /* ARM_COMPUTE_CL */
155  auto threads = parser.add_option<utils::SimpleOption<int>>("threads", 1);
156  threads->set_help("Number of threads to use");
157 
158  try
159  {
160  parser.parse(argc, argv);
161 
162  if(options.help->is_set() && options.help->value())
163  {
164  parser.print_help(argv[0]);
165  return 0;
166  }
167 
168  std::vector<std::unique_ptr<framework::Printer>> printers = options.create_printers();
169 
170  Scheduler::get().set_num_threads(threads->value());
171 #ifdef ARM_COMPUTE_CL
172  if(enable_tuner->is_set())
173  {
174  cl_tuner.set_tune_new_kernels(enable_tuner->value());
175 
176  //set tuner mode
177  cl_tuner.set_tuner_mode(tuner_mode->value());
178 
179  // If that's the first run then the file won't exist yet
180  if(file_exists(tuner_file->value()))
181  {
182  cl_tuner.load_from_file(tuner_file->value());
183  }
184  }
185  else if(!tuner_file->value().empty())
186  {
187  //If we're not tuning and the file doesn't exist then we should raise an error:
188  cl_tuner.load_from_file(tuner_file->value());
189  }
190 #endif /* ARM_COMPUTE_CL */
191  if(options.log_level->value() > framework::LogLevel::NONE)
192  {
193  for(auto &p : printers)
194  {
195  p->print_global_header();
196  }
197  }
198 
199  if(options.log_level->value() >= framework::LogLevel::CONFIG)
200  {
201  for(auto &p : printers)
202  {
203  p->print_entry("Version", build_information());
204  p->print_entry("CommandLine", command_line(argc, argv));
205  p->print_entry("Seed", support::cpp11::to_string(seed->value()));
206 #ifdef ARM_COMPUTE_CL
207  if(opencl_is_available())
208  {
209  p->print_entry("CL_DEVICE_VERSION", CLKernelLibrary::get().get_device_version());
210  }
211  else
212  {
213  p->print_entry("CL_DEVICE_VERSION", "Unavailable");
214  }
215 #endif /* ARM_COMPUTE_CL */
216  const arm_compute::CPUInfo &cpu_info = Scheduler::get().cpu_info();
217  const unsigned int num_cpus = cpu_info.get_cpu_num();
218  p->print_entry("cpu_has_fp16", support::cpp11::to_string(cpu_info.has_fp16()));
219  p->print_entry("cpu_has_dotprod", support::cpp11::to_string(cpu_info.has_dotprod()));
220 
221  for(unsigned int j = 0; j < num_cpus; ++j)
222  {
223  const CPUModel model = cpu_info.get_cpu_model(j);
224  p->print_entry("CPU" + support::cpp11::to_string(j), cpu_model_to_string(model));
225  }
226  p->print_entry("Iterations", support::cpp11::to_string(options.iterations->value()));
227  p->print_entry("Threads", support::cpp11::to_string(threads->value()));
228  {
229  using support::cpp11::to_string;
230  p->print_entry("Dataset mode", to_string(dataset_mode->value()));
231  }
232  }
233  }
234 
235  framework.init(options.instruments->value(), options.iterations->value(), dataset_mode->value(), filter->value(), filter_id->value(), options.log_level->value());
236  for(auto &p : printers)
237  {
238  framework.add_printer(p.get());
239  }
240  framework.set_throw_errors(options.throw_errors->value());
241  framework.set_stop_on_error(stop_on_error->value());
242  framework.set_error_on_missing_assets(error_on_missing_assets->value());
243 
244  bool success = true;
245 
246  if(list_tests->value())
247  {
248  for(auto &p : printers)
249  {
250  p->print_list_tests(framework.test_infos());
251  p->print_global_footer();
252  }
253 
254  return 0;
255  }
256 
257  if(test_instruments->value())
258  {
259  framework::Profiler profiler = framework.get_profiler();
260  profiler.start();
261  profiler.stop();
262  for(auto &p : printers)
263  {
264  p->print_measurements(profiler.measurements());
265  }
266 
267  return 0;
268  }
269 
270  library = support::cpp14::make_unique<AssetsLibrary>(assets->value(), seed->value());
271 
272  if(!parser.validate())
273  {
274  return 1;
275  }
276 
277  success = framework.run();
278 
279  if(options.log_level->value() > framework::LogLevel::NONE)
280  {
281  for(auto &p : printers)
282  {
283  p->print_global_footer();
284  }
285  }
286 
287 #ifdef ARM_COMPUTE_CL
288  if(opencl_is_available())
289  {
290  CLScheduler::get().sync();
291  if(enable_tuner->is_set() && enable_tuner->value() && tuner_file->is_set())
292  {
293  cl_tuner.save_to_file(tuner_file->value());
294  }
295  }
296 #endif /* ARM_COMPUTE_CL */
297 
298  return (success ? 0 : 1);
299  }
300  catch(const std::exception &error)
301  {
302  std::cerr << error.what() << "\n";
303 
304  if(options.throw_errors->value())
305  {
306  throw;
307  }
308 
309  return 1;
310  }
311  return 0;
312 }
