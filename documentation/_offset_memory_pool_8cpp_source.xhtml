44 void OffsetMemoryPool::acquire(MemoryMappings &handles)
45 {
46  ARM_COMPUTE_ERROR_ON(_blob == nullptr);
47 
48  // Set memory to handlers
49  for(auto &handle : handles)
50  {
51  ARM_COMPUTE_ERROR_ON(handle.first == nullptr);
52  handle.first->set_owned_region(_blob->extract_subregion(handle.second, _blob_info.size - handle.second));
53  }
54 }
