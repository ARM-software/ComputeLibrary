44 void NEConvolutionLayer::configure(ITensor *input, const ITensor *weights, const ITensor *biases, ITensor *output, const PadStrideInfo &conv_info, const WeightsInfo &weights_info,
45  const Size2D &dilation, const ActivationLayerInfo &act_info, bool enable_fast_math, unsigned int num_groups)
46 {
47  // Perform validate step
48  ARM_COMPUTE_ERROR_ON_NULLPTR(input, weights, output);
49  ARM_COMPUTE_UNUSED(num_groups);
50  ARM_COMPUTE_ERROR_THROW_ON(NEConvolutionLayer::validate(input->info(), weights->info(), ((biases != nullptr) ? biases->info() : nullptr), output->info(), conv_info, weights_info, dilation, act_info,
51  enable_fast_math));
52 
53  switch(NEConvolutionLayer::get_convolution_method(input->info(), weights->info(), output->info(), conv_info, weights_info, dilation, act_info))
54  {
55  case ConvolutionMethod::WINOGRAD:
56  {
57  auto f = arm_compute::support::cpp14::make_unique<NEWinogradConvolutionLayer>(_memory_manager);
58  f->configure(input, weights, biases, output, conv_info, act_info, enable_fast_math);
59  _function = std::move(f);
60  break;
61  }
62  case ConvolutionMethod::GEMM:
63  {
64  auto f = arm_compute::support::cpp14::make_unique<NEGEMMConvolutionLayer>(_memory_manager);
65  f->configure(input, weights, biases, output, conv_info, weights_info, dilation, act_info);
66  _function = std::move(f);
67  break;
68  }
69  case ConvolutionMethod::DIRECT:
70  {
71  auto f = arm_compute::support::cpp14::make_unique<NEDirectConvolutionLayer>(_memory_manager);
72  f->configure(input, weights, biases, output, conv_info, act_info);
73  _function = std::move(f);
74  break;
75  }
76  case ConvolutionMethod::FFT:
77  {
78  auto f = arm_compute::support::cpp14::make_unique<NEFFTConvolutionLayer>(_memory_manager);
79  f->configure(input, weights, biases, output, conv_info, act_info);
80  _function = std::move(f);
81  break;
82  }
83  default:
84  ARM_COMPUTE_ERROR("Not supported.");
85  break;
86  }
87 }

89 Status NEConvolutionLayer::validate(const ITensorInfo *input, const ITensorInfo *weights, const ITensorInfo *biases, const ITensorInfo *output, const PadStrideInfo &conv_info,
90  const WeightsInfo &weights_info, const Size2D &dilation, const ActivationLayerInfo &act_info, bool enable_fast_math, unsigned int num_groups)
91 {
92  ARM_COMPUTE_RETURN_ERROR_ON_MSG((num_groups != 1), "Grouping (num_groups != 1) is not supported on NEON");
93 
94  switch(NEConvolutionLayer::get_convolution_method(input, weights, output, conv_info, weights_info, dilation, act_info))
95  {
96  case ConvolutionMethod::WINOGRAD:
97  //Validate Winograd
98  ARM_COMPUTE_RETURN_ON_ERROR(NEWinogradConvolutionLayer::validate(input, weights, biases, output, conv_info, act_info, enable_fast_math));
99  break;
100  case ConvolutionMethod::GEMM:
101  //Validate Gemm-based Convolution
102  ARM_COMPUTE_RETURN_ON_ERROR(NEGEMMConvolutionLayer::validate(input, weights, biases, output, conv_info, weights_info, dilation, act_info));
103  break;
104  case ConvolutionMethod::DIRECT:
105  //Validate Gemm-based Convolution
106  ARM_COMPUTE_RETURN_ON_ERROR(NEDirectConvolutionLayer::validate(input, weights, biases, output, conv_info, act_info));
107  case ConvolutionMethod::FFT:
108  // Validate FFT-based convolution layer
109  ARM_COMPUTE_RETURN_ON_ERROR(NEFFTConvolutionLayer::validate(input, weights, nullptr, output, conv_info, act_info));
110  break;
111  default:
112  ARM_COMPUTE_ERROR("Not supported.");
113  break;
114  }
115 
116  return Status{};
117 }

119 ConvolutionMethod NEConvolutionLayer::get_convolution_method(const ITensorInfo *input, const ITensorInfo *weights,
120  const ITensorInfo *output, const PadStrideInfo &conv_info,
121  const WeightsInfo &weights_info, const Size2D &dilation, const ActivationLayerInfo &act_info, bool enable_fast_math)
122 {
123  ARM_COMPUTE_ERROR_ON_NULLPTR(input, output, weights);
124  ARM_COMPUTE_UNUSED(weights_info);
125 
126  const size_t idx_w = get_data_layout_dimension_index(input->data_layout(), DataLayoutDimension::WIDTH);
127  const size_t idx_h = get_data_layout_dimension_index(input->data_layout(), DataLayoutDimension::HEIGHT);
128  const size_t idx_c = get_data_layout_dimension_index(input->data_layout(), DataLayoutDimension::CHANNEL);
129 
130  /* Input spatial dims, kernel size, IFM/OFM, conv info*/
131  using ConvolutionConfiguration = std::tuple<Size2D, Size2D, Size2D, PadStrideInfo>;
132  using ConfigurationMethod = std::pair<ConvolutionConfiguration, ConvolutionMethod>;
133 
134  const std::vector<ConfigurationMethod> known_configs =
135  {
136  // Alexnet
137  ConfigurationMethod(ConvolutionConfiguration(Size2D(27U, 27U), Size2D(5U, 5U), Size2D(48U, 128U), PadStrideInfo(1U, 1U, 2U, 2U)), ConvolutionMethod::GEMM),
138  // VGG16 / VGG19
139  ConfigurationMethod(ConvolutionConfiguration(Size2D(224U, 224U), Size2D(3U, 3U), Size2D(3U, 64U), PadStrideInfo(1U, 1U, 1U, 1U)), ConvolutionMethod::GEMM),
140  // Mobilenet 224
141  ConfigurationMethod(ConvolutionConfiguration(Size2D(224U, 224U), Size2D(3U, 3U), Size2D(3U, 32U), PadStrideInfo(2U, 2U, 0U, 1U, 0U, 1U, DimensionRoundingType::FLOOR)), ConvolutionMethod::GEMM),
142  // Mobilenet 160
143  ConfigurationMethod(ConvolutionConfiguration(Size2D(160U, 160U), Size2D(3U, 3U), Size2D(3U, 24U), PadStrideInfo(2U, 2U, 0U, 1U, 0U, 1U, DimensionRoundingType::FLOOR)), ConvolutionMethod::GEMM)
144  };
145 
146  const auto find_config = [&](ConfigurationMethod c)
147  {
148  const ConvolutionConfiguration config = c.first;
149  const PadStrideInfo info = std::get<3>(config);
150 
151  return std::get<0>(config) == Size2D(input->dimension(idx_w), input->dimension(idx_h)) && std::get<1>(config) == Size2D(weights->dimension(idx_w), weights->dimension(idx_h))
152  && std::get<2>(config) == Size2D(weights->dimension(idx_c), weights->dimension(3)) && info.pad_top() == conv_info.pad_top() && info.pad_right() == conv_info.pad_right()
153  && info.pad_bottom() == conv_info.pad_bottom() && info.pad_left() == conv_info.pad_left() && info.stride() == conv_info.stride();
154  };
155 
156  std::vector<ConfigurationMethod>::const_iterator found;
157  if((found = std::find_if(known_configs.begin(), known_configs.end(), find_config)) != known_configs.end())
158  {
159  return (*found).second;
160  }
161 
162  if(dilation != Size2D(1U, 1U))
163  {
164  return ConvolutionMethod::GEMM;
165  }
166  else
167  {
168  if((weights->dimension(idx_h) > 7) && (input->dimension(idx_c) > output->dimension(idx_c)) && (NEFFTConvolutionLayer::validate(input, weights, nullptr, output, conv_info, act_info)))
169  {
170  return ConvolutionMethod::FFT;
171  }
172  if(input->dimension(idx_c) < 16)
173  {
174  return ConvolutionMethod::GEMM;
175  }
176  return bool(NEWinogradConvolutionLayer::validate(input, weights, nullptr, output, conv_info, act_info, enable_fast_math)) ? ConvolutionMethod::WINOGRAD : ConvolutionMethod::GEMM;
177  }
178 }
