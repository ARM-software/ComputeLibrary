88  /* We will need to walk through the blocks of B in a few contexts, so
89  * factor that out. */
90  class blockwalker {
91  private:
92  /* Size loops, etc. based on our parent's configuration */
93  const GemmInterleaved<strategy, To, Tr> &_parent;
94 
95  /* K, X and multi parameters for current iteration. */
96  unsigned int _k0=0, _x0=0, _multi=0;
97 
98  unsigned int _index=0;
99  bool _done=false;
100  bool _newkblock=true;
101  bool _newmulti=true;
102 
103  public:
104  blockwalker(const GemmInterleaved<strategy, To, Tr> &parent) : _parent(parent) { }
105 
106  unsigned int xmax() {
107  return std::min(_x0 + _parent._x_block, _parent._Nsize);
108  }
109 
110  unsigned int kmax() {
111  return std::min(_k0 + _parent._k_block, _parent._Ksize);
112  }
113 
114  /* Advance to the next block, return false at the end. */
115  bool advance(void) {
116  if (_done) {
117  return false;
118  }
119 
120  _newkblock=false;
121  _x0 += _parent._x_block;
122  if (_x0 >= _parent._Nsize) {
123  _x0=0;
124  _k0 += _parent._k_block;
125  if (_k0 >= _parent._Ksize) {
126  _k0=0;
127  _multi++;
128  if (_multi >= _parent._nmulti) {
129  _done=true;
130  return false;
131  }
132  _newmulti=true;
133  }
134  _newkblock=true;
135  }
136  _index++;
137 
138  return true;
139  }
140 
141  unsigned int k0(void) { return _k0; }
142  unsigned int x0(void) { return _x0; }
143  unsigned int multi(void) { return _multi; }
144  unsigned int index(void) { return _index; }
145  bool done(void) { return _done; }
146  bool newkblock(void) { return _newkblock; }
147  };

149  // A working size: One of these needed, regardless of thread count. Divided according to window.
150  size_t get_a_working_size() const {
151  return ROUND_UP(sizeof(Toi) * _k_block * _Mround * _nbatches);
152  }

154  // B working size: 0, 1 or 3 of these needed depending on pretransposed and threading settings.
155  size_t get_b_working_size() const {
156  return ROUND_UP(sizeof(Toi) * _x_block * _k_block);
157  }

160  size_t get_c_working_size() const {

164  // Internal execute function.
165  // This supports both the "pretransposed" and "standard" interfaces via the template parameter.
166  template<bool pretransposed>
167  void execute_internal(unsigned int start, unsigned int end, int threadid) {
168 #ifdef CYCLE_PROFILING
169  profiler prof;
170 #endif
171  strategy strat(_ci);
172 
173  blockwalker current(*this);
174  blockwalker next=current;
175 
176  /* Translate 'start' and 'end' into a position within the batches and rows. */
177  const unsigned int window_per_batch = _Mround / strategy::out_height();
178  unsigned int batch_0 = start / window_per_batch;
179  unsigned int batch_end = end / window_per_batch;
180 
181  /* Compute the M values to operate on */
182  unsigned int m_0 = (start - (batch_0 * window_per_batch)) * strategy::out_height();
183  unsigned int m_max = (end - (batch_end * window_per_batch)) * strategy::out_height();
184 
185  /* Make sure we've been set up correctly. */
186  if (pretransposed) {
187  assert(_B_transposed);
188  } else {
189  assert(_bm);
190  }
191 
192  assert(_working_space);
193  int8_t *working_space_bytes = reinterpret_cast<int8_t *>(_working_space);
194 
195  // Private buffers. Treat working_space as an array of C buffers
196  // (one per thread) first, followed by the (window-divided) A
197  // buffer.
198  // Set a_panel to the base of the A buffers - compute offsets into it based on M/batches later.
199  Toi * const a_panel = reinterpret_cast<Toi *>(working_space_bytes + (_maxthreads * get_c_working_size()));
200  Tri * const c_panel = reinterpret_cast<Tri *>(working_space_bytes + (threadid * get_c_working_size()));
201 
202  // Shared buffers - these come either from BufferManager or _B_transposed.
203  const Toi *b_panel;
204 
205  if (pretransposed) {
206  b_panel = _B_transposed;
207  }
208 
209  //printf("Starting GEMM loop, x_block=%d, k_block=%d\n", _x_block, _k_block);
210 
211  // newkblock() is always true on the first iteration, so this will be set properly on the first loop.
212  int kern_k = 0;
213 
214  for (;!current.done();current.advance()) {
215  if (current.newkblock()) {
216 #ifdef CYCLE_PROFILING
217  auto p=prof.ScopedProfiler(PROFILE_PREPA, (end - start) * strategy::out_height() * (current.kmax()-current.k0()) * sizeof(Toi));
218 #endif
219  for (unsigned int batch = batch_0; batch <= batch_end; batch++) {
220  unsigned int first_m = (batch == batch_0) ? m_0 : 0;
221  unsigned int last_m = (batch == batch_end) ? m_max : _Msize;
222 
223  if (first_m >= last_m)
224  continue;
225 
226  strat.transforms.PrepareA(a_panel + ((batch * _Mround + first_m) * _k_block),
227  this->_Aptr + (batch * this->_A_batch_stride) + (current.multi() * this->_A_multi_stride),
228  this->_lda, first_m, last_m, current.k0(), current.kmax(), _trA);
229  }
230 
231  // Figure out how many "K" the kernel will actually process.
232  kern_k = iceildiv(current.kmax() - current.k0(), strategy::k_unroll());
233  kern_k *= strat.k_unroll();
234  }
235 
236  int bblocks = iceildiv(current.xmax() - current.x0(), strategy::out_width());
237 
238  if (!pretransposed) {
239  /* Look ahead to the next block and populate it if necessary.
240  * This avoids the populate operation becoming a bottleneck, and
241  * helps keep the threads synchronized (the first thread to get
242  * here will populate while the rest will advance).
243  *
244  * If we are running single threaded, bm->try_populate() will do
245  * nothing.
246  */
247  if (next.advance()) {
248  _bm->try_populate(next.index(), [&](void *buffer) {
249 #ifdef CYCLE_PROFILING
250  auto p=prof.ScopedProfiler(PROFILE_PREPB, (next.xmax()-next.x0()) * (next.kmax()-next.k0()) * sizeof(Toi));
251 #endif
252 
253  Toi *b_panel = reinterpret_cast<Toi *>(buffer);
254 
255  strat.transforms.PrepareB(b_panel, this->_Bptr + (next.multi() * this->_B_multi_stride), this->_ldb,
256  next.x0(), next.xmax(), next.k0(), next.kmax(), _trB);
257  });
258  }
259 
260  /* Get the buffer for this iteration from the BufferManager. */
261  b_panel = reinterpret_cast<Toi *>(_bm->get(current.index(), [&](void *bpv) {
262 #ifdef CYCLE_PROFILING
263  auto p=prof.ScopedProfiler(PROFILE_PREPB, (current.xmax()-current.x0()) * (current.kmax()-current.k0()) * sizeof(Toi));
264 #endif
265 
266  Toi *b_panel = reinterpret_cast<Toi *>(bpv);
267 
268  strat.transforms.PrepareB(b_panel, this->_Bptr + (current.multi() * this->_B_multi_stride), this->_ldb,
269  current.x0(), current.xmax(), current.k0(), current.kmax(), _trB);
270  }));
271  }
272 
273  /* Do the actual work. */
274  for (unsigned int batch = batch_0; batch <= batch_end; batch++) {
275  unsigned int first_m = (batch == batch_0) ? m_0 : 0;
276  unsigned int last_m = (batch == batch_end) ? m_max : _Msize;
277 
278  const Toi *a_ptr = a_panel + (batch * _Mround + first_m) * _k_block;
279 
280  if (first_m >= last_m)
281  continue;
282 
283  for (unsigned int y=first_m; y<last_m; y+=strategy::out_height()) {
284  unsigned int ymax = std::min(_Msize, y + strategy::out_height());
285 
286  {
287 #ifdef CYCLE_PROFILING
288  auto p=prof.ScopedProfiler(PROFILE_KERNEL, (strategy::out_height() * bblocks * strategy::out_width() * kern_k));
289 #endif
290 
291  strat.kernel(a_ptr, b_panel, c_panel, 1, bblocks, kern_k);
292 
293  a_ptr += (strategy::out_height() * kern_k);
294  }
295 
296  {
297 #ifdef CYCLE_PROFILING
298  auto p=prof.ScopedProfiler(PROFILE_MERGE, (strategy::out_height() * bblocks * strategy::out_width() * sizeof(Tr)));
299 #endif
300  strat.transforms.Merge(this->_Cptr + (batch * this->_C_batch_stride) + (current.multi() * this->_C_multi_stride),
301  c_panel, this->_ldc, y, ymax, current.x0(), current.xmax(),
302  _alpha, (current.k0()==0 ? _beta : static_cast<Tr>(1)));
303  }
304  }
305  }
306 
307  if (pretransposed) {
308  b_panel += (bblocks * strat.out_width() * kern_k);
309  } else {
310  _bm->release(current.index());
311  }
312  }
313  }

319  /* Constructor */
320  GemmInterleaved(const GemmArgs<Tr> &args)
321  : _ci(args._ci), _Msize(args._Msize), _Nsize(args._Nsize), _Ksize(args._Ksize),
322  _nbatches(args._nbatches), _nmulti(args._nmulti), _trA(args._trA), _trB(args._trB),
323  _alpha(args._alpha), _beta(args._beta), _maxthreads(args._maxthreads), _nthreads(args._maxthreads),
324  _pretransposed(args._pretransposed_hint) {
325  const unsigned int L1_size = _ci->get_L1_cache_size();
326  const unsigned int L2_size = _ci->get_L2_cache_size();
327 
328  assert(_maxthreads > 0);
329 
330  // Work out blocking parameters, or override from provided GemmConfig
331  if (args._cfg && args._cfg->inner_block_size) {
332  _k_block = args._cfg->inner_block_size;
333  } else {
334  // k_block: Find out how much of the larger array can be loaded into half the cache.
335  // This should account for associative caches.
336  _k_block = (L1_size / 2) / (sizeof(Toi) * (std::max(strategy::out_width(), strategy::out_height())));
337 
338  // Needs to be (at least a single) multiple of the K unroll level.
339  _k_block /= strategy::k_unroll();
340  _k_block = std::max(_k_block, 1U) * strategy::k_unroll();
341 
342  // Now tune to presented problem size; this is how many blocks we need.
343  unsigned int num_k_blocks = iceildiv(_Ksize, _k_block);
344 
345  // So divide the space equally into that many blocks.
346  _k_block = iceildiv(_Ksize, num_k_blocks);
347 
348  // And round UP to the K unroll level required.
349  _k_block = iceildiv(_k_block, strategy::k_unroll());
350  _k_block *= strategy::k_unroll();
351  }
352 
353  if (args._cfg && args._cfg->outer_block_size) {
354  _x_block = args._cfg->outer_block_size;
355  } else {
356  // x_block: Work out how many rows (of length k_block) will fit in the L2
357  // Don't allocate more than 90% of the L2 to allow for overheads, and subtract off the L1 contents.
358  _x_block = (((L2_size * 9) / 10) - (_k_block * sizeof(Toi) * (strategy::out_width() + strategy::out_height()))) /
359  (sizeof(Toi) * _k_block);
360 
361  // Needs to be (at least a single) multiple of the kernel output width.
362  _x_block /= strategy::out_width();
363  _x_block = std::max(_x_block, 1U) * strategy::out_width();
364 
365  // And tune to the presented problem size.
366  unsigned int num_x_blocks = iceildiv(_Nsize, _x_block);
367  _x_block = iceildiv(_Nsize, num_x_blocks);
368 
369  _x_block = iceildiv(_x_block, strategy::out_width());
370  _x_block *= strategy::out_width();
371  }
372 
373  // Work out the rounded size of M - needed for some buffers.
374  _Mround = iceildiv(_Msize, strategy::out_height());
375  _Mround *= strategy::out_height();
376  }

406  // Interface implementation - working space
407  size_t get_working_size() const override {
408  // In all cases, we need one A buffer plus a C buffer per thread.
409  size_t size = get_a_working_size() + (get_c_working_size() * _maxthreads);
410 
411  // For pretransposed case, there is no working space needed for B.
412  // Otherwise, we need a BufferManager.
413  if (!_pretransposed) {
414  size += BufferManager::get_storage_requirement(_maxthreads, get_b_working_size());
415  }
416 
417  size += 64; // Add on a cache line extra for alignment.
418 
419  return size;
420  }

422  void set_working_space(void *working_space) override {
423  // Make sure everything ends up cache line aligned
424  int8_t *working_space_bytes = reinterpret_cast<int8_t *>(working_space);
425  intptr_t working_space_int = reinterpret_cast<intptr_t>(working_space);
426 
427  size_t diff=0;
428 
429  if (working_space_int & 0x3F) {
430  diff = 0x40 - (working_space_int & 0x3F);
431  }
432 
433  working_space_bytes += diff;
434 
435  if (_pretransposed) {
436  // Pretransposed case: just set internal pointer to parameter value.
437  _working_space = reinterpret_cast<void *>(working_space_bytes);
438  } else {
439  // Otherwise, use the first part of the working space for the buffer manager.
440  // It's legal to call this again so don't leak a buffer manager if it already existed.
441  delete _bm;
442 
443  _bm = new BufferManager(_nthreads, get_b_working_size(), reinterpret_cast<void *>(working_space_bytes));
444 
445  working_space_bytes += BufferManager::get_storage_requirement(_maxthreads, get_b_working_size());
446 
447  _working_space = reinterpret_cast<void *>(working_space_bytes);
448  }
449  }
