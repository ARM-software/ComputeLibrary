33 // Some macros used to decide how much working space to allocate.
34 // Round allocations up to the next cache line.
35 #define ALLOC_ROUND 64
36 #define ROUND_UP(x) ((((x) + ALLOC_ROUND-1) / ALLOC_ROUND) * ALLOC_ROUND)

38 // Implementation of the GemmCommon abstract class.
39 //
40 // This implementation interleaves the source matrices in blocks - good for
41 // larger matrices.
42 template<typename strategy, typename To, typename Tr>
43 class GemmInterleaved : public GemmCommon<To, Tr> {
44  typedef typename strategy::operand_type Toi;
45  typedef typename strategy::result_type Tri;
46 
47  const unsigned int M;
48  const unsigned int N;
49  const unsigned int K;
50 
51  const bool trA;
52  const bool trB;

54  const strategy strat;

56  unsigned int k_block = 0;
57  unsigned int x_block = 0;
58  unsigned int Mround = 0;
59 
60  size_t get_a_working_size() const {
61  return ROUND_UP(sizeof(Toi) * k_block * Mround);
62  }
63 
64  size_t get_b_working_size() const {
65  return ROUND_UP(sizeof(Toi) * x_block * k_block);
66  }
67 
68  size_t get_c_working_size() const {
69  return ROUND_UP(sizeof(Tri) * x_block * strat.out_height);
70  }

72 public:
73  size_t get_working_size() const override {
74  return get_a_working_size() + get_b_working_size() + get_c_working_size();

77  GemmInterleaved(const CPUInfo *ci, const unsigned int M, const unsigned int N, const unsigned int K, const bool trA, const bool trB) : M(M), N(N), K(K), trA(trA), trB(trB), strat(ci) {
78  const unsigned int L1_size = ci->L1_size;
79  const unsigned int L2_size = ci->L2_size;
80 
81  // Work out blocking parameters
82  // k_block: Each iteration will consume (out_width + out_height)
83  // operands - so how many iterations will fill the L1?
84  k_block = L1_size / (sizeof(Toi) * (strat.out_width + strat.out_height));
85 
86  // Needs to be a multiple of the K unroll level.
87  k_block /= strat.k_unroll;
88  k_block *= strat.k_unroll;
89 
90  // Now tune to presented problem size; this is how many blocks we need.
91  int num_k_blocks = (K + (k_block - 1)) / k_block;
92 
93  // So divide the space equally into that many blocks.
94  k_block = (K + num_k_blocks - 1) / num_k_blocks;
95 
96  // And round UP to the K unroll level required.
97  k_block = (k_block + strat.k_unroll - 1) / strat.k_unroll;
98  k_block *= strat.k_unroll;
99 
100  // x_block: Work out how many rows (of length k_block) will fit in the L2
101  x_block = L2_size / (sizeof(Toi) * k_block);
102 
103  // Needs to be a multiple of the kernel output width.
104  x_block /= strat.out_width;
105  x_block *= strat.out_width;
106 
107  // And tune to the presented problem size.
108  int num_x_blocks = (N + (x_block - 1)) / x_block;
109  x_block = (N + num_x_blocks - 1) / num_x_blocks;
110 
111  x_block = (x_block + strat.out_width - 1) / strat.out_width;
112  x_block *= strat.out_width;
113 
114  // Work out the rounded size of M - needed for some buffers.
115  Mround = (M + (strat.out_height - 1)) / strat.out_height;
116  Mround *= strat.out_height;
117  }

119  // Actually execute the GEMM.
120  void execute(const To *A, const int lda, const To *B, const int ldb, Tr *C, const int ldc, const Tr alpha, const Tr beta, void *working_space) const override {
121  profiler prof;
122 
123  int8_t *working_space_bytes = reinterpret_cast<int8_t *>(working_space);
124  intptr_t working_space_int = reinterpret_cast<intptr_t>(working_space_bytes);
125  size_t diff = 0;
126 
127  if (working_space_int & 0xF) {
128  diff = 0x10 - (working_space_int & 0xF);
129  }
130 
131  Toi * const a_panel = reinterpret_cast<Toi *>(working_space_bytes + diff);
132  Toi * const b_panel = reinterpret_cast<Toi *>(working_space_bytes + get_a_working_size() + diff);
133  Tri * const c_panel = reinterpret_cast<Tri *>(working_space_bytes + get_a_working_size() + get_b_working_size() + diff);
134 
135  for (unsigned int k0=0; k0<K; k0 += k_block) {
136  unsigned int kmax = k0 + k_block;
137  if (kmax > K) kmax = K;
138 
139  // Figure out how many "K" the kernel will actually process.
140  int kern_k = ((kmax - k0) + (strat.k_unroll - 1)) / strat.k_unroll;
141  kern_k *= strat.k_unroll;
142 
143  prof(PROFILE_PREPA, [&](void) {
144  if (trA ^ strategy::A_transpose) {
145  Transform<strategy::A_interleave, strategy::A_block, true>(a_panel, A, lda, 0, M, k0, kmax);
146  } else {
147  Transform<strategy::A_interleave, strategy::A_block, false>(a_panel, A, lda, 0, M, k0, kmax);
148  }
149  });
150 
151  for (unsigned int x0=0; x0<N; x0 += x_block) {
152  unsigned int xmax = x0 + x_block;
153  if (xmax > N) xmax = N;
154 
155  int bblocks = (xmax - x0 + strat.out_width - 1) / strat.out_width;
156 
157  prof(PROFILE_PREPB, [&](void) {
158  if (trB ^ strategy::B_transpose) {
159  Transform<strategy::B_interleave, strategy::B_block, true>(b_panel, B, ldb, x0, xmax, k0, kmax);
160  } else {
161  Transform<strategy::B_interleave, strategy::B_block, false>(b_panel, B, ldb, x0, xmax, k0, kmax);
162  }
163  });
164 
165  for (unsigned int y=0; y<M; y+=strat.out_height) {
166  unsigned int ymax = y + strat.out_height;
167  if (ymax > M) ymax = M;
168 
169  prof(PROFILE_KERNEL, [&](void) { strat.kernel(a_panel + (y * kern_k), b_panel, c_panel, 1, bblocks, kern_k); });
170  prof(PROFILE_MERGE, [&](void) { MergeResults<strategy::out_width, strategy::out_height>(C, c_panel, ldc, y, ymax, x0, xmax, alpha, (k0==0 ? beta : static_cast<Tr>(1))); });
171  }
172  }
173  }
174  }
175 };
