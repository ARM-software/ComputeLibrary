157 void JSONPrinter::print_measurements(const Profiler::MeasurementsMap &measurements)
158 {
159  print_separator(_first_test_entry);
160 
161  *_stream << R"("measurements" : {)";
162 
163  for(auto i_it = measurements.cbegin(), i_end = measurements.cend(); i_it != i_end;)
164  {
165  *_stream << R"(")" << i_it->first << R"(" : {)";
166 
167  auto add_measurements = [](double a, const Measurement & b)
168  {
169  return a + b.value;
170  };
171 
172  auto cmp_measurements = [](const Measurement & a, const Measurement & b)
173  {
174  return a.value < b.value;
175  };
176 
177  double sum_values = std::accumulate(i_it->second.cbegin(), i_it->second.cend(), 0., add_measurements);
178  int num_values = i_it->second.size();
179  const auto minmax_values = std::minmax_element(i_it->second.begin(), i_it->second.end(), cmp_measurements);
180 
181  if(num_values > 2)
182  {
183  sum_values -= minmax_values.first->value + minmax_values.second->value;
184  num_values -= 2;
185  }
186 
187  auto measurement_to_string = [](const Measurement & measurement)
188  {
189  return support::cpp11::to_string(measurement.value);
190  };
191 
192  *_stream << R"("avg" : )" << (sum_values / num_values) << ",";
193  *_stream << R"("min" : )" << minmax_values.first->value << ",";
194  *_stream << R"("max" : )" << minmax_values.second->value << ",";
195  *_stream << R"("raw" : [)" << join(i_it->second.begin(), i_it->second.end(), ",", measurement_to_string) << "],";
196  *_stream << R"("unit" : ")" << minmax_values.first->unit << R"(")";
197  *_stream << "}";
198 
199  if(++i_it != i_end)
200  {
201  *_stream << ",";
202  }
203  }
204 
205  *_stream << "}";
206 }
