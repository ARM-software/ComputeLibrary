37 std::pair<GEMMLHSMatrixInfo, GEMMRHSMatrixInfo> configure_gemm_reshaped(unsigned int m, unsigned int n, unsigned int m0, unsigned int n0, unsigned int k0, unsigned int v0, unsigned int h0,
38  bool lhs_interleave, bool rhs_interleave)
39 {
40  GEMMLHSMatrixInfo lhs_info;
41  GEMMRHSMatrixInfo rhs_info;
42 
43  // Configure GEMMLHSMatrixInfo
44  lhs_info.m0 = m0;
45  lhs_info.k0 = k0;
46  lhs_info.v0 = ((m / (lhs_info.m0 * v0)) == 0) ? 1 : v0;
47  lhs_info.interleave = lhs_interleave;
48  lhs_info.transpose = false;
49 
50  // Configure GEMMRHSMatrixInfo
51  rhs_info.n0 = n0;
52  rhs_info.k0 = lhs_info.k0;
53  rhs_info.h0 = ((n / (rhs_info.n0 * h0)) == 0) ? 1 : h0;
54  rhs_info.interleave = rhs_interleave;
55  rhs_info.transpose = true;
56 
57  return std::make_pair(lhs_info, rhs_info);
58 }
