47 template <typename T1, typename T2, unsigned int stridex>
48 class convolver_3x3
49 {
50 public:
51  static void convolve(const Window &window, unsigned int num_elems_written_per_iteration,
52  const ITensor *input, const ITensor *weights, ITensor *output, const PadStrideInfo &conv_info, unsigned int depth_multiplier, const Size2D &dilation)
53  {
54  const int input_offset = -input->info()->quantization_info().uniform().offset;
55  const int weights_offset = -weights->info()->quantization_info().uniform().offset;
56 
57  const int input_stride_x = input->info()->strides_in_bytes().x();
58  const int input_stride_y = input->info()->strides_in_bytes().y();
59  const int input_stride_z = input->info()->strides_in_bytes().z();
60  const int input_stride_w = input->info()->strides_in_bytes()[3];
61  const int output_stride_y = output->info()->strides_in_bytes().y();
62  const int kernel_stride_y = weights->info()->strides_in_bytes().y();
63  const int kernel_stride_z = weights->info()->strides_in_bytes().z();
64  const int output_w = output->info()->dimension(0);
65  const int output_h = output->info()->dimension(1);
66  const int delta_input = detail::get_input_num_elems_processed<stridex>(num_elems_written_per_iteration);
67  const unsigned int conv_stride_y = std::get<1>(conv_info.stride());
68  const unsigned int conv_pad_x = conv_info.pad_left();
69  const unsigned int conv_pad_y = conv_info.pad_top();
70 
71  // setup output window for the iterator
72  Window window_out = window;
73  window_out.set(Window::DimX, Window::Dimension(0, output->info()->dimension(Window::DimX), output->info()->dimension(Window::DimX)));
74  window_out.set(Window::DimY, Window::Dimension(0, output->info()->dimension(Window::DimY), output->info()->dimension(Window::DimY)));
75 
76  // setup input window for the iterator
77  Window window_in = window;
78  // Iteration of input is taken care of in execute_window_loop
79  window_in.set(Window::DimX, Window::Dimension(0, 0, 0));
80  window_in.set(Window::DimY, Window::Dimension(0, 0, 0));
81  window_in.set(Window::DimZ, Window::Dimension(0, 0, 0));
82 
83  Window window_k = calculate_max_window(*weights->info(), Steps(1u));
84 
85  Iterator in(input, window_in);
86  Iterator out(output, window_out);
87  Iterator w(weights, window_k);
88 
89  const uint8_t *weights_ptr = w.ptr();
90 
91  execute_window_loop(window_out, [&](const Coordinates & id)
92  {
93  int ih = 0;
94  int oh = 0;
95 
96  const uint8_t *input_ptr = in.ptr() - conv_pad_x * input_stride_x - conv_pad_y * input_stride_y + (id.z() / depth_multiplier) * input_stride_z + input_stride_w * id[3];
97  const uint8_t *ptr_weights_base = weights_ptr + id.z() * kernel_stride_z;
98 
99  const auto ptr_weights_r0 = reinterpret_cast<const T1 *>(ptr_weights_base);
100  const auto ptr_weights_r1 = reinterpret_cast<const T1 *>(ptr_weights_base + kernel_stride_y);
101  const auto ptr_weights_r2 = reinterpret_cast<const T1 *>(ptr_weights_base + kernel_stride_y * 2);
102  const auto vw_r0 = detail::load_matrix_row(ptr_weights_r0, weights_offset);
103  const auto vw_r1 = detail::load_matrix_row(ptr_weights_r1, weights_offset);
104  const auto vw_r2 = detail::load_matrix_row(ptr_weights_r2, weights_offset);
105 
106  for(ih = 0, oh = 0; oh < output_h; ++oh, ih += conv_stride_y)
107  {
108  auto in_top = reinterpret_cast<const T1 *>(input_ptr + (ih + 0) * input_stride_y);
109  auto in_mid = reinterpret_cast<const T1 *>(input_ptr + (ih + dilation.y()) * input_stride_y);
110  auto in_low = reinterpret_cast<const T1 *>(input_ptr + (ih + 2 * dilation.y()) * input_stride_y); //uint8
111  auto p_out = reinterpret_cast<T2 *>(out.ptr() + oh * output_stride_y); //int32
112 
113  for(int ow = 0; ow < output_w; ow += num_elems_written_per_iteration,
114  in_top += delta_input, in_mid += delta_input, in_low += delta_input,
115  p_out += num_elems_written_per_iteration)
116  {
117  if(dilation == Size2D(1U, 1U))
118  {
119  auto vres = detail::convolve_3x3<stridex>(in_top, in_mid, in_low, vw_r0, vw_r1, vw_r2, input_offset);
120  detail::store_results<stridex>(p_out, vres);
121  }
122  else
123  {
124  auto vres = detail::convolve_3x3_dilation<stridex>(in_top, in_mid, in_low, vw_r0, vw_r1, vw_r2, dilation.x(), input_offset);
125  detail::store_results<stridex>(p_out, vres);
126  }
127  }
128  }
129  },
130  out);
131  }
132 };

134 template <typename T1, typename T2>
135 inline void convolve_3x3(const Window &window, unsigned int num_elems_written_per_iteration,
136  const ITensor *input, const ITensor *weights, ITensor *output,
137  const PadStrideInfo &conv_info, unsigned int depth_multiplier, const Size2D &dilation)
138 {
139  const unsigned int conv_stride_x = std::get<0>(conv_info.stride());
140  switch(conv_stride_x)
141  {
142  case 1:
143  convolver_3x3<T1, T2, 1>::convolve(window, num_elems_written_per_iteration, input, weights, output, conv_info, depth_multiplier, dilation);
144  break;
145  case 2:
146  convolver_3x3<T1, T2, 2>::convolve(window, num_elems_written_per_iteration, input, weights, output, conv_info, depth_multiplier, dilation);
147  break;
148  case 3:
149  convolver_3x3<T1, T2, 3>::convolve(window, num_elems_written_per_iteration, input, weights, output, conv_info, depth_multiplier, dilation);
150  break;
151  default:
152  ARM_COMPUTE_ERROR("Not implemented");
153  }
154 }

156 Status validate_arguments(const ITensorInfo *input, const ITensorInfo *weights, const ITensorInfo *output, const PadStrideInfo &conv_info, unsigned int depth_multiplier, const Size2D &dilation)
157 {
158  ARM_COMPUTE_RETURN_ERROR_ON_CPU_F16_UNSUPPORTED(input);
159  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(input, 1, DataType::QASYMM8, DataType::F16, DataType::F32);
160  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input, weights);
161 
162  const DataLayout data_layout = input->data_layout();
163  const unsigned int width_idx = get_data_layout_dimension_index(data_layout, DataLayoutDimension::WIDTH);
164  const unsigned int height_idx = get_data_layout_dimension_index(data_layout, DataLayoutDimension::HEIGHT);
165 
166  ARM_COMPUTE_RETURN_ERROR_ON(weights->dimension(width_idx) != 3 || weights->dimension(height_idx) != 3);
167  ARM_COMPUTE_RETURN_ERROR_ON(conv_info.stride().first < 1 || conv_info.stride().first > 3);
168 
169  if(output->total_size() != 0)
170  {
171  const TensorShape output_shape = misc::shape_calculator::compute_depthwise_convolution_shape(*input, *weights, conv_info, depth_multiplier, dilation);
172  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DIMENSIONS(output->tensor_shape(), output_shape);
173 
174  if(is_data_type_quantized_asymmetric(input->data_type()))
175  {
176  ARM_COMPUTE_RETURN_ERROR_ON(output->data_type() != DataType::S32);
177  }
178  else
179  {
180  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input, output);
181  }
182  }
183 
184  return Status{};
185 }

187 std::pair<Status, Window> validate_and_configure_window(ITensorInfo *input, ITensorInfo *weights, ITensorInfo *output, const PadStrideInfo &conv_info, unsigned int depth_multiplier,
188  const Size2D &dilation)
189 {
190  Window win;
191  bool window_changed = false;
192 
193  // Get convolved dimensions
194  const TensorShape output_shape = misc::shape_calculator::compute_depthwise_convolution_shape(*input, *weights, conv_info, depth_multiplier, dilation);
195  const DataType output_dt = (input->data_type() == DataType::QASYMM8) ? DataType::S32 : input->data_type();
196 
197  // Output auto inizialitation if not yet initialized
198  auto_init_if_empty(*output, input->clone()->set_is_resizable(true).reset_padding().set_tensor_shape(output_shape).set_data_type(output_dt).set_quantization_info(output->quantization_info()));
199 
200  // Configure kernel window (generic)
201  const unsigned int conv_stride_x = conv_info.stride().first;
202  const unsigned int conv_stride_y = conv_info.stride().second;
203  const unsigned int conv_pad_top = conv_info.pad_top();
204  const unsigned int conv_pad_left = conv_info.pad_left();
205 
206  unsigned int num_elems_written_per_iteration = 16 >> conv_stride_x;
207  unsigned int num_elems_read_per_iteration = 0;
208 
209  switch(input->data_type())
210  {
211  case DataType::QASYMM8:
212  num_elems_read_per_iteration = 16 + 15 * (dilation.x() - 1);
213  break;
214 #ifdef __ARM_FEATURE_FP16_VECTOR_ARITHMETIC
215  case DataType::F16:
216  num_elems_written_per_iteration = 32 >> conv_stride_x;
217  num_elems_read_per_iteration = 24 + 23 * (dilation.x() - 1);
218  break;
219 #endif // __ARM_FEATURE_FP16_VECTOR_ARITHMETIC
220  case DataType::F32:
221  num_elems_read_per_iteration = 12 + 11 * (dilation.x() - 1);
222  break;
223  default:
224  ARM_COMPUTE_ERROR("Data type not supported.");
225  }
226 
227  // Configure kernel window
228  win = calculate_max_window(*output, Steps(num_elems_written_per_iteration));
229 
230  AccessWindowRectangle input_access(input, -conv_pad_left, -conv_pad_top, num_elems_read_per_iteration, 3 + 2 * (dilation.y() - 1), conv_stride_x, conv_stride_y);
231  AccessWindowStatic weights_access(weights, 0, 0, 3, 3);
232  AccessWindowHorizontal output_access(output, 0, num_elems_written_per_iteration);
233 
234  window_changed = update_window_and_padding(win, input_access, weights_access, output_access);
235  output_access.set_valid_region(win, ValidRegion(Coordinates(), output->tensor_shape()));
236 
237  Status err = (window_changed) ? ARM_COMPUTE_CREATE_ERROR(ErrorCode::RUNTIME_ERROR, "Insufficient Padding!") : Status{};
238  return std::make_pair(err, win);
239 }

252 void NEDepthwiseConvolutionLayer3x3Kernel::configure(const ITensor *input, const ITensor *weights, ITensor *output, const PadStrideInfo &conv_info, unsigned int depth_multiplier,
253  const Size2D &dilation)
254 {
255  ARM_COMPUTE_ERROR_ON_NULLPTR(input, weights, output);
256  ARM_COMPUTE_ERROR_THROW_ON(validate_arguments(input->info(), weights->info(), output->info(), conv_info, depth_multiplier, dilation));
257 
258  _input = input;
259  _output = output;
260  _weights = weights;
261  _conv_info = conv_info;
262  _depth_multiplier = depth_multiplier;
263  switch(input->info()->data_type())
264  {
265  case DataType::QASYMM8:
266  case DataType::F32:
267  _num_elems_written_per_iteration = 16 >> _conv_info.stride().first;
268  break;
269  case DataType::F16:
270  _num_elems_written_per_iteration = 32 >> _conv_info.stride().first;
271  break;
272  default:
273  ARM_COMPUTE_ERROR("Data type not supported.");
274  }
275  _border_size = BorderSize(_conv_info.pad_top(), _conv_info.pad_right(), _conv_info.pad_bottom(), _conv_info.pad_left());
276  _dilation = dilation;
277  auto win_config = validate_and_configure_window(_input->info(), _weights->info(), _output->info(), _conv_info, _depth_multiplier, dilation);
278  ARM_COMPUTE_ERROR_THROW_ON(win_config.first);
279  INEKernel::configure(win_config.second);
280 }

290 void NEDepthwiseConvolutionLayer3x3Kernel::run(const Window &window, const ThreadInfo &info)
291 {
292  ARM_COMPUTE_ERROR_ON_UNCONFIGURED_KERNEL(this);
293  ARM_COMPUTE_UNUSED(info);
294 
295  ARM_COMPUTE_UNUSED(info);
296 
297  switch(_input->info()->data_type())
298  {
299 #ifdef __ARM_FEATURE_FP16_VECTOR_ARITHMETIC
300  case DataType::F16:
301  convolve_3x3<float16_t, float16_t>(window, _num_elems_written_per_iteration, _input, _weights, _output, _conv_info, _depth_multiplier, _dilation);
302  break;
303 #endif // __ARM_FEATURE_FP16_VECTOR_ARITHMETIC
304  case DataType::F32:
305  convolve_3x3<float, float>(window, _num_elems_written_per_iteration, _input, _weights, _output, _conv_info, _depth_multiplier, _dilation);
306  break;
307  case DataType::QASYMM8:
308  convolve_3x3<uint8_t, int32_t>(window, _num_elems_written_per_iteration, _input, _weights, _output, _conv_info, _depth_multiplier, _dilation);
309  break;
310  default:
311  ARM_COMPUTE_ERROR("Not implemented");
312  }
