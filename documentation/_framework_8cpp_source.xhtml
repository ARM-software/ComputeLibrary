265 void Framework::run_test(const TestInfo &info, TestCaseFactory &test_factory)
266 {
267  if(test_factory.status() == TestCaseFactory::Status::DISABLED)
268  {
269  log_test_skipped(info);
270  set_test_result(info, TestResult(TestResult::Status::DISABLED));
271  return;
272  }
273 
274  log_test_start(info);
275 
276  Profiler profiler = get_profiler();
277  TestResult result(TestResult::Status::NOT_RUN);
278 
279  _current_test_info = &info;
280  _current_test_result = &result;
281 
282  if(_log_level >= LogLevel::ERRORS)
283  {
284  func_on_all_printers([](Printer * p)
285  {
286  p->print_errors_header();
287  });
288  }
289 
290  const bool is_expected_failure = info.status == TestCaseFactory::Status::EXPECTED_FAILURE;
291 
292  try
293  {
294  std::unique_ptr<TestCase> test_case = test_factory.make();
295 
296  try
297  {
298  test_case->do_setup();
299 
300  for(int i = 0; i < _num_iterations; ++i)
301  {
302  //Start the profiler if:
303  //- there is only one iteration
304  //- it's not the first iteration of a multi-iterations run.
305  //
306  //Reason: if the CLTuner is enabled then the first run will be really messy
307  //as each kernel will be executed several times, messing up the instruments like OpenCL timers.
308  if(_num_iterations == 1 || i != 0)
309  {
310  profiler.start();
311  }
312  test_case->do_run();
313 #ifdef ARM_COMPUTE_CL
314  if(opencl_is_available())
315  {
316  CLScheduler::get().sync();
317  }
318 #endif /* ARM_COMPUTE_CL */
319 #ifdef ARM_COMPUTE_GC
320  if(opengles31_is_available())
321  {
322  GCScheduler::get().sync();
323  }
324 #endif /* ARM_COMPUTE_GC */
325  if(_num_iterations == 1 || i != 0)
326  {
327  profiler.stop();
328  }
329  }
330 
331  test_case->do_teardown();
332 
333  // Change status to success if no error has happend
334  if(result.status == TestResult::Status::NOT_RUN)
335  {
336  result.status = TestResult::Status::SUCCESS;
337  }
338  }
339  catch(const FileNotFound &error)
340  {
341  if(_error_on_missing_assets)
342  {
343  if(_log_level >= LogLevel::ERRORS)
344  {
345  TestError test_error(error.what(), LogLevel::ERRORS);
346  func_on_all_printers([&](Printer * p)
347  {
348  p->print_error(test_error, is_expected_failure);
349  });
350  }
351 
352  result.status = TestResult::Status::FAILED;
353 
354  if(_throw_errors)
355  {
356  throw;
357  }
358  }
359  else
360  {
361  if(_log_level >= LogLevel::DEBUG)
362  {
363  func_on_all_printers([&](Printer * p)
364  {
365  p->print_info(error.what());
366  });
367  }
368 
369  result.status = TestResult::Status::NOT_RUN;
370  }
371  }
372  catch(const TestError &error)
373  {
374  if(_log_level >= error.level())
375  {
376  func_on_all_printers([&](Printer * p)
377  {
378  p->print_error(error, is_expected_failure);
379  });
380  }
381 
382  result.status = TestResult::Status::FAILED;
383 
384  if(_throw_errors)
385  {
386  throw;
387  }
388  }
389 #ifdef ARM_COMPUTE_CL
390  catch(const ::cl::Error &error)
391  {
392  if(_log_level >= LogLevel::ERRORS)
393  {
394  std::stringstream stream;
395  stream << "Error code: " << error.err();
396  TestError test_error(error.what(), LogLevel::ERRORS, stream.str());
397  func_on_all_printers([&](Printer * p)
398  {
399  p->print_error(test_error, is_expected_failure);
400  });
401  }
402 
403  result.status = TestResult::Status::FAILED;
404 
405  if(_throw_errors)
406  {
407  throw;
408  }
409  }
410 #endif /* ARM_COMPUTE_CL */
411  catch(const std::exception &error)
412  {
413  if(_log_level >= LogLevel::ERRORS)
414  {
415  func_on_all_printers([&](Printer * p)
416  {
417  p->print_error(error, is_expected_failure);
418  });
419  }
420 
421  result.status = TestResult::Status::CRASHED;
422 
423  if(_throw_errors)
424  {
425  throw;
426  }
427  }
428  catch(...)
429  {
430  if(_log_level >= LogLevel::ERRORS)
431  {
432  func_on_all_printers([&](Printer * p)
433  {
434  p->print_error(TestError("Received unknown exception"), is_expected_failure);
435  });
436  }
437 
438  result.status = TestResult::Status::CRASHED;
439 
440  if(_throw_errors)
441  {
442  throw;
443  }
444  }
445  }
446  catch(const std::exception &error)
447  {
448  if(_log_level >= LogLevel::ERRORS)
449  {
450  func_on_all_printers([&](Printer * p)
451  {
452  p->print_error(error, is_expected_failure);
453  });
454  }
455 
456  result.status = TestResult::Status::CRASHED;
457 
458  if(_throw_errors)
459  {
460  throw;
461  }
462  }
463  catch(...)
464  {
465  if(_log_level >= LogLevel::ERRORS)
466  {
467  func_on_all_printers([&](Printer * p)
468  {
469  p->print_error(TestError("Received unknown exception"), is_expected_failure);
470  });
471  }
472 
473  result.status = TestResult::Status::CRASHED;
474 
475  if(_throw_errors)
476  {
477  throw;
478  }
479  }
480 
481  if(_log_level >= LogLevel::ERRORS)
482  {
483  func_on_all_printers([](Printer * p)
484  {
485  p->print_errors_footer();
486  });
487  }
488 
489  _current_test_info = nullptr;
490  _current_test_result = nullptr;
491 
492  if(result.status == TestResult::Status::FAILED)
493  {
494  if(info.status == TestCaseFactory::Status::EXPECTED_FAILURE)
495  {
496  result.status = TestResult::Status::EXPECTED_FAILURE;
497  }
498  }
499 
500  if(result.status == TestResult::Status::FAILED || result.status == TestResult::Status::CRASHED)
501  {
502  if(_stop_on_error)
503  {
504  throw std::runtime_error("Abort on first error.");
505  }
506  }
507 
508  result.measurements = profiler.measurements();
509 
510  set_test_result(info, result);
511  log_test_end(info);
512 }

514 bool Framework::run()
515 {
516  // Clear old test results
517  _test_results.clear();
518 
519  if(_log_level >= LogLevel::TESTS)
520  {
521  func_on_all_printers([](Printer * p)
522  {
523  p->print_run_header();
524  });
525  }
526 
527  const std::chrono::time_point<std::chrono::high_resolution_clock> start = std::chrono::high_resolution_clock::now();
528 
529  int id = 0;
530 
531  for(auto &test_factory : _test_factories)
532  {
533  const std::string test_case_name = test_factory->name();
534  const TestInfo test_info{ id, test_case_name, test_factory->mode(), test_factory->status() };
535 
536  if(_test_filter.is_selected(test_info))
537  {
538  run_test(test_info, *test_factory);
539  }
540 
541  ++id;
542  }
543 
544  const std::chrono::time_point<std::chrono::high_resolution_clock> end = std::chrono::high_resolution_clock::now();
545 
546  if(_log_level >= LogLevel::TESTS)
547  {
548  func_on_all_printers([](Printer * p)
549  {
550  p->print_run_footer();
551  });
552  }
553 
554  auto runtime = std::chrono::duration_cast<std::chrono::seconds>(end - start);
555  std::map<TestResult::Status, int> results = count_test_results();
556 
557  if(_log_level > LogLevel::NONE)
558  {
559  std::cout << "Executed " << _test_results.size() << " test(s) ("
560  << results[TestResult::Status::SUCCESS] << " passed, "
561  << results[TestResult::Status::EXPECTED_FAILURE] << " expected failures, "
562  << results[TestResult::Status::FAILED] << " failed, "
563  << results[TestResult::Status::CRASHED] << " crashed, "
564  << results[TestResult::Status::DISABLED] << " disabled) in " << runtime.count() << " second(s)\n";
565  }
566 
567  int num_successful_tests = results[TestResult::Status::SUCCESS] + results[TestResult::Status::EXPECTED_FAILURE] + results[TestResult::Status::DISABLED];
568 
569  return (static_cast<unsigned int>(num_successful_tests) == _test_results.size());
570 }

591 Profiler Framework::get_profiler() const
592 {
593  Profiler profiler;
594 
595  const bool all_instruments = std::any_of(
596  _instruments.begin(),
597  _instruments.end(),
598  [](InstrumentsDescription type) -> bool { return type.first == InstrumentType::ALL; });
599 
600  auto is_selected = [&](InstrumentsDescription instrument) -> bool
601  {
602  return std::find_if(_instruments.begin(), _instruments.end(), [&](InstrumentsDescription type) -> bool {
603  const auto group = static_cast<InstrumentType>(static_cast<uint64_t>(type.first) & 0xFF00);
604  return (group == instrument.first) && (instrument.second == type.second);
605  })
606  != _instruments.end();
607  };
608 
609  for(const auto &instrument : _available_instruments)
610  {
611  if(all_instruments || is_selected(instrument.first))
612  {
613  profiler.add(instrument.second());
614  }
615  }
616 
617  return profiler;
618 }
