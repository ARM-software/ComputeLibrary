281 void Framework::run_test(const TestInfo &info, TestCaseFactory &test_factory)
282 {
283  if(test_factory.status() == TestCaseFactory::Status::DISABLED)
284  {
285  log_test_skipped(info);
286  set_test_result(info, TestResult(TestResult::Status::DISABLED));
287  return;
288  }
289 
290  log_test_start(info);
291 
292  Profiler profiler = get_profiler();
293  TestResult result(TestResult::Status::NOT_RUN);
294 
295  _current_test_info = &info;
296  _current_test_result = &result;
297 
298  if(_log_level >= LogLevel::ERRORS)
299  {
300  func_on_all_printers([](Printer * p)
301  {
302  p->print_errors_header();
303  });
304  }
305 
306  const bool is_expected_failure = info.status == TestCaseFactory::Status::EXPECTED_FAILURE;
307 
308  try
309  {
310  std::unique_ptr<TestCase> test_case = test_factory.make();
311 
312  try
313  {
314  profiler.test_start();
315 
316  test_case->do_setup();
317 
318  for(int i = 0; i < _num_iterations; ++i)
319  {
320  //Start the profiler if:
321  //- there is only one iteration
322  //- it's not the first iteration of a multi-iterations run.
323  //
324  //Reason: if the CLTuner is enabled then the first run will be really messy
325  //as each kernel will be executed several times, messing up the instruments like OpenCL timers.
326  if(_num_iterations == 1 || i != 0)
327  {
328  profiler.start();
329  }
330  test_case->do_run();
331  test_case->do_sync();
332  if(_num_iterations == 1 || i != 0)
333  {
334  profiler.stop();
335  }
336  }
337 
338  test_case->do_teardown();
339 
340  profiler.test_stop();
341 
342  // Change status to success if no error has happend
343  if(result.status == TestResult::Status::NOT_RUN)
344  {
345  result.status = TestResult::Status::SUCCESS;
346  }
347  }
348  catch(const FileNotFound &error)
349  {
350  profiler.test_stop();
351  if(_error_on_missing_assets)
352  {
353  if(_log_level >= LogLevel::ERRORS)
354  {
355  TestError test_error(error.what(), LogLevel::ERRORS);
356  func_on_all_printers([&](Printer * p)
357  {
358  p->print_error(test_error, is_expected_failure);
359  });
360  }
361 
362  result.status = TestResult::Status::FAILED;
363 
364  if(_throw_errors)
365  {
366  throw;
367  }
368  }
369  else
370  {
371  if(_log_level >= LogLevel::DEBUG)
372  {
373  func_on_all_printers([&](Printer * p)
374  {
375  p->print_info(error.what());
376  });
377  }
378 
379  result.status = TestResult::Status::NOT_RUN;
380  }
381  }
382  catch(const TestError &error)
383  {
384  profiler.test_stop();
385  if(_log_level >= error.level())
386  {
387  func_on_all_printers([&](Printer * p)
388  {
389  p->print_error(error, is_expected_failure);
390  });
391  }
392 
393  result.status = TestResult::Status::FAILED;
394 
395  if(_throw_errors)
396  {
397  throw;
398  }
399  }
400 #ifdef ARM_COMPUTE_CL
401  catch(const ::cl::Error &error)
402  {
403  profiler.test_stop();
404  if(_log_level >= LogLevel::ERRORS)
405  {
406  std::stringstream stream;
407  stream << "Error code: " << error.err();
408  TestError test_error(error.what(), LogLevel::ERRORS, stream.str());
409  func_on_all_printers([&](Printer * p)
410  {
411  p->print_error(test_error, is_expected_failure);
412  });
413  }
414 
415  result.status = TestResult::Status::FAILED;
416 
417  if(_throw_errors)
418  {
419  throw;
420  }
421  }
422 #endif /* ARM_COMPUTE_CL */
423  catch(const std::exception &error)
424  {
425  profiler.test_stop();
426  if(_log_level >= LogLevel::ERRORS)
427  {
428  func_on_all_printers([&](Printer * p)
429  {
430  p->print_error(error, is_expected_failure);
431  });
432  }
433 
434  result.status = TestResult::Status::CRASHED;
435 
436  if(_throw_errors)
437  {
438  throw;
439  }
440  }
441  catch(...)
442  {
443  profiler.test_stop();
444  if(_log_level >= LogLevel::ERRORS)
445  {
446  func_on_all_printers([&](Printer * p)
447  {
448  p->print_error(TestError("Received unknown exception"), is_expected_failure);
449  });
450  }
451 
452  result.status = TestResult::Status::CRASHED;
453 
454  if(_throw_errors)
455  {
456  throw;
457  }
458  }
459  }
460  catch(const std::exception &error)
461  {
462  if(_log_level >= LogLevel::ERRORS)
463  {
464  func_on_all_printers([&](Printer * p)
465  {
466  p->print_error(error, is_expected_failure);
467  });
468  }
469 
470  result.status = TestResult::Status::CRASHED;
471 
472  if(_throw_errors)
473  {
474  throw;
475  }
476  }
477  catch(...)
478  {
479  if(_log_level >= LogLevel::ERRORS)
480  {
481  func_on_all_printers([&](Printer * p)
482  {
483  p->print_error(TestError("Received unknown exception"), is_expected_failure);
484  });
485  }
486 
487  result.status = TestResult::Status::CRASHED;
488 
489  if(_throw_errors)
490  {
491  throw;
492  }
493  }
494 
495  if(_log_level >= LogLevel::ERRORS)
496  {
497  func_on_all_printers([](Printer * p)
498  {
499  p->print_errors_footer();
500  });
501  }
502 
503  _current_test_info = nullptr;
504  _current_test_result = nullptr;
505 
506  if(result.status == TestResult::Status::FAILED)
507  {
508  if(info.status == TestCaseFactory::Status::EXPECTED_FAILURE)
509  {
510  result.status = TestResult::Status::EXPECTED_FAILURE;
511  }
512  }
513 
514  if(result.status == TestResult::Status::FAILED || result.status == TestResult::Status::CRASHED)
515  {
516  if(_stop_on_error)
517  {
518  throw std::runtime_error("Abort on first error.");
519  }
520  }
521 
522  result.measurements = profiler.measurements();
523 
524  set_test_result(info, result);
525  log_test_end(info);
526 }

528 bool Framework::run()
529 {
530  // Clear old test results
531  _test_results.clear();
532 
533  if(_log_level >= LogLevel::TESTS)
534  {
535  func_on_all_printers([](Printer * p)
536  {
537  p->print_run_header();
538  });
539  }
540 
541  const std::chrono::time_point<std::chrono::high_resolution_clock> start = std::chrono::high_resolution_clock::now();
542 
543  int id = 0;
544  int id_run_test = 0;
545 
546  for(auto &test_factory : _test_factories)
547  {
548  const std::string test_case_name = test_factory->name();
549  const TestInfo test_info{ id, test_case_name, test_factory->mode(), test_factory->status() };
550 
551  if(_test_filter.is_selected(test_info))
552  {
553 #ifdef ARM_COMPUTE_CL
554  // Every 100 tests, reset the OpenCL context to release the allocated memory
555  if(opencl_is_available() && (id_run_test % 100) == 0)
556  {
557  auto ctx_properties = CLScheduler::get().context().getInfo<CL_CONTEXT_PROPERTIES>(nullptr);
558  auto queue_properties = CLScheduler::get().queue().getInfo<CL_QUEUE_PROPERTIES>(nullptr);
559 
560  cl::Context new_ctx = cl::Context(CL_DEVICE_TYPE_DEFAULT, ctx_properties.data());
561  cl::CommandQueue new_queue = cl::CommandQueue(new_ctx, CLKernelLibrary::get().get_device(), queue_properties);
562 
563  CLKernelLibrary::get().clear_programs_cache();
564  CLScheduler::get().set_context(new_ctx);
565  CLScheduler::get().set_queue(new_queue);
566  }
567 #endif // ARM_COMPUTE_CL
568  run_test(test_info, *test_factory);
569 
570  ++id_run_test;
571  }
572 
573  ++id;
574  }
575 
576  const std::chrono::time_point<std::chrono::high_resolution_clock> end = std::chrono::high_resolution_clock::now();
577 
578  if(_log_level >= LogLevel::TESTS)
579  {
580  func_on_all_printers([](Printer * p)
581  {
582  p->print_run_footer();
583  });
584  }
585 
586  auto runtime = std::chrono::duration_cast<std::chrono::seconds>(end - start);
587  std::map<TestResult::Status, int> results = count_test_results();
588 
589  if(_log_level > LogLevel::NONE)
590  {
591  std::cout << "Executed " << _test_results.size() << " test(s) ("
592  << results[TestResult::Status::SUCCESS] << " passed, "
593  << results[TestResult::Status::EXPECTED_FAILURE] << " expected failures, "
594  << results[TestResult::Status::FAILED] << " failed, "
595  << results[TestResult::Status::CRASHED] << " crashed, "
596  << results[TestResult::Status::DISABLED] << " disabled) in " << runtime.count() << " second(s)\n";
597  }
598 
599  int num_successful_tests = results[TestResult::Status::SUCCESS] + results[TestResult::Status::EXPECTED_FAILURE] + results[TestResult::Status::DISABLED];
600 
601  return (static_cast<unsigned int>(num_successful_tests) == _test_results.size());
602 }

623 Profiler Framework::get_profiler() const
624 {
625  Profiler profiler;
626 
627  const bool all_instruments = std::any_of(
628  _instruments.begin(),
629  _instruments.end(),
630  [](InstrumentsDescription type) -> bool { return type.first == InstrumentType::ALL; });
631 
632  auto is_selected = [&](InstrumentsDescription instrument) -> bool
633  {
634  return std::find_if(_instruments.begin(), _instruments.end(), [&](InstrumentsDescription type) -> bool {
635  const auto group = static_cast<InstrumentType>(static_cast<uint64_t>(type.first) & 0xFF00);
636  return (group == instrument.first) && (instrument.second == type.second);
637  })
638  != _instruments.end();
639  };
640 
641  for(const auto &instrument : _available_instruments)
642  {
643  if(all_instruments || is_selected(instrument.first))
644  {
645  profiler.add(instrument.second());
646  }
647  }
648 
649  return profiler;
