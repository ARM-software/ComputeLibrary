104 __kernel void reduction_operation_x(
105  IMAGE_DECLARATION(src),
106  IMAGE_DECLARATION(partial_res),
107  __local DATA_TYPE *local_results)
108 {
109  Image src = CONVERT_TO_IMAGE_STRUCT(src);
110  Image partial_res = CONVERT_TO_IMAGE_STRUCT(partial_res);
111 
112  unsigned int lsize = get_local_size(0);
113  unsigned int lid = get_local_id(0);
114 
115  for(unsigned int y = 0; y < get_local_size(1); ++y)
116  {
117  local_results[lid] = OPERATION((__global DATA_TYPE *)offset(&src, 0, y));
118  barrier(CLK_LOCAL_MEM_FENCE);
119 
120  // Perform parallel reduction
121  for(unsigned int i = lsize >> 1; i > 0; i >>= 1)
122  {
123  if(lid < i)
124  {
125 #if defined(PROD)
126  local_results[lid] *= local_results[lid + i];
127 #else
128  local_results[lid] += local_results[lid + i];
129 #endif //defined(PROD)
130  }
131  barrier(CLK_LOCAL_MEM_FENCE);
132  }
133 
134  if(lid == 0)
135  {
136 #if defined(MEAN) && defined(WIDTH)
137  if(y == get_local_size(1) - 1)
138  {
139  local_results[0] /= WIDTH;
140  }
141 #endif /* defined(MEAN) && defined(WIDTH) */
142  ((__global DATA_TYPE *)offset(&partial_res, get_group_id(0), y))[0] = local_results[0];
143  }
144  }
145 }

173
