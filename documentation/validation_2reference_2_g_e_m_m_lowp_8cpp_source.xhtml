42 void quantize_down_int32_to_uint8_scale(const SimpleTensor<T> *in, const SimpleTensor<T> *bias, SimpleTensor<uint8_t> *dst, int32_t result_offset, int32_t result_mult_int, int32_t result_shift,
43  int32_t min, int32_t max)
44 {
45  const int cols_in = in->shape().x();
46 
47  for(int i = 0; i < in->num_elements(); ++i)
48  {
49  int32_t result = ((*in)[i] + result_offset);
50 
51  if(bias != nullptr)
52  {
53  result += (*bias)[i % cols_in];
54  }
55 
56  result *= result_mult_int;
57 
58  result >>= result_shift;
59 
60  // Bounded ReLu
61  if(min != max)
62  {
63  result = std::max(min, std::min(max, result));
64  }
65 
66  (*dst)[i] = static_cast<uint8_t>(std::max(0, std::min(255, result)));
67  }

100 template <typename T_out, typename T_in>
101 SimpleTensor<T_out> gemmlowp_matrix_multiply_core(const SimpleTensor<T_in> &a, const SimpleTensor<T_in> &b, int32_t a_offset, int32_t b_offset)
102 {
103  static_assert(std::is_same<typename std::decay<T_out>::type, int32_t>::value, "Only int32_t is allowed for the output");
104 
105  TensorShape shape(b.shape()[0], a.shape()[1]);
106  DataType dt = std::is_same<T_out, int32_t>::value ? DataType::S32 : DataType::U32;
107  SimpleTensor<T_out> c(shape, dt);
108 
109  const int K = a.shape().x();
110  const int b_width = b.shape().x();
111  const int rows = c.shape().y(); //M
112  const int cols = c.shape().x(); //N
113 
114  std::vector<T_out> acc;
115  acc.resize(cols);
116 
117  for(int i = 0; i < rows; ++i)
118  {
119  for(int j = 0; j < cols; ++j)
120  {
121  acc[j] = 0;
122  }
123  for(int k = 0; k < K; ++k)
124  {
125  const T_out tmp_a = a_offset + static_cast<T_out>(a[k + i * K]);
126  for(int j = 0; j < b_width; ++j)
127  {
128  const T_out tmp_b = b_offset + static_cast<T_out>(b[j + k * b_width]);
129  const T_out mult_as_int = tmp_a * tmp_b;
130  acc[j] += mult_as_int;
131  }
132  }
133  for(int j = 0; j < cols; ++j)
134  {
135  c[j + i * cols] = acc[j];
136  }
137  }
138 
139  return c;
140 }
