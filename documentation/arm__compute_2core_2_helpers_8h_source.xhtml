99 template <typename T>
100 inline T delta_bilinear_c1(const T *pixel_ptr, size_t stride, float dx, float dy)
101 {
102  ARM_COMPUTE_ERROR_ON(pixel_ptr == nullptr);
103 
104  const float dx1 = 1.0f - dx;
105  const float dy1 = 1.0f - dy;
106 
107  const T a00 = *pixel_ptr;
108  const T a01 = *(pixel_ptr + 1);
109  const T a10 = *(pixel_ptr + stride);
110  const T a11 = *(pixel_ptr + stride + 1);
111 
112  const float w1 = dx1 * dy1;
113  const float w2 = dx * dy1;
114  const float w3 = dx1 * dy;
115  const float w4 = dx * dy;
116 
117  return static_cast<T>(a00 * w1 + a01 * w2 + a10 * w3 + a11 * w4);
118 }

209 template <typename T>
210 inline uint8_t pixel_bilinear_c1_clamp(const T *first_pixel_ptr, size_t stride, size_t width, size_t height, float x, float y)
211 {
212  ARM_COMPUTE_ERROR_ON(first_pixel_ptr == nullptr);
213 
214  x = std::max(-1.f, std::min(x, static_cast<float>(width)));
215  y = std::max(-1.f, std::min(y, static_cast<float>(height)));
216 
217  const float xi = std::floor(x);
218  const float yi = std::floor(y);
219 
220  const float dx = x - xi;
221  const float dy = y - yi;
222 
223  if(dx == 0.0f)
224  {
225  if(dy == 0.0f)
226  {
227  return static_cast<T>(first_pixel_ptr[static_cast<int32_t>(xi) + static_cast<int32_t>(yi) * stride]);
228  }
229  return delta_linear_c1_y(first_pixel_ptr + static_cast<int32_t>(xi) + static_cast<int32_t>(yi) * stride, stride, dy);
230  }
231  if(dy == 0.0f)
232  {
233  return delta_linear_c1_x(first_pixel_ptr + static_cast<int32_t>(xi) + static_cast<int32_t>(yi) * stride, dx);
234  }
235  return delta_bilinear_c1(first_pixel_ptr + static_cast<int32_t>(xi) + static_cast<int32_t>(yi) * stride, stride, dx, dy);
236 }

466 template <typename... Ts>
467 ValidRegion intersect_valid_regions(Ts &&... regions)
468 {
469  auto intersect = [](const ValidRegion & r1, const ValidRegion & r2) -> ValidRegion
470  {
471  ValidRegion region;
472 
473  for(size_t d = 0; d < std::min(r1.anchor.num_dimensions(), r2.anchor.num_dimensions()); ++d)
474  {
475  region.anchor.set(d, std::max(r1.anchor[d], r2.anchor[d]));
476  }
477 
478  for(size_t d = 0; d < std::min(r1.shape.num_dimensions(), r2.shape.num_dimensions()); ++d)
479  {
480  region.shape.set(d, std::min(r1.shape[d], r2.shape[d]));
481  }
482 
483  return region;
484  };
485 
486  return foldl(intersect, std::forward<Ts>(regions)...);
487 }
