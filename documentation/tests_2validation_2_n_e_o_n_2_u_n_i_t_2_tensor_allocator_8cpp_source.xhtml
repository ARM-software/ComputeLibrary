46 TEST_CASE(ImportMemory, framework::DatasetMode::ALL)
47 {
48  // Init tensor info
49  TensorInfo info(TensorShape(24U, 16U, 3U), 1, DataType::F32);
50 
51  // Allocate memory buffer
52  const size_t total_size = info.total_size();
53  auto data = support::cpp14::make_unique<uint8_t[]>(total_size);
54 
55  // Negative case : Import pointer with zero size
56  Tensor t1;
57  t1.allocator()->init(info);
58  ARM_COMPUTE_EXPECT(!bool(t1.allocator()->import_memory(data.get(), 0)), framework::LogLevel::ERRORS);
59  ARM_COMPUTE_EXPECT(t1.info()->is_resizable(), framework::LogLevel::ERRORS);
60 
61  // Negative case : Import nullptr
62  Tensor t2;
63  t2.allocator()->init(info);
64  ARM_COMPUTE_EXPECT(!bool(t2.allocator()->import_memory(nullptr, total_size)), framework::LogLevel::ERRORS);
65  ARM_COMPUTE_EXPECT(t2.info()->is_resizable(), framework::LogLevel::ERRORS);
66 
67  // Negative case : Import memory to a tensor that is memory managed
68  Tensor t3;
69  MemoryGroup mg;
70  t3.allocator()->set_associated_memory_group(&mg);
71  ARM_COMPUTE_EXPECT(!bool(t3.allocator()->import_memory(data.get(), total_size)), framework::LogLevel::ERRORS);
72  ARM_COMPUTE_EXPECT(t3.info()->is_resizable(), framework::LogLevel::ERRORS);
73 
74  // Positive case : Set raw pointer
75  Tensor t4;
76  t4.allocator()->init(info);
77  ARM_COMPUTE_EXPECT(bool(t4.allocator()->import_memory(data.get(), total_size)), framework::LogLevel::ERRORS);
78  ARM_COMPUTE_EXPECT(!t4.info()->is_resizable(), framework::LogLevel::ERRORS);
79  ARM_COMPUTE_EXPECT(t4.buffer() == reinterpret_cast<uint8_t *>(data.get()), framework::LogLevel::ERRORS);
80  t4.allocator()->free();
81  ARM_COMPUTE_EXPECT(t4.info()->is_resizable(), framework::LogLevel::ERRORS);
82  ARM_COMPUTE_EXPECT(t4.buffer() == nullptr, framework::LogLevel::ERRORS);
83 }
