53 TEST_CASE(ImportMemory, framework::DatasetMode::ALL)
54 {
55  // Init tensor info
56  TensorInfo info(TensorShape(24U, 16U, 3U), 1, DataType::F32);
57 
58  // Allocate memory buffer
59  const size_t total_size = info.total_size();
60  auto data = support::cpp14::make_unique<uint8_t[]>(total_size);
61 
62  // Negative case : Import nullptr
63  Tensor t1;
64  t1.allocator()->init(info);
65  ARM_COMPUTE_EXPECT(!bool(t1.allocator()->import_memory(nullptr)), framework::LogLevel::ERRORS);
66  ARM_COMPUTE_EXPECT(t1.info()->is_resizable(), framework::LogLevel::ERRORS);
67 
68  // Negative case : Import misaligned pointer
69  Tensor t2;
70  const size_t required_alignment = 339;
71  t2.allocator()->init(info, required_alignment);
72  ARM_COMPUTE_EXPECT(!bool(t2.allocator()->import_memory(data.get())), framework::LogLevel::ERRORS);
73  ARM_COMPUTE_EXPECT(t2.info()->is_resizable(), framework::LogLevel::ERRORS);
74 
75  // Negative case : Import memory to a tensor that is memory managed
76  Tensor t3;
77  MemoryGroup mg;
78  t3.allocator()->set_associated_memory_group(&mg);
79  ARM_COMPUTE_EXPECT(!bool(t3.allocator()->import_memory(data.get())), framework::LogLevel::ERRORS);
80  ARM_COMPUTE_EXPECT(t3.info()->is_resizable(), framework::LogLevel::ERRORS);
81 
82  // Positive case : Set raw pointer
83  Tensor t4;
84  t4.allocator()->init(info);
85  ARM_COMPUTE_EXPECT(bool(t4.allocator()->import_memory(data.get())), framework::LogLevel::ERRORS);
86  ARM_COMPUTE_EXPECT(!t4.info()->is_resizable(), framework::LogLevel::ERRORS);
87  ARM_COMPUTE_EXPECT(t4.buffer() == reinterpret_cast<uint8_t *>(data.get()), framework::LogLevel::ERRORS);
88  t4.allocator()->free();
89  ARM_COMPUTE_EXPECT(t4.info()->is_resizable(), framework::LogLevel::ERRORS);
90  ARM_COMPUTE_EXPECT(t4.buffer() == nullptr, framework::LogLevel::ERRORS);
91 }

93 TEST_CASE(ImportMemoryMalloc, framework::DatasetMode::ALL)
94 {
95  const ActivationLayerInfo act_info(ActivationLayerInfo::ActivationFunction::RELU);
96  const TensorShape shape = TensorShape(24U, 16U, 3U);
97  const DataType data_type = DataType::F32;
98 
99  // Create tensor
100  const TensorInfo info(shape, 1, data_type);
101  const size_t required_alignment = 64;
102  Tensor tensor;
103  tensor.allocator()->init(info, required_alignment);
104 
105  // Create and configure activation function
106  NEActivationLayer act_func;
107  act_func.configure(&tensor, nullptr, act_info);
108 
109  // Allocate and import tensor
110  const size_t total_size_in_elems = tensor.info()->tensor_shape().total_size();
111  const size_t total_size_in_bytes = tensor.info()->total_size();
112  size_t space = total_size_in_bytes + required_alignment;
113  auto raw_data = support::cpp14::make_unique<uint8_t[]>(space);
114 
115  void *aligned_ptr = raw_data.get();
116  support::cpp11::align(required_alignment, total_size_in_bytes, aligned_ptr, space);
117 
118  ARM_COMPUTE_EXPECT(bool(tensor.allocator()->import_memory(aligned_ptr)), framework::LogLevel::ERRORS);
119  ARM_COMPUTE_EXPECT(!tensor.info()->is_resizable(), framework::LogLevel::ERRORS);
120 
121  // Fill tensor
122  std::uniform_real_distribution<float> distribution(-5.f, 5.f);
123  std::mt19937 gen(library->seed());
124  auto *typed_ptr = reinterpret_cast<float *>(aligned_ptr);
125  for(unsigned int i = 0; i < total_size_in_elems; ++i)
126  {
127  typed_ptr[i] = distribution(gen);
128  }
129 
130  // Execute function and sync
131  act_func.run();
132 
133  // Validate result by checking that the input has no negative values
134  for(unsigned int i = 0; i < total_size_in_elems; ++i)
135  {
136  ARM_COMPUTE_EXPECT(typed_ptr[i] >= 0, framework::LogLevel::ERRORS);
137  }
138 
139  // Release resources
140  tensor.allocator()->free();
141  ARM_COMPUTE_EXPECT(tensor.info()->is_resizable(), framework::LogLevel::ERRORS);
142 }
