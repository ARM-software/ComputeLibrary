39 // Implementation of the GemmCommon abstract class.
40 //
41 // This is implementation is for GEMV with pretransposition.
42 //
43 // batches are not supported as a batched GEMV makes no sense (can be converted to a GEMM).
44 template<typename strategy, typename To, typename Tr>
45 class GemvPretransposed : public GemmCommon<To, Tr> {
46  typedef typename strategy::operand_type Toi;
47  typedef typename strategy::result_type Tri;
48

67 public:
68  GemvPretransposed(GemvPretransposed &) = delete;
69  GemvPretransposed & operator= (GemvPretransposed &) = delete;

71  GemvPretransposed(const GemmArgs<Tr> &args)
72  : _Nsize(args._Nsize), _Ksize(args._Ksize), _nmultis(args._nmulti), _trB(args._trB), _beta(args._beta), _ci(args._ci),
73  _buffer_per_multi(_Ksize * iceildiv(_Nsize, strategy::A_interleave()) * strategy::A_interleave()) {
74  /* For now don't do any blocking. TODO: figure out if we should. */
75  if (args._cfg && args._cfg->inner_block_size) {
76  m_block = args._cfg->inner_block_size;
77  } else {
78  m_block = _Ksize;
79  }
80 
81  if (args._cfg && args._cfg->outer_block_size) {
82  n_block = args._cfg->outer_block_size;
83  } else {
84  n_block = _Nsize;
85  }
86  }

88  // Window is number of out_width blocks, times number of multis.
89  unsigned int get_window_size() const override {
90  return iceildiv(_Nsize, strategy::out_width()) * _nmultis;
91  }
92 
93  // Actually execute the GEMV.
94  void execute(unsigned int start, unsigned int end, int) override {
95 #ifdef CYCLE_PROFILING
96  profiler prof;
97 #endif
98  strategy strat(_ci);
99 
100  /* Break the window values down into multis of interest... */
101  const unsigned int window_per_multi = iceildiv(_Nsize, strategy::out_width());
102  const unsigned int multi_0 = start / window_per_multi;
103  const unsigned int multi_end = end / window_per_multi;
104 
105  /* ... and figure out where we start and end in the first and last multi. */
106  const unsigned int n_0 = (start - (multi_0 * window_per_multi)) * strategy::out_width();
107  const unsigned int n_max = (end - (multi_end * window_per_multi)) * strategy::out_width();
108 
109  static_assert(std::is_same<Tr, Tri>::value, "GemvPretransposed: Result types must be the same.");
110 
111  for (unsigned int multi=multi_0; multi<=multi_end; multi++) {
112  const unsigned int n_start = (multi==multi_0) ? n_0 : 0;
113  const unsigned int n_end = (multi==multi_end) ? n_max : _Nsize;
114 
115  if (n_end <= n_start)
116  continue;
117 
118  for (unsigned int m0=0; m0<_Ksize; m0+=m_block) {
119  unsigned int mmax = std::min(m0 + m_block, _Ksize);
120 
121  for (unsigned int n=n_start; n<n_end; n+=n_block) {
122  unsigned int nmax = std::min(n + n_block, n_end);
123 #ifdef CYCLE_PROFILING
124  auto p = prof.ScopedProfiler(PROFILE_KERNEL, (mmax-m0) * (nmax-n));
125 #endif
126  /* This assumes that the underlying call was a GEMM with M=1; for the N=1 case we would have to pick up this->_Bptr below instead */
127  strat.kernel(_A_pretransposed + (multi * _buffer_per_multi) + (n * _Ksize) + (m0 * strategy::A_interleave()),
128  (_Ksize * strategy::A_interleave()),
129  this->_Aptr + (multi * this->_A_multi_stride) + m0,
130  this->_Cptr + (multi * this->_C_multi_stride) + n,
131  _beta, (mmax-m0), (nmax-n));
132  }
133  }
134  }
135  }

151  using GemmCommon<To, Tr>::pretranspose_B_array;
152  void pretranspose_B_array(void *buffer, const To *B, const int ldb, const int B_multi_stride) override {
153  Toi *A_buffer = reinterpret_cast<Toi *>(buffer);
154 
155  for (unsigned int multi=0; multi<_nmultis; multi++) {
156  /* Reverse sense here as we are dealing with B rather than A. So if
157  * strategy::A_transpose is false and _trB is false, we still
158  * transpose. */
159  if (_trB ^ strategy::A_transpose()) {
160  Transform<strategy::A_interleave(), strategy::A_block(), false>(A_buffer + (multi * _buffer_per_multi), B + (multi * B_multi_stride), ldb, 0, _Nsize, 0, _Ksize);
161  } else {
162  Transform<strategy::A_interleave(), strategy::A_block(), true>(A_buffer + (multi * _buffer_per_multi), B + (multi * B_multi_stride), ldb, 0, _Nsize, 0, _Ksize);
163  }
164  }
165 
166  _A_pretransposed = A_buffer;
167  }
168 
169  void set_pretransposed_B_data(void *buffer) override {
170  _A_pretransposed = reinterpret_cast<Toi *>(buffer);
171  }
172 };
