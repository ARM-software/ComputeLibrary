251 template <typename U>
252 struct compare<AbsoluteTolerance<U>> : public compare_base<AbsoluteTolerance<U>>
253 {
254  using compare_base<AbsoluteTolerance<U>>::compare_base;
255 
256  operator bool() const
257  {
258  if(!std::isfinite(this->_target) || !std::isfinite(this->_reference))
259  {
260  return false;
261  }
262  else if(this->_target == this->_reference)
263  {
264  return true;
265  }
266 
267  using comparison_type = typename std::conditional<std::is_integral<U>::value, int64_t, U>::type;
268 
269  const comparison_type abs_difference(std::abs(static_cast<comparison_type>(this->_target) - static_cast<comparison_type>(this->_reference)));
270 
271  return abs_difference <= static_cast<comparison_type>(this->_tolerance);
272  }
273 };

275 template <typename U>
276 struct compare<RelativeTolerance<U>> : public compare_base<RelativeTolerance<U>>
277 {
278  using compare_base<RelativeTolerance<U>>::compare_base;
279 
280  operator bool() const
281  {
282  if(!std::isfinite(this->_target) || !std::isfinite(this->_reference))
283  {
284  return false;
285  }
286  else if(this->_target == this->_reference)
287  {
288  return true;
289  }
290 
291  const U epsilon = (std::is_same<half, typename std::remove_cv<U>::type>::value || (this->_reference == 0)) ? static_cast<U>(0.01) : static_cast<U>(1e-05);
292 
293  if(std::abs(static_cast<double>(this->_reference) - static_cast<double>(this->_target)) <= epsilon)
294  {
295  return true;
296  }
297  else
298  {
299  if(static_cast<double>(this->_reference) == 0.0f) // We have checked whether _reference and _target is closing. If _reference is 0 but not closed to _target, it should return false
300  {
301  return false;
302  }
303 
304  const double relative_change = std::abs(static_cast<double>(this->_target) - static_cast<double>(this->_reference)) / this->_reference;
305 
306  return relative_change <= static_cast<U>(this->_tolerance);
307  }
308  }

318 template <typename T, typename U>
319 void validate(const IAccessor &tensor, const SimpleTensor<T> &reference, const ValidRegion &valid_region, U tolerance_value, float tolerance_number)
320 {
321  int64_t num_mismatches = 0;
322  int64_t num_elements = 0;
323 
324  ARM_COMPUTE_EXPECT_EQUAL(tensor.element_size(), reference.element_size(), framework::LogLevel::ERRORS);
325  ARM_COMPUTE_EXPECT_EQUAL(tensor.data_type(), reference.data_type(), framework::LogLevel::ERRORS);
326 
327  if(reference.format() != Format::UNKNOWN)
328  {
329  ARM_COMPUTE_EXPECT_EQUAL(tensor.format(), reference.format(), framework::LogLevel::ERRORS);
330  }
331 
332  ARM_COMPUTE_EXPECT_EQUAL(tensor.num_channels(), reference.num_channels(), framework::LogLevel::ERRORS);
333  ARM_COMPUTE_EXPECT(compare_dimensions(tensor.shape(), reference.shape()), framework::LogLevel::ERRORS);
334 
335  const int min_elements = std::min(tensor.num_elements(), reference.num_elements());
336  const int min_channels = std::min(tensor.num_channels(), reference.num_channels());
337 
338  // Iterate over all elements within valid region, e.g. U8, S16, RGB888, ...
339  for(int element_idx = 0; element_idx < min_elements; ++element_idx)
340  {
341  const Coordinates id = index2coord(reference.shape(), element_idx);
342 
343  if(is_in_valid_region(valid_region, id))
344  {
345  // Iterate over all channels within one element
346  for(int c = 0; c < min_channels; ++c)
347  {
348  const T &target_value = reinterpret_cast<const T *>(tensor(id))[c];
349  const T &reference_value = reinterpret_cast<const T *>(reference(id))[c];
350 
351  if(!compare<U>(target_value, reference_value, tolerance_value))
352  {
353  ARM_COMPUTE_TEST_INFO("id = " << id);
354  ARM_COMPUTE_TEST_INFO("channel = " << c);
355  ARM_COMPUTE_TEST_INFO("target = " << std::setprecision(5) << framework::make_printable(target_value));
356  ARM_COMPUTE_TEST_INFO("reference = " << std::setprecision(5) << framework::make_printable(reference_value));
357  ARM_COMPUTE_TEST_INFO("tolerance = " << std::setprecision(5) << framework::make_printable(static_cast<typename U::value_type>(tolerance_value)));
358  framework::ARM_COMPUTE_PRINT_INFO();
359 
360  ++num_mismatches;
361  }
362 
363  ++num_elements;
364  }
365  }
366  }
367 
368  if(num_elements > 0)
369  {
370  const int64_t absolute_tolerance_number = tolerance_number * num_elements;
371  const float percent_mismatches = static_cast<float>(num_mismatches) / num_elements * 100.f;
372 
373  ARM_COMPUTE_TEST_INFO(num_mismatches << " values (" << std::fixed << std::setprecision(2) << percent_mismatches
374  << "%) mismatched (maximum tolerated " << std::setprecision(2) << tolerance_number << "%)");
375  ARM_COMPUTE_EXPECT(num_mismatches <= absolute_tolerance_number, framework::LogLevel::ERRORS);
376  }

380 template <typename T, typename U, typename V>
381 std::pair<int64_t, int64_t> compare_keypoints(T first1, T last1, U first2, U last2, V tolerance)
382 {
383  int64_t num_missing = 0;
384  int64_t num_mismatches = 0;
385 
386  while(first1 != last1)
387  {
388  const auto point = std::find_if(first2, last2, [&](KeyPoint point)
389  {
390  return point.x == first1->x && point.y == first1->y;
391  });
392 
393  if(point == last2)
394  {
395  ++num_missing;
396  ARM_COMPUTE_TEST_INFO("keypoint1 = " << *first1)
397  ARM_COMPUTE_EXPECT_FAIL("Key point not found", framework::LogLevel::DEBUG);
398  }
399  else if(!validate(point->tracking_status, first1->tracking_status) || !validate(point->strength, first1->strength, tolerance) || !validate(point->scale, first1->scale)
400  || !validate(point->orientation, first1->orientation) || !validate(point->error, first1->error))
401  {
402  ++num_mismatches;
403  ARM_COMPUTE_TEST_INFO("keypoint1 = " << *first1)
404  ARM_COMPUTE_TEST_INFO("keypoint2 = " << *point)
405  ARM_COMPUTE_EXPECT_FAIL("Mismatching keypoint", framework::LogLevel::DEBUG);
406  }
407 
408  ++first1;
409  }
410 
411  return std::make_pair(num_missing, num_mismatches);
412 }

414 template <typename T, typename U, typename V>
415 void validate_keypoints(T target_first, T target_last, U reference_first, U reference_last, V tolerance)
416 {
417  const int64_t num_elements_target = std::distance(target_first, target_last);
418  const int64_t num_elements_reference = std::distance(reference_first, reference_last);
419 
420  ARM_COMPUTE_EXPECT_EQUAL(num_elements_target, num_elements_reference, framework::LogLevel::ERRORS);
421 
422  int64_t num_missing = 0;
423  int64_t num_mismatches = 0;
424 
425  if(num_elements_reference > 0)
426  {
427  std::tie(num_missing, num_mismatches) = compare_keypoints(reference_first, reference_last, target_first, target_last, tolerance);
428 
429  const float percent_missing = static_cast<float>(num_missing) / num_elements_reference * 100.f;
430  const float percent_mismatches = static_cast<float>(num_mismatches) / num_elements_reference * 100.f;
431 
432  ARM_COMPUTE_TEST_INFO(num_missing << " keypoints (" << std::fixed << std::setprecision(2) << percent_missing << "%) are missing in target");
433  ARM_COMPUTE_EXPECT_EQUAL(num_missing, 0, framework::LogLevel::ERRORS);
434 
435  ARM_COMPUTE_TEST_INFO(num_mismatches << " keypoints (" << std::fixed << std::setprecision(2) << percent_mismatches << "%) mismatched");
436  ARM_COMPUTE_EXPECT_EQUAL(num_mismatches, 0, framework::LogLevel::ERRORS);
437  }
438 
439  if(num_elements_target > 0)
440  {
441  std::tie(num_missing, num_mismatches) = compare_keypoints(target_first, target_last, reference_first, reference_last, tolerance);
442 
443  const float percent_missing = static_cast<float>(num_missing) / num_elements_target * 100.f;
444 
445  ARM_COMPUTE_TEST_INFO(num_missing << " keypoints (" << std::fixed << std::setprecision(2) << percent_missing << "%) are not part of target");
446  ARM_COMPUTE_EXPECT_EQUAL(num_missing, 0, framework::LogLevel::ERRORS);
447  }
