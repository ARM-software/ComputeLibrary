41 TensorDescriptor FullyConnectedLayerNode::compute_weights_descriptor(const TensorDescriptor &input_descriptor,
42  unsigned int num_outputs,
43  FullyConnectedLayerInfo fc_info,
44  const QuantizationInfo &weights_quant_info)
45 {
46  unsigned int num_weights = 1;
47  unsigned int num_dimensions = input_descriptor.shape.num_dimensions();
48  // Ignore the batch dimension if there is one:
49  if(num_dimensions == 2 || num_dimensions == 4)
50  {
51  num_dimensions--;
52  }
53  for(unsigned int i = 0; i < num_dimensions; i++)
54  {
55  num_weights *= input_descriptor.shape[i];
56  }
57 
58  TensorDescriptor weights_descriptor = input_descriptor;
59  weights_descriptor.shape = TensorShape(num_weights, num_outputs);
60 
61  // If weights are tranposed, use tranposed shape
62  if(!fc_info.transpose_weights)
63  {
64  weights_descriptor.shape = TensorShape(num_outputs, num_weights);
65  }
66 
67  // Set quantization info if present
68  if(!weights_quant_info.empty())
69  {
70  weights_descriptor.quant_info = weights_quant_info;
71  }
72 
73  return weights_descriptor;
74 }

76 TensorDescriptor FullyConnectedLayerNode::compute_output_descriptor(const TensorDescriptor &input_descriptor,
77  unsigned int num_outputs,
78  const QuantizationInfo &out_quant_info)
79 {
80  // Note: Only 1D batch space is supported at the moment
81  unsigned int batches = input_descriptor.shape[1];
82  if(input_descriptor.shape.num_dimensions() > 2)
83  {
84  batches = input_descriptor.shape[3];
85  }
86 
87  // Set descriptor shape
88  TensorDescriptor output_descriptor = input_descriptor;
89  output_descriptor.shape = TensorShape(num_outputs, batches);
90 
91  // Set quantization info if present
92  if(!out_quant_info.empty())
93  {
94  output_descriptor.quant_info = out_quant_info;
95  }
96 
97  return output_descriptor;
98 }
