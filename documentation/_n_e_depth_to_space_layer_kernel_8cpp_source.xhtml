41 Status validate_arguments(const ITensorInfo *input, const ITensorInfo *output, int32_t block_shape)
42 {
43  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(input, output);
44  ARM_COMPUTE_RETURN_ERROR_ON(input->num_dimensions() > 4);
45  ARM_COMPUTE_RETURN_ERROR_ON(block_shape < 2);
46 
47  const DataLayout data_layout = input->data_layout();
48  const int idx_channel = get_data_layout_dimension_index(data_layout, DataLayoutDimension::CHANNEL);
49  ARM_COMPUTE_RETURN_ERROR_ON(input->tensor_shape()[idx_channel] % (block_shape * block_shape) != 0);
50  // Validate output if initialized
51  if(output->total_size() != 0)
52  {
53  const int idx_width = get_data_layout_dimension_index(data_layout, DataLayoutDimension::WIDTH);
54  const int idx_height = get_data_layout_dimension_index(data_layout, DataLayoutDimension::HEIGHT);
55  ARM_COMPUTE_RETURN_ERROR_ON(output->tensor_shape()[idx_width] != (block_shape * input->tensor_shape()[idx_width]));
56  ARM_COMPUTE_RETURN_ERROR_ON(output->tensor_shape()[idx_height] != (block_shape * input->tensor_shape()[idx_height]));
57  ARM_COMPUTE_RETURN_ERROR_ON(output->num_dimensions() > 4);
58  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input, output);
59  }
60 
61  return Status{};
62 }

70 void NEDepthToSpaceLayerKernel::configure(const ITensor *input, ITensor *output, int32_t block_shape)
71 {
72  ARM_COMPUTE_ERROR_ON_NULLPTR(input, output);
73  TensorShape output_shape = compute_depth_to_space_shape(input->info(), block_shape);
74  // Output auto inizialitation if not yet initialized
75  auto_init_if_empty(*output->info(), input->info()->clone()->set_tensor_shape(output_shape));
76 
77  // Perform validation step
78  ARM_COMPUTE_ERROR_THROW_ON(validate_arguments(input->info(), output->info(), block_shape));
79 
80  _input = input;
81  _output = output;
82  _block_shape = block_shape;
83  _data_layout = input->info()->data_layout();
84 
85  // Configure kernel window
86  Window win = calculate_max_window(*input->info(), Steps());
87  ICPPKernel::configure(win);
88 }

97 void NEDepthToSpaceLayerKernel::run(const Window &window, const ThreadInfo &info)
98 {
99  ARM_COMPUTE_UNUSED(info);
100  ARM_COMPUTE_ERROR_ON_UNCONFIGURED_KERNEL(this);
101  ARM_COMPUTE_ERROR_ON_INVALID_SUBWINDOW(ICPPKernel::window(), window);
102 
103  const int idx_channel = get_data_layout_dimension_index(_data_layout, DataLayoutDimension::CHANNEL);
104  const int depth_size = _input->info()->dimension(idx_channel);
105  const int r = (depth_size / (_block_shape * _block_shape));
106  const int element_size = _input->info()->element_size();
107 
108  Window slice_out = window.first_slice_window_3D();
109 
110  // The slice_out slice does not move
111  slice_out.set(Window::DimX, Window::Dimension(0, 0, 0));
112  slice_out.set(Window::DimY, Window::Dimension(0, 0, 0));
113  slice_out.set(Window::DimZ, Window::Dimension(0, 0, 0));
114 
115  // Main loop for NCHW and NHWC
116  if(_data_layout == DataLayout::NCHW)
117  {
118  Window slice_in = window.first_slice_window_2D();
119  do
120  {
121  Iterator in(_input, slice_in);
122  execute_window_loop(slice_in, [&](const Coordinates & id)
123  {
124  const int x = id.x();
125  const int y = id.y();
126 
127  const int z = id.z() % r;
128  const int out_x = x * _block_shape + (id.z() / r) % _block_shape;
129  const int out_y = y * _block_shape + (id.z() / r) / _block_shape;
130  Coordinates output_coords{ out_x, out_y, z, id[3] };
131  memcpy(_output->ptr_to_element(output_coords), in.ptr(), element_size);
132  },
133  in);
134  }
135  while(window.slide_window_slice_2D(slice_in));
136  }
137  else
138  {
139  Window slice_in = window.first_slice_window_3D();
140  do
141  {
142  Iterator in(_input, slice_in);
143  execute_window_loop(slice_in, [&](const Coordinates & id)
144  {
145  const int x = id.y();
146  const int y = id.z();
147 
148  const int z = id.x() % r;
149  const int out_x = x * _block_shape + (id.x() / r) % _block_shape;
150  const int out_y = y * _block_shape + (id.x() / r) / _block_shape;
151  Coordinates output_coords{ z, out_x, out_y, id[3] };
152  memcpy(_output->ptr_to_element(output_coords), in.ptr(), element_size);
153  },
154  in);
155  }
156  while(window.slide_window_slice_3D(slice_in));
157  }
158 }
