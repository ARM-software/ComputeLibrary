71 template <>
72 std::unique_ptr<IFunction> create_convolution_layer<GCConvolutionLayerFunctions, GCTargetInfo>(ConvolutionLayerNode &node, GraphContext &ctx)
73 {
74  validate_node<GCTargetInfo>(node, 3 /* expected inputs */, 1 /* expected outputs */);
75 
76  // Extract IO and info
77  GCTargetInfo::TensorType *input = get_backing_tensor<GCTargetInfo>(node.input(0));
78  GCTargetInfo::TensorType *weights = get_backing_tensor<GCTargetInfo>(node.input(1));
79  GCTargetInfo::TensorType *biases = get_backing_tensor<GCTargetInfo>(node.input(2));
80  GCTargetInfo::TensorType *output = get_backing_tensor<GCTargetInfo>(node.output(0));
81 
82  if(is_data_type_quantized_asymmetric(input->info()->data_type()))
83  {
84  biases->info()->set_data_type(DataType::S32);
85  }
86 
87  const PadStrideInfo conv_info = node.convolution_info();
88  const ConvolutionMethod conv_algorithm = node.convolution_method();
89  const ActivationLayerInfo fused_act = node.fused_activation();
90 
91  // Create and configure function (we assume that functions have been validated before creation)
92  std::shared_ptr<IMemoryManager> mm = get_memory_manager(ctx, GCTargetInfo::TargetType);
93  std::unique_ptr<IFunction> func;
94  std::string func_name;
95 
96  if(conv_algorithm == ConvolutionMethod::Direct)
97  {
98  std::tie(func, func_name) = create_named_function<GCConvolutionLayerFunctions::DirectConvolutionLayer>(
99  std::string("DirectConvolutionLayer"),
100  input, weights, biases, output, conv_info, fused_act);
101  }
102  else
103  {
104  std::tie(func, func_name) = create_named_memory_managed_function<GCConvolutionLayerFunctions::GenericConvolutionLayer>(
105  std::string("ConvolutionLayer"), mm,
106  input, weights, biases, output, conv_info, WeightsInfo(), Size2D(1U, 1U), fused_act);
107  }
108 
109  // Log info
110  ARM_COMPUTE_LOG_GRAPH_INFO("Instantiated "
111  << node.name()
112  << " Type: " << func_name
113  << " Data Type: " << input->info()->data_type()
114  << " Input QuantInfo: " << input->info()->quantization_info()
115  << " Weights QuantInfo: " << weights->info()->quantization_info()
116  << " Input shape: " << input->info()->tensor_shape()
117  << " Weights shape: " << weights->info()->tensor_shape()
118  << " Output shape: " << output->info()->tensor_shape()
119  << (fused_act.enabled() ? " " + to_string(fused_act.activation()) : "")
120  << std::endl);
121  return func;
122 }

139

141  const DepthwiseConvolutionMethod dwc_algorithm = node.depthwise_convolution_method();
142  const ActivationLayerInfo fused_act = node.fused_activation();

144 
145  // Create and configure function (we assume that functions have been validated before creation)
146  std::unique_ptr<IFunction> func;
147  std::string func_name;
148  if(dwc_algorithm == DepthwiseConvolutionMethod::Optimized3x3)
149  {

153  }

168  << " Weights shape: " << weights->info()->tensor_shape()
169  << " Output shape: " << output->info()->tensor_shape()
170  << " Depth multiplier: " << depth_multiplier

176 template <>
177 std::unique_ptr<IFunction> create_eltwise_layer<GCEltwiseFunctions, GCTargetInfo>(EltwiseLayerNode &node)
178 {
179  ARM_COMPUTE_LOG_GRAPH_VERBOSE(
180  "Creating GC EltwiseLayer node with ID : " << node.id() << " and Name: " << node.name() << std::endl);
181  ARM_COMPUTE_ERROR_ON(node.num_inputs() != 2);
182  ARM_COMPUTE_ERROR_ON(node.num_outputs() != 1);
183 
184  // Extract IO and info
185  GCTargetInfo::TensorType *input1 = get_backing_tensor<GCTargetInfo>(node.input(0));
186  GCTargetInfo::TensorType *input2 = get_backing_tensor<GCTargetInfo>(node.input(1));
187  GCTargetInfo::TensorType *output = get_backing_tensor<GCTargetInfo>(node.output(0));
188  const EltwiseOperation eltwise_op = node.eltwise_operation();
189  const ConvertPolicy convert_policy = node.convert_policy();
190  ARM_COMPUTE_ERROR_ON(input1 == nullptr);
191  ARM_COMPUTE_ERROR_ON(input2 == nullptr);
192  ARM_COMPUTE_ERROR_ON(output == nullptr);
193 
194  std::unique_ptr<IFunction> func = nullptr;
195  std::string func_name;
196  if(eltwise_op == EltwiseOperation::Add)
197  {
198  std::tie(func, func_name) = create_named_function<GCEltwiseFunctions::Addition>(
199  std::string("GCArithmeticAddition"),
200  input1, input2, output, convert_policy);
201  }
202  else if(eltwise_op == EltwiseOperation::Sub)
203  {
204  ARM_COMPUTE_ERROR("Arithmetic subtraction is not supported in GLES backend");
205  }
206  else if(eltwise_op == EltwiseOperation::Mul)
207  {
208  std::tie(func, func_name) = create_named_function<GCEltwiseFunctions::Multiplication>(
209  std::string("PixelWiseMultiplication"),
210  input1, input2, output, 1.f);
211  }
212  else
213  {
214  ARM_COMPUTE_ERROR("Unsupported element-wise operation!");
215  }
216 
217  // Log info
218  ARM_COMPUTE_LOG_GRAPH_INFO("Instantiated "
219  << node.name()
220  << " Type: " << node.type()
221  << " Target: " << GCTargetInfo::TargetType
222  << " Operation: " << func_name
223  << " Data Type: " << input1->info()->data_type()
224  << " Shape: " << input1->info()->tensor_shape()
225  << std::endl);
226 
227  return func;
228 }

231 std::unique_ptr<IFunction> GCFunctionFactory::create(INode *node, GraphContext &ctx)
232 {
233  if(node == nullptr)
234  {
235  return nullptr;
236  }
237 
238  NodeType type = node->type();
239  switch(type)
240  {
241  case NodeType::ActivationLayer:
242  return detail::create_activation_layer<GCActivationLayer, GCTargetInfo>(*polymorphic_downcast<ActivationLayerNode *>(node));
243  case NodeType::BatchNormalizationLayer:
244  return detail::create_batch_normalization_layer<GCBatchNormalizationLayer, GCTargetInfo>(*polymorphic_downcast<BatchNormalizationLayerNode *>(node));
245  case NodeType::ConvolutionLayer:
246  return detail::create_convolution_layer<GCConvolutionLayerFunctions, GCTargetInfo>(*polymorphic_downcast<ConvolutionLayerNode *>(node), ctx);
247  case NodeType::ConcatenateLayer:
248  return detail::create_concatenate_layer<GCConcatenateLayer, GCTargetInfo>(*polymorphic_downcast<ConcatenateLayerNode *>(node));
249  case NodeType::DepthwiseConvolutionLayer:
250  return detail::create_depthwise_convolution_layer<GCDepthwiseConvolutionLayerFunctions, GCTargetInfo>(*polymorphic_downcast<DepthwiseConvolutionLayerNode *>(node));
251  case NodeType::EltwiseLayer:
252  return detail::create_eltwise_layer<GCEltwiseFunctions, GCTargetInfo>(*polymorphic_downcast<EltwiseLayerNode *>(node));
253  case NodeType::FullyConnectedLayer:
254  return detail::create_fully_connected_layer<GCFullyConnectedLayer, GCTargetInfo>(*polymorphic_downcast<FullyConnectedLayerNode *>(node), ctx);
255  case NodeType::NormalizationLayer:
256  return detail::create_normalization_layer<GCNormalizationLayer, GCTargetInfo>(*polymorphic_downcast<NormalizationLayerNode *>(node), ctx);
257  case NodeType::NormalizePlanarYUVLayer:
258  return detail::create_normalize_planar_yuv_layer<GCNormalizePlanarYUVLayer, GCTargetInfo>(*polymorphic_downcast<NormalizePlanarYUVLayerNode *>(node));
259  case NodeType::PoolingLayer:
260  return detail::create_pooling_layer<GCPoolingLayer, GCTargetInfo>(*polymorphic_downcast<PoolingLayerNode *>(node));
261  case NodeType::ResizeLayer:
262  return detail::create_resize_layer<GCScale, GCTargetInfo>(*polymorphic_downcast<ResizeLayerNode *>(node));
263  case NodeType::SoftmaxLayer:
264  return detail::create_softmax_layer<GCSoftmaxLayer, GCTargetInfo>(*polymorphic_downcast<SoftmaxLayerNode *>(node), ctx);
265  default:
266  return nullptr;
267  }
268 }
