44 Status validate_arguments(const ITensorInfo *input, const ITensorInfo *weights, const ITensorInfo *biases, const ITensorInfo *output, const PadStrideInfo &conv_info, unsigned int depth_multiplier,
45  const ActivationLayerInfo &act_info, const Size2D &dilation)
46 {
47  ARM_COMPUTE_RETURN_ERROR_ON_F16_UNSUPPORTED(input);
48  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(input, 1, DataType::F16, DataType::F32, DataType::QASYMM8);
49  ARM_COMPUTE_RETURN_ERROR_ON_MSG((act_info.enabled()) && (input->data_type() == DataType::QASYMM8) && (act_info.activation() != ActivationLayerInfo::ActivationFunction::LU_BOUNDED_RELU)
50  && (act_info.activation() != ActivationLayerInfo::ActivationFunction::BOUNDED_RELU)
51  && (act_info.activation() != ActivationLayerInfo::ActivationFunction::RELU)
52  && (act_info.activation() != ActivationLayerInfo::ActivationFunction::LOGISTIC),
53  "For QASYMM8 only logistic, relu, lower bounded relu and lower-upper bounded relu are supported");
54  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input, weights);
55  ARM_COMPUTE_RETURN_ERROR_ON(depth_multiplier > 1); // COMPMID-1071 Add depth multiplier support for NHWC
56 
57  ARM_COMPUTE_RETURN_ERROR_ON(conv_info.stride().first < 1);
58  ARM_COMPUTE_RETURN_ERROR_ON(std::max(conv_info.pad_top(), conv_info.pad_bottom()) > 1);
59 
60  ARM_COMPUTE_RETURN_ERROR_ON((dilation.x() < 1) || (dilation.y() < 1));
61 
62  const bool is_qasymm = is_data_type_quantized_asymmetric(input->data_type());
63  const size_t weights_width = 3;
64  const size_t weights_height = 3;
65 
66  if(is_qasymm)
67  {
68  DepthwiseConvolutionReshapeInfo info;
69  info.c0 = 4;
70  ARM_COMPUTE_RETURN_ERROR_ON((weights->dimension(0) / info.c0) != weights_width * weights_height);
71  }
72  else
73  {
74  ARM_COMPUTE_RETURN_ERROR_ON((weights->dimension(1) != weights_width) || (weights->dimension(2) != weights_height));
75  }
76 
77  if(biases != nullptr)
78  {
79  if(is_qasymm)
80  {
81  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(biases, 1, DataType::S32);
82  }
83  else
84  {
85  ARM_COMPUTE_RETURN_ERROR_ON(biases->dimension(0) != weights->dimension(0));
86  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(weights, biases);
87  }
88 
89  ARM_COMPUTE_RETURN_ERROR_ON(biases->num_dimensions() > 1);
90  }
91 
92  if(output->total_size() != 0)
93  {
94  const TensorShape output_shape = arm_compute::misc::shape_calculator::compute_depthwise_convolution_shape(
95  *input, TensorInfo(TensorShape(weights_width, weights_height), 1, weights->data_type()).set_data_layout(DataLayout::NCHW), conv_info, depth_multiplier, dilation);
96  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DIMENSIONS(output->tensor_shape(), output_shape);
97  }
98 
99  return Status{};
100 }

102 std::pair<Status, Window> validate_and_configure_window(ITensorInfo *input, ITensorInfo *weights, ITensorInfo *bias, ITensorInfo *output,
103  const PadStrideInfo &conv_info, unsigned int depth_multiplier, const Size2D &dilation)
104 {
105  const size_t weights_width = 3;
106  const size_t weights_height = 3;
107 
108  // Get convolved dimensions
109  const TensorShape output_shape = arm_compute::misc::shape_calculator::compute_depthwise_convolution_shape(
110  *input, TensorInfo(TensorShape(weights_width, weights_height), 1, weights->data_type()).set_data_layout(DataLayout::NCHW), conv_info, depth_multiplier, dilation);
111 
112  // Output auto inizialitation if not yet initialized
113  auto_init_if_empty(*output,
114  output_shape,
115  1,
116  input->data_type(),
117  input->quantization_info());
118 
119  const bool is_qasymm = is_data_type_quantized_asymmetric(input->data_type());
120  const bool is_stride_1_dilation_1 = ((conv_info.stride().first == conv_info.stride().second) && (conv_info.stride().first == 1) && dilation.x() == 1 && dilation.y() == 1);
121 
122  const unsigned int num_rows_processed_per_iteration = is_stride_1_dilation_1 ? 2 : 1;
123  const unsigned int num_elems_accessed_per_iteration = is_qasymm ? 4 : (8 / input->element_size());
124  const unsigned int num_rows_read_per_iteration = num_rows_processed_per_iteration + 2;
125  const unsigned int num_rows_written_per_iteration = std::ceil(num_rows_processed_per_iteration / static_cast<float>(conv_info.stride().first));
126 
127  BorderSize border_size;
128  border_size = BorderSize(conv_info.pad_left(), 0, std::max(std::max(conv_info.pad_right(), conv_info.pad_bottom()), conv_info.pad_top()), 0);
129 
130  // Configure kernel window
131  Window win = calculate_max_window(*output, Steps(num_elems_accessed_per_iteration, num_rows_written_per_iteration));
132 
133  AccessWindowStatic input_access(input, 0, -border_size.top, ceil_to_multiple(input->dimension(0), num_elems_accessed_per_iteration),
134  ceil_to_multiple(input->dimension(1) + border_size.bottom, num_rows_read_per_iteration));
135  AccessWindowRectangle output_access(output, 0, 0, num_elems_accessed_per_iteration, num_rows_written_per_iteration);
136 
137  bool window_changed = false;
138 
139  if(is_qasymm)
140  {
141  window_changed = update_window_and_padding(win, input_access, output_access);
142  }
143  else
144  {
145  AccessWindowStatic weights_access(weights, 0, 0, ceil_to_multiple(weights->dimension(0), num_elems_accessed_per_iteration), weights->dimension(1));
146  window_changed = update_window_and_padding(win, input_access, weights_access, output_access);
147  }
148 
149  if(bias != nullptr)
150  {
151  AccessWindowHorizontal bias_access(bias, 0, num_elems_accessed_per_iteration);
152  window_changed = window_changed || update_window_and_padding(win, bias_access);
153  }
154 
155  output_access.set_valid_region(win, ValidRegion(Coordinates(), output->tensor_shape()));
156 
157  Status err = (window_changed) ? ARM_COMPUTE_CREATE_ERROR(ErrorCode::RUNTIME_ERROR, "Insufficient Padding!") : Status{};
158  return std::make_pair(err, win);
159 }

181  const bool is_stride_1 = ((conv_info.stride().first == conv_info.stride().second) && (conv_info.stride().first == 1));
182  const bool is_stride_1_dilation_1 = (is_stride_1 && dilation.x() == 1 && dilation.y() == 1);

187  _output = output;
188  _weights = weights;
189  _biases = biases;
190  _conv_stride_y = conv_info.stride().second;

193

195  if(is_dot8_supported && is_qasymm)
196  {
197  _num_planes_processed_per_iteration = 1;
198  }
199 
200  _border_size = BorderSize(is_qasymm && is_stride_1 ? 0 : conv_info.pad_left(), 0, std::max(std::max(conv_info.pad_right(), conv_info.pad_bottom()), conv_info.pad_top()), 0);
201 
202  const unsigned int num_elems_accessed_per_iteration = is_qasymm ? 4 : (8 / input->info()->element_size());
203 
204  CLBuildOptions build_opts;
205  build_opts.add_option_if(act_info.enabled(), "-DFUSED_ACTIVATION=" + lower_string(string_from_activation_func(act_info.activation())));
206  build_opts.add_option_if(_biases != nullptr, "-DHAS_BIAS");

209  build_opts.add_option("-DCONV_PAD_TOP=" + support::cpp11::to_string(conv_info.pad_top()));
210  build_opts.add_option("-DCONV_PAD_LEFT=" + support::cpp11::to_string(conv_info.pad_left()));
211  build_opts.add_option("-DDILATION_X=" + support::cpp11::to_string(dilation.x()));
212  build_opts.add_option("-DDILATION_Y=" + support::cpp11::to_string(dilation.y()));
213 
214  if(is_qasymm)
215  {
216  float multiplier = _input->info()->quantization_info().scale * _weights->info()->quantization_info().scale / _output->info()->quantization_info().scale;
217  int output_multiplier = 0;
218  int output_shift = 0;
219  quantization::calculate_quantized_multiplier_less_than_one(multiplier, &output_multiplier, &output_shift);
220 
221  build_opts.add_option("-DSRC_DIM_1=" + support::cpp11::to_string(_input->info()->dimension(1)));
222  build_opts.add_option("-DINPUT_OFFSET=" + support::cpp11::to_string(-_input->info()->quantization_info().offset));
223  build_opts.add_option("-DWEIGHTS_OFFSET=" + support::cpp11::to_string(-_weights->info()->quantization_info().offset));
224  build_opts.add_option("-DOUTPUT_OFFSET=" + support::cpp11::to_string(_output->info()->quantization_info().offset));
225  build_opts.add_option("-DK_OFFSET=" + support::cpp11::to_string(9 * input->info()->quantization_info().offset * weights->info()->quantization_info().offset));
226  build_opts.add_option("-DOUTPUT_MULTIPLIER=" + support::cpp11::to_string(output_multiplier));
227  build_opts.add_option("-DOUTPUT_SHIFT=" + support::cpp11::to_string(output_shift));
228 
229  if(act_info.enabled())
230  {
231  const int a_val = output->info()->quantization_info().quantize(act_info.a(), RoundingPolicy::TO_NEAREST_UP);
232  const int b_val = output->info()->quantization_info().quantize(act_info.b(), RoundingPolicy::TO_NEAREST_UP);
233  const int o1 = output->info()->quantization_info().offset;
234 
235  build_opts.add_option("-DA_VAL=" + support::cpp11::to_string(a_val));
236  build_opts.add_option("-DB_VAL=" + support::cpp11::to_string(b_val));
237  build_opts.add_option("-DCONST_0=" + support::cpp11::to_string(o1));
238 
239  const float s1 = input->info()->quantization_info().scale;
240  build_opts.add_option("-DS1_VAL=" + float_to_string_with_full_precision(s1));
241  build_opts.add_option("-DO1_VAL=" + support::cpp11::to_string(o1));
242  }
243  }
244  else
245  {
246  build_opts.add_option_if(act_info.enabled(), "-DA_VAL=" + float_to_string_with_full_precision(act_info.a()));
247  build_opts.add_option_if(act_info.enabled(), "-DB_VAL=" + float_to_string_with_full_precision(act_info.b()));
248  build_opts.add_option_if(act_info.enabled(), "-DSELECT_DATA_TYPE=" + get_cl_select_type_from_data_type(input->info()->data_type()));
249  build_opts.add_option("-DDATA_TYPE=" + get_cl_type_from_data_type(_input->info()->data_type()));
250  }
251 
252  if(is_stride_1_dilation_1)
253  {
254  build_opts.add_option("-DNUM_ROWS_PROCESSED=" + support::cpp11::to_string(_num_rows_processed_per_iteration));
255  build_opts.add_option("-DNUM_PLANES_PROCESSED=" + support::cpp11::to_string(_num_planes_processed_per_iteration));
256  build_opts.add_option("-DDST_DIM_2=" + support::cpp11::to_string(_output->info()->dimension(2)));
257  }
258  else
259  {
260  build_opts.add_option("-DCONV_STRIDE_X=" + support::cpp11::to_string(conv_info.stride().first));
261  build_opts.add_option("-DCONV_STRIDE_Y=" + support::cpp11::to_string(_conv_stride_y));
262  }
263  build_opts.add_option_if(_input->info()->tensor_shape().total_size_upper(3) > 1,
264  "-DDST_DEPTH=" + support::cpp11::to_string(static_cast<int>(std::ceil(_output->info()->dimension(2) / static_cast<float>(_num_planes_processed_per_iteration)))));
265 
266  std::string kernel_name;
267  // Create kernel
268  if(is_qasymm)
269  {
270  kernel_name = std::string("dwc_3x3_reshaped_qasymm8");
271  kernel_name += (is_dot8_supported && is_stride_1_dilation_1 ? "_dot8" : "");
272  kernel_name += (is_stride_1_dilation_1 ? "_stride1" : "");
273  kernel_name += "_nhwc";
274  }
275  else
276  {
277  kernel_name = std::string("depthwise_convolution_3x3_nhwc");
278  kernel_name += (is_stride_1_dilation_1 ? "_stride1" : "");
279  }
280 
281  build_opts.add_option_if(input->info()->data_type() == DataType::F16, "-DIS_F16");
282  build_opts.add_option_if(input->info()->data_type() == DataType::F32, "-DIS_F32");
283 
284  ICLKernel::configure_internal(win_config.second);
285  _kernel = static_cast<cl::Kernel>(CLKernelLibrary::get().create_kernel(kernel_name, build_opts.options()));
286 
287  // Set config_id for enabling LWS tuning
288  _config_id = kernel_name;
289  _config_id += "_";
290  _config_id += support::cpp11::to_string(input->info()->dimension(0));
291  _config_id += "_";
292  _config_id += support::cpp11::to_string(input->info()->dimension(1));
293  _config_id += "_";
294  _config_id += support::cpp11::to_string(input->info()->dimension(2));
295  _config_id += "_";

315 void CLDepthwiseConvolutionLayer3x3NHWCKernel::run(const Window &window, cl::CommandQueue &queue)
316 {
317  ARM_COMPUTE_ERROR_ON_UNCONFIGURED_KERNEL(this);
318  ARM_COMPUTE_ERROR_ON_INVALID_SUBWINDOW(IKernel::window(), window);
319 
320  // Collapse window
321  Window window_collapsed = window.collapse_if_possible(ICLKernel::window(), Window::DimZ);
322  const size_t total_batches = _input->info()->tensor_shape().total_size_upper(3);
323  const bool is_qasymm = is_data_type_quantized_asymmetric(_input->info()->data_type());
324 
325  Window win = window_collapsed;
326  win.set(Window::DimZ, Window::Dimension(0, std::ceil(_output->info()->dimension(2) / static_cast<float>(_num_planes_processed_per_iteration)) * total_batches, 1));
327 
328  // Create input window and adjust
329  Window win_in = win;
330  win_in.set_dimension_step(Window::DimY, _num_rows_processed_per_iteration);
331  win_in.set_dimension_step(Window::DimZ, _conv_stride_y);
332 
333  ARM_COMPUTE_ERROR_ON((win_in.y().step() < window.y().step()) || (win_in.z().step() < window.z().step()));
334 
335  Window slice_in = win_in.first_slice_window_4D();
336  Window slice_out = win.first_slice_window_4D();
337 
338  unsigned int idx = 2 * num_arguments_per_4D_tensor() + (is_qasymm ? num_arguments_per_2D_tensor() : num_arguments_per_3D_tensor());
339 
340  if(_biases != nullptr)
341  {
342  Window win_biases;
343  win_biases.use_tensor_dimensions(_biases->info()->tensor_shape());
344  win_biases.set_dimension_step(Window::DimX, window.x().step());
345  add_1D_tensor_argument(idx, _biases, win_biases);
346  }
347 
348  const int max_offset = _input->info()->strides_in_bytes().z() * _input->info()->dimension(2) - (_input->info()->padding().bottom + _input->info()->padding().top) *
349  _input->info()->strides_in_bytes().y();
350  _kernel.setArg(idx, max_offset);
351 
352  do
353  {
354  unsigned int idx = 0;
355  add_4D_tensor_argument(idx, _input, slice_in);
356  add_4D_tensor_argument(idx, _output, slice_out);
357  if(is_qasymm)
358  {
359  add_2D_tensor_argument(idx, _weights, slice_out);
360  }
361  else
362  {
363  add_3D_tensor_argument(idx, _weights, slice_out);
364  }
365  enqueue(queue, *this, slice_out, lws_hint());
366  }
367  while(win.slide_window_slice_4D(slice_out) && win_in.slide_window_slice_4D(slice_in));
368 }
