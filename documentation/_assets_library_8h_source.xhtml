387 template <typename T, typename D>
388 void AssetsLibrary::fill_borders_with_garbage(T &&tensor, D &&distribution, std::random_device::result_type seed_offset) const
389 {
390  const PaddingSize padding_size = tensor.padding();
391 
392  Window window;
393  window.set(0, Window::Dimension(-padding_size.left, tensor.shape()[0] + padding_size.right, 1));
394  if(tensor.shape().num_dimensions() > 1)
395  {
396  window.set(1, Window::Dimension(-padding_size.top, tensor.shape()[1] + padding_size.bottom, 1));
397  }
398 
399  std::mt19937 gen(_seed);
400 
401  execute_window_loop(window, [&](const Coordinates & id)
402  {
403  TensorShape shape = tensor.shape();
404 
405  // If outside of valid region
406  if(id.x() < 0 || id.x() >= static_cast<int>(shape.x()) || id.y() < 0 || id.y() >= static_cast<int>(shape.y()))
407  {
408  using ResultType = typename std::remove_reference<D>::type::result_type;
409  const ResultType value = distribution(gen);
410  void *const out_ptr = tensor(id);
411  store_value_with_data_type(out_ptr, value, tensor.data_type());
412  }
413  });
414 }

684 template <typename T>
685 void AssetsLibrary::fill_layer_data(T &&tensor, std::string name) const
686 {
687 #ifdef _WIN32
688  const std::string path_separator("\\");
689 #else /* _WIN32 */
690  const std::string path_separator("/");
691 #endif /* _WIN32 */
692  const std::string path = _library_path + path_separator + name;
693 
694  std::vector<unsigned long> shape;
695 
696  // Open file
697  std::ifstream stream(path, std::ios::in | std::ios::binary);
698  if(!stream.good())
699  {
700  throw framework::FileNotFound("Could not load npy file: " + path);
701  }
702  std::string header = npy::read_header(stream);
703 
704  // Parse header
705  bool fortran_order = false;
706  std::string typestr;
707  npy::parse_header(header, typestr, fortran_order, shape);
708 
709  // Check if the typestring matches the given one
710  std::string expect_typestr = get_typestring(tensor.data_type());
711  ARM_COMPUTE_ERROR_ON_MSG(typestr != expect_typestr, "Typestrings mismatch");
712 
713  // Validate tensor shape
714  ARM_COMPUTE_ERROR_ON_MSG(shape.size() != tensor.shape().num_dimensions(), "Tensor ranks mismatch");
715  if(fortran_order)
716  {
717  for(size_t i = 0; i < shape.size(); ++i)
718  {
719  ARM_COMPUTE_ERROR_ON_MSG(tensor.shape()[i] != shape[i], "Tensor dimensions mismatch");
720  }
721  }
722  else
723  {
724  for(size_t i = 0; i < shape.size(); ++i)
725  {
726  ARM_COMPUTE_ERROR_ON_MSG(tensor.shape()[i] != shape[shape.size() - i - 1], "Tensor dimensions mismatch");
727  }
728  }
729 
730  // Read data
731  if(tensor.padding().empty())
732  {
733  // If tensor has no padding read directly from stream.
734  stream.read(reinterpret_cast<char *>(tensor.data()), tensor.size());
735  }
736  else
737  {
738  // If tensor has padding accessing tensor elements through execution window.
739  Window window;
740  window.use_tensor_dimensions(tensor.shape());
741 
742  execute_window_loop(window, [&](const Coordinates & id)
743  {
744  stream.read(reinterpret_cast<char *>(tensor(id)), tensor.element_size());
745  });
746  }
