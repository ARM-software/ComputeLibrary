41 class BufferManagerMultipleThreads final : public IBufferManager
42 {
43 public:
45  static constexpr unsigned int NUM_BUFFERS = 3;
46 
47  explicit BufferManagerMultipleThreads(unsigned int max_num_users)
48  : _max_num_users(max_num_users)
49  {
50  }
51  unsigned int num_buffers() const override
52  {
53  return NUM_BUFFERS;
54  }
55  /* - Lock the requested index if it's free and return true if it needs reshaping.
56  * - Return false without acquiring the lock if the buffer at the index is already reshaped / being reshaped.
57  * - Block if the corresponding buffer for the given index is still being used by a different index.
58  */
59  bool lock_to_reshape_if_needed(unsigned int index) override
60  {
61  Buffer &buf = get_buffer_from_index(index);
62  while(true)
63  {
64  if(buf.index == index && buf.state != State::FREE)
65  {
66  //Another thread already is reshaping / has reshaped this block: nothing to do
67  return false;
68  }
69  else
70  {
71  std::unique_lock<std::mutex> lock(buf.mutex);
72  //If the buffer is free then lock it for reshaping:
73  if(buf.state == State::FREE)
74  {
75  buf.index = index;
76  buf.state = State::BEING_RESHAPED;
77  return true;
78  }
79  // Check again just in case it changed while we were acquiring the lock:
80  if(buf.index == index)
81  {
82  //Another thread is reshaping this block already, nothing to do
83  return false;
84  }
85  // buf.index != index: Buffer still being used by another block, need to wait
86  buf.sem.wait(lock);
87  }
88  }
89  }
90  /* Mark the buffer at the given index as reshaped and release the lock acquired via lock_to_reshape_if_needed() */
91  void mark_as_reshaped(unsigned int index) override
92  {
93  Buffer &buf = get_buffer_from_index(index);
94  {
95  std::lock_guard<std::mutex> lock(buf.mutex);
96  buf.users = _max_num_users;
97  buf.state = State::IN_USE;
98  }
99  buf.sem.notify_all();
100  }
101 
102  /* Block until the buffer at the given index is reshaped */
103  void wait_for_reshaping(unsigned int index) override
104  {
105  Buffer &buf = get_buffer_from_index(index);
106  ARM_COMPUTE_ERROR_ON(buf.index != index); // Should have blocked in lock_to_reshape_if_needed()
107  // Check if it's already ready to use:
108  if(buf.state == State::IN_USE)
109  return;
110  std::unique_lock<std::mutex> lock(buf.mutex);
111  //Double check it didn't change while we were acquiring the lock:
112  if(buf.state == State::IN_USE)
113  return;
114  buf.sem.wait(lock);
115  }
116  /* Mark the buffer at the given index as not used by this thread anymore.
117  * Once all the threads have called this method then the buffer is marked as free again.
118  */
119  void mark_as_unused(unsigned int index) override
120  {
121  Buffer &buf = get_buffer_from_index(index);
122  ARM_COMPUTE_ERROR_ON(buf.index != index); // Should have blocked in lock_to_reshape_if_needed()
123  if(--buf.users == 0)
124  {
125  std::unique_lock<std::mutex> lock(buf.mutex);
126  buf.state = State::FREE;
127  lock.unlock();
128  buf.sem.notify_all();
129  }
130  }
131 
132 private:
133  enum class State
134  {
135  FREE,
136  BEING_RESHAPED,
137  IN_USE
138  };
139  struct Buffer
140  {
141  unsigned int index{};
142  std::atomic_uint users{};
143  State state{ State::FREE };
144  std::mutex mutex{};
145  std::condition_variable sem{};
146  } _buffers[NUM_BUFFERS];
147  Buffer &get_buffer_from_index(unsigned int index)
148  {
149  return _buffers[index % NUM_BUFFERS];
150  }
151  unsigned int _max_num_users;
152 };

190 void NEGEMMInterleavedWrapper::prepare()
191 {
192  if(!_is_prepared)
193  {
194  if(_pretranspose_b)
195  {
196  _transformed_b.allocator()->allocate();
197  NEScheduler::get().schedule(_prepare_b.get(), Window::DimX);
198  _b->mark_as_unused();
199  }
200  else
201  {
202  _prepare_b->create_workloads(_b_workloads);
203  }
204  _transform_a->create_workloads(_a_workloads);
205  _matrix_multiply->create_workloads(_mm_workloads);
206 
207  //Maximum number of workloads to create:
208  const unsigned int num_threads = NEScheduler::get().num_threads();
209  const unsigned int max_iterations = std::max(num_threads, _num_windows);
210  //Maximum number of iterations the parameters allow:
211  const unsigned int num_iterations = _batch_window.num_iterations_total();
212  // Keep the smallest of the two:
213  const unsigned int num_windows = std::min(num_iterations, max_iterations);
214  const TensorShape window_shape = _batch_window.shape();
215  const unsigned int num_x_blocks = _block_walker.num_iterations(Window::DimX);
216 
217  // Create a 1D window to dynamically split the batch window:
218  Window win_1D;
219  win_1D.set(0, Window::Dimension(0, num_iterations));
220 
221  // Create one workload for each sub-window:
222  for(unsigned int w = 0; w < num_windows; w++)
223  {
224  Window win = win_1D.split_window(0, w, num_windows);
225  const Coordinates start_offset = index2coords(window_shape, win.x().start());
226  const Coordinates end_offset = index2coords(window_shape, win.x().end() - 1);
227 
228  if(_pretranspose_b)
229  {
230  auto workload = [start_offset, end_offset, num_x_blocks, this](const ThreadInfo & info)
231  {
232  //For each block of rows in "M"
233  auto workload_mm = this->_mm_workloads.begin();
234  for(auto workload_a = this->_a_workloads.begin(); workload_a != this->_a_workloads.end(); workload_a++)
235  {
236  // Transform one k_block from A:
237  this->_transform_a->transform(*workload_a, info, this->_batch_window, start_offset, end_offset);
238  // Then perform the matrix multiplication for each x block along N:
239  for(unsigned int i = 0; i < num_x_blocks; i++)
240  {
241  ARM_COMPUTE_ERROR_ON(workload_mm == this->_mm_workloads.end());
242  this->_matrix_multiply->transform(*workload_mm++, info, this->_batch_window, start_offset, end_offset);
243  }
244  }
245  };
246  _workloads.push_back(workload);
247  }
248  else
249  {
250  auto workload = [num_threads, start_offset, end_offset, num_x_blocks, this](const ThreadInfo & info)
251  {
252  //For each block of rows in "M"
253  auto workload_mm = this->_mm_workloads.begin();
254  unsigned int workload_b = 0;
255  //If there is only one thread then only reshape the B blocks as you need them:
256  unsigned int workload_b_next = num_threads == 1 ? this->_b_workloads.size() : 1;
257 
258  for(auto workload_a = this->_a_workloads.begin(); workload_a != this->_a_workloads.end(); workload_a++)
259  {
260  // Transform one k_block from A:
261  this->_transform_a->transform(*workload_a, info, this->_batch_window, start_offset, end_offset);
262  // Then perform the matrix multiplication for each x block along N:
263  for(unsigned int i = 0; i < num_x_blocks; i++)
264  {
265  ARM_COMPUTE_ERROR_ON(workload_mm == this->_mm_workloads.end());
266  if(workload_b_next < this->_b_workloads.size())
267  {
268  //Lock on BufferManager: need to run it ?
269  if(this->_buffer_manager->lock_to_reshape_if_needed(workload_b_next))
270  {
271  this->_prepare_b->transform(this->_b_workloads[workload_b_next], info);
272  this->_buffer_manager->mark_as_reshaped(workload_b_next);
273  }
274  workload_b_next++;
275  }
276  ARM_COMPUTE_ERROR_ON(workload_b >= this->_b_workloads.size());
277  // Run if needed or wait
278  if(this->_buffer_manager->lock_to_reshape_if_needed(workload_b))
279  {
280  this->_prepare_b->transform(this->_b_workloads[workload_b], info);
281  this->_buffer_manager->mark_as_reshaped(workload_b);
282  }
283  this->_buffer_manager->wait_for_reshaping(workload_b);
284  this->_matrix_multiply->transform(*workload_mm++, info, this->_batch_window, start_offset, end_offset);
285  this->_buffer_manager->mark_as_unused(workload_b);
286  workload_b++;
287  }
288  }
289  };
290  _workloads.push_back(workload);
291  }
292  }
293  if(!_pretranspose_b && num_windows > 1 && num_windows % num_threads != 0)
294  {
295  //Make sure the number of workloads is a multiple of the number of threads to avoid dead locks:
296  for(unsigned int leftover = num_windows % num_threads; leftover != num_threads; leftover++)
297  {
298  auto workload = [this](const ThreadInfo & info)
299  {
300  unsigned int workload_b = 0;
301  //If there is only one thread then only reshape the B blocks as you need them:
302  unsigned int workload_b_next = 1;
303 
304  for(unsigned int iteration = 0; iteration < this->_mm_workloads.size(); iteration++)
305  {
306  if(workload_b_next < this->_b_workloads.size())
307  {
308  //Lock on BufferManager: need to run it ?
309  if(this->_buffer_manager->lock_to_reshape_if_needed(workload_b_next))
310  {
311  this->_prepare_b->transform(this->_b_workloads[workload_b_next], info);
312  this->_buffer_manager->mark_as_reshaped(workload_b_next);
313  }
314  workload_b_next++;
315  }
316  ARM_COMPUTE_ERROR_ON(workload_b >= this->_b_workloads.size());
317  // Run if needed or wait
318  if(this->_buffer_manager->lock_to_reshape_if_needed(workload_b))
319  {
320  this->_prepare_b->transform(this->_b_workloads[workload_b], info);
321  this->_buffer_manager->mark_as_reshaped(workload_b);
322  }
323  this->_buffer_manager->wait_for_reshaping(workload_b);
324  this->_buffer_manager->mark_as_unused(workload_b);
325  workload_b++;
326  }
327  };
328  _workloads.push_back(workload);
329  }
330  }
331 
332  _is_prepared = true;
333  }

336 void NEGEMMInterleavedWrapper::configure(const ITensor *a, const ITensor *b, ITensor *c, float alpha, float beta, bool pretranspose_b)
337 {
338  _params = INEGEMMWrapperKernel::extract_parameters(a, b, c);
339  _a = a;
340  _b = b;
341  _c = c;
342  _pretranspose_b = pretranspose_b;
343 
344  const DataType input_type = a->info()->data_type();
345  const CPUInfo &ci = NEScheduler::get().cpu_info();
346  const unsigned int num_threads = NEScheduler::get().num_threads();
347 
348  const arm_gemm::KernelDescription gemm_kernel_info = get_gemm_info(input_type, ci, num_threads, _params, alpha, beta, pretranspose_b);
349  ARM_COMPUTE_ERROR_ON(gemm_kernel_info.method != arm_gemm::GemmMethod::GEMM_INTERLEAVED);
350 
351  // Forcing 128-byte alignment (required by 32-bit kernels)
352  const unsigned int alignment = 128;
353  _transformed_b.allocator()->init(TensorInfo{}, alignment);
354  _tmp_c.allocator()->init(TensorInfo{}, alignment);
355  _tag = "NEGEMMInterleaved_" + gemm_kernel_info.name;
356 
357  // Get strategy
358  std::unique_ptr<detail::IInterleavedStrategy> strategy = detail::create_strategy(gemm_kernel_info.name);
359  _num_windows = iceildiv(_params.M, strategy->out_height()) * _params.batches;
360  ARM_COMPUTE_ERROR_ON(strategy == nullptr);
361 
362  if(!_pretranspose_b)
363  {
364  _block_sizes = strategy->calculate_block_sizes_for_strategy(ci, _params);
365  _batch_window.set(Window::DimX, Window::Dimension(0, ceil_to_multiple(_block_sizes.m_round, _block_sizes.strategy_out_height), _block_sizes.strategy_out_height));
366  _batch_window.set(Window::DimY, Window::Dimension(0, _params.batches));
367  // If the execution is single threaded or has only one window then the buffer manager only needs 1 buffer else we will use NUM_BUFFERS buffers and ping pong between them:
368  const unsigned int num_iterations = _batch_window.num_iterations_total();
369  if(NEScheduler::get().num_threads() == 1 || num_iterations == 1)
370  {
371  _buffer_manager = support::cpp14::make_unique<BufferManagerSingleThread>();
372  }
373  else
374  {
375 #ifdef NO_MULTI_THREADING
376  ARM_COMPUTE_ERROR("Can't have more than 1 buffer without multiple threads");
377 #else /* NO_MULTI_THREADING */
378  _buffer_manager = support::cpp14::make_unique<BufferManagerMultipleThreads>(NEScheduler::get().num_threads());
379 #endif /* NO_MULTI_THREADING */
380  }
381  // If B is transposed at every iteration then transformed_B can be managed:
382  _memory_group.manage(&_transformed_b);
383  auto_init_if_empty(*_transformed_b.info(), _b->info()->clone()->set_tensor_shape(TensorShape(_block_sizes.x_block * _block_sizes.k_block, _buffer_manager->num_buffers())));
384  }
385  else
386  {
387  _tag += "_preB";
388  }
389 
390  _prepare_b = strategy->instantiate_prepareB(b, &_transformed_b, _params, ci);
391  ARM_COMPUTE_ERROR_ON(_prepare_b == nullptr);
392 
393  if(_pretranspose_b)
394  {
395  _block_sizes = _prepare_b->block_sizes();
396  _batch_window.set(Window::DimX, Window::Dimension(0, ceil_to_multiple(_block_sizes.m_round, _block_sizes.strategy_out_height), _block_sizes.strategy_out_height));
397  _batch_window.set(Window::DimY, Window::Dimension(0, _params.batches));
398  }
399 
400  _block_walker.set(Window::DimX, Window::Dimension(0, ceil_to_multiple(_params.N, _block_sizes.x_block), _block_sizes.x_block));
401  _block_walker.set(Window::DimY, Window::Dimension(0, ceil_to_multiple(_params.K, _block_sizes.k_block), _block_sizes.k_block));
402  _block_walker.set(Window::DimZ, Window::Dimension(0, _params.multis));
403 
404  _transformed_a.allocator()->init(TensorInfo(TensorShape{ _block_sizes.k_block, _block_sizes.m_round, _params.batches }, 1, input_type), alignment);
405  _memory_group.manage(&_transformed_a);
406  _memory_group.manage(&_tmp_c);
407 
408  _transform_a = strategy->instantiate_transformA(_a, &_transformed_a, _block_walker, _params);
409  _matrix_multiply = strategy->instantiate_matrix_multiply(&_transformed_a, &_transformed_b, &_tmp_c, c, _block_walker, _block_sizes, _params, alpha, beta, pretranspose_b, num_threads);
410  ARM_COMPUTE_ERROR_ON(_transform_a == nullptr);
411  ARM_COMPUTE_ERROR_ON(_matrix_multiply == nullptr);
412 
413  _transformed_a.allocator()->allocate();
414  _tmp_c.allocator()->allocate();
415  if(!_pretranspose_b)
416  {
417  _transformed_b.allocator()->allocate();
418  }
419 }
