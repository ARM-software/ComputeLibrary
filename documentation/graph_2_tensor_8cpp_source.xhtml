75 bool Tensor::call_accessor()
76 {
77  // Early exit guard
78  if(!_accessor || !_handle)
79  {
80  return false;
81  }
82 
83  // Map tensor
84  _handle->map(true);
85 
86  // Return in case of null backend buffer
87  if(_handle->tensor().buffer() == nullptr)
88  {
89  return false;
90  }
91 
92  // Call accessor
93  bool retval = _accessor->access_tensor(_handle->tensor());
94 
95  // Unmap tensor
96  _handle->unmap();
97 
98  return retval;
99 }
