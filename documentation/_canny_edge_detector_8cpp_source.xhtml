47 const auto MARK_ZERO = 0u;
48 const auto MARK_MAYBE = 127u;
49 const auto MARK_EDGE = 255u;

51 template <typename T>
52 void trace_edge(SimpleTensor<T> &dst, const ValidRegion &valid_region)
53 {
54  std::stack<Coordinates> pixels_stack;
55  for(auto i = 0; i < dst.num_elements(); ++i)
56  {
57  if(dst[i] == MARK_EDGE)
58  {
59  pixels_stack.push(index2coord(dst.shape(), i));
60  }
61  }
62 
63  while(!pixels_stack.empty())
64  {
65  const Coordinates pixel_coord = pixels_stack.top();
66  pixels_stack.pop();
67 
68  std::array<Coordinates, 8> neighbours =
69  {
70  {
71  Coordinates(pixel_coord.x() - 1, pixel_coord.y() + 0),
72  Coordinates(pixel_coord.x() + 1, pixel_coord.y() + 0),
73  Coordinates(pixel_coord.x() - 1, pixel_coord.y() - 1),
74  Coordinates(pixel_coord.x() + 1, pixel_coord.y() + 1),
75  Coordinates(pixel_coord.x() + 0, pixel_coord.y() - 1),
76  Coordinates(pixel_coord.x() + 0, pixel_coord.y() + 1),
77  Coordinates(pixel_coord.x() + 1, pixel_coord.y() - 1),
78  Coordinates(pixel_coord.x() - 1, pixel_coord.y() + 1)
79  }
80  };
81 
82  // Mark MAYBE neighbours as edges since they are next to an EDGE
83  std::for_each(neighbours.begin(), neighbours.end(), [&](Coordinates & coord)
84  {
85  if(is_in_valid_region(valid_region, coord))
86  {
87  const size_t pixel_index = coord2index(dst.shape(), coord);
88  const T pixel = dst[pixel_index];
89  if(pixel == MARK_MAYBE)
90  {
91  dst[pixel_index] = MARK_EDGE;
92  pixels_stack.push(coord);
93  }
94  }
95  });
96  }
97 
98  // Mark all remaining MAYBE pixels as ZERO (not edges)
99  for(auto i = 0; i < dst.num_elements(); ++i)
100  {
101  if(dst[i] == MARK_MAYBE)
102  {
103  dst[i] = MARK_ZERO;
104  }
105  }
106 }

108 template <typename U, typename T>
109 SimpleTensor<T> canny_edge_detector_impl(const SimpleTensor<T> &src, int32_t upper, int32_t lower, int gradient_size, MagnitudeType norm_type,
110  BorderMode border_mode, T constant_border_value)
111 {
112  ARM_COMPUTE_ERROR_ON(gradient_size != 3 && gradient_size != 5 && gradient_size != 7);
113  ARM_COMPUTE_ERROR_ON(lower < 0 || lower >= upper);
114 
115  // Output: T == uint8_t
116  SimpleTensor<T> dst{ src.shape(), src.data_type() };
117  ValidRegion valid_region = shape_to_valid_region(src.shape(), border_mode == BorderMode::UNDEFINED, BorderSize(gradient_size / 2 + 1));
118 
119  // Sobel computation: U == int16_t or int32_t
120  SimpleTensor<U> gx, gy;
121  std::tie(gx, gy) = sobel<U>(src, gradient_size, border_mode, constant_border_value, GradientDimension::GRAD_XY);
122 
123  using unsigned_U = typename traits::make_unsigned_conditional_t<U>::type;
124  using promoted_U = typename common_promoted_signed_type<U>::intermediate_type;
125 
126  // Gradient magnitude and phase (edge direction)
127  const DataType mag_data_type = gx.data_type() == DataType::S16 ? DataType::U16 : DataType::U32;
128  SimpleTensor<unsigned_U> grad_mag{ gx.shape(), mag_data_type };
129  SimpleTensor<uint8_t> grad_dir{ gy.shape(), DataType::U8 };
130 
131  for(auto i = 0; i < grad_mag.num_elements(); ++i)
132  {
133  double mag = 0.f;
134 
135  if(norm_type == MagnitudeType::L2NORM)
136  {
137  mag = support::cpp11::round(std::sqrt(static_cast<promoted_U>(gx[i]) * gx[i] + static_cast<promoted_U>(gy[i]) * gy[i]));
138  }
139  else // MagnitudeType::L1NORM
140  {
141  mag = static_cast<promoted_U>(std::abs(gx[i])) + static_cast<promoted_U>(std::abs(gy[i]));
142  }
143 
144  float angle = 180.f * std::atan2(static_cast<float>(gy[i]), static_cast<float>(gx[i])) / M_PI;
145  grad_dir[i] = support::cpp11::round(angle < 0.f ? 180 + angle : angle);
146  grad_mag[i] = saturate_cast<unsigned_U>(mag);
147  }
148 
149  /*
150  Quantise the phase into 4 directions
151  0° dir=0 0.0 <= p < 22.5 or 157.5 <= p < 180
152  45° dir=1 22.5 <= p < 67.5
153  90° dir=2 67.5 <= p < 112.5
154  135° dir=3 112.5 <= p < 157.5
155  */
156  for(auto i = 0; i < grad_dir.num_elements(); ++i)
157  {
158  const auto direction = std::fabs(grad_dir[i]);
159  grad_dir[i] = (direction < 22.5 || direction >= 157.5) ? 0 : (direction < 67.5) ? 1 : (direction < 112.5) ? 2 : 3;
160  }
161 
162  // Non-maximum suppression
163  std::vector<int> strong_edges;
164  const auto upper_thresh = static_cast<uint32_t>(upper);
165  const auto lower_thresh = static_cast<uint32_t>(lower);
166 
167  const auto pixel_at_offset = [&](const SimpleTensor<unsigned_U> &tensor, const Coordinates & coord, int xoffset, int yoffset)
168  {
169  return tensor_elem_at(tensor, Coordinates{ coord.x() + xoffset, coord.y() + yoffset }, border_mode, static_cast<unsigned_U>(constant_border_value));
170  };
171 
172  for(auto i = 0; i < dst.num_elements(); ++i)
173  {
174  const auto coord = index2coord(dst.shape(), i);
175  if(!is_in_valid_region(valid_region, coord) || grad_mag[i] <= lower_thresh)
176  {
177  dst[i] = MARK_ZERO;
178  continue;
179  }
180 
181  unsigned_U mag_90, mag90;
182  switch(grad_dir[i])
183  {
184  case 0: // North/South edge direction, compare against East/West pixels (left & right)
185  mag_90 = pixel_at_offset(grad_mag, coord, -1, 0);
186  mag90 = pixel_at_offset(grad_mag, coord, 1, 0);
187  break;
188  case 1: // NE/SW edge direction, compare against NW/SE pixels (top-left & bottom-right)
189  mag_90 = pixel_at_offset(grad_mag, coord, -1, -1);
190  mag90 = pixel_at_offset(grad_mag, coord, +1, +1);
191  break;
192  case 2: // East/West edge direction, compare against North/South pixels (top & bottom)
193  mag_90 = pixel_at_offset(grad_mag, coord, 0, -1);
194  mag90 = pixel_at_offset(grad_mag, coord, 0, +1);
195  break;
196  case 3: // NW/SE edge direction, compare against NE/SW pixels (top-right & bottom-left)
197  mag_90 = pixel_at_offset(grad_mag, coord, +1, -1);
198  mag90 = pixel_at_offset(grad_mag, coord, -1, +1);
199  break;
200  default:
201  ARM_COMPUTE_ERROR("Invalid gradient phase provided");
202  break;
203  }
204 
205  // Potential edge if greater than both pixels at +/-90° on either side
206  if(grad_mag[i] > mag_90 && grad_mag[i] > mag90)
207  {
208  // Double thresholding and edge tracing
209  if(grad_mag[i] > upper_thresh)
210  {
211  dst[i] = MARK_EDGE; // Definite edge pixel
212  strong_edges.emplace_back(i);
213  }
214  else
215  {
216  dst[i] = MARK_MAYBE;
217  }
218  }
219  else
220  {
221  dst[i] = MARK_ZERO; // Since not greater than neighbours
222  }
223  }
224 
225  // Final edge tracing
226  trace_edge<T>(dst, valid_region);
227  return dst;
228 }

232 SimpleTensor<T> canny_edge_detector(const SimpleTensor<T> &src,
233  int32_t upper_thresh, int32_t lower_thresh, int gradient_size, MagnitudeType norm_type,
234  BorderMode border_mode, T constant_border_value)
