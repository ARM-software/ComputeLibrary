37 template <typename T>
38 void generate_range(SimpleTensor<T> &dst, float start, const size_t num_of_elements, float step)
39 {
40  float val = start;
41  for(size_t index = 0; index < num_of_elements; index++)
42  {
43  dst[index] = static_cast<T>(val);
44  val += step;
45  }
46 }
