141  template <typename T>
142  void fill_image(T &image)
143  {
144  ARM_COMPUTE_ERROR_ON(!is_open());
145  ARM_COMPUTE_ERROR_ON(image.info()->dimension(0) != _width || image.info()->dimension(1) != _height);
146  ARM_COMPUTE_ERROR_ON_FORMAT_NOT_IN(&image, Format::U8, Format::RGB888);
147  try
148  {
149 #ifdef ARM_COMPUTE_CL
150  // Map buffer if creating a CLTensor
151  if(std::is_same<typename std::decay<T>::type, CLImage>::value)
152  {
153  image.map();
154  }
155 #endif
156  // Check if the file is large enough to fill the image
157  const size_t current_position = _fs.tellg();
158  _fs.seekg(0, std::ios_base::end);
159  const size_t end_position = _fs.tellg();
160  _fs.seekg(current_position, std::ios_base::beg);
161 
162  ARM_COMPUTE_ERROR_ON_MSG((end_position - current_position) < image.info()->tensor_shape().total_size() * image.info()->element_size(),
163  "Not enough data in file");
164  ARM_COMPUTE_UNUSED(end_position);
165 
166  switch(image.info()->format())
167  {
168  case Format::U8:
169  {
170  Window window;
171  window.set(Window::DimX, Window::Dimension(0, _width, 1));
172  window.set(Window::DimY, Window::Dimension(0, _height, 1));
173 
174  Iterator out(&image, window);
175 
176  unsigned char red = 0;
177  unsigned char green = 0;
178  unsigned char blue = 0;
179 
180  execute_window_loop(window, [&](const Coordinates & id)
181  {
182  red = _fs.get();
183  green = _fs.get();
184  blue = _fs.get();
185 
186  *out.ptr() = 0.2126f * red + 0.7152f * green + 0.0722f * blue;
187  },
188  out);
189 
190  break;
191  }
192  case Format::RGB888:
193  {
194  Window window;
195  window.set(Window::DimX, Window::Dimension(0, _width, _width));
196  window.set(Window::DimY, Window::Dimension(0, _height, 1));
197 
198  Iterator out(&image, window);
199 
200  execute_window_loop(window, [&](const Coordinates & id)
201  {
202  _fs.read(reinterpret_cast<std::fstream::char_type *>(out.ptr()), _width * image.info()->element_size());
203  },
204  out);
205 
206  break;
207  }
208  default:
209  ARM_COMPUTE_ERROR("Unsupported format");
210  }
211 
212 #ifdef ARM_COMPUTE_CL
213  // Unmap buffer if creating a CLTensor
214  if(std::is_same<typename std::decay<T>::type, CLTensor>::value)
215  {
216  image.unmap();
217  }
218 #endif
219  }
220  catch(const std::ifstream::failure &e)
221  {
222  ARM_COMPUTE_ERROR("Loading PPM file: %s", e.what());
223  }
224  }
225 
226 private:
227  std::ifstream _fs;
228  unsigned int _width, _height;
229 };

241 void save_to_ppm(T &tensor, const std::string &ppm_filename)
242 {
243  ARM_COMPUTE_ERROR_ON_FORMAT_NOT_IN(&tensor, Format::RGB888, Format::U8);
244  ARM_COMPUTE_ERROR_ON(tensor.info()->num_dimensions() > 2);
245 
246  std::ofstream fs;
247 
248  try
249  {
250  fs.exceptions(std::ofstream::failbit | std::ofstream::badbit | std::ofstream::eofbit);
251  fs.open(ppm_filename, std::ios::out | std::ios::binary);
252 
253  const unsigned int width = tensor.info()->tensor_shape()[0];
254  const unsigned int height = tensor.info()->tensor_shape()[1];
255 
256  fs << "P6\n"
257  << width << " " << height << " 255\n";
258 
259 #ifdef ARM_COMPUTE_CL
260  // Map buffer if creating a CLTensor
261  if(std::is_same<typename std::decay<T>::type, CLTensor>::value)
262  {
263  tensor.map();
264  }
265 #endif
266 
267  switch(tensor.info()->format())
268  {
269  case Format::U8:
270  {
271  Window window;
272  window.set(Window::DimX, Window::Dimension(0, width, 1));
273  window.set(Window::DimY, Window::Dimension(0, height, 1));
274 
275  Iterator in(&tensor, window);
276 
277  execute_window_loop(window, [&](const Coordinates & id)
278  {
279  const unsigned char value = *in.ptr();
280 
281  fs << value << value << value;
282  },
283  in);
284 
285  break;
286  }
287  case Format::RGB888:
288  {
289  Window window;
290  window.set(Window::DimX, Window::Dimension(0, width, width));
291  window.set(Window::DimY, Window::Dimension(0, height, 1));
292 
293  Iterator in(&tensor, window);
294 
295  execute_window_loop(window, [&](const Coordinates & id)
296  {
297  fs.write(reinterpret_cast<std::fstream::char_type *>(in.ptr()), width * tensor.info()->element_size());
298  },
299  in);
300 
301  break;
302  }
303  default:
304  ARM_COMPUTE_ERROR("Unsupported format");
305  }
306 #ifdef ARM_COMPUTE_CL
307  // Unmap buffer if creating a CLTensor
308  if(std::is_same<typename std::decay<T>::type, CLTensor>::value)
309  {
310  tensor.unmap();
311  }
312 #endif
313  }
314  catch(const std::ofstream::failure &e)
315  {
316  ARM_COMPUTE_ERROR("Writing %s: (%s)", ppm_filename.c_str(), e.what());
317  }
318 }
