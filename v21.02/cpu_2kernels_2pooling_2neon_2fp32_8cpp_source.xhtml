104  const uint32x4_t voffset_x1 = { offset_x1, offset_x1 + 1, offset_x1 + 2, offset_x1 + 3 };

108  const uint32x4_t tmp_indices1 = vbslq_u32(vcgeq_f32(v_x2, v_x3), voffset_x2, voffset_x3);
109  const uint32x4_t tmp_indices2 = vbslq_u32(vcgeq_f32(vmaxq_f32(v_x0, v_x1), vmaxq_f32(v_x2, v_x3)), tmp_indices0, tmp_indices1);
110 
111  // Store indices
112  vst1q_u32(reinterpret_cast<uint32_t *>(indices.ptr()) + x_off, tmp_indices2);
113  }
114 
115  // Left-overs loop
116  for(; x_off < window_end_x; ++x_off)
117  {
118  const auto x0 = *(reinterpret_cast<const float *>(in.ptr() + in_x0_offset) + x_off);
119  const auto x1 = *(reinterpret_cast<const float *>(in.ptr() + in_x1_offset) + x_off);
120  const auto x2 = *(reinterpret_cast<const float *>(in.ptr() + in_x2_offset) + x_off);
121  const auto x3 = *(reinterpret_cast<const float *>(in.ptr() + in_x3_offset) + x_off);
122  res = std::max(std::max(x2, x3), std::max(x0, x1));
123 
124  // Store result
125  *(reinterpret_cast<float *>(out.ptr()) + x_off) = res;
126 
127  const uint32_t offset_base = offset_no_padding<float>(in.offset(), id, *src->info(), pool_stride_x, pool_stride_y);
128  const uint32_t offset_x0 = (uint32_t)offset_base / sizeof(float) + x_off;
129  const uint32_t offset_x1 = (uint32_t)offset_x0 + in_stride_y / sizeof(float) - pad_right;
130  const uint32_t offset_x2 = (uint32_t)offset_x0 + in_stride_z / sizeof(float) - pad_right * src->info()->tensor_shape()[1];
131  const uint32_t offset_x3 = (uint32_t)offset_x2 + in_stride_y / sizeof(float) - pad_right;
132  const uint32_t tmp_idx0 = (x0 >= x1) ? offset_x0 : offset_x1;
133  const uint32_t tmp_idx1 = (x2 >= x3) ? offset_x2 : offset_x3;
134  const uint32_t tmp_idx2 = (std::max(x0, x1) >= std::max(x2, x3)) ? tmp_idx0 : tmp_idx1;
135 
136  // Store indices

138  }
139  },
140  in, out, indices);

144 void poolingMxN_fp32_neon_nhwc(const ITensor *src, ITensor *dst0, ITensor *dst1, PoolingLayerInfo &pool_info, const Window &window_src, const Window &window)
145 {
146  if(pool_info.pool_size == Size2D(2, 2) && pool_info.pool_type == PoolingType::MAX && dst1)
147  {
148  pooling2_f32_maxpool_indices(src, dst0, dst1, pool_info, window_src, window);
149  }
150  else
151  {
152  const int window_start_x = window.x().start();
153  const int window_end_x = window.x().end();
154  const int window_step_x = 4;
155 
156  Window window_out = window;
157  window_out.set(Window::DimX, Window::Dimension(0, 1, 1));
158 
159  Iterator in(src, window_src);
160  Iterator out(dst0, window_out);
161 
162  const int pool_size_x = pool_info.is_global_pooling ? src->info()->tensor_shape().y() : pool_info.pool_size.width;
163  const int pool_size_y = pool_info.is_global_pooling ? src->info()->tensor_shape().z() : pool_info.pool_size.height;
164  const int pool_pad_right = pool_info.pad_stride_info.pad_right();
165  const int pool_pad_top = pool_info.pad_stride_info.pad_top();
166  const int pool_pad_left = pool_info.pad_stride_info.pad_left();
167  const int pool_pad_bottom = pool_info.pad_stride_info.pad_bottom();
168  int pool_stride_x = 0;
169  int pool_stride_y = 0;
170  std::tie(pool_stride_x, pool_stride_y) = pool_info.pad_stride_info.stride();
171  const int upper_bound_w = src->info()->dimension(1) + (pool_info.exclude_padding ? 0 : pool_pad_right);
172  const int upper_bound_h = src->info()->dimension(2) + (pool_info.exclude_padding ? 0 : pool_pad_bottom);
173 
174  float32x4_t vres;
175 
176  execute_window_loop(window_out, [&](const Coordinates & id)
177  {
178  const int idx_width = id.y() * pool_stride_x;
179  const int idx_height = id.z() * pool_stride_y;
180  const int pool_limit_y = pool_pad_top - idx_height;
181  const int pool_limit_x = pool_pad_left - idx_width;
182 
183  const int pool_start_y = std::max(0, window_src.z().start() + pool_limit_y);
184  const int pool_end_y = std::min(pool_size_y, window_src.z().end() + pool_limit_y);
185  const int pool_start_x = std::max(0, window_src.y().start() + pool_limit_x);
186  const int pool_end_x = std::min(pool_size_x, window_src.y().end() + pool_limit_x);
187 
188  int x_off = window_start_x;
189  for(; x_off <= (window_end_x - window_step_x); x_off += window_step_x)
190  {
191  if(pool_info.pool_type != PoolingType::MAX)
192  {
193  // Calculate scale
194  const float scale = calculate_avg_scale(pool_info.exclude_padding, DataLayout::NHWC, id, pool_size_x, pool_size_y, upper_bound_w, upper_bound_h, pool_pad_left, pool_pad_top, pool_stride_x,
195  pool_stride_y);
196  const float32x4_t scale_v = vdupq_n_f32(scale);
197 
198  // Perform pooling
199  vres = vdupq_n_f32(0.0f);
200 
201  for(int y = pool_start_y; y < pool_end_y; ++y)
202  {
203  for(int x = pool_start_x; x < pool_end_x; ++x)
204  {
205  const float32x4_t data = vld1q_f32(reinterpret_cast<const float *>(in.ptr() + (x - pool_pad_left) * static_cast<int>(src->info()->strides_in_bytes().y()) + (y - pool_pad_top) * static_cast<int>
206  (src->info()->strides_in_bytes().z())) + x_off);
207 
208  // Get power of 2 in case of l2 pooling and accumulate
209  if(pool_info.pool_type == PoolingType::L2)
210  {
211  vres = vmlaq_f32(vres, data, data);
212  }
213  else
214  {
215  vres = vaddq_f32(vres, data);
216  }
217  }
218  }
219  // Divide by scale
220  vres = vmulq_f32(vres, scale_v);
221  }
222  else
223  {
224  vres = vdupq_n_f32(std::numeric_limits<float>::lowest());
225  for(int y = pool_start_y; y < pool_end_y; ++y)
226  {
227  for(int x = pool_start_x; x < pool_end_x; ++x)
228  {
229  const float32x4_t data = vld1q_f32(reinterpret_cast<const float *>(in.ptr() + (x - pool_pad_left) * static_cast<int>(src->info()->strides_in_bytes().y()) + (y - pool_pad_top) * static_cast<int>
230  (src->info()->strides_in_bytes().z())) + x_off);
231  vres = vmaxq_f32(vres, data);
232  }
233  }
234  }
235 
236  // Calculate square-root in case of l2 pooling
237  if(pool_info.pool_type == PoolingType::L2)
238  {
239  float32x4_t l2_res = { static_cast<float>(sqrt(vgetq_lane_f32(vres, 0))),
240  static_cast<float>(sqrt(vgetq_lane_f32(vres, 1))),
241  static_cast<float>(sqrt(vgetq_lane_f32(vres, 2))),
242  static_cast<float>(sqrt(vgetq_lane_f32(vres, 3)))
243  };
244  vres = l2_res;
245  }
246 
247  // Store result
248  vst1q_f32(reinterpret_cast<float *>(out.ptr()) + x_off, vres);
249  }
250 
251  // Left-overs loop
252  for(; x_off < window_end_x; ++x_off)
253  {
254  float res = 0.0f;
255 
256  if(pool_info.pool_type != PoolingType::MAX)
257  {
258  // Calculate scale
259  const float scale = calculate_avg_scale(pool_info.exclude_padding, DataLayout::NHWC, id, pool_size_x, pool_size_y, upper_bound_w, upper_bound_h, pool_pad_left, pool_pad_top, pool_stride_x,
260  pool_stride_y);
261 
262  for(int y = pool_start_y; y < pool_end_y; ++y)
263  {
264  for(int x = pool_start_x; x < pool_end_x; ++x)
265  {
266  const float data = *(reinterpret_cast<const float *>(in.ptr() + (x - pool_pad_left) * static_cast<int>(src->info()->strides_in_bytes().y()) + (y - pool_pad_top) * static_cast<int>
267  (src->info()->strides_in_bytes().z())) + x_off);
268 
269  // Get power of 2 in case of l2 pooling and accumulate
270  if(pool_info.pool_type == PoolingType::L2)
271  {
272  res += data * data;
273  }
274  else
275  {
276  res += data;
277  }
278  }
279  }
280 
281  // Divide by scale
282  res *= scale;
283  }
284  else
285  {
286  res = std::numeric_limits<float>::lowest();
287  for(int y = pool_start_y; y < pool_end_y; ++y)
288  {
289  for(int x = pool_start_x; x < pool_end_x; ++x)
290  {
291  const float data = *(reinterpret_cast<const float *>(in.ptr() + (x - pool_pad_left) * static_cast<int>(src->info()->strides_in_bytes().y()) + (y - pool_pad_top) * static_cast<int>
292  (src->info()->strides_in_bytes().z())) + x_off);
293  res = std::max(res, data);
294  }
295  }
296  }
297 
298  // Calculate square-root in case of l2 pooling
299  if(pool_info.pool_type == PoolingType::L2)
300  {
301  res = std::sqrt(res);
302  }
303 
304  // Store result
305  *(reinterpret_cast<float *>(out.ptr()) + x_off) = res;
306  }
307  },
308  in, out);
309  }
