44 void NEFFT2D::configure(const ITensor *input, ITensor *output, const FFT2DInfo &config)
45 {
46  ARM_COMPUTE_ERROR_ON_NULLPTR(input, output);
47  ARM_COMPUTE_ERROR_THROW_ON(NEFFT2D::validate(input->info(), output->info(), config));
48  ARM_COMPUTE_LOG_PARAMS(input, output, config);
49 
50  // Setup first pass
51  FFT1DInfo first_pass_config;
52  first_pass_config.axis = config.axis0;
53  first_pass_config.direction = config.direction;
54  _memory_group.manage(&_first_pass_tensor);
55  _first_pass_func.configure(input, &_first_pass_tensor, first_pass_config);
56 
57  // Setup second pass
58  FFT1DInfo second_pass_config;
59  second_pass_config.axis = config.axis1;
60  second_pass_config.direction = config.direction;
61  _second_pass_func.configure(&_first_pass_tensor, output, second_pass_config);
62  _first_pass_tensor.allocator()->allocate();

65 Status NEFFT2D::validate(const ITensorInfo *input, const ITensorInfo *output, const FFT2DInfo &config)
66 {
67  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(input, output);
68 
69  // Create intermediate tensor info
70  TensorInfo first_pass_tensor(input->clone()->set_is_resizable(true).reset_padding().set_num_channels(2));
71 
72  // Validate first pass
73  FFT1DInfo first_pass_config;
74  first_pass_config.axis = config.axis0;
75  first_pass_config.direction = config.direction;
76  ARM_COMPUTE_RETURN_ON_ERROR(NEFFT1D::validate(input, &first_pass_tensor, first_pass_config));
77 
78  // Validate second pass
79  FFT1DInfo second_pass_config;
80  second_pass_config.axis = config.axis1;
81  second_pass_config.direction = config.direction;
82  ARM_COMPUTE_RETURN_ON_ERROR(NEFFT1D::validate(&first_pass_tensor, output, second_pass_config));
83 
84  // Checks performed when output is configured
85  if ((output != nullptr) && (output->total_size() != 0))
86  {
87  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_SHAPES(input, output);
88  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input, output);
89  }
90 
91  return Status{};
92 }
