36 void mul_F32_F32_F32(const ITensor *src1, const ITensor *src2, ITensor *out, const Window &window, float scale)
37 {
38  // Create input windows
39  Window input1_win = window.broadcast_if_dimension_le_one(src1->info()->tensor_shape());
40  Window input2_win = window.broadcast_if_dimension_le_one(src2->info()->tensor_shape());
41 
42  // Clear X Dimension on execution window as we handle manually
43  Window win = window;
44  win.set(Window::DimX, Window::Dimension(0, 1, 1));
45 
46  constexpr int window_step_x = 16 / sizeof(float);
47  const auto window_start_x = static_cast<int>(window.x().start());
48  const auto window_end_x = static_cast<int>(window.x().end());
49  const bool is_broadcast_across_x = src1->info()->tensor_shape().x() != src2->info()->tensor_shape().x();
50 
51  using ExactTagType = typename wrapper::traits::neon_vector<float, window_step_x>::tag_type;
52 
53  if (is_broadcast_across_x)
54  {
55  const bool is_broadcast_input_2 = input2_win.x().step() == 0;
56  Window broadcast_win = is_broadcast_input_2 ? input2_win : input1_win;
57  Window non_broadcast_win = !is_broadcast_input_2 ? input2_win : input1_win;
58  const ITensor *broadcast_tensor = is_broadcast_input_2 ? src2 : src1;
59  const ITensor *non_broadcast_tensor = !is_broadcast_input_2 ? src2 : src1;
60 
61  // Clear X Dimension on execution window as we handle manually
62  non_broadcast_win.set(Window::DimX, Window::Dimension(0, 1, 1));
63 
64  Iterator broadcast_input(broadcast_tensor, broadcast_win);
65  Iterator non_broadcast_input(non_broadcast_tensor, non_broadcast_win);
66  Iterator dst(out, win);
67 
68  execute_window_loop(
69  win,
70  [&](const Coordinates &)
71  {
72  const auto non_broadcast_input_ptr = reinterpret_cast<const float *>(non_broadcast_input.ptr());
73  const auto output_ptr = reinterpret_cast<float *>(dst.ptr());
74 
75  const float broadcast_value = *reinterpret_cast<const float *>(broadcast_input.ptr());
76  const auto broadcast_value_vec = wrapper::vdup_n(broadcast_value, ExactTagType{});
77  const auto scale_vec = wrapper::vdup_n(scale, ExactTagType{});
78 
79  // Compute window_step_x elements per iteration
80  int x = window_start_x;
81  for (; x <= (window_end_x - window_step_x); x += window_step_x)
82  {
83  const auto non_broadcast_v = wrapper::vloadq(non_broadcast_input_ptr + x);
84  auto res = wrapper::vmul(wrapper::vmul(broadcast_value_vec, non_broadcast_v), scale_vec);
85  wrapper::vstore(output_ptr + x, res);
86  }
87 
88  // Compute left-over elements
89  for (; x < window_end_x; ++x)
90  {
91  const auto non_broadcast_v = *(non_broadcast_input_ptr + x);
92  *(output_ptr + x) = broadcast_value * non_broadcast_v * scale;
93  }
94  },
95  broadcast_input, non_broadcast_input, dst);
96  }
97  else
98  {
99  // Clear X Dimension on execution window as we handle manually
100  input1_win.set(Window::DimX, Window::Dimension(0, 1, 1));
101  input2_win.set(Window::DimX, Window::Dimension(0, 1, 1));
102 
103  Iterator input1(src1, input1_win);
104  Iterator input2(src2, input2_win);
105  Iterator dst(out, win);
106 
107  execute_window_loop(
108  win,
109  [&](const Coordinates &)
110  {
111  const auto input1_ptr = reinterpret_cast<const float *>(input1.ptr());
112  const auto input2_ptr = reinterpret_cast<const float *>(input2.ptr());
113  const auto output_ptr = reinterpret_cast<float *>(dst.ptr());
114 
115  // Compute window_step_x elements per iteration
116  int x = window_start_x;
117  for (; x <= (window_end_x - window_step_x); x += window_step_x)
118  {
119  const auto ta1 = wrapper::vloadq(input1_ptr + x);
120  const auto ta2 = wrapper::vloadq(input2_ptr + x);
121  const auto scale_vec = wrapper::vdup_n(scale, ExactTagType{});
122  const auto res = wrapper::vmul(wrapper::vmul(ta1, ta2), scale_vec);
123  wrapper::vstore(output_ptr + x, res);
124  }
125 
126  // Compute left-over elements
127  for (; x < window_end_x; ++x)
128  {
129  const auto ta1 = *(input1_ptr + x);
130  const auto ta2 = *(input2_ptr + x);
131  *(output_ptr + x) = ta1 * ta2 * scale;
132  }
133  },
134  input1, input2, dst);
135  }
136 }
