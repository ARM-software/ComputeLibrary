67 Status is_supported_op_helper(const GpuWorkloadContext &context,
68  const ITensorInfo *lhs,
69  const ITensorInfo *rhs,
70  const ITensorInfo *dst,
71  const MatMulAttributes &attributes,
72  const GpuMatMulSettings &settings)
73 {
74  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(lhs, rhs);
75 
76  TensorInfo dst_info_to_validate;
77  const ITensorInfo *dst_info_to_validate_ptr = &dst_info_to_validate;
78 
79  if (dst != nullptr)
80  {
81  dst_info_to_validate_ptr = dst;
82  }
83 
84  calculate_and_init_dst_if_empty(&dst_info_to_validate, lhs, rhs, attributes, settings);
85 
86  // Check support level
87  // Data type
88  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(lhs, 1, DataType::F16, DataType::F32);
89 
90  // Check components
91  if (context.gpu_language() == GpuLanguage::OpenCL)
92  {
93  const auto cl_compile_ctx = context.cl_compile_context();
94  ARM_COMPUTE_RETURN_ERROR_ON(cl_compile_ctx == nullptr);
95  // Validate MatMul Component
96  {
97  const auto properties =
98  IGpuKernelComponent::Properties().stage(UnitWorkloadStage{UnitWorkloadStage::Stage::Run});
99 
100  ArgumentPack<ITensorInfo> arguments;
101  arguments.add_const_tensor(ACL_SRC_0, lhs);
102  arguments.add_const_tensor(ACL_SRC_1, rhs);
103  arguments.add_const_tensor(ACL_DST_0, dst_info_to_validate_ptr);
104 
105  ARM_COMPUTE_RETURN_ON_ERROR(ClComponentMatMul::validate(properties, arguments, attributes, settings));
106  }
107  }
108  else
109  {
110  ARM_COMPUTE_RETURN_ERROR_MSG("Unimplemented Gpu language");
111  }
112  return Status{};
113 }

160 Status GpuMatMul::validate_op(const GpuWorkloadSketch &sketch,
161  const ITensorInfo *lhs,
162  const ITensorInfo *rhs,
163  const MatMulAttributes &attributes,
164  const GpuMatMulSettings &settings)
165 {
166  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(lhs, rhs);
167 
168  // Check if tensors have valid id. I.e. they are created from a sketch
169  ARM_COMPUTE_RETURN_ERROR_ON(!lhs->has_valid_id() || !rhs->has_valid_id());
170 
171  // Refer to GpuMatmul::validate_op() for id-validness of this TensorInfo object
172  TensorInfo dst_info_to_validate;
173 
174  // Auto initialize dst tensor info
175  calculate_and_init_dst_if_empty(&dst_info_to_validate, lhs, rhs, attributes, settings);
176 
177  // Perform fusion test
178  // Check if operator meets fusion constraints
179  ArgumentPack<ITensorInfo> tensors;
180  tensors.add_const_tensor(ACL_SRC_0, lhs);
181  tensors.add_const_tensor(ACL_SRC_1, rhs);
182  tensors.add_const_tensor(ACL_DST_0, &dst_info_to_validate);
183  const auto op = sketch.implementation().operator_group().new_operator(operator_type, tensors);
184  ARM_COMPUTE_RETURN_ERROR_ON_MSG(!sketch.implementation().operator_group().try_add_operator(op),
185  "Operator fusion test failed. This operator cannot be fused into the workload");
186 
187  // Check if configuration is supported
188  return is_supported_op_helper(*sketch.gpu_context(), lhs, rhs, &dst_info_to_validate, attributes, settings);
189 }

191 ITensorInfo *GpuMatMul::create_op(GpuWorkloadSketch &sketch,
192  ITensorInfo *lhs,
193  ITensorInfo *rhs,
194  const Attributes &attributes,
195  const Settings &settings)
196 {
197  ARM_COMPUTE_LOG_PARAMS(lhs, rhs, attributes, settings);
198 
199  ITensorInfo *dst = sketch.implementation().create_virtual_tensor();
200 
201  // Assert validation
202  ARM_COMPUTE_ERROR_THROW_ON(GpuMatMul::validate_op(sketch, lhs, rhs, attributes, settings));
203  ARM_COMPUTE_ERROR_ON_NULLPTR(lhs, rhs, dst);
204 
205  // Auto initialize dst tensor
206  calculate_and_init_dst_if_empty(dst, lhs, rhs, attributes, settings);
207 
208  // Translate into components and add to component graph
209  auto &comp_graph = sketch.implementation().component_graph();
210  const auto sketch_ctx = sketch.implementation().context();
211 
212  if (sketch_ctx->gpu_language() == GpuLanguage::OpenCL)
213  {
214  auto properties = IGpuKernelComponent::Properties();
215  properties.stage(UnitWorkloadStage{UnitWorkloadStage::Stage::Run});
216 
217  ArgumentPack<ITensorInfo> arguments;
218  arguments.add_const_tensor(ACL_SRC_0, lhs);
219  arguments.add_const_tensor(ACL_SRC_1, rhs);
220  arguments.add_const_tensor(ACL_DST_0, dst);
221  comp_graph.add_new_component<ClComponentMatMul>(properties, arguments, attributes, settings);
222  }
223  else
224  {
225  ARM_COMPUTE_ERROR("Unimplemented Gpu language");
226  }
227 
228  // Set up fusion test by adding to the Operator Group
229  // Note this has to be performed after all the components have been successfully added to the component graph
230 
231  // Pack tensor infos
232  ArgumentPack<ITensorInfo> tensors;
233  tensors.add_const_tensor(ACL_SRC_0, lhs);
234  tensors.add_const_tensor(ACL_SRC_1, rhs);
235  tensors.add_const_tensor(ACL_DST_0, dst);
236 
237  const auto op = sketch.implementation().operator_group().new_operator(operator_type, tensors);
238  sketch.implementation().operator_group().add_operator(op);
239 
240  return dst;
241 }
