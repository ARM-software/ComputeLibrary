101 void GraphContext::finalize()
102 {
103  const size_t num_pools = 1;
104  for (auto &mm_obj : _memory_managers)
105  {
106  ARM_COMPUTE_ERROR_ON(!mm_obj.second.allocator);
107 
108  // Finalize intra layer memory manager
109  if (mm_obj.second.intra_mm != nullptr)
110  {
111  mm_obj.second.intra_mm->populate(*mm_obj.second.allocator, num_pools);
112  }
113  // Finalize cross layer memory manager
114  if (mm_obj.second.cross_mm != nullptr)
115  {
116  mm_obj.second.cross_mm->populate(*mm_obj.second.allocator, num_pools);
117  }
118  }
119 }
