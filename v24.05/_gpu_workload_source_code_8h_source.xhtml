41 /** Extract kernel arguments of one tensor from a flat list of kernel arguments.
42  *
43  * @param[in] flat_kernel_args
44  * @return GpuKernelArgumentList
45  */
46 GpuKernelArgumentList extract_kernel_args_for_one_tensor(GpuKernelArgumentList &flat_kernel_args)
47 {
48  if (flat_kernel_args.empty())
49  {
50  return {};
51  }
52  GpuKernelArgumentList tensor_kargs{};
53 
54  const GpuKernelArgumentBinding &karg_head = flat_kernel_args.front();
55  tensor_kargs.push_back(karg_head);
56  flat_kernel_args.pop_front();
57  const auto tensor_id = karg_head.id();
58 
59  while (!flat_kernel_args.empty())
60  {
61  const GpuKernelArgumentBinding &karg = flat_kernel_args.front();
62  if (karg.id() != tensor_id) // Encounter the next tensor, return the current tensor's kernel arguments
63  {
64  return tensor_kargs;
65  }
66  tensor_kargs.push_back(karg);
67  flat_kernel_args.pop_front();
68  }
69  return tensor_kargs;
70 }

203 /** Hold the generated kernel source code and other information required to compile and run the workload.
204  */
205 class GpuWorkloadSourceCode
206 {
207 public:
208  /** Default constructor */
209  GpuWorkloadSourceCode() = default;
210  /** Add a unit workload to the workload code
211  *
212  * @param[in] kernel_code @ref GpuKernelSourceCode to be contained within the unit workload
213  * @param[in] stage Stage of the unit workload
214  * @param[in] mem_map @ref MemoryDescriptor map for all tensors within the unit workload
215  * @param[in] context @ref GpuWorkloadContext associated with the unit workload
216  *
217  * @return UnitWorkloadId Allocated unit workload id
218  */
219  UnitWorkloadId add_unit_workload(const GpuKernelSourceCode &kernel_code,
220  const UnitWorkloadStage &stage,
221  const MemoryDescriptorMap &mem_map,
222  const GpuWorkloadContext *context)
223  {
224  // Use the size of the kernel codes as Id
225  const auto uwk_id = static_cast<UnitWorkloadId>(_unit_workloads.size());
226  const auto unit_work = GpuUnitWorkload(uwk_id, kernel_code, stage);
227  _unit_workloads.push_back(unit_work);
228 
229  GpuKernelArgumentList flat_kernel_args = kernel_code.arguments();
230  GpuKernelArgumentList tensor_kargs{};
231  while (true)
232  {
233  tensor_kargs = extract_kernel_args_for_one_tensor(flat_kernel_args);
234  if (tensor_kargs.empty())
235  {
236  break;
237  }
238  else
239  {
240  const auto tensor_id = tensor_kargs.at(0).id();
241  _workload_arguments[tensor_id] = GpuWorkloadArgument{
242  *context->implementation().get_tensor_info(tensor_id), mem_map.at(tensor_id), tensor_kargs};
243  if (_tensor_uwork_map.find(tensor_id) == _tensor_uwork_map.end())
244  {
245  _tensor_uwork_map[tensor_id] = std::set<UnitWorkloadId>();
246  }
247  _tensor_uwork_map[tensor_id].insert(uwk_id);
248  }
249  }
250 
251  return uwk_id;
252  }
253  /** Get a unit workload from its id */
254  const GpuUnitWorkload &query_unit_workload(UnitWorkloadId id) const
255  {
256  ARM_COMPUTE_ERROR_ON(id < 0);
257  return _unit_workloads.at(id);
258  }
259  /** Get all unit workloads sorted in topological order */
260  std::vector<UnitWorkloadId> unit_workloads() const
261  {
262  std::vector<UnitWorkloadId> ids{};
263 
264  for (const auto &uwk : _unit_workloads)
265  {
266  ids.push_back(uwk.id());
267  }
268  return ids;
269  }
270  /** Get a @ref GpuWorkloadArgument from its associated tensor id */
271  const GpuWorkloadArgument *query_tensor(ITensorInfo::Id t_id) const
272  {
273  return &_workload_arguments.at(t_id);
274  }
275  /** Get all tensors in the entire workload */
276  std::vector<ITensorInfo::Id> tensors() const
277  {
278  std::vector<ITensorInfo::Id> ids{};
279  for (const auto &id_tensor : _workload_arguments)
280  {
281  ids.push_back(id_tensor.first);
282  }
283  return ids;
284  }
285  /** Get all unit workloads connected to the tensor with @p t_id */
286  std::vector<UnitWorkloadId> get_unit_workloads_from_tensor(ITensorInfo::Id t_id) const
287  {
288  const auto unit_work_set = _tensor_uwork_map.at(t_id);
289  return std::vector<UnitWorkloadId>(unit_work_set.begin(), unit_work_set.end());
290  }
291 
292 private:
293  std::vector<GpuUnitWorkload> _unit_workloads{};
294  std::map<ITensorInfo::Id, GpuWorkloadArgument> _workload_arguments{};
295  std::map<ITensorInfo::Id, std::set<UnitWorkloadId>> _tensor_uwork_map{};
296 };
