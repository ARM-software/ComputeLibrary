53  {
54  Tensor *dst = output(0);

62 TensorDescriptor PadLayerNode::configure_output(size_t idx) const
63 {
64  ARM_COMPUTE_UNUSED(idx);
65  ARM_COMPUTE_ERROR_ON(idx >= _outputs.size());
66 
67  const Tensor *src = input(0);
68  ARM_COMPUTE_ERROR_ON(src == nullptr);
69 
70  TensorDescriptor output_desc = src->desc();
71  const TensorShape input_shape = src->desc().shape;
72  for (size_t dim = 0; dim < _padding.size(); ++dim)
73  {
74  output_desc.shape.set(dim, _padding[dim].first + input_shape[dim] + _padding[dim].second);
75  }
76 
77  return output_desc;
78 }
