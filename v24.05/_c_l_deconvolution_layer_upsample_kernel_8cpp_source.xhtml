44 Status CLDeconvolutionLayerUpsampleKernel::validate(const ITensorInfo *input,
45  const ITensorInfo *output,
46  const PadStrideInfo &info)
47 {
48  ARM_COMPUTE_UNUSED(info);
49  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(input, output);
50  ARM_COMPUTE_RETURN_ERROR_ON_F16_UNSUPPORTED(input);
51  ARM_COMPUTE_RETURN_ERROR_ON(input->data_type() == DataType::UNKNOWN);
52  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input, output);
53  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_QUANTIZATION_INFO(input, output);
54 
55  const DataLayout data_layout = input->data_layout();
56 
57  const size_t idx_w = get_data_layout_dimension_index(data_layout, DataLayoutDimension::WIDTH);
58  const size_t idx_h = get_data_layout_dimension_index(data_layout, DataLayoutDimension::HEIGHT);
59  const size_t idx_c = get_data_layout_dimension_index(data_layout, DataLayoutDimension::CHANNEL);
60 
61  ARM_COMPUTE_RETURN_ERROR_ON(output->dimension(idx_w) == 0);
62  ARM_COMPUTE_RETURN_ERROR_ON(output->dimension(idx_h) == 0);
63 
64  ARM_COMPUTE_RETURN_ERROR_ON(input->dimension(idx_c) != output->dimension(idx_c));
65  for (size_t i = 3; i < Coordinates::num_max_dimensions; ++i)
66  {
67  ARM_COMPUTE_RETURN_ERROR_ON(input->dimension(i) != output->dimension(i));
68  }
69 
70  return Status{};
71 }

78 void CLDeconvolutionLayerUpsampleKernel::configure(const CLCompileContext &compile_context,
79  const ICLTensor *input,
80  ICLTensor *output,
81  const PadStrideInfo &info)
82 {
83  ARM_COMPUTE_ERROR_ON_NULLPTR(input, output);
84 
85  // Perform validation step
86  ARM_COMPUTE_ERROR_THROW_ON(CLDeconvolutionLayerUpsampleKernel::validate(input->info(), output->info(), info));
87  auto padding_info = get_padding_info({input, output});
88 
89  _input = input;

93 
94  // Create kernel
95  CLBuildOptions build_opts;
96  build_opts.add_option(("-DDATA_TYPE=" + get_cl_unsigned_type_from_element_size(input->info()->element_size())));
97  _kernel = create_kernel(compile_context, "deconvolution_upsample", build_opts.options());
98 
99  constexpr unsigned int num_elems_processed_per_iteration = 1;
100 
101  // Configure kernel window
102  Window win = calculate_max_window(*output->info(), Steps(num_elems_processed_per_iteration));
103  AccessWindowHorizontal output_access(output->info(), 0, num_elems_processed_per_iteration);
104 
105  ICLKernel::configure_internal(win);
106  ARM_COMPUTE_ERROR_ON(has_padding_changed(padding_info));

109 void CLDeconvolutionLayerUpsampleKernel::run(const Window &window, cl::CommandQueue &queue)
110 {
111  ARM_COMPUTE_ERROR_ON_UNCONFIGURED_KERNEL(this);
112  ARM_COMPUTE_ERROR_ON_INVALID_SUBWINDOW(ICLKernel::window(), window);
113 
114  const size_t idx_w = get_data_layout_dimension_index(_data_layout, DataLayoutDimension::WIDTH);
115  const size_t idx_h = get_data_layout_dimension_index(_data_layout, DataLayoutDimension::HEIGHT);
116 
117  const int out_start_x = _info.pad_left();
118  const int out_end_x = _output->info()->dimension(idx_w) - _info.pad_right() + _info.stride().first - 1;
119  const int out_step_x = _info.stride().first;
120 
121  const int out_start_y = _info.pad_top();
122  const int out_end_y = _output->info()->dimension(idx_h) - _info.pad_bottom() + _info.stride().second - 1;
123  const int out_step_y = _info.stride().second;
124 
125  switch (_data_layout)
126  {
127  case DataLayout::NCHW:
128  {
129  Window collapsed = window.collapse_if_possible(ICLKernel::window(), Window::DimZ);
130 
131  Window slice_out = collapsed.first_slice_window_3D();
132  slice_out.set(Window::DimX, Window::Dimension(out_start_x, out_end_x, out_step_x));
133  slice_out.set(Window::DimY, Window::Dimension(out_start_y, out_end_y, out_step_y));
134 
135  Window slice_in = collapsed.first_slice_window_3D();
136 
137  do
138  {
139  unsigned int idx = 0;
140  add_3D_tensor_argument(idx, _input, slice_in);
141  add_3D_tensor_argument(idx, _output, slice_out);
142  enqueue(queue, *this, slice_out, lws_hint());
143  } while (collapsed.slide_window_slice_3D(slice_in) && collapsed.slide_window_slice_3D(slice_out));
144  break;
145  }
146  case DataLayout::NHWC:
147  {
148  // NOTE: not collapsing in NHWC
149  Window slice_out = window.first_slice_window_3D();
150  slice_out.set(Window::DimY, Window::Dimension(out_start_x, out_end_x, out_step_x));
151  slice_out.set(Window::DimZ, Window::Dimension(out_start_y, out_end_y, out_step_y));
152 
153  Window slice_in = window.first_slice_window_3D();
154 
155  do
156  {
157  unsigned int idx = 0;
158  add_3D_tensor_argument(idx, _input, slice_in);
159  add_3D_tensor_argument(idx, _output, slice_out);
160  enqueue(queue, *this, slice_out, lws_hint());
161  } while (window.slide_window_slice_3D(slice_in) && window.slide_window_slice_3D(slice_out));
162  break;
163  }
164  default:
165  ARM_COMPUTE_ERROR("Unsupported data layout");
166  }
167 }
