252 Status validate_arguments(const ITensorInfo *input, const ITensorInfo *output, unsigned int axis, ReductionOperation op)
253 {
254  ARM_COMPUTE_UNUSED(op);
255 
256  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(input, output);
257  ARM_COMPUTE_RETURN_ERROR_ON_CPU_F16_UNSUPPORTED(input);
258 
259  if (input->num_channels() == 1)
260  {
261  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(input, 1, DataType::QASYMM8_SIGNED, DataType::QASYMM8,
262  DataType::S32, DataType::F16, DataType::F32);
263  }
264  else
265  {
266  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(input, 2, DataType::F32);
267  ARM_COMPUTE_RETURN_ERROR_ON(op != ReductionOperation::SUM);
268  ARM_COMPUTE_RETURN_ERROR_ON(axis != 2);
269  }
270 
271  ARM_COMPUTE_RETURN_ERROR_ON_MSG(axis >= TensorShape::num_max_dimensions,
272  "Reduction axis greater than max number of dimensions");
273  ARM_COMPUTE_RETURN_ERROR_ON_MSG(axis > 3, "Unsupported reduction axis");
274 
275  if (output->total_size() != 0)
276  {
277  bool is_arg_min_max = (op == ReductionOperation::ARG_IDX_MAX || op == ReductionOperation::ARG_IDX_MIN);
278  if (!is_arg_min_max)
279  {
280  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input, output);
281  ARM_COMPUTE_RETURN_ERROR_ON(input->num_channels() != output->num_channels());
282  }
283  else
284  {
285  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(output, 1, DataType::U32, DataType::S32);
286  }
287 
288  const TensorShape output_shape =
289  arm_compute::misc::shape_calculator::compute_reduced_shape(input->tensor_shape(), axis);
290  const TensorInfo tensor_info_reshaped = input->clone()->set_tensor_shape(output_shape);
291  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_SHAPES(output, &tensor_info_reshaped);
292  }
293 
294  return Status{};
295 }

302 void NEReductionOperationKernel::configure(const ITensor *input,
303  ITensor *output,
304  unsigned int axis,
305  ReductionOperation op)
306 {
307  ARM_COMPUTE_ERROR_ON_NULLPTR(input, output);
308 
309  ARM_COMPUTE_ERROR_THROW_ON(validate_arguments(input->info(), output->info(), axis, op));
310 
311  _input = input;
312  _output = output;
313  _op = op;
314  _reduction_axis = axis;
315 
316  // Configure kernel window
317  Window win = calculate_max_window(*input->info(), Steps());
318  INEKernel::configure(win);
319 
320  // Calculate output shape and set if empty
321  const TensorShape output_shape =
322  arm_compute::misc::shape_calculator::compute_reduced_shape(input->info()->tensor_shape(), axis);
323  // Output auto initialization if not yet initialized
324  const bool is_arg_min_max = (op == ReductionOperation::ARG_IDX_MIN || op == ReductionOperation::ARG_IDX_MAX);
325  DataType output_data_type = is_arg_min_max ? DataType::S32 : input->info()->data_type();
326  auto_init_if_empty(*output->info(), input->info()
327  ->clone()
328  ->set_tensor_shape(output_shape)
329  .set_data_type(output_data_type)
330  .reset_padding()
331  .set_is_resizable(true));
332  // Determine the reduction function
333  NEReductionOperationKernel::reduce_op();
334 }
