376 template <typename T>
377 static inline int mali_ioctl(int fd, T &arg)
378 {
379  auto *hdr = &arg.header;
380  const int cmd = _IOC(_IOC_READ | _IOC_WRITE, LINUX_UK_BASE_MAGIC, hdr->id, sizeof(T));
381 
382  if(ioctl(fd, cmd, &arg))
383  return -1;
384  if(hdr->ret)
385  return -1;
386 
387  return 0;
388 }
