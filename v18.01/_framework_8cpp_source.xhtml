255 void Framework::run_test(const TestInfo &info, TestCaseFactory &test_factory)
256 {
257  if(test_factory.status() == TestCaseFactory::Status::DISABLED)
258  {
259  log_test_skipped(info);
260  set_test_result(info, TestResult(TestResult::Status::DISABLED));
261  return;
262  }
263 
264  log_test_start(info);
265 
266  Profiler profiler = get_profiler();
267  TestResult result(TestResult::Status::NOT_RUN);
268 
269  _current_test_info = &info;
270  _current_test_result = &result;
271 
272  if(_log_level >= LogLevel::ERRORS)
273  {
274  func_on_all_printers([](Printer * p)
275  {
276  p->print_errors_header();
277  });
278  }
279 
280  const bool is_expected_failure = info.status == TestCaseFactory::Status::EXPECTED_FAILURE;
281 
282  try
283  {
284  std::unique_ptr<TestCase> test_case = test_factory.make();
285 
286  try
287  {
288  test_case->do_setup();
289 
290  for(int i = 0; i < _num_iterations; ++i)
291  {
292  //Start the profiler if:
293  //- there is only one iteration
294  //- it's not the first iteration of a multi-iterations run.
295  //
296  //Reason: if the CLTuner is enabled then the first run will be really messy
297  //as each kernel will be executed several times, messing up the instruments like OpenCL timers.
298  if(_num_iterations == 1 || i != 0)
299  {
300  profiler.start();
301  }
302  test_case->do_run();
303  test_case->do_sync();
304  if(_num_iterations == 1 || i != 0)
305  {
306  profiler.stop();
307  }
308  }
309 
310  test_case->do_teardown();
311 
312  // Change status to success if no error has happend
313  if(result.status == TestResult::Status::NOT_RUN)
314  {
315  result.status = TestResult::Status::SUCCESS;
316  }
317  }
318  catch(const FileNotFound &error)
319  {
320  if(_error_on_missing_assets)
321  {
322  if(_log_level >= LogLevel::ERRORS)
323  {
324  TestError test_error(error.what(), LogLevel::ERRORS);
325  func_on_all_printers([&](Printer * p)
326  {
327  p->print_error(test_error, is_expected_failure);
328  });
329  }
330 
331  result.status = TestResult::Status::FAILED;
332 
333  if(_throw_errors)
334  {
335  throw;
336  }
337  }
338  else
339  {
340  if(_log_level >= LogLevel::DEBUG)
341  {
342  func_on_all_printers([&](Printer * p)
343  {
344  p->print_info(error.what());
345  });
346  }
347 
348  result.status = TestResult::Status::NOT_RUN;
349  }
350  }
351  catch(const TestError &error)
352  {
353  if(_log_level >= error.level())
354  {
355  func_on_all_printers([&](Printer * p)
356  {
357  p->print_error(error, is_expected_failure);
358  });
359  }
360 
361  result.status = TestResult::Status::FAILED;
362 
363  if(_throw_errors)
364  {
365  throw;
366  }
367  }
368 #ifdef ARM_COMPUTE_CL
369  catch(const ::cl::Error &error)
370  {
371  if(_log_level >= LogLevel::ERRORS)
372  {
373  std::stringstream stream;
374  stream << "Error code: " << error.err();
375  TestError test_error(error.what(), LogLevel::ERRORS, stream.str());
376  func_on_all_printers([&](Printer * p)
377  {
378  p->print_error(test_error, is_expected_failure);
379  });
380  }
381 
382  result.status = TestResult::Status::FAILED;
383 
384  if(_throw_errors)
385  {
386  throw;
387  }
388  }
389 #endif /* ARM_COMPUTE_CL */
390  catch(const std::exception &error)
391  {
392  if(_log_level >= LogLevel::ERRORS)
393  {
394  func_on_all_printers([&](Printer * p)
395  {
396  p->print_error(error, is_expected_failure);
397  });
398  }
399 
400  result.status = TestResult::Status::CRASHED;
401 
402  if(_throw_errors)
403  {
404  throw;
405  }
406  }
407  catch(...)
408  {
409  if(_log_level >= LogLevel::ERRORS)
410  {
411  func_on_all_printers([&](Printer * p)
412  {
413  p->print_error(TestError("Received unknown exception"), is_expected_failure);
414  });
415  }
416 
417  result.status = TestResult::Status::CRASHED;
418 
419  if(_throw_errors)
420  {
421  throw;
422  }
423  }
424  }
425  catch(const std::exception &error)
426  {
427  if(_log_level >= LogLevel::ERRORS)
428  {
429  func_on_all_printers([&](Printer * p)
430  {
431  p->print_error(error, is_expected_failure);
432  });
433  }
434 
435  result.status = TestResult::Status::CRASHED;
436 
437  if(_throw_errors)
438  {
439  throw;
440  }
441  }
442  catch(...)
443  {
444  if(_log_level >= LogLevel::ERRORS)
445  {
446  func_on_all_printers([&](Printer * p)
447  {
448  p->print_error(TestError("Received unknown exception"), is_expected_failure);
449  });
450  }
451 
452  result.status = TestResult::Status::CRASHED;
453 
454  if(_throw_errors)
455  {
456  throw;
457  }
458  }
459 
460  if(_log_level >= LogLevel::ERRORS)
461  {
462  func_on_all_printers([](Printer * p)
463  {
464  p->print_errors_footer();
465  });
466  }
467 
468  _current_test_info = nullptr;
469  _current_test_result = nullptr;
470 
471  if(result.status == TestResult::Status::FAILED)
472  {
473  if(info.status == TestCaseFactory::Status::EXPECTED_FAILURE)
474  {
475  result.status = TestResult::Status::EXPECTED_FAILURE;
476  }
477  }
478 
479  if(result.status == TestResult::Status::FAILED || result.status == TestResult::Status::CRASHED)
480  {
481  if(_stop_on_error)
482  {
483  throw std::runtime_error("Abort on first error.");
484  }
485  }
486 
487  result.measurements = profiler.measurements();
488 
489  set_test_result(info, result);
490  log_test_end(info);
491 }

493 bool Framework::run()
494 {
495  // Clear old test results
496  _test_results.clear();
497 
498  if(_log_level >= LogLevel::TESTS)
499  {
500  func_on_all_printers([](Printer * p)
501  {
502  p->print_run_header();
503  });
504  }
505 
506  const std::chrono::time_point<std::chrono::high_resolution_clock> start = std::chrono::high_resolution_clock::now();
507 
508  int id = 0;
509 
510  for(auto &test_factory : _test_factories)
511  {
512  const std::string test_case_name = test_factory->name();
513  const TestInfo test_info{ id, test_case_name, test_factory->mode(), test_factory->status() };
514 
515  if(_test_filter.is_selected(test_info))
516  {
517  run_test(test_info, *test_factory);
518  }
519 
520  ++id;
521  }
522 
523  const std::chrono::time_point<std::chrono::high_resolution_clock> end = std::chrono::high_resolution_clock::now();
524 
525  if(_log_level >= LogLevel::TESTS)
526  {
527  func_on_all_printers([](Printer * p)
528  {
529  p->print_run_footer();
530  });
531  }
532 
533  auto runtime = std::chrono::duration_cast<std::chrono::seconds>(end - start);
534  std::map<TestResult::Status, int> results = count_test_results();
535 
536  if(_log_level > LogLevel::NONE)
537  {
538  std::cout << "Executed " << _test_results.size() << " test(s) ("
539  << results[TestResult::Status::SUCCESS] << " passed, "
540  << results[TestResult::Status::EXPECTED_FAILURE] << " expected failures, "
541  << results[TestResult::Status::FAILED] << " failed, "
542  << results[TestResult::Status::CRASHED] << " crashed, "
543  << results[TestResult::Status::DISABLED] << " disabled) in " << runtime.count() << " second(s)\n";
544  }
545 
546  int num_successful_tests = results[TestResult::Status::SUCCESS] + results[TestResult::Status::EXPECTED_FAILURE] + results[TestResult::Status::DISABLED];
547 
548  return (static_cast<unsigned int>(num_successful_tests) == _test_results.size());
549 }

570 Profiler Framework::get_profiler() const
571 {
572  Profiler profiler;
573 
574  const bool all_instruments = std::any_of(
575  _instruments.begin(),
576  _instruments.end(),
577  [](InstrumentsDescription type) -> bool { return type.first == InstrumentType::ALL; });
578 
579  auto is_selected = [&](InstrumentsDescription instrument) -> bool
580  {
581  return std::find_if(_instruments.begin(), _instruments.end(), [&](InstrumentsDescription type) -> bool {
582  const auto group = static_cast<InstrumentType>(static_cast<uint64_t>(type.first) & 0xFF00);
583  return (group == instrument.first) && (instrument.second == type.second);
584  })
585  != _instruments.end();
586  };
587 
588  for(const auto &instrument : _available_instruments)
589  {
590  if(all_instruments || is_selected(instrument.first))
591  {
592  profiler.add(instrument.second());
593  }
594  }
595 
596  return profiler;
597 }
