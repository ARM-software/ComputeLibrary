61 {

63  // [ b_height * (16 / element_size), ceil(b_width / (16.0f / element_size) ]
64  TensorShape shape_transposed1xW_b{ b.tensor_shape() };

125 {
126  TensorShape scale_out_shape(input.tensor_shape());
127  const unsigned int out_x = input.dimension(0) + (input.dimension(0) - 1) * (sx - 1) + inner_border_right + 2 * info.pad().first;
128  const unsigned int out_y = input.dimension(1) + (input.dimension(1) - 1) * (sy - 1) + inner_border_top + 2 * info.pad().second;
129  scale_out_shape.set(0, out_x);
130  scale_out_shape.set(1, out_y);
131
