294 template <typename T>
295 int SimpleTensor<T>::num_channels() const
296 {
297  switch(_format)
298  {
299  case Format::U8:
300  case Format::S16:
301  case Format::U16:
302  case Format::S32:
303  case Format::U32:
304  case Format::F32:
305  return 1;
306  case Format::RGB888:
307  return 3;
308  case Format::UNKNOWN:
309  return _num_channels;
310  default:
311  ARM_COMPUTE_ERROR("NOT SUPPORTED!");
312  }
313 }
