40 template <typename TensorType, typename AccessorType, typename T>
41 void compute_target_impl(const TensorShape &shape, DataType dt, FixedPointOp op, int fixed_point_position, TensorType &src, TensorType &dst)
42 {
43  Window window;
44 
45  switch(dt)
46  {
47  case DataType::QS8:
48  {
49  constexpr unsigned int num_elems_processed_per_iteration = 16;
50  window = calculate_max_window(*src.info(), Steps(num_elems_processed_per_iteration));
51  AccessWindowHorizontal input_access(src.info(), 0, num_elems_processed_per_iteration);
52  AccessWindowHorizontal output_access(dst.info(), 0, num_elems_processed_per_iteration);
53  update_window_and_padding(window, input_access, output_access);
54  break;
55  }
56  case DataType::QS16:
57  {
58  constexpr unsigned int num_elems_processed_per_iteration = 8;
59  window = calculate_max_window(*src.info(), Steps(num_elems_processed_per_iteration));
60  AccessWindowHorizontal input_access(src.info(), 0, num_elems_processed_per_iteration);
61  AccessWindowHorizontal output_access(dst.info(), 0, num_elems_processed_per_iteration);
62  update_window_and_padding(window, input_access, output_access);
63  break;
64  }
65  default:
66  ARM_COMPUTE_ERROR("Not Supported");
67  break;
68  }
69 
70  int min;
71  int max;
72  switch(op)
73  {
74  case FixedPointOp::EXP:
75  {
76  // Fill tensors. Keep the range between [-1.0, 1.0) so the result won't
77  // overflow.
78  min = -(1 << (fixed_point_position - 1));
79  max = (1 << (fixed_point_position - 1));
80  break;
81  }
82  case FixedPointOp::INV_SQRT:
83  {
84  if(dt == DataType::QS8)
85  {
86  // Fill tensors. Keep the range between [1, 127).
87  min = 1;
88  max = 127;
89  }
90  else
91  {
92  // Fill tensors. Keep the range between [1, 0x7FFF)
93  min = 1;
94  max = 0x7FFF;
95  }
96  break;
97  }
98  case FixedPointOp::LOG:
99  {
100  if(dt == DataType::QS8)
101  {
102  // Fill tensors. Keep the range between [(1 << (fixed_point_position - 1), 63) so the result won't
103  // overflow. E.g. for Q2.5 ln(0.001) = -6.9, which cannot be represented.
104  min = (1 << (fixed_point_position - 1));
105  max = 0x3F;
106  }
107  else
108  {
109  // Fill tensors. Keep the range between [(1 << (fixed_point_position - 1), 0x3FFF) so the result won't
110  // overflow.
111  min = (1 << (fixed_point_position - 1));
112  max = 0x3FFF;
113  }
114  break;
115  }
116  case FixedPointOp::RECIPROCAL:
117  {
118  if(dt == DataType::QS8)
119  {
120  // Fill tensors. Keep the range between [15, 100) so the result won't
121  // overflow. E.g. for Q2.5 reciprocal(0.001) = 1000, which cannot be represented.
122  min = 15;
123  max = 0x7F;
124  }
125  else
126  {
127  // Fill tensors. Keep the range between [15, 0x7FFF) so the result won't
128  // overflow.
129  min = 15;
130  max = 0x7FFF;
131  }
132  break;
133  }
134  default:
135  ARM_COMPUTE_ERROR("Not Supported");
136  break;
137  }
138 
139  std::uniform_int_distribution<> distribution(min, max);
140  library->fill(AccessorType(src), distribution, 0);
141 
142  Iterator input(&src, window);
143  Iterator output(&dst, window);
144 
145  const auto loop_function = [&](const Coordinates & id)
146  {
147  switch(dt)
148  {
149  case DataType::QS8:
150  {
151  const qint8x16_t qs8in = vld1q_s8(reinterpret_cast<const qint8_t *>(input.ptr()));
152  switch(op)
153  {
154  case FixedPointOp::EXP:
155  {
156  // Use saturated exp
157  vst1q_s8(reinterpret_cast<qint8_t *>(output.ptr()), vqexpq_qs8(qs8in, fixed_point_position));
158  break;
159  }
160  case FixedPointOp::INV_SQRT:
161  {
162  vst1q_s8(reinterpret_cast<qint8_t *>(output.ptr()), vqinvsqrtq_qs8(qs8in, fixed_point_position));
163  break;
164  }
165  case FixedPointOp::LOG:
166  {
167  vst1q_s8(reinterpret_cast<qint8_t *>(output.ptr()), vlogq_qs8(qs8in, fixed_point_position));
168  break;
169  }
170  case FixedPointOp::RECIPROCAL:
171  {
172  vst1q_s8(reinterpret_cast<qint8_t *>(output.ptr()), vrecipq_qs8(qs8in, fixed_point_position));
173  break;
174  }
175  default:
176  ARM_COMPUTE_ERROR("Not Supported");
177  break;
178  }
179  break;
180  }
181  case DataType::QS16:
182  {
183  const qint16x8_t qs16in = vld1q_qs16(reinterpret_cast<const qint16_t *>(input.ptr()));
184  switch(op)
185  {
186  case FixedPointOp::EXP:
187  {
188  // Use saturated exp
189  vst1q_qs16(reinterpret_cast<qint16_t *>(output.ptr()), vqexpq_qs16(qs16in, fixed_point_position));
190  break;
191  }
192  case FixedPointOp::INV_SQRT:
193  {
194  vst1q_qs16(reinterpret_cast<qint16_t *>(output.ptr()), vqinvsqrtq_qs16(qs16in, fixed_point_position));
195  break;
196  }
197  case FixedPointOp::LOG:
198  {
199  vst1q_qs16(reinterpret_cast<qint16_t *>(output.ptr()), vlogq_qs16(qs16in, fixed_point_position));
200  break;
201  }
202  case FixedPointOp::RECIPROCAL:
203  {
204  vst1q_qs16(reinterpret_cast<qint16_t *>(output.ptr()), vqrecipq_qs16(qs16in, fixed_point_position));
205  break;
206  }
207  default:
208  ARM_COMPUTE_ERROR("Not Supported");
209  break;
210  }
211  break;
212  }
213  default:
214  ARM_COMPUTE_ERROR("Not Supported");
215  break;
216  }
217  };
218 
219  execute_window_loop(window, loop_function, input, output);
220 }
