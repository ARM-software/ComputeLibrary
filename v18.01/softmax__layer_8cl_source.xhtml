114 __kernel void softmax_layer_max(
115  TENSOR3D_DECLARATION(src),
116  TENSOR3D_DECLARATION(dst),
117  uint width)
118 {
119  Image src = CONVERT_TENSOR3D_TO_IMAGE_STRUCT(src);
120  Image dst = CONVERT_TENSOR3D_TO_IMAGE_STRUCT(dst);
121 
122  // Initialize local maximum
123  VEC_DATA_TYPE(DATA_TYPE, 16)
124  max_val = (VEC_DATA_TYPE(DATA_TYPE, 16))type_min;
125 
126  // Calculate max of row
127  const uint width4 = width >> 4;
128  for(uint i = 0; i < width4; i++)
129  {
130  VEC_DATA_TYPE(DATA_TYPE, 16)
131  data = vload16(0, (__global DATA_TYPE *)offset(&src, i << 4, 0));
132  max_val = MAX_OP(data, max_val, DATA_TYPE, 16);
133  }
134 
135 #ifdef NON_MULTIPLE_OF_16
136  // Handle non multiple of 16
137  VEC_DATA_TYPE(DATA_TYPE, 16)
138  data = vload16(0, (__global DATA_TYPE *)offset(&src, width4 << 4, 0));
139  VEC_DATA_TYPE(SELECT_DATA_TYPE, 16)
140  widx = CONVERT(((uint16)(width4 << 4) + idx16) < width, VEC_DATA_TYPE(SELECT_DATA_TYPE, 16));
141  max_val = MAX_OP(max_val, select(type_min, data, widx), DATA_TYPE, 16);
142 #endif /* NON_MULTIPLE_OF_16 */
143 
144  // Perform max reduction
145  max_val.s01234567 = MAX_OP(max_val.s01234567, max_val.s89ABCDEF, DATA_TYPE, 8);
146  max_val.s0123 = MAX_OP(max_val.s0123, max_val.s4567, DATA_TYPE, 4);
147  max_val.s01 = MAX_OP(max_val.s01, max_val.s23, DATA_TYPE, 2);
148  max_val.s0 = MAX_OP(max_val.s0, max_val.s1, DATA_TYPE, 1);
149 
150  // Store result
151  *((__global DATA_TYPE *)dst.ptr) = max_val.s0;
152 }

500 __kernel void softmax_layer_max_shift_exp_sum_parallel(

512  const uint lid = get_local_id(0);
513 
514 #ifdef BETA
515  // Initialize beta
516  VEC_DATA_TYPE(DATA_TYPE, 4)
517  beta = (VEC_DATA_TYPE(DATA_TYPE, 4))BETA;
518 #endif /* BETA */
519 
520  // Define one temporary vector per work-item.
521  __local VEC_DATA_TYPE(DATA_TYPE, 4) tmp_local[GRID_SIZE];
522  __local DATA_TYPE max_local;
523 
524  __constant VEC_DATA_TYPE(DATA_TYPE, 4) type_min4 = (VEC_DATA_TYPE(DATA_TYPE, 4))(MINVAL);
525  VEC_DATA_TYPE(DATA_TYPE, 4)
526  max_val_vec = (VEC_DATA_TYPE(DATA_TYPE, 4))type_min4;
527  // Number of elements per work-item.
528  const uint row = width / GRID_SIZE;
529  // Number of iterations per work-item.
530  const uint width_ = row >> 2;
531  // Calculate max of row
532  uint i = 0;
533  for(; i < width_; i++)
534  {
535  VEC_DATA_TYPE(DATA_TYPE, 4)
536  data_max = VLOAD(4)(0, (__global DATA_TYPE *)offset(&src, i * GRID_SIZE * 4, 0));
537  max_val_vec = MAX_OP(data_max, max_val_vec, DATA_TYPE, 4);
538  }
539 #ifdef NON_MULTIPLE_OF_GRID_SIZE
540  // How many work-items needed to complete the computation.

542  if(lid < boundary_workitems)
543  {
544  VEC_DATA_TYPE(DATA_TYPE, 4)
545  data_max = VLOAD(4)(0, (__global DATA_TYPE *)offset(&src, i * GRID_SIZE * 4, 0));
546  max_val_vec = MAX_OP(data_max, max_val_vec, DATA_TYPE, 4);
547  }
548 #ifdef NON_MULTIPLE_OF_VECTOR_SIZE
549  if(boundary_workitems == 0)
550  {
551  boundary_workitems = GRID_SIZE;
552  i--;
553  }
554  if(lid == (boundary_workitems - 1))
555  {
556  // Handle non multiple of 4
557  VEC_DATA_TYPE(DATA_TYPE, 4)
558  data_max = VLOAD(4)(0, (__global DATA_TYPE *)offset(&src, (GRID_SIZE * i * 4) + 4, 0));
559  VEC_DATA_TYPE(SELECT_DATA_TYPE, 4)
560  widx = CONVERT(((uint4)(GRID_SIZE * i * 4) + boundary_workitems * 4 + idx4) < width, VEC_DATA_TYPE(SELECT_DATA_TYPE, 4));
561  max_val_vec = MAX_OP(max_val_vec, select(type_min_, data_max, widx), DATA_TYPE, 4);
562  }
563 #endif /* NON_MULTIPLE_OF_VECTOR_SIZE */
564 #endif /* NON_MULTIPLE_OF_GRID_SIZE */
565  tmp_local[lid] = max_val_vec;
566 
567  barrier(CLK_LOCAL_MEM_FENCE);
568 
569  if(GRID_SIZE >= 256)
570  {
571  if(lid < 128)
572  {
573  tmp_local[lid] = MAX_OP(tmp_local[lid + 128], tmp_local[lid], DATA_TYPE, 4);
574  }
575  barrier(CLK_LOCAL_MEM_FENCE);
576  }
577  if(GRID_SIZE >= 128)
578  {
579  if(lid < 64)
580  {
581  tmp_local[lid] = MAX_OP(tmp_local[lid + 64], tmp_local[lid], DATA_TYPE, 4);
582  }
583  barrier(CLK_LOCAL_MEM_FENCE);
584  }
585  if(GRID_SIZE >= 64)
586  {
587  if(lid < 32)
588  {
589  tmp_local[lid] = MAX_OP(tmp_local[lid + 32], tmp_local[lid], DATA_TYPE, 4);
590  }
591  barrier(CLK_LOCAL_MEM_FENCE);
592  }
593  if(GRID_SIZE >= 32)
594  {
595  if(lid < 16)
596  {
597  tmp_local[lid] = MAX_OP(tmp_local[lid + 16], tmp_local[lid], DATA_TYPE, 4);
598  }
599  barrier(CLK_LOCAL_MEM_FENCE);
600  }
601  if(GRID_SIZE >= 16)
602  {
603  if(lid < 8)
604  {
605  tmp_local[lid] = MAX_OP(tmp_local[lid + 8], tmp_local[lid], DATA_TYPE, 4);
606  }
607  barrier(CLK_LOCAL_MEM_FENCE);
608  }
609  if(GRID_SIZE >= 8)
610  {
611  if(lid < 4)
612  {
613  tmp_local[lid] = MAX_OP(tmp_local[lid + 4], tmp_local[lid], DATA_TYPE, 4);
614  }
615  barrier(CLK_LOCAL_MEM_FENCE);
616  }
617  if(GRID_SIZE >= 4)
618  {
619  if(lid < 2)
620  {
621  tmp_local[lid] = MAX_OP(tmp_local[lid + 2], tmp_local[lid], DATA_TYPE, 4);
622  }
623  barrier(CLK_LOCAL_MEM_FENCE);
624  }
625  if(lid == 0)
626  {
627  max_val_vec = MAX_OP(tmp_local[lid + 1], tmp_local[lid], DATA_TYPE, 4);
628  max_val_vec.s01 = MAX_OP(max_val_vec.s01, max_val_vec.s23, DATA_TYPE, 2);
629  max_val_vec.s0 = MAX_OP(max_val_vec.s0, max_val_vec.s1, DATA_TYPE, 1);
630  max_local = max_val_vec.s0;
631  }
632  barrier(CLK_LOCAL_MEM_FENCE);
633 
634  /* Second section */
635 
636  // Set sum vector
637  VEC_DATA_TYPE(DATA_TYPE, 4)
638  sum1D = 0;
639  DATA_TYPE max_val = max_local;
640 
641  // Shift values, exp and sum
642  for(i = 0; i < width_; i++)
643  {
644  VEC_DATA_TYPE(DATA_TYPE, 4)
645  data = VLOAD(4)(0, (__global DATA_TYPE *)offset(&src, i * GRID_SIZE * 4, 0));
646  data = SUB_OP(data, max_val, DATA_TYPE, 4);
647 #ifdef BETA
648  data = MUL_OP(data, beta, DATA_TYPE, 4);
649 #endif /* BETA */
650  data = EXP_OP(data, DATA_TYPE, 4);
651  VSTORE(4)
652  (data, 0, (__global DATA_TYPE *)offset(&dst, i * GRID_SIZE * 4, 0));
653  sum1D = ADD_OP(sum1D, data, DATA_TYPE, 4);
654  }
655 #ifdef NON_MULTIPLE_OF_GRID_SIZE
656  boundary_workitems = (width % (GRID_SIZE * 4)) / 4;
657  if(lid < boundary_workitems)
658  {
659  VEC_DATA_TYPE(DATA_TYPE, 4)
660  data = VLOAD(4)(0, (__global DATA_TYPE *)offset(&src, i * GRID_SIZE * 4, 0));
661  data = SUB_OP(data, max_val, DATA_TYPE, 4);
662 #ifdef BETA
