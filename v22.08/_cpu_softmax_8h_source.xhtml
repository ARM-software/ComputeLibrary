42 /** Basic function to compute a SoftmaxLayer and a Log SoftmaxLayer.
43  *
44  * Softmax is calculated by :
45  * @f[ out = exp((x - max(x)) * beta) / sum(exp((x - max(x)) * beta)) @f]
46  *
47  * Log Softmax is calculated by :
48  * @f[ out = (x - max(x) * beta) - log(\sum{e^{x - max(x) * beta}}) @f]
49  *
50  * This function runs the following function/kernels:
51  * -# If axis is not 0:
52  * -# @ref CpuPermute
53  * -# @ref kernels::CpuLogits1DMaxKernel
54  * -# @ref kernels::CpuLogits1DSoftmaxKernel
55  */
56 template <bool IS_LOG = false>
57 class CpuSoftmaxGeneric : public ICpuOperator
58 {
59 public:
60  CpuSoftmaxGeneric();
61  /** Set the input and output tensors.
62  *
63  * @param[in,out] src Source tensor info. Data types supported: QASYMM8/QASYMM8_SIGNED/F16/F32.
64  * last value of each row to the nearest multiple.
65  * @param[out] dst Destination tensor ifo. Data types supported: same as @p input.
66  * @param[in] beta (Optional) A scaling factor for the exponent.
67  * @param[in] axis (Optional) The dimension in which to apply the function. E.g. for input of shape 4x5x6 and
68  * axis=1, softmax will be applied to 4x6=24 vectors of size 5. Defaults to 0
69  */
