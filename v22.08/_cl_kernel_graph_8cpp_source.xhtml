61 Status ClDirectConv2dKernel::validate(const ITensorInfo *src, const ITensorInfo *weights, const ITensorInfo *biases, const ITensorInfo *dst, const ClDirectConv2dKernelDescriptor &conv2d_desc)
62 {
63  // 1. Check validity
64  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(src, weights, dst);
65  // Matching data type
66  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(src, weights);
67  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(src, dst);
68  if(biases != nullptr)
69  {
70  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(src, biases);
71  }
72 
73  // Matching data layout
74  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_LAYOUT(src, weights);
75  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_LAYOUT(src, dst);
76  if(biases != nullptr)
77  {
78  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_LAYOUT(src, biases);
79  }
80 
81  // All tensor infos are initialized
82  ARM_COMPUTE_RETURN_ERROR_ON(src->tensor_shape().total_size() == 0);
83  ARM_COMPUTE_RETURN_ERROR_ON(weights->tensor_shape().total_size() == 0);
84  ARM_COMPUTE_RETURN_ERROR_ON(dst->tensor_shape().total_size() == 0);
85  if(biases != nullptr)
86  {
87  ARM_COMPUTE_RETURN_ERROR_ON(biases->tensor_shape().total_size() == 0);
88  }
89  // Device requirements are met
90  ARM_COMPUTE_RETURN_ERROR_ON_F16_UNSUPPORTED(src);
91  // weights shape is correct
92  const DataLayout data_layout = src->data_layout();
93  const int channel_idx = get_data_layout_dimension_index(data_layout, DataLayoutDimension::CHANNEL);
94  ARM_COMPUTE_RETURN_ERROR_ON_MSG(weights->dimension(channel_idx) != src->dimension(channel_idx), "Weights feature map dimension should match the respective src's one");
95  ARM_COMPUTE_RETURN_ERROR_ON_MSG(weights->num_dimensions() > 4, "Weights can be at most 4 dimensional");
96 
97  // dst shape is correct
98  PadStrideInfo legacy_pad_stride(conv2d_desc.conv2d.stride.x(), conv2d_desc.conv2d.stride.y(), conv2d_desc.conv2d.pad.left, conv2d_desc.conv2d.pad.right, conv2d_desc.conv2d.pad.top,
99  conv2d_desc.conv2d.pad.bottom, DimensionRoundingType{});
100  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DIMENSIONS(dst->tensor_shape(),
101  misc::shape_calculator::compute_deep_convolution_shape(*src, *weights, legacy_pad_stride));
102 
103  // biases shape is correct
104  if(biases != nullptr)
105  {
106  ARM_COMPUTE_RETURN_ERROR_ON_MSG(biases->dimension(0) != weights->dimension(3),
107  "Biases size and number of dst feature maps should match");
108  ARM_COMPUTE_RETURN_ERROR_ON_MSG(biases->num_dimensions() > 1,
109  "Biases should be one dimensional");
110  }
111 
112  // 2. Check support level
113  // Data type
114  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(src, 1, DataType::F16, DataType::F32);
115  // Data layout
116  ARM_COMPUTE_RETURN_ERROR_ON_DATA_LAYOUT_NOT_IN(src, DataLayout::NHWC);
117 
118  return Status{};
119 }

144 Status ClElementwiseKernel::validate(const ITensorInfo *lhs, const ITensorInfo *rhs, const ITensorInfo *dst)
145 {
146  // 1. Check validity
147  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(lhs, rhs, dst);
148 
149  // Matching data type
150  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(lhs, rhs);
151  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(lhs, dst);
152 
153  // Matching data layout
154  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_LAYOUT(lhs, rhs);
155  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_LAYOUT(lhs, dst);
156 
157  // All tensor infos are initialized
158  ARM_COMPUTE_RETURN_ERROR_ON(lhs->tensor_shape().total_size() == 0);
159  ARM_COMPUTE_RETURN_ERROR_ON(rhs->tensor_shape().total_size() == 0);
160  ARM_COMPUTE_RETURN_ERROR_ON(dst->tensor_shape().total_size() == 0);
161 
162  // Device requirements are met
163  ARM_COMPUTE_RETURN_ERROR_ON_F16_UNSUPPORTED(lhs);
164 
165  const bool in_place = (lhs == dst) || (rhs == dst);
166  const bool src0_in_place = in_place && (lhs == dst);
167 
168  // dst shape is correct
169  const TensorShape out_shape = TensorShape::broadcast_shape(lhs->tensor_shape(), rhs->tensor_shape());
170  ARM_COMPUTE_RETURN_ERROR_ON_MSG(out_shape.total_size() == 0, "Inputs are not broadcast compatible");
171  ARM_COMPUTE_RETURN_ERROR_ON_MSG(detail::have_different_dimensions(out_shape, dst->tensor_shape(), 0), "Wrong shape for dst");
172  if(in_place)
173  {
174  ARM_COMPUTE_RETURN_ERROR_ON_MSG(detail::have_different_dimensions(out_shape, src0_in_place ? lhs->tensor_shape() : rhs->tensor_shape(), 0),
175  "Wrong shape for dst, cannot do in_place calculation");
176  }
177 
178  // 2. Check support level
179 
180  // Data type
181  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(lhs, 1, DataType::F32, DataType::F16);
182 
183  // Data layout
184  ARM_COMPUTE_RETURN_ERROR_ON_DATA_LAYOUT_NOT_IN(lhs, DataLayout::NHWC);
185 
186  return Status{};
187 }
