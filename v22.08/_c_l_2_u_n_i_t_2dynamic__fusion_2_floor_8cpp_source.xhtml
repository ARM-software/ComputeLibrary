58 TEST_CASE(Operator_Floor_1_F32, framework::DatasetMode::ALL)
59 {
60  /* Computation:
61  * out = floor(input)
62  */
63  const auto data_type = DataType::F32;
64  const auto data_layout = DataLayout::NHWC;
65  const auto t_shape = TensorShape(32, 16);
66  auto t_input_info = TensorInfo(t_shape, 1, data_type, data_layout);
67  auto t_dst_info = TensorInfo();
68 
69  FloorDescriptor floor_desc{};
70 
71  // Create reference
72  SimpleTensor<float> ref_t_input{ t_shape, data_type, 1, QuantizationInfo(), DataLayout::NHWC };
73 
74  // Fill reference
75  fill<float>(ref_t_input, 0, library.get());
76 
77  auto ref_t_input_nchw = reference::permute(ref_t_input, PermutationVector(1U, 2U, 0U));
78  auto t_dst_shape_nchw = t_shape;
79  permute(t_dst_shape_nchw, PermutationVector(1U, 2U, 0U));
80 
81  auto ref_t_dst_nchw = reference::floor_layer(ref_t_input_nchw);
82  const auto ref_t_dst = reference::permute(ref_t_dst_nchw, PermutationVector(2U, 0U, 1U));
83 
84  CLScheduler::get().default_reinit();
85  const auto cl_compile_ctx = CLKernelLibrary::get().get_compile_context();
86  OperatorGraph op_graph;
87 
88  const auto op_t_input = add_tensor(op_graph, t_input_info);
89  const auto op_t_dst = add_tensor(op_graph, t_dst_info);
90 
91  add_op_floor(op_graph, floor_desc, op_t_input, op_t_dst);
92 
93  const ClWorkloadContext workload_ctx{ GpuInfo{ CLScheduler::get().target() } };
94  ClWorkload workload;
95  build(workload, op_graph, workload_ctx);
96 
97  ClCompositeOperator op;
98  op.configure(cl_compile_ctx, workload);
99 
100  // Construct tensors
101  CLTensor t_input{};
102  CLTensor t_dst{};
103 
104  // Init tensors
105  t_input.allocator()->init(t_input_info);
106  t_dst.allocator()->init(t_dst_info);
107 
108  // Allocate and fill tensors
109  t_input.allocator()->allocate();
110  t_dst.allocator()->allocate();
111  fill<float>(CLAccessor(t_input), 0, library.get());
112  // "Pack" tensors
113  OpTensorBinding bp_tensors({ { op_t_input, &t_input },
114  { op_t_dst, &t_dst }
115  });
116 
117  // Populate prepare and run pack-maps (including allocating aux tensors)
118  ClAuxTensorData aux_tensor_data{};
119  TensorPackMap prepare_pack_map{};
120  TensorPackMap run_pack_map{};
121  bind_tensors(aux_tensor_data, prepare_pack_map, run_pack_map, workload, bp_tensors);
122 
123  op.prepare(prepare_pack_map);
124  op.run(run_pack_map);
125  RelativeTolerance<float> tolerance_f32(0.001f); /**< Tolerance value for comparing reference's output against implementation's output for floating point data types */
126  validate(CLAccessor(t_dst), ref_t_dst_nchw, tolerance_f32);
127 }
