39 template <typename T>
40 SimpleTensor<T> gather(const SimpleTensor<T> &src, const SimpleTensor<uint32_t> &indices, uint32_t actual_axis)
41 {
42  const auto *indices_ptr = static_cast<const uint32_t *>(indices.data());
43  const TensorShape dst_shape = arm_compute::misc::shape_calculator::compute_gather_shape(src.shape(), indices.shape(), actual_axis);
44  SimpleTensor<T> dst(dst_shape, src.data_type());
45 
46  Window win;
47  win.use_tensor_dimensions(dst_shape);
48  if(indices.shape().num_dimensions() == 1u)
49  {
50  execute_window_loop(win, [&](const Coordinates & id)
51  {
52  Coordinates offset;
53  for(unsigned int dim = 0; dim < id.num_dimensions(); ++dim)
54  {
55  if(dim == actual_axis)
56  {
57  offset.set(dim, indices_ptr[id[dim]]);
58  }
59  else
60  {
61  offset.set(dim, id[dim]);
62  }
63  }
64  *reinterpret_cast<T *>(dst(id)) = *reinterpret_cast<const T *>(src(offset));
65  });
66  }
67  else
68  {
69  if(actual_axis == 1)
70  {
71  win.set(Window::DimX, Window::Dimension(0, 1, 1));
72  execute_window_loop(win, [&](const Coordinates & id)
73  {
74  auto *dst_ptr = dst(id);
75  Coordinates index_offset;
76  for(uint32_t k = 0; k < indices.shape().num_dimensions(); ++k)
77  {
78  index_offset.set(k, id[k + 1]);
79  }
80  const uint32_t row = *reinterpret_cast<const uint32_t *>(indices(index_offset));
81  Coordinates src_offset;
82  src_offset.set(0, 0);
83  src_offset.set(1, row);
84  for(uint32_t j = 2; j < src.shape().num_dimensions(); ++j)
85  {
86  src_offset.set(j, id[1 + indices.shape().num_dimensions() + (j - 2)]);
87  }
88  const auto in_ptr_row = src(src_offset);
89  memcpy(dst_ptr, in_ptr_row, src.shape()[0] * src.element_size());
90  });
91  }
92  else
93  {
94  ARM_COMPUTE_ERROR("Not implemented.");
95  }
96  }
97 
98  return dst;
99 }
