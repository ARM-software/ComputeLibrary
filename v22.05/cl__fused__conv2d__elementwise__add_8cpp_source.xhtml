56 #define TICK(clock_name) \

59  auto clock_name##_tock = std::chrono::high_resolution_clock::now(); \
60  measurement_map["\"" #clock_name "\""] = duration_cast<microseconds>(clock_name##_tock - clock_name##_tick);

165  /// @page example_dynamic_fusion_cl_conv2d_elementwise_add
166  /// @subsection add_conv2d Add the first operator (root operator) Conv2d
167  /// The first operator to be added to the graph is called the "root operator" of the entire graph.
168  /// @note As of now, operators need to be inserted according to their dependency order. This is because output tensor auto-initialization occurs during construction time.
169  /// Later this might be changed to allow out-of-order insertion.
170 
171  /// Before we insert the operator, we need to initialize the required TensorInfo objects.
172  /// We can choose not to initialize an output TensorInfo; if so, they will be auto-initialized during the construction of the OperatorGraph
173  /// The "t_acc_info" is the TensorInfo of the accumulator tensor, which is the output tensor of our first operator conv2d
174  /// @snippet dynamic_fusion/cl_fused_conv2d_elementwise_add.cpp Initialize Conv2d TensorInfo

176  auto t_input_info = TensorInfo(t_input_shape, 1, data_type, data_layout);
177  auto t_weight_info = TensorInfo(t_weight_shape, 1, data_type, data_layout);
178  auto t_bias_info = TensorInfo(t_bias_shape, 1, data_type, data_layout);
179  auto t_acc_info = TensorInfo();

182  /// @page example_dynamic_fusion_cl_conv2d_elementwise_add
183  /// Next we associate the TensorInfo with the OpTensor s created in the op_graph.
184  /// @note The associated TensorInfo objects must be in scope and remain valid until the ClWorkload building is completed

188  /// @snippet dynamic_fusion/cl_fused_conv2d_elementwise_add.cpp Add OpTensors

191  const auto op_t_weight = add_tensor(op_graph, t_weight_info);
192  const auto op_t_bias = add_tensor(op_graph, t_bias_info);

196  /// @page example_dynamic_fusion_cl_conv2d_elementwise_add
197  /// Finally we add the Conv2d operator to op_graph. The Conv2dDescriptor contains all the TOSA-compliant attribute parameters
198  /// The add_op... group of functions accept the OpTensors created by the add_tensor function, and return an Operator handle.
199  /// This handle can be used to further query and modify the operator inside the OperatorGraph after its creation

210  /// @page example_dynamic_fusion_cl_conv2d_elementwise_add
211  /// @subsection add_elementwise_add Add the second operator Elementwise Add
212  /// This is similar to adding the first operator to op_graph, except that we link the two operators together by their common tensor,
213  /// namely the accumulator tensor op_t_acc, which is the output of conv2d and the input (lhs) of the addition
214  /// @note At the moment, it is recommended to always declare a separate TensorInfo (even if empty) for each OpTensor.
215  /// For example, here op_t_dst could be associated with op_t_acc info as they are the same,
216  /// but we still recommend creating a separate object.
217 
218  /// @snippet dynamic_fusion/cl_fused_conv2d_elementwise_add.cpp Add Elementwise Add Operator
219  // [Add Elementwise Add Operator]
220  auto t_l1_addend_info = TensorInfo(t_l1_addend_shape, 1, data_type, data_layout);
221  auto t_dst_info = TensorInfo();
222  const auto op_t_l1_addend = add_tensor(op_graph, t_l1_addend_info);
223  const auto op_t_dst = add_tensor(op_graph, t_dst_info);
224  AddDescriptor add_desc{};
225  add_op_elementwise_add(op_graph, add_desc, op_t_acc, op_t_l1_addend, op_t_dst);
