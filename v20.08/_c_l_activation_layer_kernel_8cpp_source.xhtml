44 Status validate_arguments(const ITensorInfo *input, const ITensorInfo *output, const ActivationLayerInfo &act_info)
45 {
46  ARM_COMPUTE_RETURN_ERROR_ON_F16_UNSUPPORTED(input);
47  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(input, 1, DataType::QASYMM8, DataType::QASYMM8_SIGNED, DataType::QSYMM16, DataType::F16, DataType::F32);
48 
49  static std::set<ActivationLayerInfo::ActivationFunction> quantized_supported_activations =
50  {
51  ActivationLayerInfo::ActivationFunction::RELU,
52  ActivationLayerInfo::ActivationFunction::LU_BOUNDED_RELU,
53  ActivationLayerInfo::ActivationFunction::BOUNDED_RELU,
54  ActivationLayerInfo::ActivationFunction::LOGISTIC,
55  ActivationLayerInfo::ActivationFunction::TANH,
56  ActivationLayerInfo::ActivationFunction::HARD_SWISH
57  };
58  const DataType data_type = input->data_type();
59  const QuantizationInfo &oq_info = (output != nullptr) ? output->quantization_info() : input->quantization_info();
60  const ActivationLayerInfo::ActivationFunction f_act = act_info.activation();
61 
62  ARM_COMPUTE_RETURN_ERROR_ON_MSG(is_data_type_quantized(data_type) && (quantized_supported_activations.count(f_act) == 0),
63  "For Quantized data type only tanh, logistic, relu and lower/upper bounded relu are supported");
64 
65  ARM_COMPUTE_RETURN_ERROR_ON(data_type == DataType::QASYMM8 && (f_act == ActivationLayerInfo::ActivationFunction::TANH) && (oq_info != QuantizationInfo(1.f / 128.f, 128)));
66  ARM_COMPUTE_RETURN_ERROR_ON(data_type == DataType::QASYMM8 && (f_act == ActivationLayerInfo::ActivationFunction::LOGISTIC) && (oq_info != QuantizationInfo(1.f / 256.f, 0)));
67 
68  ARM_COMPUTE_RETURN_ERROR_ON(is_data_type_quantized_symmetric(data_type) && (f_act == ActivationLayerInfo::ActivationFunction::TANH) && (oq_info != QuantizationInfo(1.f / 32768.f, 0)));
69  ARM_COMPUTE_RETURN_ERROR_ON(is_data_type_quantized_symmetric(data_type) && (f_act == ActivationLayerInfo::ActivationFunction::LOGISTIC) && (oq_info != QuantizationInfo(1.f / 32768.f, 0)));
70 
71  ARM_COMPUTE_RETURN_ERROR_ON(data_type == DataType::QASYMM8_SIGNED && (f_act == ActivationLayerInfo::ActivationFunction::TANH) && (oq_info != QuantizationInfo(1.f / 128.f, 0)));
72  ARM_COMPUTE_RETURN_ERROR_ON(data_type == DataType::QASYMM8_SIGNED && (f_act == ActivationLayerInfo::ActivationFunction::LOGISTIC) && (oq_info != QuantizationInfo(1.f / 256.f, -128)));
73 
74  // Checks performed when output is configured
75  if((output != nullptr) && (output->total_size() != 0))
76  {
77  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_SHAPES(input, output);
78  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input, output);
79  }
80 
81  return Status{};
82 }

84 std::pair<Status, Window> validate_and_configure_window(ITensorInfo *input, ITensorInfo *output)
85 {
86  if(output != nullptr)
87  {
88  ARM_COMPUTE_ERROR_ON_NULLPTR(input, output);
89  // Output auto inizialitation if not yet initialized
90  auto_init_if_empty(*output, *input);
91  }
92 
93  const unsigned int num_elems_processed_per_iteration = 16 / input->element_size();
94 
95  Window win = calculate_max_window(*input, Steps(num_elems_processed_per_iteration));
96  bool window_changed = false;
97 
98  if(output != nullptr)
99  {
100  AccessWindowHorizontal input_access(input, 0, num_elems_processed_per_iteration);
101  AccessWindowHorizontal output_access(output, 0, num_elems_processed_per_iteration);
102  window_changed = update_window_and_padding(win, input_access, output_access);
103  output_access.set_valid_region(win, input->valid_region());
104  }
105  else
106  {
107  window_changed = update_window_and_padding(win,
108  AccessWindowHorizontal(input, 0, num_elems_processed_per_iteration));
109  }
110 
111  Status err = (window_changed) ? ARM_COMPUTE_CREATE_ERROR(ErrorCode::RUNTIME_ERROR, "Insufficient Padding!") : Status{};
112  return std::make_pair(err, win);
113 }

121 void CLActivationLayerKernel::configure(const CLCompileContext &compile_context, ITensorInfo *input, ITensorInfo *output, ActivationLayerInfo act_info)
122 {
123  ARM_COMPUTE_ERROR_ON_NULLPTR(input);
124 
125  _run_in_place = (output == nullptr) || (output == input);
126 
127  if(output != nullptr)
128  {
129  // Output auto inizialitation if not yet initialized
130  auto_init_if_empty(*output, *input->clone());
131  }
132 
133  ARM_COMPUTE_ERROR_THROW_ON(validate_arguments(input, (output != nullptr) ? output : nullptr, act_info));
134 
135  const unsigned int num_elems_processed_per_iteration = 16 / input->element_size();
136  const DataType dt = input->data_type();
137  float a_const = act_info.a();
138  float b_const = act_info.b();
139 
140  const ActivationLayerInfo::ActivationFunction f_act = act_info.activation();
141  const bool is_quantized = is_data_type_quantized(dt);
142  const bool perform_activation_in_float =
143  (f_act == ActivationLayerInfo::ActivationFunction::LOGISTIC) || (f_act == ActivationLayerInfo::ActivationFunction::TANH) || (f_act == ActivationLayerInfo::ActivationFunction::HARD_SWISH);
144 
145  // Set build options
146  CLBuildOptions build_opts;
147  build_opts.add_option_if(perform_activation_in_float, "-DFLOAT_DOMAIN");
148  build_opts.add_option_if(_run_in_place, "-DIN_PLACE");
149  build_opts.add_option(("-DACT=" + lower_string(string_from_activation_func(f_act))));
150  build_opts.add_option(("-DDATA_TYPE=" + get_cl_type_from_data_type(dt)));
151  build_opts.add_option(("-DVEC_SIZE=" + support::cpp11::to_string(num_elems_processed_per_iteration)));
152 
153  std::string kernel_name = std::string("activation_layer");
154 
155  // Set quantization info build options
156  if(is_quantized)
157  {
158  const UniformQuantizationInfo iq_info = input->quantization_info().uniform();
159 
160  if(!perform_activation_in_float)
161  {
162  int a_const_int = 0;
163  int b_const_int = 0;
164 
165  // Create quantized version of constants a, b if needed
166  switch(dt)
167  {
168  case DataType::QASYMM8:
169  {
170  a_const_int = quantize_qasymm8(a_const, iq_info);
171  b_const_int = quantize_qasymm8(b_const, iq_info);
172  }
173  break;
174  case DataType::QASYMM8_SIGNED:
175  {
176  a_const_int = quantize_qasymm8_signed(a_const, iq_info);
177  b_const_int = quantize_qasymm8_signed(b_const, iq_info);
178  }
179  break;
180  case DataType::QSYMM16:
181  {
182  a_const_int = quantize_qsymm16(a_const, iq_info);
183  b_const_int = quantize_qsymm16(b_const, iq_info);
184  }
185  break;
186  default:
187  break;
188  }
189  build_opts.add_option(("-DA_VAL=" + support::cpp11::to_string(a_const_int)));
190  build_opts.add_option(("-DB_VAL=" + support::cpp11::to_string(b_const_int)));
191  }
192  else
193  {
194  build_opts.add_option(("-DA_VAL=" + float_to_string_with_full_precision(a_const)));
195  build_opts.add_option(("-DB_VAL=" + float_to_string_with_full_precision(b_const)));
196  }
197 
198  // Quantized value of 0 corresponds to the offset o1
199  build_opts.add_option(("-DCONST_0=" + (is_data_type_quantized_asymmetric(dt) ? support::cpp11::to_string(iq_info.offset) : "0")));
200  build_opts.add_option(("-DS1_VAL=" + float_to_string_with_full_precision(iq_info.scale)));
201  build_opts.add_option_if(is_data_type_quantized_asymmetric(dt), "-DO1_VAL=" + support::cpp11::to_string(iq_info.offset));
202 
203  // Set correct kernel name
204  kernel_name += perform_activation_in_float ? std::string("_quant_f32") : std::string("_quant");
205 
206  // Set scale and offset of the input and output if they have different quantization info
207  if(output != nullptr)
208  {
209  const UniformQuantizationInfo oq_info = output->quantization_info().uniform();
210 
211  if(iq_info != oq_info)
212  {
213  build_opts.add_option(("-DS2_VAL=" + float_to_string_with_full_precision(oq_info.scale)));
214  build_opts.add_option_if(is_data_type_quantized_asymmetric(dt), "-DO2_VAL=" + support::cpp11::to_string(oq_info.offset));
215  }
216  }
217  }
218  else
219  {
220  // Set A, B constants in build options for float types
221  build_opts.add_option(("-DA_VAL=" + float_to_string_with_full_precision(a_const)));
222  build_opts.add_option(("-DB_VAL=" + float_to_string_with_full_precision(b_const)));
223  }
224 
225  // Create kernel
226  _kernel = create_kernel(compile_context, kernel_name, build_opts.options());
227 
228  // Configure kernel window
229  auto win_config = validate_and_configure_window(input, (_run_in_place) ? nullptr : output);
230  ARM_COMPUTE_ERROR_THROW_ON(win_config.first);
231  ICLKernel::configure_internal(win_config.second);
232 
233  // Set config_id for enabling LWS tuning
234  _config_id = "activation_layer_";
235  _config_id += lower_string(string_from_data_type(dt));
236  _config_id += "_";
237  _config_id += support::cpp11::to_string(input->dimension(0));
238  _config_id += "_";
239  _config_id += support::cpp11::to_string(input->dimension(1));
240 }

242 Status CLActivationLayerKernel::validate(const ITensorInfo *input, const ITensorInfo *output, const ActivationLayerInfo &act_info)
243 {
244  const bool run_in_place = (output == nullptr) || (output == input);
245  ARM_COMPUTE_RETURN_ON_ERROR(validate_arguments(input, output, act_info));
246  ARM_COMPUTE_RETURN_ON_ERROR(validate_and_configure_window(input->clone().get(), (run_in_place) ? nullptr : output->clone().get()).first);
247 
248  return Status{};

251 void CLActivationLayerKernel::run_op(ITensorPack &tensors, const Window &window, cl::CommandQueue &queue)
252 {
253  ARM_COMPUTE_ERROR_ON_UNCONFIGURED_KERNEL(this);
254  ARM_COMPUTE_ERROR_ON_INVALID_SUBWINDOW(ICLKernel::window(), window);
255 
256  const auto src = utils::cast::polymorphic_downcast<const ICLTensor *>(tensors.get_const_tensor(TensorType::ACL_SRC));
257  auto dst = utils::cast::polymorphic_downcast<ICLTensor *>(tensors.get_tensor(TensorType::ACL_DST));
258  ARM_COMPUTE_ERROR_ON(_run_in_place && src != dst);
259 
260  Window collapsed = window.collapse_if_possible(ICLKernel::window(), Window::DimZ);
261  Window slice = collapsed.first_slice_window_3D();
262 
263  do
264  {
265  unsigned int idx = 0;
266  add_3D_tensor_argument(idx, src, slice);
267  if(!_run_in_place)
268  {
269  add_3D_tensor_argument(idx, dst, slice);
270  }
271  enqueue(queue, *this, slice, lws_hint());
272  }
273  while(collapsed.slide_window_slice_3D(slice));
274 }
