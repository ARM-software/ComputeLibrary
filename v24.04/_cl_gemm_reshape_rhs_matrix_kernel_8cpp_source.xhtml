51 Status validate_arguments(const ITensorInfo *src, const ITensorInfo *dst, const GEMMRHSMatrixInfo &rhs_info)
52 {
53  ARM_COMPUTE_RETURN_ERROR_ON(rhs_info.n0 == 0);
54  ARM_COMPUTE_RETURN_ERROR_ON(rhs_info.k0 == 0);
55  ARM_COMPUTE_RETURN_ERROR_ON(rhs_info.h0 == 0);
56  ARM_COMPUTE_RETURN_ERROR_ON_MSG(((rhs_info.n0 & (rhs_info.n0 - 1)) && rhs_info.n0 != 3),
57  "Only 2,3,4,8,16 are supported for n0");
58  ARM_COMPUTE_RETURN_ERROR_ON_MSG(((rhs_info.k0 & (rhs_info.k0 - 1)) && (rhs_info.k0 != 1) && (rhs_info.k0 != 3)),
59  "Only 1,2,3,4,8,16 are supported for k0");
60  ARM_COMPUTE_RETURN_ERROR_ON(rhs_info.n0 > 16);
61  ARM_COMPUTE_RETURN_ERROR_ON(rhs_info.k0 > 16);
62  ARM_COMPUTE_RETURN_ERROR_ON((rhs_info.k0 == 1) && (rhs_info.transpose));
63 
64  ARM_COMPUTE_RETURN_ERROR_ON_F16_UNSUPPORTED(src);
65  ARM_COMPUTE_RETURN_ERROR_ON(src->data_type() == DataType::UNKNOWN);
66 
67  if (rhs_info.export_to_cl_image)
68  {
69  const TensorInfo tensor_reshaped_info(misc::shape_calculator::compute_rhs_reshaped_shape(*src, rhs_info), 1,
70  src->data_type());
71  ARM_COMPUTE_RETURN_ON_ERROR(gemm::validate_image2d_support_on_rhs(tensor_reshaped_info, rhs_info));
72  }
73 
74  if (dst->total_size() != 0)
75  {
76  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DIMENSIONS(
77  dst->tensor_shape(), misc::shape_calculator::compute_rhs_reshaped_shape(*src, rhs_info));
78  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(src, dst);
79  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_QUANTIZATION_INFO(src, dst);
80  }
81 
82  return Status{};
83 }

85 std::pair<Status, Window>
86 validate_and_configure_window(ITensorInfo *src, ITensorInfo *dst, const GEMMRHSMatrixInfo &rhs_info)
87 {
88  const unsigned int num_elems_processed_per_iteration_x = rhs_info.n0;
89  const unsigned int num_elems_processed_per_iteration_y = rhs_info.k0;
90  bool window_changed = false;
91 
92  // dst auto initialization if not yet initialized
93  auto_init_if_empty(
94  *dst, src->clone()->set_tensor_shape(misc::shape_calculator::compute_rhs_reshaped_shape(*src, rhs_info)));
95 
96  // Configure window
97  Window win =
98  calculate_max_window(*src, Steps(num_elems_processed_per_iteration_x, num_elems_processed_per_iteration_y));
99 
100  AccessWindowRectangle src_access(src, 0, 0, num_elems_processed_per_iteration_x,
101  num_elems_processed_per_iteration_y);
102 
103  window_changed = update_window_and_padding(win, src_access);
104 
105  if (rhs_info.export_to_cl_image)
106  {
107  gemm::update_padding_for_cl_image(dst);
108  }
109 
110  // Collapse along the Z direction
111  // This collapse needs to be here in order to tune the Z dimension of LWS
112  Window collapsed = win.collapse(win, Window::DimZ);
113 
114  Status err =
115  (window_changed) ? ARM_COMPUTE_CREATE_ERROR(ErrorCode::RUNTIME_ERROR, "Insufficient Padding!") : Status{};
116  return std::make_pair(err, collapsed);
117 }

125 void ClGemmReshapeRhsMatrixKernel::configure(const CLCompileContext &compile_context,
126  ITensorInfo *src,
127  ITensorInfo *dst,
128  const GEMMRHSMatrixInfo &rhs_info)
129 {
130  ARM_COMPUTE_ERROR_ON_NULLPTR(src, dst);
131 
132  // Perform validate step
133  ARM_COMPUTE_ERROR_THROW_ON(validate_arguments(src, dst, rhs_info));
134 
135  // Create build options
136  CLBuildOptions build_opts;
137  build_opts.add_option("-DN0=" + support::cpp11::to_string(rhs_info.n0));
138  build_opts.add_option("-DK0=" + support::cpp11::to_string(rhs_info.k0));
139  build_opts.add_option_if(rhs_info.interleave, "-DINTERLEAVE");
140  build_opts.add_option_if(rhs_info.transpose, "-DRESHAPE_RHS_T");
141  build_opts.add_option_if(!rhs_info.transpose, "-DRESHAPE_RHS_NT");
142  build_opts.add_option("-DDATA_TYPE=" + get_cl_unsigned_type_from_element_size(src->element_size()));
143 
144  std::string kernel_name("gemm_reshape_rhs_matrix_");
145  kernel_name += rhs_info.transpose ? "t" : "nt";
146 
147  // Create kernel
148  _kernel = create_kernel(compile_context, kernel_name, build_opts.options());
149 
150  // Configure kernel window
151  auto win_config = validate_and_configure_window(src, dst, rhs_info);
152  ARM_COMPUTE_ERROR_THROW_ON(win_config.first);
153  ICLKernel::configure_internal(win_config.second);
154 
155  unsigned int idx = 2 * num_arguments_per_3d_tensor_nhw();
156  _kernel.setArg<cl_int>(idx++, rhs_info.h0);
157 }

169 void ClGemmReshapeRhsMatrixKernel::run_op(ITensorPack &tensors, const Window &window, cl::CommandQueue &queue)
170 {
171  ARM_COMPUTE_ERROR_ON_UNCONFIGURED_KERNEL(this);
172  ARM_COMPUTE_ERROR_ON_INVALID_SUBWINDOW(ICLKernel::window(), window);
173 
174  const auto src =
175  utils::cast::polymorphic_downcast<const ICLTensor *>(tensors.get_const_tensor(TensorType::ACL_SRC));
176  auto dst = utils::cast::polymorphic_downcast<ICLTensor *>(tensors.get_tensor(TensorType::ACL_DST));
177 
178  ARM_COMPUTE_ERROR_ON_NULLPTR(src, dst);
179 
180  Window slice = window.first_slice_window_3D();
181 
182  do
183  {
184  unsigned int idx = 0;
185  add_3d_tensor_nhw_argument(idx, src);
186  add_3d_tensor_nhw_argument(idx, dst);
187  enqueue(queue, *this, slice, lws_hint());
188  } while (window.slide_window_slice_3D(slice));
189 }
