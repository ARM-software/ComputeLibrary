161 /** Configuration test */
162 bool validate_configuration(unsigned int m_value, unsigned int n_value, unsigned int k_value, unsigned int b_value,
163  unsigned int m0_value, unsigned int n0_value, unsigned int k0_value, unsigned int h0_value,
164  bool i_value_rhs, bool t_value_rhs, bool export_to_cl_image, bool broadcast_bias, bool input_as_3d, unsigned int depth_output_gemm3d, const ActivationLayerInfo &act_info,
165  DataType dt_input0, DataType dt_input1, DataType dt_input2, DataType dt_output, float alpha, float beta)
166 {
167  const unsigned int M = m_value;
168  const unsigned int N = n_value;
169  const unsigned int K = k_value;
170 
171  GEMMLHSMatrixInfo lhs_info;
172  lhs_info.m0 = m0_value;
173  lhs_info.k0 = k0_value;
174 
175  GEMMRHSMatrixInfo rhs_info;
176  rhs_info.n0 = n0_value;
177  rhs_info.k0 = k0_value;
178  rhs_info.h0 = h0_value;
179  rhs_info.interleave = i_value_rhs;
180  rhs_info.transpose = t_value_rhs;
181  rhs_info.export_to_cl_image = export_to_cl_image;
182 
183  GEMMKernelInfo kernel_info;
184  kernel_info.m = M;
185  kernel_info.n = N;
186  kernel_info.k = K;
187  kernel_info.depth_output_gemm3d = depth_output_gemm3d;
188  kernel_info.reinterpret_input_as_3d = input_as_3d;
189  kernel_info.broadcast_bias = broadcast_bias;
190  kernel_info.activation_info = act_info;
191 
192  const TensorShape lhs_shape(K, M, b_value);
193  const TensorShape rhs_shape(N, K, b_value);
194  const TensorShape rhs_shape_reshaped = compute_rhs_reshaped_shape(TensorInfo(rhs_shape, 1, dt_input1),
195  rhs_info);
196 
197  const TensorShape dst_shape = compute_mm_shape(TensorInfo(lhs_shape, 1, dt_input0),
198  TensorInfo(rhs_shape_reshaped, 1, dt_input1),
199  kernel_info);
200 
201  const TensorShape bias_shape(N,
202  M, // Correct calculation should be: broadcast_bias? 1 : M, it's wrong here on purpose just for validation test
203  broadcast_bias? 1 : b_value);
204 
205  // Create tensor info
206  TensorInfo lhs = TensorInfo(lhs_shape, 1, dt_input0);
207  TensorInfo rhs_reshaped = TensorInfo(rhs_shape_reshaped, 1, dt_input1);
208  TensorInfo bias = TensorInfo(bias_shape, 1, dt_input2);
209  TensorInfo dst = TensorInfo(dst_shape, 1, dt_output);
210 
211  // Create and configure function
212  CLGEMMMatrixMultiplyReshapedOnlyRHS gemm;
213  return bool(gemm.validate(&lhs, &rhs_reshaped, &bias, &dst, alpha, beta, lhs_info, rhs_info, kernel_info));
214 }

223  * A series of validation tests on configurations which according to the API specification
224  * the function should fail against.

226  * Checks performed in order:
227  * - Mismachting data type: input1, input2 and output need to have same data type as input0. Support data type: F32/F16.
228  * - Unsupported M0: MO can only be 1,2,3,4,5,6,7,8
229  * - Unsupported N0: NO can only be 2,3,4,8,16
230  * - Unsupported K0: KO can only be 2,3,4,8,16
231  * - Unsupported bias addition: bias broadcast mode is 0 if the input or output has to be reinterpreted as 3D
232  * - Incorrect bias diemension when bias broadcast mode is 1 and beta is not 0.0f, should be (n, 1), not (n, m)
233  * - Incorrect input0 dimension when input is reinterpreted as 3D: input0->dimension(1) * input0->dimension(2) != m
234  * - Correct support for creating an OpenCL image object from buffer
235  * - Incorrect support for creating an OpenCL image object from buffer. N0 is 2 but it can only be 4,8 and 16
236  * - Correct F16 support for creating an OpenCL image object from buffer.
