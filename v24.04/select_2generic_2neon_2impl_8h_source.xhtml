40 template <typename ScalarType, typename VectorType>
41 void select_op(const ITensor *cond,
42  const ITensor *in1,
43  const ITensor *in2,
44  ITensor *out,
45  const Window &window,
46  const int window_step_x,
47  const int window_start_x,
48  const int window_end_x,
49  const int limit,
50  VectorType (*condition_conversion)(const uint8_t *))
51 {
52  Window win = window;
53  win.set(Window::DimX, Window::Dimension(0, 1, 1));
54 
55  Iterator condition(cond, win);
56  Iterator input1(in1, win);
57  Iterator input2(in2, win);
58  Iterator output(out, win);
59 
60  execute_window_loop(
61  win,
62  [&](const Coordinates &)
63  {
64  auto output_ptr = reinterpret_cast<ScalarType *>(output.ptr());
65  const auto condition_ptr = reinterpret_cast<const uint8_t *>(condition.ptr());
66  const auto input1_ptr = reinterpret_cast<const ScalarType *>(input1.ptr());
67  const auto input2_ptr = reinterpret_cast<const ScalarType *>(input2.ptr());
68 
69  int x = window_start_x;
70  for (; x <= limit; x += window_step_x)
71  {
72  const auto c = (*condition_conversion)(condition_ptr + x);
73  const auto a = wrapper::vloadq(input1_ptr + x);
74  const auto b = wrapper::vloadq(input2_ptr + x);
75  wrapper::vstore(output_ptr + x, wrapper::vbsl(c, a, b));
76  }
77  for (; x < window_end_x; ++x)
78  {
79  const auto c = *(condition_ptr + x);
80  const auto a = *(input1_ptr + x);
81  const auto b = *(input2_ptr + x);
82  *(output_ptr + x) = static_cast<bool>(c) ? a : b;
83  }
84  },
85  condition, input1, input2, output);
86 }

139 template <typename ScalarType>
140 void select_op_not_same_rank(
141  const ITensor *cond, const ITensor *in1, const ITensor *in2, ITensor *out, const Window &window)
142 {
143  ARM_COMPUTE_UNUSED(window);
144 
145  auto output_ptr = reinterpret_cast<ScalarType *>(out->buffer());
146  const auto condition_ptr = reinterpret_cast<const uint8_t *>(cond->buffer());
147  const auto input1_ptr = reinterpret_cast<const ScalarType *>(in1->buffer());
148  const auto input2_ptr = reinterpret_cast<const ScalarType *>(in2->buffer());
149 
150  const int outer_size = cond->info()->total_size() / cond->info()->element_size();
151  const int inner_size = (in1->info()->total_size() / in1->info()->element_size()) / outer_size;
152  int offset = 0;
153  const int step = 16 / in1->info()->element_size();
154 
155  for (int i = 0; i < outer_size; ++i)
156  {
157  int x = offset;
158  const auto input_ptr = static_cast<bool>(*(condition_ptr + i)) ? input1_ptr : input2_ptr;
159  for (; x <= offset + inner_size - step; x += step)
160  {
161  wrapper::vstore(output_ptr + x, wrapper::vloadq(input_ptr + x));
162  }
163  if (x <= offset + inner_size - (step / 2))
164  {
165  wrapper::vstore(output_ptr + x, wrapper::vload(input_ptr + x));
166  x += step / 2;
167  }
168  for (; x < offset + inner_size; ++x)
169  {
170  *(output_ptr + x) = *(input_ptr + x);
171  }
172  offset += inner_size;
173  }
174 }
