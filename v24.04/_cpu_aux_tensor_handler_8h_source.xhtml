38 /** Tensor handler to wrap and handle tensor allocations on workspace buffers
39  *
40  * @note Important: Despite the impression given by its name, the handler owns, rather than merely points to, the
41  * underlying tensor memory.
42  *
43  * @note About memory handling using bypass_* flags
44  * The bypass_alloc / bypass_import flags are meant to skip the expensive auxiliary tensor memory allocations or
45  * imports that are not needed during runtime, e.g. when the handler is not used at all in some branch of execution.
46  *
47  * If not handled correctly, these two flags can lead to performance issues (not bypass when needed to), or memory
48  * bugs (bypass when should not to).
49  *
50  * Make sure:
51  *
52  * 1. The aux tensor handlers must always be declared at the root level, or the same level as the run/prepare
53  * methods that potentially use them.
54  *
55  * Once the handler is destroyed (e.g. when going out of scope), the memory it owns (returned by the get()
56  * method) will also be destroyed.
57  *
58  * Thus it's important to ensure the handler is always in-scope when it is being used by a operator / kernel.
59  *
60  * 2. The handler's bypass_alloc and bypass_import flags should always be inverse of whether the handler is used in
61  * its surrounding scope by run/prepare. (This usually means being added to some tensor pack)
62  *
63  * This ensures we only bypass if and only if the aux tensor is not used by the op / kernel later.
64  *
65  *
66  * So the general usage pattern goes like this:
67  *
68  * bool use_aux_tensor = some_condition_about_when_to_use_the_aux_tensor
69  *
70  * CpuAuxTensorHandler aux_handler {..., !use_aux_tensor || bypass_alloc / bypass_import ||};
71  *
72  * if (use_aux_tensor)
73  * {
74  * tensor_pack.add_tensor(aux_handler.get());
75  * }
76  * op.run(tensor_pack);
77  */
78 class CpuAuxTensorHandler
79 {
80 public:
81  /** Create a temporary tensor handle, by either important an existing tensor from a tensor pack, or allocating a
82  * new one.
83  *
84  * @param[in] slot_id Slot id of the tensor to be retrieved in the tensor pack
85  * If no such tensor exists in the tensor pack, a new tensor will be allocated.
86  * @param[in] info Tensor info containing requested size of the new tensor.
87  * If requested size is larger than the tensor retrieved from the tensor pack,
88  * a new tensor will be allocated.
89  * @param[in,out] pack Tensor pack to retrieve the old tensor. When @p pack_inject is true, the new
90  * tensor will also be added here.
91  * @param[in] pack_inject In case of a newly allocated tensor, whether to add this tensor back to the
92  * @p pack
93  * @param[in] bypass_alloc Bypass allocation in case of a new tensor
94  * This is to prevent unnecessary memory operations when the handler object is not
95  * used
96  * @param[in] bypass_import Bypass importation in case of a retrieved tensor
97  * This is to prevent unnecessary memory operations when the handler object is not
98  * used
99  */
100  CpuAuxTensorHandler(int slot_id,
101  TensorInfo &info,
102  ITensorPack &pack,
103  bool pack_inject = false,
104  bool bypass_alloc = false,
105  bool bypass_import = false)
106  : _tensor()
107  {
108  if (info.total_size() == 0)
109  {
110  return;
111  }
112  _tensor.allocator()->soft_init(info);
113 
114  ITensor *packed_tensor = utils::cast::polymorphic_downcast<ITensor *>(pack.get_tensor(slot_id));
115  if ((packed_tensor == nullptr) || (info.total_size() > packed_tensor->info()->total_size()))
116  {
117  if (!bypass_alloc)
118  {
119  _tensor.allocator()->allocate();
120  ARM_COMPUTE_LOG_INFO_WITH_FUNCNAME_ACL("Allocating auxiliary tensor");
121  }
122 
123  if (pack_inject)
124  {
125  pack.add_tensor(slot_id, &_tensor);
126  _injected_tensor_pack = &pack;
127  _injected_slot_id = slot_id;
128  }
129  }
130  else
131  {
132  if (!bypass_import)
133  {
134  _tensor.allocator()->import_memory(packed_tensor->buffer());
135  }
136  }
137  }
138 
139  /** Create a temporary handle to the original tensor with a new @ref TensorInfo
140  * This is useful if we want to change a tensor's tensor info at run time without modifying the original tensor
141  *
142  * @param[in] info New tensor info to "assign" to @p tensor
143  * @param[in] tensor Tensor to be assigned a new @ref TensorInfo
144  * @param[in] bypass_import Bypass importing @p tensor's memory into the handler.
145  * This is to prevent unnecessary memory operations when the handler object is not used
146  */
147  CpuAuxTensorHandler(TensorInfo &info, const ITensor &tensor, bool bypass_import = false) : _tensor()
148  {
149  _tensor.allocator()->soft_init(info);
150  if (!bypass_import)
151  {
152  ARM_COMPUTE_ERROR_ON(tensor.info() == nullptr);
153  if (info.total_size() <= tensor.info()->total_size())
154  {
155  _tensor.allocator()->import_memory(tensor.buffer());
156  }
157  }
158  }
159 
160  CpuAuxTensorHandler(const CpuAuxTensorHandler &) = delete;
161  CpuAuxTensorHandler &operator=(const CpuAuxTensorHandler) = delete;
162 
163  ~CpuAuxTensorHandler()
164  {
165  if (_injected_tensor_pack)
166  {
167  _injected_tensor_pack->remove_tensor(_injected_slot_id);
168  }
169  }
170 
171  ITensor *get()
172  {
173  return &_tensor;
174  }
175 
176  ITensor *operator()()
177  {
178  return &_tensor;
179  }
180 
181 private:
182  Tensor _tensor{};
183  ITensorPack *_injected_tensor_pack{nullptr};
184  int _injected_slot_id{TensorType::ACL_UNKNOWN};
185 };
