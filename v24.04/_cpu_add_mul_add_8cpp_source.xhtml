38 void CpuAddMulAdd::configure(const ITensorInfo *input1,
39  const ITensorInfo *input2,
40  const ITensorInfo *bn_mul,
41  const ITensorInfo *bn_add,
42  ITensorInfo *add_output,
43  ITensorInfo *final_output,
44  ConvertPolicy policy,
45  const ActivationLayerInfo &act_info)
46 {
47  ARM_COMPUTE_LOG_PARAMS(input1, input2, bn_mul, bn_add, add_output, final_output, policy, act_info);
48 
49  auto k = std::make_unique<kernels::CpuAddMulAddKernel>();
50 
51  const DataType data_type = input1->data_type();
52  if (is_data_type_quantized(data_type))
53  {
54  _dequantize_bn_mul.configure(bn_mul, &_dequantized_bn_mul);
55  _dequantize_bn_add.configure(bn_add, &_dequantized_bn_add);
56 
57  k->configure(input1, input2, &_dequantized_bn_mul, &_dequantized_bn_add, add_output, final_output, policy,
58  act_info);
59 
60  // Save auxilary memory requirements after configuration
61  _aux_mem[DequantizedBnMul] =
62  experimental::MemoryInfo(offset_int_vec(DequantizedBnMul), experimental::MemoryLifetime::Temporary,
63  _dequantized_bn_mul.total_size());
64  _aux_mem[DequantizedBnAdd] =
65  experimental::MemoryInfo(offset_int_vec(DequantizedBnAdd), experimental::MemoryLifetime::Temporary,
66  _dequantized_bn_add.total_size());
67  }
68  else
69  {
70  k->configure(input1, input2, bn_mul, bn_add, add_output, final_output, policy, act_info);
71  }
72 
73  _kernel = std::move(k);
74 }

76 Status CpuAddMulAdd::validate(const ITensorInfo *input1,
77  const ITensorInfo *input2,
78  const ITensorInfo *bn_mul,
79  const ITensorInfo *bn_add,
80  const ITensorInfo *add_output,
81  const ITensorInfo *final_output,
82  ConvertPolicy policy,
83  const ActivationLayerInfo &act_info)
84 {
85  const DataType data_type = input1->data_type();
86  if (is_data_type_quantized(data_type))
87  {
88  TensorInfo dequantized_bn_mul = bn_mul->clone()->set_data_type(DataType::F32);
89  TensorInfo dequantized_bn_add = bn_add->clone()->set_data_type(DataType::F32);
90 
91  ARM_COMPUTE_RETURN_ON_ERROR(CpuDequantize::validate(bn_mul, &dequantized_bn_mul));
92  ARM_COMPUTE_RETURN_ON_ERROR(CpuDequantize::validate(bn_add, &dequantized_bn_add));
93 
94  return kernels::CpuAddMulAddKernel::validate(input1, input2, &dequantized_bn_mul, &dequantized_bn_add,
95  add_output, final_output, policy, act_info);
96  }
97  else
98  {
99  return kernels::CpuAddMulAddKernel::validate(input1, input2, bn_mul, bn_add, add_output, final_output, policy,
100  act_info);
101  }

104 void CpuAddMulAdd::run(ITensorPack &tensors)
105 {
106  const DataType data_type = tensors.get_const_tensor(TensorType::ACL_SRC_0)->info()->data_type();
107 
108  if (is_data_type_quantized(data_type))
109  {
110  const ITensor *bn_mul = tensors.get_const_tensor(TensorType::ACL_SRC_2);
111  const ITensor *bn_add = tensors.get_const_tensor(TensorType::ACL_SRC_3);
112 
113  CpuAuxTensorHandler dequantized_bn_mul_handler(offset_int_vec(DequantizedBnMul), _dequantized_bn_mul, tensors,
114  true);
115  CpuAuxTensorHandler dequantized_bn_add_handler(offset_int_vec(DequantizedBnAdd), _dequantized_bn_add, tensors,
116  true);
117 
118  ITensorPack dequantize_mul_pack = {{TensorType::ACL_SRC_0, bn_mul},
119  {TensorType::ACL_DST_0, dequantized_bn_mul_handler.get()}};
120 
121  ITensorPack dequantize_add_pack = {{TensorType::ACL_SRC_0, bn_add},
122  {TensorType::ACL_DST_0, dequantized_bn_add_handler.get()}};
123 
124  _dequantize_bn_mul.run(dequantize_mul_pack);
125  _dequantize_bn_add.run(dequantize_add_pack);
126 
127  ITensorPack add_mul_add_pack = {
128  {TensorType::ACL_SRC_0, tensors.get_const_tensor(TensorType::ACL_SRC_0)},
129  {TensorType::ACL_SRC_1, tensors.get_const_tensor(TensorType::ACL_SRC_1)},
130  {TensorType::ACL_SRC_2, dequantized_bn_mul_handler.get()},
131  {TensorType::ACL_SRC_3, dequantized_bn_add_handler.get()},
132  {TensorType::ACL_DST_0, tensors.get_tensor(TensorType::ACL_DST_0)},
133  {TensorType::ACL_DST_1, tensors.get_tensor(TensorType::ACL_DST_1)},
134  };
135 
136  NEScheduler::get().schedule_op(_kernel.get(), Window::DimY, _kernel->window(), add_mul_add_pack);
137  }
138  else
139  {
140  NEScheduler::get().schedule_op(_kernel.get(), Window::DimY, _kernel->window(), tensors);
141  }
142 }
