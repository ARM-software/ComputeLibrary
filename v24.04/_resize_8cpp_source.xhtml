104 {

123  { DataType::F32, true },
124  { DataType::F64, false },
125  { DataType::SIZET, false },
126  };
127 
128  for (auto &kv : supported_data_types)
129  {
130  const TensorInfo input_info = TensorInfo{default_input_shape, 1, kv.first, default_data_layout};
131 
132  CLCompileContext cl_compile_ctx = CLKernelLibrary::get().get_compile_context();
133  GpuWorkloadContext context = GpuWorkloadContext{&cl_compile_ctx};
134  GpuWorkloadSketch sketch{&context};
135 
136  const ITensorInfo *sketch_input_info = context.create_tensor_info(input_info);
137 
138  ResizeAttributes attributes;
139  attributes.output_width(default_output_shape[0]); // shape is not important unless it's empty
140  attributes.output_height(default_output_shape[1]);
141 
142  Status status = GpuResize::validate_op(sketch, sketch_input_info, attributes);
143  ARM_COMPUTE_EXPECT(bool(status) == kv.second, framework::LogLevel::ERRORS);
144  }
145 }
