70 void NEDeviceBackend::setup_backend_context(GraphContext &ctx)
71 {
72  // Set number of threads
73  if (ctx.config().num_threads >= 0)
74  {
75  Scheduler::get().set_num_threads(ctx.config().num_threads);
76  }
77 
78  // Create function level memory manager
79  if (ctx.memory_management_ctx(Target::NEON) == nullptr)
80  {
81  MemoryManagerContext mm_ctx;
82  mm_ctx.target = Target::NEON;
83  mm_ctx.intra_mm = create_memory_manager(MemoryManagerAffinity::Offset);
84  mm_ctx.cross_mm = create_memory_manager(MemoryManagerAffinity::Offset);
85  mm_ctx.cross_group = std::make_shared<MemoryGroup>(mm_ctx.cross_mm);
86  mm_ctx.allocator = &_allocator;
87 
88  ctx.insert_memory_management_ctx(std::move(mm_ctx));
89  }
90 
91  // Create function level weights manager
92  if (ctx.weights_management_ctx(Target::NEON) == nullptr)
93  {
94  WeightsManagerContext wm_ctx;
95  wm_ctx.target = Target::NEON;
96  wm_ctx.wm = create_weights_manager();
97 
98  ctx.insert_weights_management_ctx(std::move(wm_ctx));
99  }
100 }

153 std::shared_ptr<arm_compute::IMemoryManager> NEDeviceBackend::create_memory_manager(MemoryManagerAffinity affinity)
154 {
155  std::shared_ptr<ILifetimeManager> lifetime_mgr = nullptr;
156  if (affinity == MemoryManagerAffinity::Buffer)
157  {
158  lifetime_mgr = std::make_shared<BlobLifetimeManager>();
159  }
160  else
161  {
162  lifetime_mgr = std::make_shared<OffsetLifetimeManager>();
163  }
164  auto pool_mgr = std::make_shared<PoolManager>();
165  auto mm = std::make_shared<MemoryManagerOnDemand>(lifetime_mgr, pool_mgr);
166 
167  return mm;
168 }
