151 Status validate_arguments(const ITensorInfo *src,
152  const ITensorInfo *dst,
153  const PoolingLayerInfo &pool_info,
154  const ITensorInfo *indices,
155  Size2D pool_size)
156 {
157  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(src, dst);
158  ARM_COMPUTE_RETURN_ERROR_ON(pool_size.x() == 0);
159  ARM_COMPUTE_RETURN_ERROR_ON(pool_size.y() == 0);
160 
161  int pool_stride_x = 0;
162  int pool_stride_y = 0;
163  int output_width = 0;
164  int output_height = 0;
165  PoolingType pool_type = pool_info.pool_type;
166  const PadStrideInfo pad_stride_info = pool_info.pad_stride_info;
167  const auto data_layout = pool_info.data_layout == DataLayout::UNKNOWN ? src->data_layout() : pool_info.data_layout;
168  const int idx_width = get_data_layout_dimension_index(data_layout, DataLayoutDimension::WIDTH);
169  const int idx_height = get_data_layout_dimension_index(data_layout, DataLayoutDimension::HEIGHT);
170 
171  ARM_COMPUTE_RETURN_ERROR_ON_MSG(
172  (!is_data_type_float(src->data_type())) && (is_pool_region_entirely_outside_input(pool_info)),
173  "Pooling region that is entirely outside input tensor is unsupported for non-float types");
174 
175  std::tie(output_width, output_height) =
176  scaled_dimensions_signed(src->tensor_shape()[idx_width], src->tensor_shape()[idx_height], pool_size.x(),
177  pool_size.y(), pool_info.pad_stride_info);
178  ARM_COMPUTE_RETURN_ERROR_ON_MSG((output_width < 1 || output_height < 1),
179  "Calculated output dimension size is invalid");
180 
181  TensorInfo out_info(TensorInfo(compute_pool_shape(*src, pool_info), 1, dst->data_type()));
182  std::tie(pool_stride_x, pool_stride_y) = pad_stride_info.stride();
183 
184  ARM_COMPUTE_RETURN_ERROR_ON_CPU_F16_UNSUPPORTED(src);
185  if (indices)
186  {
187  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(src, 1, DataType::F32, DataType::F16);
188  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(indices, 1, DataType::U32);
189  ARM_COMPUTE_RETURN_ERROR_ON_MSG(pool_type != PoolingType::MAX,
190  "Pooling indices only supported for MAX pooling method");
191  }
192  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(src, 1, DataType::QASYMM8, DataType::QASYMM8_SIGNED,
193  DataType::F16, DataType::F32);
194  ARM_COMPUTE_RETURN_ERROR_ON(pool_type == PoolingType::L2 && is_data_type_quantized(src->data_type()));
195  ARM_COMPUTE_RETURN_ERROR_ON_MSG(
196  is_data_type_quantized(src->data_type()) && !pool_info.exclude_padding &&
197  (pool_info.pool_type == PoolingType::AVG) && pool_info.pad_stride_info.has_padding() &&
198  (src->data_layout() == DataLayout::NHWC),
199  "exclude_padding equal false is not supported for AVG Pooling with padding on quantized types");
200 
201  if (dst->total_size() != 0)
202  {
203  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(src, dst);
204  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_LAYOUT(src, dst);
205  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_SHAPES(dst, &out_info);
206  if (indices)
207  {
208  ARM_COMPUTE_RETURN_ERROR_ON_MSG(
209  ((pool_size != Size2D(2, 2)) && !pool_info.use_kernel_indices),
210  "Pooling indices returning source tensor coordinates is only supported for pool size 2x2");
211  ARM_COMPUTE_RETURN_ERROR_ON_MSG(pool_info.use_kernel_indices && (src->data_layout() != DataLayout::NHWC),
212  "Pooling kernel indices only supported for NHWC");
213  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_SHAPES(indices, &out_info);
214  }
215  }
216 
217  const auto *uk = CpuPool2dKernel::get_implementation(PoolDataTypeISASelectorData{
218  src->data_type(), src->data_layout(), pool_stride_x, pool_size, CPUInfo::get().get_isa()});
219  ARM_COMPUTE_RETURN_ERROR_ON(uk == nullptr || uk->ukernel == nullptr);
220 
221  return Status{};
222 }

224 std::pair<Status, Window> validate_and_configure_window(ITensorInfo *src,
225  ITensorInfo *dst,
226  ITensorInfo *indices,
227  const PoolingLayerInfo &pool_info,
228  unsigned int &num_elems_processed_per_iteration,
229  int pool_size_x,
230  int pool_size_y)
231 {
232  // dst auto inizialitation if not yet initialized
233  auto_init_if_empty(*dst, src->clone()->set_tensor_shape(compute_pool_shape(*src, pool_info)));
234  if (indices)
235  {
236  // Indices auto inizialitation if not yet initialized
237  auto_init_if_empty(*indices, (src->clone()->set_tensor_shape(compute_pool_shape(*src, pool_info)))
238  .set_data_type(DataType::U32) /* we store the offset to the element */);
239  }
240  const auto data_layout = pool_info.data_layout == DataLayout::UNKNOWN ? src->data_layout() : pool_info.data_layout;
241 
242  int pool_stride_x = 0;
243  int pool_stride_y = 0;
244  const int idx_width = get_data_layout_dimension_index(data_layout, DataLayoutDimension::WIDTH);
245  const int idx_height = get_data_layout_dimension_index(data_layout, DataLayoutDimension::HEIGHT);
246  const PadStrideInfo pad_stride_info = pool_info.pad_stride_info;
247 
248  std::tie(pool_stride_x, pool_stride_y) = pad_stride_info.stride();
249  const bool is_square = pool_size_x == pool_size_y;
250  const unsigned int pooled_w = dst->dimension(idx_width);
251  const unsigned int pooled_h = dst->dimension(idx_height);
252 
253  //If it's not squared and optimized will be executed the MxN
254  num_elems_processed_per_iteration = 1;
255 
256  if (is_square)
257  {
258  switch (src->data_type())
259  {
260  case DataType::QASYMM8:
261  case DataType::QASYMM8_SIGNED:
262  switch (pool_size_x)
263  {
264  case 2:
265  num_elems_processed_per_iteration = (pool_stride_x == 2) ? 8 : 15;
266  break;
267  case 3:
268  num_elems_processed_per_iteration = (pool_stride_x == 2) ? 7 : 14;
269  break;
270  default:
271  break;
272  }
273  break;
274  case DataType::F16:
275  num_elems_processed_per_iteration = 1;
276  break;
277  case DataType::F32:
278  num_elems_processed_per_iteration = 1;
279  break;
280  default:
281  ARM_COMPUTE_ERROR("Element size not supported");
282  break;
283  }
284  }
285 
286  bool window_changed = false;
287  Window win{};
288  // Upper limit for the number of right/bottom border elements that are accessed
289  TensorShape dst_shape{src->tensor_shape()};
290  dst_shape.set(0, pooled_w);
291  dst_shape.set(1, pooled_h);
292  TensorInfo dst_info(src->clone()->set_tensor_shape(dst_shape));
293  win = calculate_max_window(dst_info, Steps(num_elems_processed_per_iteration));
294 
295  Status err =
296  (window_changed) ? ARM_COMPUTE_CREATE_ERROR(ErrorCode::RUNTIME_ERROR, "Insufficient Padding!") : Status{};
297  return std::make_pair(err, win);
298 }

301 void CpuPool2dKernel::configure(ITensorInfo *src,
302  ITensorInfo *dst,
303  const PoolingLayerInfo &pool_info,
304  ITensorInfo *indices)
305 {
306  ARM_COMPUTE_ERROR_ON_NULLPTR(src, dst);
307  const PadStrideInfo pad_stride_info = pool_info.pad_stride_info;
308  const bool is_global_pooling = pool_info.is_global_pooling;
309 
310  // Get data layout
311  const auto data_layout = pool_info.data_layout == DataLayout::UNKNOWN ? src->data_layout() : pool_info.data_layout;
312  const int idx_width = get_data_layout_dimension_index(data_layout, DataLayoutDimension::WIDTH);
313  const int idx_height = get_data_layout_dimension_index(data_layout, DataLayoutDimension::HEIGHT);
314 
315  // Update pool size in case of global pooling
316  const Size2D pool_size(is_global_pooling ? src->dimension(idx_width) : pool_info.pool_size.width,
317  is_global_pooling ? src->dimension(idx_height) : pool_info.pool_size.height);
318 
319  // Perform validation step
320  ARM_COMPUTE_ERROR_THROW_ON(validate_arguments(src, dst, pool_info, indices, pool_size));
321 
322  const auto *uk = CpuPool2dKernel::get_implementation(
323  PoolDataTypeISASelectorData{src->data_type(), src->data_layout(), (int)pad_stride_info.stride().first,
324  pool_size, CPUInfo::get().get_isa()});
325  ARM_COMPUTE_ERROR_ON(uk == nullptr);
326 
327  // Set instance variables
328  _pool_info = pool_info;
329  _data_layout = src->data_layout();
330  _pool_size = pool_size;
331  _pool_stride_x = pad_stride_info.stride().first;
332  _run_method = uk->ukernel;
333  _name = std::string("CpuPool2dKernel").append("/").append(uk->name);
334 
335  if (_data_layout == DataLayout::NHWC)
336  {
337  // Configure kernel window
338  Window win = calculate_max_window(*dst, Steps());
339  ICpuKernel::configure(win);
340  }
341  else
342  {
343  // Configure kernel window
344  auto win_config = validate_and_configure_window(
345  src, dst, indices, pool_info, _num_elems_processed_per_iteration, pool_size.x(), pool_size.y());
346  ARM_COMPUTE_ERROR_THROW_ON(win_config.first);
347  ICpuKernel::configure(win_config.second);
348  }
349 }

375  .first);

380 void CpuPool2dKernel::run_op(ITensorPack &tensors, const Window &window, const ThreadInfo &info)
381 {
382  ARM_COMPUTE_UNUSED(info);
383  ARM_COMPUTE_ERROR_ON_UNCONFIGURED_KERNEL(this);
384  ARM_COMPUTE_ERROR_ON_INVALID_SUBWINDOW(ICpuKernel::window(), window);
385  ARM_COMPUTE_ERROR_ON(_run_method == nullptr);
386 
387  const ITensor *src = tensors.get_const_tensor(TensorType::ACL_SRC_0);
388  ITensor *dst = tensors.get_tensor(TensorType::ACL_DST_0);
389  ITensor *indices = tensors.get_tensor(TensorType::ACL_DST_1);
390 
391  const unsigned int pool_stride_x = _pool_info.pad_stride_info.stride().first;
392  const unsigned int pool_stride_y = _pool_info.pad_stride_info.stride().second;
393  const unsigned int pool_size = _pool_info.pool_size.width;
394 
395  Window window_src(window);
396  if (_data_layout == DataLayout::NCHW)
397  {
398  // Set step for src in x and y direction for the src
399  unsigned int window_x_inc = 0;
400  switch (src->info()->data_type())
401  {
402  case DataType::QASYMM8:
403  case DataType::QASYMM8_SIGNED:
404  {
405  window_x_inc = pool_stride_x;
406  if ((pool_size == 2 || pool_size == 3) && pool_stride_x < 3)
407  {
408  window_x_inc = (pool_stride_x == 2) ? _num_elems_processed_per_iteration * 2
409  : _num_elems_processed_per_iteration;
410  }
411  break;
412  }
413 
414  case DataType::F16:
415  case DataType::F32:
416  {
417  window_x_inc = pool_stride_x;
418  break;
419  }
420  default:
421  {
422  ARM_COMPUTE_ERROR("Not supported");
423  }
424  }
425  window_src.set(Window::DimX, Window::Dimension(window.x().start() * pool_stride_x,
426  window.x().end() * pool_stride_x, window_x_inc));
427  window_src.set(Window::DimY, Window::Dimension(window.y().start() * pool_stride_y,
428  window.y().end() * pool_stride_y, pool_stride_y));
429  }
430  else
431  {
432  window_src.set(Window::DimX, Window::Dimension(0, 1, 1));
433  window_src.set(Window::DimY, Window::Dimension(0, src->info()->dimension(1), pool_stride_x));
434  window_src.set(Window::DimZ, Window::Dimension(0, src->info()->dimension(2), pool_stride_y));
435  }
436  _run_method(src, dst, indices, _pool_info, window_src, window);
437 }
