43 GpuWorkloadSourceCode GpuKernelComponentStream::write_workload_code()
44 {
45  GpuWorkloadSourceCode source_code;
46  // Traverse through component groups and assemble workload together
47  for (auto &&group : _component_groups)
48  {
49  group.finalize();
50 
51  // Write kernel code
52  GpuLogicalKernel logical_kernel(_services, group);
53  const GpuKernelSourceCode kernel_code = logical_kernel.write_kernel_code();
54  // The whole unit workload stage is determined by the root component
55  const auto unit_workload_stage = group.get_root_component()->properties().stage();
56  source_code.add_unit_workload(kernel_code, unit_workload_stage, _mem_map, _context);
57  }
58  return source_code;
59 }
