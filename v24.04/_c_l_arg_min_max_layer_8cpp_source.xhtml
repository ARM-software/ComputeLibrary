52 Status
53 CLArgMinMaxLayer::validate(const ITensorInfo *input, int axis, const ITensorInfo *output, const ReductionOperation &op)
54 {
55  ARM_COMPUTE_ERROR_ON_NULLPTR(input, output);
56  ARM_COMPUTE_RETURN_ERROR_ON_F16_UNSUPPORTED(input);
57  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(input, 1, DataType::QASYMM8, DataType::QASYMM8_SIGNED,
58  DataType::S32, DataType::F16, DataType::F32);
59  ARM_COMPUTE_RETURN_ERROR_ON_MSG(op != ReductionOperation::ARG_IDX_MAX && op != ReductionOperation::ARG_IDX_MIN,
60  "Invalid reduction operation");
61  ARM_COMPUTE_RETURN_ERROR_ON_MSG(axis >= static_cast<int>(TensorShape::num_max_dimensions),
62  "Reduction axis greater than max number of dimensions");
63  ARM_COMPUTE_RETURN_ERROR_ON_MSG(axis > 3, "Unsupported reduction axis");
64 
65  DataType output_data_type = DataType::S32;
66  TensorInfo not_reshaped_output;
67  const auto input_num_channles = input->num_channels();
68  const auto input_qinfo = input->quantization_info();
69 
70  if (output->total_size() != 0)
71  {
72  output_data_type = output->data_type();
73  const TensorInfo expected_output_shape = output->clone()->set_tensor_shape(
74  arm_compute::misc::shape_calculator::compute_reduced_shape(input->tensor_shape(), axis, false));
75  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_SHAPES(&expected_output_shape, output);
76  }
77 
78  auto shape_before_reshape = input->tensor_shape();
79  shape_before_reshape.set(axis, 1);
80  auto initialize_tensorinfo = [](TensorInfo &ti, TensorShape shape, DataType data_type, int num_channels,
81  QuantizationInfo qinfo) {
82  ti.set_data_type(data_type).set_tensor_shape(shape).set_num_channels(num_channels).set_quantization_info(qinfo);
83  };
84 
85  initialize_tensorinfo(not_reshaped_output, shape_before_reshape, output_data_type, input_num_channles, input_qinfo);
86 
87  ARM_COMPUTE_RETURN_ON_ERROR(CLArgMinMaxLayerKernel::validate(input, &not_reshaped_output, axis, op));
88  ARM_COMPUTE_RETURN_ON_ERROR(CLReshapeLayer::validate(&not_reshaped_output, output));
89  return Status{};
90 }

97 void CLArgMinMaxLayer::configure(const CLCompileContext &compile_context,
98  const ICLTensor *input,
99  int axis,
100  ICLTensor *output,
101  const ReductionOperation &op)
102 {
103  ARM_COMPUTE_ERROR_ON_NULLPTR(input, output);
104  ARM_COMPUTE_LOG_PARAMS(input, axis, output, op);
105 
106  _reduction_axis = axis;
107 
108  const TensorShape output_shape =
109  arm_compute::misc::shape_calculator::compute_reduced_shape(input->info()->tensor_shape(), axis, false);
110  DataType output_data_type =
111  (output->info()->data_type() == DataType::UNKNOWN) ? DataType::S32 : output->info()->data_type();
112  auto_init_if_empty(*output->info(), input->info()
113  ->clone()
114  ->set_tensor_shape(output_shape)
115  .set_data_type(output_data_type)
116  .reset_padding()
117  .set_is_resizable(true));
118 
119  TensorShape not_reshaped_output_shape{input->info()->tensor_shape()};
120  not_reshaped_output_shape.set(axis, 1);
121  auto_init_if_empty(*_not_reshaped_output.info(), input->info()
122  ->clone()
123  ->set_tensor_shape(not_reshaped_output_shape)
124  .set_data_type(output_data_type)
125  .reset_padding()
126  .set_is_resizable(true));
127 
128  _arg_min_max_kernel = std::make_unique<CLArgMinMaxLayerKernel>();
129  _arg_min_max_kernel->configure(compile_context, input, &_not_reshaped_output, axis, op);
130 
131  _memory_group.manage(&_not_reshaped_output);
132 
133  _reshape.configure(compile_context, &_not_reshaped_output, output);
134  _not_reshaped_output.allocator()->allocate();
135 }
