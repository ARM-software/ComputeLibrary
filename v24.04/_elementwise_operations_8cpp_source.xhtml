39 template <typename T>
40 T arithm_op(ArithmeticOperation op, T src1, T src2, ConvertPolicy convert_policy)
41 {
42  using intermediate_type = typename common_promoted_signed_type<T>::intermediate_type;
43 
44  intermediate_type val;
45 
46  switch(op)
47  {
48  case ArithmeticOperation::ADD:
49  {
50  val = static_cast<intermediate_type>(src1) + static_cast<intermediate_type>(src2);
51  break;
52  }
53  case ArithmeticOperation::SUB:
54  {
55  val = static_cast<intermediate_type>(src1) - static_cast<intermediate_type>(src2);
56  break;
57  }
58  case ArithmeticOperation::MIN:
59  {
60  val = std::min(static_cast<intermediate_type>(src1), static_cast<intermediate_type>(src2));
61  break;
62  }
63  case ArithmeticOperation::MAX:
64  {
65  val = std::max(static_cast<intermediate_type>(src1), static_cast<intermediate_type>(src2));
66  break;
67  }
68  case ArithmeticOperation::SQUARED_DIFF:
69  {
70  intermediate_type tmp = (static_cast<intermediate_type>(src1) - static_cast<intermediate_type>(src2));
71  val = tmp * tmp;
72  break;
73  }
74  case ArithmeticOperation::DIV:
75  {
76  val = (static_cast<intermediate_type>(src1) / static_cast<intermediate_type>(src2));
77  break;
78  }
79  case ArithmeticOperation::POWER:
80  {
81  val = std::pow(static_cast<intermediate_type>(src1), static_cast<intermediate_type>(src2));
82  break;
83  }
84  case ArithmeticOperation::PRELU:
85  {
86  const T x = static_cast<intermediate_type>(src1);
87  const T alpha = static_cast<intermediate_type>(src2);
88  val = (x > 0 ? x : alpha * x);
89  break;
90  }
91  default:
92  {
93  ARM_COMPUTE_ERROR("Not handled");
94  }
95  }
96  T result;
97  if(op == ArithmeticOperation::ADD || op == ArithmeticOperation::SUB || op == ArithmeticOperation::DIV || op == ArithmeticOperation::POWER)
98  {
99  result = (convert_policy == ConvertPolicy::SATURATE) ? saturate_cast<T>(val) : static_cast<T>(val);
100  }
101  else
102  {
103  result = static_cast<T>(val);
104  }
105  return result;
106 }

108 template <size_t dim>
109 struct BroadcastUnroll
110 {
111  template <typename T>
112  static void unroll(ArithmeticOperation op, const SimpleTensor<T> &src1, const SimpleTensor<T> &src2, SimpleTensor<T> &dst,
113  ConvertPolicy convert_policy, Coordinates &id_src1, Coordinates &id_src2, Coordinates &id_dst)
114  {
115  const bool src1_is_broadcast = (src1.shape()[dim - 1] != dst.shape()[dim - 1]);
116  const bool src2_is_broadcast = (src2.shape()[dim - 1] != dst.shape()[dim - 1]);
117 
118  id_src1.set(dim - 1, 0);
119  id_src2.set(dim - 1, 0);
120  id_dst.set(dim - 1, 0);
121 
122  for(size_t i = 0; i < dst.shape()[dim - 1]; ++i, ++id_dst[dim - 1])
123  {
124  BroadcastUnroll < dim - 1 >::unroll(op, src1, src2, dst, convert_policy, id_src1, id_src2, id_dst);
125 
126  id_src1[dim - 1] += !src1_is_broadcast;
127  id_src2[dim - 1] += !src2_is_broadcast;
128  }
129  }
130 };
