36 /** Basic function to split a tensor along a given axis */
37 template <typename SliceType, typename TensorInterfaceType = ITensor>
38 class CPPSplit : public IFunction
39 {
40 public:
41  CPPSplit() : _outputs_vector(), _slice_functions(), _num_outputs(0)
42  {
43  }
44  /** Static function to check if given info will lead to a valid configuration of @ref CPPSplit
45  *
46  * @param[in] input The input tensor info. Data types supported: All.
47  * @param[in] outputs A vector containing the output tensors' info. Data types supported: same as @p input.
48  * The output tensors should match the input tensor dimensions for all shape dimensions apart
49  * from the split dimension
50  * @param[in] axis Axis on which to split the input.
51  *
52  * @return a status
53  */
54  static Status validate(const ITensorInfo *input, const std::vector<ITensorInfo *> &outputs, unsigned int axis)
55  {
56  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(input);
57  ARM_COMPUTE_RETURN_ERROR_ON(axis >= input->num_dimensions());
58  ARM_COMPUTE_RETURN_ERROR_ON(outputs.size() < 2);
59 
60  // Get output shape
61  TensorShape output_shape{};
62  unsigned int total_output_shape_size = 0;
63 
64  // Sum the output sizes and fall back to evenly-sized splits if any are zero
65  const bool using_split_shapes = std::none_of(outputs.begin(), outputs.end(),
66  [&total_output_shape_size](ITensorInfo *info)
67  {
68  unsigned int output_shape_size =
69  info->tensor_shape().total_size();
70  total_output_shape_size += output_shape_size;
71  return output_shape_size == 0;
72  });
73 
74  if (using_split_shapes)
75  {
76  ARM_COMPUTE_RETURN_ERROR_ON(input->tensor_shape().total_size() != total_output_shape_size);
77  }
78  else
79  {
80  output_shape = arm_compute::misc::shape_calculator::compute_split_shape(input, axis, outputs.size());
81  ARM_COMPUTE_RETURN_ERROR_ON(output_shape.total_size() == 0);
82  }
83 
84  // Validate output tensors
85  unsigned int axis_offset = 0;
86  for (const auto &output : outputs)
87  {
88  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(output);
89  if (using_split_shapes)
90  {
91  output_shape = output->tensor_shape();
92  ARM_COMPUTE_RETURN_ERROR_ON(output_shape.total_size() == 0);
93  }
94 
95  const size_t axis_split_step = output_shape[axis];
96 
97  // Start/End coordinates
98  Coordinates start_coords;
99  Coordinates end_coords;
100  for (unsigned int d = 0; d < output_shape.num_dimensions(); ++d)
101  {
102  end_coords.set(d, -1);
103  }
104 
105  // Output auto inizialitation if not yet initialized
106  TensorInfo tmp_output_info = *output->clone();
107  if (tmp_output_info.tensor_shape().total_size() == 0)
108  {
109  tmp_output_info = input->clone()->set_is_resizable(true).set_tensor_shape(output_shape);
110  }
111 
112  // Update coordinate on axis
113  start_coords.set(axis, axis_offset);
114  end_coords.set(axis, axis_offset + axis_split_step);
115 
116  ARM_COMPUTE_RETURN_ON_ERROR(SliceType::validate(input, output, start_coords, end_coords));
117  axis_offset += axis_split_step;
118  }
119 
120  return Status{};
121  }
122 
123  /** Initialise the kernel's input and outputs.
124  *
125  * @param[in] input The input tensor. Data types supported: All
126  * @param[out] outputs A vector containing the output tensors. Data types supported: Same as @p input.
127  * The output tensors should match the input tensor dimensions for all shape dimensions apart
128  * from the split dimension.
129  * @param[in] axis Axis on which to split the input.
130  */
131  void
132  configure(const TensorInterfaceType *input, const std::vector<TensorInterfaceType *> &outputs, unsigned int axis)
133  {
134  // Create Slice functions
135  _num_outputs = outputs.size();
136  _slice_functions.resize(_num_outputs);
137 
138  // Extract output tensor info
139  std::vector<ITensorInfo *> outputs_info;
140  for (auto &output : outputs)
141  {
142  ARM_COMPUTE_ERROR_ON_NULLPTR(output);
143  outputs_info.emplace_back(output->info());
144  }
145 
146  // If any of the outputs have a zero size, fall-back to using evenly-sized output splits
147  const bool outputs_have_sizes =
148  std::none_of(outputs_info.begin(), outputs_info.end(),
149  [](ITensorInfo *info) { return info->tensor_shape().total_size() == 0; });
150 
151  // Validate
152  ARM_COMPUTE_ERROR_THROW_ON(CPPSplit::validate(input->info(), outputs_info, axis));
153 
154  unsigned int axis_offset = 0;
155  unsigned int i = 0;
156 
157  for (const auto &output_info : outputs_info)
158  {
159  // Get output shape
160  TensorShape output_shape =
161  (outputs_have_sizes
162  ? output_info->tensor_shape()
163  : arm_compute::misc::shape_calculator::compute_split_shape(input->info(), axis, _num_outputs));
164 
165  const size_t axis_split_step = output_shape[axis];
166 
167  // Start/End coordinates
168  Coordinates start_coords;
169  Coordinates end_coords;
170 
171  for (unsigned int d = 0; d < output_shape.num_dimensions(); ++d)
172  {
173  end_coords.set(d, -1);
174  }
175 
176  // Update coordinate on axis
177  start_coords.set(axis, axis_offset);
178  end_coords.set(axis, axis_offset + axis_split_step);
179 
180  // Configure slice function
181  _slice_functions[i].configure(input, outputs[i], start_coords, end_coords);
182 
183  // Set valid region from shape
184  outputs[i]->info()->set_valid_region(ValidRegion(Coordinates(), output_shape));
185 
186  // Update axis offset
187  axis_offset += axis_split_step;
188  ++i;
189  }
190  }
191 
192 protected:
193  std::vector<TensorInterfaceType *> _outputs_vector;
194  std::vector<SliceType> _slice_functions;
195  unsigned int _num_outputs;
196 };
