45 Status validate_arguments(const ITensorInfo *src, const ITensorInfo *dst, const PaddingList &padding = PaddingList())
46 {
47  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(src, dst);
48  ARM_COMPUTE_RETURN_ERROR_ON(src->data_type() == DataType::UNKNOWN);
49  ARM_COMPUTE_RETURN_ERROR_ON(padding.size() > 4);
50 
51  // Validate destination if initialized
52  if (dst->total_size() != 0)
53  {
54  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DIMENSIONS(
55  misc::shape_calculator::compute_padded_shape(src->tensor_shape(), padding), dst->tensor_shape());
56  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(src, dst);
57  }
58 
59  return Status{};
60 }

70 validate_and_configure_window_with_padding(const ITensorInfo *src, ITensorInfo *dst, const PaddingList &padding)
71 {
72  const TensorShape src_shape = src->tensor_shape();
73  const TensorShape padded_shape = misc::shape_calculator::compute_padded_shape(src_shape, padding);
74  auto_init_if_empty(*dst, src->clone()->set_tensor_shape(padded_shape));
75  // Configure window

82 void CpuCopyKernel::configure(const ITensorInfo *src, ITensorInfo *dst, const PaddingList &padding)
83 {
84  ARM_COMPUTE_ERROR_ON_NULLPTR(src, dst);
85  ARM_COMPUTE_ERROR_THROW_ON(validate_arguments(src, dst, padding));
86 
87  _padding = padding;
88 
89  std::pair<Status, Window> win_config;
90  if (padding.empty())
91  {
92  win_config = validate_and_configure_window(src, dst);
93  }
94  else
95  {
96  win_config = validate_and_configure_window_with_padding(src, dst, padding);
97  }
98 
99  ARM_COMPUTE_ERROR_THROW_ON(win_config.first);
100  ICpuKernel::configure(win_config.second);

104  const arm_compute::ITensorInfo *dst,

122 void CpuCopyKernel::run_op(ITensorPack &tensors, const Window &window, const ThreadInfo &info)
123 {
124  ARM_COMPUTE_UNUSED(info);
125  ARM_COMPUTE_ERROR_ON_UNCONFIGURED_KERNEL(this);
126  ARM_COMPUTE_ERROR_ON_INVALID_SUBWINDOW(ICpuKernel::window(), window);
127 
128  const auto src = tensors.get_const_tensor(TensorType::ACL_SRC);
129  auto dst = tensors.get_tensor(TensorType::ACL_DST);
130 
131  if (_padding.empty())
132  {
133  Window dst_window{window};
134  dst_window.set(Window::DimX,
135  Window::Dimension(dst_window.x().start(), dst_window.x().end(), src->info()->dimension(0)));
136  Window out_slice = dst_window.first_slice_window_1D();
137  do
138  {
139  Iterator src_it(src, out_slice);
140  Iterator dst_it(dst, out_slice);
141 
142  execute_window_loop(
143  out_slice,
144  [&](const Coordinates &)
145  { memcpy(dst_it.ptr(), src_it.ptr(), dst->info()->dimension(0) * dst->info()->element_size()); },
146  src_it, dst_it);
147  } while (dst_window.slide_window_slice_1D(out_slice));
148  }
149  else
150  {
151  Window src_window{window};
152  src_window.set(Window::DimX,
153  Window::Dimension(0, window.x().end() - _padding[0].first, src->info()->dimension(0)));
154 
155  Iterator src_it(src, src_window);
156  Iterator dst_it(dst, window);
157  const size_t row_size_in_bytes = src->info()->dimension(0) * src->info()->element_size();
158  execute_window_loop(
159  window,
160  [&](const Coordinates &)
161  {
162  auto dst_ptr = dst_it.ptr() + _padding[0].first * dst->info()->element_size();
163  std::memcpy(dst_ptr, src_it.ptr(), row_size_in_bytes);
164  },
165  src_it, dst_it);
166  }
167 }
