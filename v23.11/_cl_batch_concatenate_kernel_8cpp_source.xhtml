47 Status validate_arguments(const ITensorInfo *src, unsigned int batch_offset, const ITensorInfo *dst)
48 {
49  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(src, dst);
50  ARM_COMPUTE_RETURN_ERROR_ON_F16_UNSUPPORTED(src);
51  ARM_COMPUTE_RETURN_ERROR_ON(src->data_type() == DataType::UNKNOWN);
52  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(src, dst);
53 
54  ARM_COMPUTE_RETURN_ERROR_ON(src->dimension(Window::DimX) != dst->dimension(Window::DimX));
55  ARM_COMPUTE_RETURN_ERROR_ON(src->dimension(Window::DimY) != dst->dimension(Window::DimY));
56  ARM_COMPUTE_RETURN_ERROR_ON(src->dimension(Window::DimZ) != dst->dimension(Window::DimZ));
57  ARM_COMPUTE_RETURN_ERROR_ON(src->dimension(3) + batch_offset > dst->dimension(3));
58  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_SHAPES(4, src, dst);
59 
60  return Status{};
61 }

69 void ClBatchConcatenateKernel::configure(const CLCompileContext &compile_context,
70  ITensorInfo *src,
71  unsigned int batch_offset,
72  ITensorInfo *dst)
73 {
74  ARM_COMPUTE_ERROR_ON_NULLPTR(src, dst);
75  ARM_COMPUTE_ERROR_THROW_ON(validate_arguments(src, batch_offset, dst));
76 
77  auto padding_info = get_padding_info({src, dst});
78 
79  _batch_offset = batch_offset;
80 
81  const unsigned int num_elems_processed_per_iteration = adjust_vec_size(16 / src->element_size(), src->dimension(0));
82 
83  // Add build options
84  CLBuildOptions build_opts;
85  build_opts.add_option("-DDATA_TYPE=" + get_cl_type_from_data_type(src->data_type()));
86  build_opts.add_option("-DVEC_SIZE=" + support::cpp11::to_string(num_elems_processed_per_iteration));
87  build_opts.add_option("-DVEC_SIZE_LEFTOVER=" +
88  support::cpp11::to_string(src->dimension(0) % num_elems_processed_per_iteration));
89  if (is_data_type_quantized_asymmetric(src->data_type()) && src->quantization_info() != dst->quantization_info())
90  {
91  const UniformQuantizationInfo iq_info = src->quantization_info().uniform();
92  const UniformQuantizationInfo oq_info = dst->quantization_info().uniform();
93 
94  build_opts.add_option("-DOFFSET_IN1=" + float_to_string_with_full_precision(iq_info.offset));
95  build_opts.add_option("-DOFFSET_OUT=" + float_to_string_with_full_precision(oq_info.offset));
96  build_opts.add_option("-DSCALE_IN1=" + float_to_string_with_full_precision(iq_info.scale));
97  build_opts.add_option("-DSCALE_OUT=" + float_to_string_with_full_precision(oq_info.scale));
98  }
99 
100  std::string kernel_name = "concatenate";
101 
102  // A macro guard to compile ONLY the kernel of interest
103  build_opts.add_option("-D" + upper_string(kernel_name));
104 
105  // Create kernel
106  _kernel = create_kernel(compile_context, kernel_name, build_opts.options());
107 
108  // Configure kernel window
109  auto win = calculate_max_window(*dst, Steps(num_elems_processed_per_iteration));
110  win.set(3, Window::Dimension(0, src->tensor_shape()[3], 1));
111  ICLKernel::configure_internal(win);
112 
113  // Set config_id for enabling LWS tuning
114  _config_id = "concatenate_";
115  _config_id += support::cpp11::to_string(3);
116  _config_id += "_";
117  _config_id += support::cpp11::to_string(batch_offset);
118  _config_id += "_";
119  _config_id += support::cpp11::to_string(src->dimension(0));
120  _config_id += "_";
121  _config_id += support::cpp11::to_string(src->dimension(1));
122  _config_id += "_";
123  _config_id += support::cpp11::to_string(src->dimension(2));
124  _config_id += "_";
125  _config_id += support::cpp11::to_string(src->dimension(3));
126 
127  ARM_COMPUTE_ERROR_ON(has_padding_changed(padding_info));
128 }

138 void ClBatchConcatenateKernel::run_op(ITensorPack &tensors, const Window &window, ::cl::CommandQueue &queue)
139 {
140  ARM_COMPUTE_ERROR_ON_UNCONFIGURED_KERNEL(this);
141  ARM_COMPUTE_ERROR_ON_INVALID_SUBWINDOW(ICLKernel::window(), window);
142 
143  const auto src =
144  utils::cast::polymorphic_downcast<const ICLTensor *>(tensors.get_const_tensor(TensorType::ACL_SRC));
145  auto dst = utils::cast::polymorphic_downcast<ICLTensor *>(tensors.get_tensor(TensorType::ACL_DST));
146 
147  Window slice = window.first_slice_window_3D();
148 
149  const int offset_to_first_elements_in_bytes = _batch_offset * dst->info()->strides_in_bytes()[3];
150 
151  unsigned int idx = 2 * num_arguments_per_3D_tensor(); // Skip the src and dst parameters
152  _kernel.setArg<cl_int>(idx, offset_to_first_elements_in_bytes);
153 
154  do
155  {
156  unsigned int idx = 0;
157  add_3D_tensor_argument(idx, src, slice);
158  add_3D_tensor_argument(idx, dst, slice);
159  enqueue(queue, *this, slice, lws_hint());
160  } while (window.slide_window_slice_3D(slice));
161 }
