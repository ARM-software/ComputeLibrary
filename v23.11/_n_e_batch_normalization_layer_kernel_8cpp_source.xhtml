104 Status validate_arguments(const ITensorInfo *input,
105  const ITensorInfo *output,
106  const ITensorInfo *mean,
107  const ITensorInfo *var,
108  const ITensorInfo *beta,
109  const ITensorInfo *gamma,
110  float epsilon,
111  ActivationLayerInfo act_info)
112 {
113  ARM_COMPUTE_UNUSED(epsilon);
114 
115  const auto *uk = get_implementation(BatchNormalizationSelectorData{input->data_type(), CPUInfo::get()});
116  ARM_COMPUTE_RETURN_ERROR_ON(uk == nullptr || uk->ukernel == nullptr);
117 
118  if (act_info.enabled())
119  {
120  ActivationLayerInfo::ActivationFunction act = act_info.activation();
121  ARM_COMPUTE_RETURN_ERROR_ON(act != ActivationLayerInfo::ActivationLayerInfo::ActivationFunction::RELU &&
122  act != ActivationLayerInfo::ActivationLayerInfo::ActivationFunction::BOUNDED_RELU &&
123  act !=
124  ActivationLayerInfo::ActivationLayerInfo::ActivationFunction::LU_BOUNDED_RELU);
125  ARM_COMPUTE_RETURN_ERROR_ON(act_info.b() > act_info.a());
126  }
127 
128  if (nullptr != output)
129  {
130  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_SHAPES(input, output);
131  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_LAYOUT(input, output);
132  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input, output);
133  }
134 
135  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input, mean, var);
136  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_SHAPES(mean, var);
137  if (beta != nullptr)
138  {
139  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input, beta);
140  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_SHAPES(mean, beta);
141  }
142  if (gamma != nullptr)
143  {
144  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input, gamma);
145  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_SHAPES(mean, gamma);
146  }
147  ARM_COMPUTE_RETURN_ERROR_ON(input->dimension(get_data_layout_dimension_index(
148  input->data_layout(), DataLayoutDimension::CHANNEL)) != mean->dimension(0));
149 
150  return Status{};
151 }

154 template <typename T, bool fused_activation, typename F>
155 void NEBatchNormalizationLayerKernel::batch_normalization_nchw(const Window &window)
156 {
157  /** SIMD vector tag type. */
158  using ExactTagType = typename wrapper::traits::neon_bitvector_tag_t<T, wrapper::traits::BitWidth::W128>;
159 
160  const int window_step_x = 16 / sizeof(T);
161  const auto window_start_x = static_cast<int>(window.x().start());
162  const auto window_end_x = static_cast<int>(window.x().end());
163 
164  Window win_to_use = window;
165  win_to_use.set(Window::DimX, Window::Dimension(0, 1, 1));
166 
167  Iterator input(_input, win_to_use);
168  Iterator output(_output, win_to_use);
169 
170  F activation_functor(_act_info);
171 
172  // Hold information about the current feature map we are iterating.
173  // Only compute denominator and constants once per feature map.
174  int slice = -1;
175 
176  const auto input_mean = reinterpret_cast<const T *>(_mean->ptr_to_element(Coordinates(0, 0)));
177  const auto input_var = reinterpret_cast<const T *>(_var->ptr_to_element(Coordinates(0, 0)));
178  const auto input_gamma =
179  (_gamma != nullptr) ? reinterpret_cast<const T *>(_gamma->ptr_to_element(Coordinates(0, 0))) : nullptr;
180  const auto input_beta =
181  (_beta != nullptr) ? reinterpret_cast<const T *>(_beta->ptr_to_element(Coordinates(0, 0))) : nullptr;
182 
183  T mean = static_cast<T>(0);
184  T var = static_cast<T>(0);
185  T gamma = static_cast<T>(1);
186  T beta = static_cast<T>(0);
187  T denominator = static_cast<T>(0);
188 
189  auto mean_vec = wrapper::vdup_n(mean, ExactTagType{});
190  auto var_vec = wrapper::vdup_n(var, ExactTagType{});
191  auto gamma_vec = wrapper::vdup_n(gamma, ExactTagType{});
192  auto beta_vec = wrapper::vdup_n(beta, ExactTagType{});
193  auto denominator_vec = wrapper::vdup_n(denominator, ExactTagType{});
194  const auto epsilon_vec = wrapper::vdup_n(static_cast<T>(_epsilon), ExactTagType{});
195  execute_window_loop(
196  win_to_use,
197  [&](const Coordinates &id)
198  {
199  const auto input_ptr = reinterpret_cast<const T *>(input.ptr());
200  const auto output_ptr = reinterpret_cast<T *>(output.ptr());
201 
202  if (slice != id.z())
203  {
204  mean = input_mean[id.z()];
205  var = input_var[id.z()];
206  mean_vec = wrapper::vdup_n(mean, ExactTagType{});
207  var_vec = wrapper::vdup_n(var, ExactTagType{});
208  if (input_gamma != nullptr)
209  {
210  gamma = input_gamma[id.z()];
211  gamma_vec = wrapper::vdup_n(gamma, ExactTagType{});
212  }
213  if (input_beta != nullptr)
214  {
215  beta = input_beta[id.z()];
216  beta_vec = wrapper::vdup_n(beta, ExactTagType{});
217  }
218 
219  // Calculate denominator
220  denominator_vec = wrapper::vinvsqrt(wrapper::vadd(var_vec, epsilon_vec));
221  denominator = wrapper::vgetlane(denominator_vec, 0);
222  slice = id.z();
223  }
224 
225  // Perform core calculations using vector operations
226  int x = window_start_x;
227  for (; x <= (window_end_x - window_step_x); x += window_step_x)
228  {
229  // Calculate x bar
230  const auto numerator = wrapper::vsub(wrapper::vloadq(input_ptr + x), mean_vec);
231  const auto x_bar = wrapper::vmul(numerator, denominator_vec);
232  auto res = wrapper::vmla(beta_vec, x_bar, gamma_vec);
233 
234  // Perform fused activation
235  if (fused_activation)
236  {
237  activation_functor(res);
238  }
239 
240  // Store results
241  wrapper::vstore(output_ptr + x, res);
242  }
243 
244  // Compute left-over elements
245  for (; x < window_end_x; ++x)
246  {
247  const T numerator = input_ptr[x] - mean;
248  const T x_bar = numerator * denominator;
249  T res = beta + x_bar * gamma;
250 
251  // Perform fused activation
252  if (fused_activation)
253  {
254  activation_functor(res);
255  }
256 
257  // Store results
258  *(output_ptr + x) = res;
259  }
260  },
261  input, output);
262 }

264 void NEBatchNormalizationLayerKernel::configure_non_fused()
265 {
266  switch (_input->info()->data_type())
267  {
268 #ifdef __ARM_FEATURE_FP16_VECTOR_ARITHMETIC
269  case DataType::F16:
270  _func = &NEBatchNormalizationLayerKernel::batch_normalization_nchw<float16_t, false,
271  detail::dummy<float16_t, 8>>;
272  break;
273 #endif // __ARM_FEATURE_FP16_VECTOR_ARITHMETIC
274  case DataType::F32:
275  _func = &NEBatchNormalizationLayerKernel::batch_normalization_nchw<float, false, detail::dummy<float, 4>>;
276  break;
277  default:
278  ARM_COMPUTE_ERROR("Element size not supported");
279  break;
280  }
281 }

283 void NEBatchNormalizationLayerKernel::configure_fused()
284 {
285  // NCHW Fused Batched Normalization with activation functions : FP32
286  static std::map<ActivationLayerInfo::ActivationFunction, BatchNormFunctionPtr> bn_fused_map_f32_nchw = {
287  {ActivationLayerInfo::ActivationFunction::RELU,
288  &NEBatchNormalizationLayerKernel::batch_normalization_nchw<float, true, detail::relu<float, 4>>},
289  {ActivationLayerInfo::ActivationFunction::BOUNDED_RELU,
290  &NEBatchNormalizationLayerKernel::batch_normalization_nchw<float, true, detail::brelu<float, 4>>},
291  {ActivationLayerInfo::ActivationFunction::LU_BOUNDED_RELU,
292  &NEBatchNormalizationLayerKernel::batch_normalization_nchw<float, true, detail::lubrelu<float, 4>>}};
293 #ifdef __ARM_FEATURE_FP16_VECTOR_ARITHMETIC
294  // NCHW Fused Batched Normalization with activation functions : FP16
295  static std::map<ActivationLayerInfo::ActivationFunction, BatchNormFunctionPtr> bn_fused_map_f16_nchw = {
296  {ActivationLayerInfo::ActivationFunction::RELU,
297  &NEBatchNormalizationLayerKernel::batch_normalization_nchw<float16_t, true, detail::relu<float16_t, 8>>},
298  {ActivationLayerInfo::ActivationFunction::BOUNDED_RELU,
299  &NEBatchNormalizationLayerKernel::batch_normalization_nchw<float16_t, true, detail::brelu<float16_t, 8>>},
300  {ActivationLayerInfo::ActivationFunction::LU_BOUNDED_RELU,
301  &NEBatchNormalizationLayerKernel::batch_normalization_nchw<float16_t, true, detail::lubrelu<float16_t, 8>>}};
302 #endif // __ARM_FEATURE_FP16_VECTOR_ARITHMETIC
303 
304  switch (_input->info()->data_type())
305  {
306 #ifdef __ARM_FEATURE_FP16_VECTOR_ARITHMETIC
307  case DataType::F16:
308  _func = bn_fused_map_f16_nchw[_act_info.activation()];
309  break;
310 #endif // __ARM_FEATURE_FP16_VECTOR_ARITHMETIC
311  case DataType::F32:
312  _func = bn_fused_map_f32_nchw[_act_info.activation()];
313  break;
314  default:
315  ARM_COMPUTE_ERROR("Element size not supported");
316  break;
317  }
318 }

333 void NEBatchNormalizationLayerKernel::configure(ITensor *input,
334  ITensor *output,
335  const ITensor *mean,
336  const ITensor *var,
337  const ITensor *beta,
338  const ITensor *gamma,
339  float epsilon,
340  ActivationLayerInfo act_info)
341 {
342  ARM_COMPUTE_ERROR_ON_NULLPTR(input, mean, var);
343 
344  ARM_COMPUTE_ERROR_THROW_ON(validate_arguments(input->info(), (output != nullptr) ? output->info() : nullptr,
345  mean->info(), var->info(), (beta != nullptr) ? beta->info() : nullptr,
346  (gamma != nullptr) ? gamma->info() : nullptr, epsilon, act_info));
347 
348  _input = input;
349  _output = input;
350  _mean = mean;
351  _var = var;
352  _gamma = gamma;
353  _beta = beta;
354  _epsilon = epsilon;
355  _act_info = act_info;
356 
357  const bool run_in_place = (output == nullptr) || (output == input);
358  if (!run_in_place)
359  {
360  _output = output;
361  }
362 
363  // Configure activation function to run
364  const bool is_nchw = _input->info()->data_layout() == DataLayout::NCHW;
365  if (is_nchw)
366  {
367  if (_act_info.enabled())
368  {
369  configure_fused();
370  }
371  else
372  {
373  configure_non_fused();
374  }
375  }
376 
377  // Configure kernel window
378  Window win = calculate_max_window(*input->info(), Steps());
379  INEKernel::configure(win);
380 
381  if (output != nullptr)
382  {
383  // Output auto initialization if not yet initialized
384  auto_init_if_empty(*output->info(), *input->info()->clone());
385  }
386 }
