46 Status validate_arguments(const ITensorInfo *input, const ITensorInfo *output, unsigned int axis, ReductionOperation op)
47 {
48  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(input, output);
49  ARM_COMPUTE_RETURN_ERROR_ON_F16_UNSUPPORTED(input);
50  if (input->num_channels() == 1)
51  {
52  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(input, 1, DataType::QASYMM8, DataType::QASYMM8_SIGNED,
53  DataType::S32, DataType::F16, DataType::F32);
54  }
55  else
56  {
57  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(input, 2, DataType::F16, DataType::F32);
58  ARM_COMPUTE_RETURN_ERROR_ON(axis == 0);
59  }
60  ARM_COMPUTE_RETURN_ERROR_ON_MSG(op == ReductionOperation::SUM_SQUARE && input->data_type() == DataType::QASYMM8,
61  "Not supported reduction operation for QASYMM8");
62  ARM_COMPUTE_RETURN_ERROR_ON_MSG(axis >= TensorShape::num_max_dimensions,
63  "Reduction axis greater than max number of dimensions");
64  ARM_COMPUTE_RETURN_ERROR_ON_MSG(axis > 3, "Unsupported reduction axis");
65  ARM_COMPUTE_RETURN_ERROR_ON((op == ReductionOperation::MEAN_SUM) && (axis == 0) && (input->dimension(0) == 0) &&
66  (input->data_type() != DataType::QASYMM8) &&
67  (input->data_type() != DataType::QASYMM8_SIGNED));
68  ARM_COMPUTE_RETURN_ERROR_ON_MSG((op == ReductionOperation::ARG_IDX_MAX) || (op == ReductionOperation::ARG_IDX_MIN),
69  "Not supported reduction operation, use CLArgMinMaxLayer");
70 
71  if (output->total_size() != 0)
72  {
73  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input, output);
74  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_QUANTIZATION_INFO(input, output);
75  }
76 
77  return Status{};
78 }

95 void CLReductionOperationKernel::configure(const CLCompileContext &compile_context,
96  const ICLTensor *input,
97  ICLTensor *output,
98  unsigned int axis,
99  ReductionOperation op)
100 {
101  ARM_COMPUTE_ERROR_ON_NULLPTR(input, output);
102 
103  ARM_COMPUTE_ERROR_THROW_ON(validate_arguments(input->info(), output->info(), axis, op));
104 
105  auto padding_info = get_padding_info({input, output});
106 
107  _input = input;
108  _output = output;
109  _reduction_axis = axis;
110  _op = op;
111 
112  const TensorShape output_shape =
113  arm_compute::misc::shape_calculator::compute_reduced_shape(input->info()->tensor_shape(), axis, true);
114  auto_init_if_empty(*output->info(),
115  input->info()->clone()->set_tensor_shape(output_shape).reset_padding().set_is_resizable(true));
116 
117  // Set build options
118  CLBuildOptions build_opts;
119  DataType data_type = input->info()->data_type();
120  std::string data_type_promoted{};
121 
122  if (is_data_type_quantized(data_type))
123  {
124  data_type_promoted = "int";
125  }
126  else
127  {
128  data_type_promoted = get_cl_type_from_data_type(data_type);
129  }
130 
131  const unsigned int width = input->info()->dimension(0) * input->info()->num_channels();
132  unsigned int vec_size = (is_data_type_quantized(input->info()->data_type()) && (axis == 0)) ? 1 : 16;
133  vec_size = adjust_vec_size(vec_size, width);
134  const unsigned int vec_size_leftover = width % vec_size;
135 
136  build_opts.add_option("-DDATA_TYPE=" + get_cl_type_from_data_type(data_type));
137  build_opts.add_option("-DDATA_TYPE_PROMOTED=" + data_type_promoted);
138  build_opts.add_option("-DVEC_SIZE=" + support::cpp11::to_string(vec_size));
139  build_opts.add_option("-DVEC_SIZE_LEFTOVER=" + support::cpp11::to_string(vec_size_leftover));
140  build_opts.add_option_if(is_data_type_float(data_type), "-DFLOAT_DATA_TYPE");
141  build_opts.add_option_if(op == ReductionOperation::SUM_SQUARE, "-DSUM_SQUARE");
142  build_opts.add_option_if(op == ReductionOperation::MEAN_SUM, "-DMEAN");
143  build_opts.add_option_if(op == ReductionOperation::SUM, "-DSUM");
144  build_opts.add_option_if(op == ReductionOperation::PROD, "-DPROD");
145  build_opts.add_option_if(op == ReductionOperation::MIN, "-DMIN");
146  build_opts.add_option_if(op == ReductionOperation::MAX, "-DMAX");
147  build_opts.add_option_if(is_data_type_quantized(data_type),
148  "-DOFFSET=" +
149  support::cpp11::to_string(input->info()->quantization_info().uniform().offset));
150  build_opts.add_option_if(
151  is_data_type_quantized(data_type),
152  "-DSCALE=" + float_to_string_with_full_precision(input->info()->quantization_info().uniform().scale));
153 
154  switch (op)
155  {
156  case ReductionOperation::SUM_SQUARE:
157  build_opts.add_option(("-DOPERATION=square_sum"));
158  break;
159  case ReductionOperation::SUM:
160  case ReductionOperation::MEAN_SUM:
161  build_opts.add_option(("-DOPERATION=sum"));
162  break;
163  case ReductionOperation::MIN:
164  build_opts.add_option(("-DOPERATION=min_"));
165  break;
166  case ReductionOperation::MAX:
167  build_opts.add_option(("-DOPERATION=max_"));
168  break;
169  case ReductionOperation::PROD:
170  build_opts.add_option(("-DOPERATION=product"));
171  break;
172  default:
173  ARM_COMPUTE_ERROR("Unsupported reduction operation");
174  }
175 
176  // Create kernel
177  std::string kernel_axis_name;
178  const bool is_serial_op = needs_serialized_reduction(_op, _input->info()->data_type(), _reduction_axis);
179 
180  switch (axis)
181  {
182  case 0:
183  {
184  build_opts.add_option("-DWIDTH=" + support::cpp11::to_string(width));
185  kernel_axis_name = ((is_serial_op) ? "non_parallel_x" : "x");
186  }
187  break;
188  case 1:
189  build_opts.add_option("-DHEIGHT=" + support::cpp11::to_string(input->info()->dimension(1)));
190  kernel_axis_name = "y";
191  break;
192  case 2:
193  build_opts.add_option("-DDEPTH=" + support::cpp11::to_string(input->info()->dimension(2)));
194  kernel_axis_name = "z";
195  break;
196  case 3:
197  build_opts.add_option("-DDEPTH=" + support::cpp11::to_string(input->info()->dimension(2)));
198  build_opts.add_option("-DBATCH=" + support::cpp11::to_string(input->info()->dimension(3)));
199  kernel_axis_name = "w";
200  break;
201  default:
202  ARM_COMPUTE_ERROR("Not supported");
203  }
204  _kernel = create_kernel(compile_context, "reduction_operation_" + kernel_axis_name, build_opts.options());
205 
206  // Configure kernel window
207  TensorShape actual_input_shape = input->info()->tensor_shape();
208  actual_input_shape[0] = width;
209 
210  Window win = calculate_max_window(actual_input_shape, Steps(vec_size));
211  ICLKernel::configure_internal(win);
212 
213  ARM_COMPUTE_ERROR_ON(has_padding_changed(padding_info));
214 }

225 void CLReductionOperationKernel::run(const Window &window, cl::CommandQueue &queue)
226 {
227  ARM_COMPUTE_ERROR_ON_UNCONFIGURED_KERNEL(this);
228  ARM_COMPUTE_ERROR_ON_INVALID_SUBWINDOW(IKernel::window(), window);
229 
230  const bool is_serial_op = needs_serialized_reduction(_op, _input->info()->data_type(), _reduction_axis);
231  switch (_reduction_axis)
232  {
233  case 0:
234  {
235  // We use parallel reduction only in non quantized types
236  if (is_serial_op)
237  {
238  // Get first input and output slices
239  Window window_in{window};
240  window_in.set(Window::DimX,
241  Window::Dimension(0, _input->info()->dimension(0), _input->info()->dimension(0)));
242 
243  Window out_window{window};
244  out_window.set(Window::DimX, Window::Dimension(0, 0, 0));
245 
246  Window in_slice = window_in.first_slice_window_1D();
247  Window out_slice = out_window.first_slice_window_1D();
248 
249  do
250  {
251  unsigned int idx = 0;
252  add_1D_tensor_argument(idx, _input, in_slice);
253  add_1D_tensor_argument(idx, _output, out_slice);
254  enqueue(queue, *this, in_slice);
255  } while (window_in.slide_window_slice_1D(in_slice) && out_window.slide_window_slice_1D(out_slice));
256  }
257  else
258  {
259  // Set out window
260  bool has_collapsed = true;
261  Window window_in = window.collapse_if_possible(window, 2, &has_collapsed);
262  ARM_COMPUTE_ERROR_ON(!has_collapsed);
263 
264  Window window_out = window_in;
265  window_out.set(0, Window::Dimension());
266 
267  unsigned int idx = 0;
268  add_3D_tensor_argument(idx, _input, window_in);
269  add_3D_tensor_argument(idx, _output, window_out);
270  enqueue(queue, *this, window_in);
271  }
272  }
273  break;
274  case 1:
275  {
276  bool has_collapsed = true;
277  Window actual_window = window.collapse_if_possible(window, 2, &has_collapsed);
278  ARM_COMPUTE_ERROR_ON(!has_collapsed);
279 
280  actual_window = actual_window.shift_dimensions(1, Window::DimY);
281 
282  const ITensorInfo *input_info = _input->info();
283  const Strides &input_strides = input_info->strides_in_bytes();
284 
285  const ITensorInfo *output_info = _output->info();
286  const Strides &output_strides = output_info->strides_in_bytes();
287 
288  unsigned int idx = 0;
289 
290  _kernel.setArg(idx++, _input->cl_buffer());
291  _kernel.setArg<cl_uint>(idx++, input_strides[1]);
292  _kernel.setArg<cl_uint>(idx++, input_strides[2]);
293  _kernel.setArg<cl_uint>(idx++, input_info->offset_first_element_in_bytes());
294 
295  _kernel.setArg(idx++, _output->cl_buffer());
296  _kernel.setArg<cl_uint>(idx++, output_strides[2]);
297  _kernel.setArg<cl_uint>(idx++, output_info->offset_first_element_in_bytes());
298 
299  enqueue(queue, *this, actual_window);
300  }
301  break;
302  case 2:
303  {
304  bool has_collapsed = true;
305  Window actual_window = window.collapse_if_possible(window, 3, &has_collapsed);
306  ARM_COMPUTE_ERROR_ON(!has_collapsed);
307 
308  actual_window = actual_window.shift_dimensions(1, Window::DimZ);
309 
310  const ITensorInfo *input_info = _input->info();
311  const Strides &input_strides = input_info->strides_in_bytes();
312 
313  const ITensorInfo *output_info = _output->info();
314  const Strides &output_strides = output_info->strides_in_bytes();
315 
316  unsigned int idx = 0;
317 
318  _kernel.setArg(idx++, _input->cl_buffer());
319  _kernel.setArg<cl_uint>(idx++, input_strides[1]);
320  _kernel.setArg<cl_uint>(idx++, input_strides[2]);
321  _kernel.setArg<cl_uint>(idx++, input_strides[3]);
322  _kernel.setArg<cl_uint>(idx++, input_info->offset_first_element_in_bytes());
323 
324  _kernel.setArg(idx++, _output->cl_buffer());
325  _kernel.setArg<cl_uint>(idx++, output_strides[1]);
326  _kernel.setArg<cl_uint>(idx++, output_strides[3]);
327  _kernel.setArg<cl_uint>(idx++, output_info->offset_first_element_in_bytes());
328 
329  enqueue(queue, *this, actual_window);
330  }
331  break;
332  case 3:
333  {
334  bool has_collapsed = true;
335  Window actual_window = window.shift_dimensions(1, Window::DimW);
336 
337  actual_window = actual_window.collapse_if_possible(actual_window, 2, &has_collapsed);
338  ARM_COMPUTE_ERROR_ON(!has_collapsed);
339 
340  const ITensorInfo *input_info = _input->info();
341  const Strides &input_strides = input_info->strides_in_bytes();
342 
343  const ITensorInfo *output_info = _output->info();
344  const Strides &output_strides = output_info->strides_in_bytes();
345 
346  unsigned int idx = 0;
347 
348  _kernel.setArg(idx++, _input->cl_buffer());
349  _kernel.setArg<cl_uint>(idx++, input_strides[1]);
350  _kernel.setArg<cl_uint>(idx++, input_strides[2]);
351  _kernel.setArg<cl_uint>(idx++, input_strides[3]);
352  _kernel.setArg<cl_uint>(idx++, input_strides[4]);
353  _kernel.setArg<cl_uint>(idx++, input_info->offset_first_element_in_bytes());
354 
355  _kernel.setArg(idx++, _output->cl_buffer());
356  _kernel.setArg<cl_uint>(idx++, output_strides[1]);
357  _kernel.setArg<cl_uint>(idx++, output_strides[2]);
358  _kernel.setArg<cl_uint>(idx++, output_strides[4]);
359  _kernel.setArg<cl_uint>(idx++, output_info->offset_first_element_in_bytes());
360 
361  enqueue(queue, *this, actual_window);
362  }
363  break;
364  default:
365  ARM_COMPUTE_ERROR("Not supported");
366  }
