32 template <typename T>
33 void bounding_box_transform(const ITensor *boxes,
34  ITensor *pred_boxes,
35  const ITensor *deltas,
36  BoundingBoxTransformInfo bbinfo,
37  const Window &window)
38 {
39  const size_t num_classes = deltas->info()->tensor_shape()[0] >> 2;
40  const size_t deltas_width = deltas->info()->tensor_shape()[0];
41  const int img_h = std::floor(bbinfo.img_height() / bbinfo.scale() + 0.5f);
42  const int img_w = std::floor(bbinfo.img_width() / bbinfo.scale() + 0.5f);
43 
44  const auto scale_after = (bbinfo.apply_scale() ? T(bbinfo.scale()) : T(1));
45  const auto scale_before = T(bbinfo.scale());
46  ARM_COMPUTE_ERROR_ON(scale_before <= 0);
47  const auto offset = (bbinfo.correct_transform_coords() ? T(1.f) : T(0.f));
48 
49  auto pred_ptr = reinterpret_cast<T *>(pred_boxes->buffer() + pred_boxes->info()->offset_first_element_in_bytes());
50  auto delta_ptr = reinterpret_cast<T *>(deltas->buffer() + deltas->info()->offset_first_element_in_bytes());
51 
52  Iterator box_it(boxes, window);
53  execute_window_loop(
54  window,
55  [&](const Coordinates &id)
56  {
57  const auto ptr = reinterpret_cast<T *>(box_it.ptr());
58  const auto b0 = *ptr;
59  const auto b1 = *(ptr + 1);
60  const auto b2 = *(ptr + 2);
61  const auto b3 = *(ptr + 3);
62  const T width = (b2 / scale_before) - (b0 / scale_before) + T(1.f);
63  const T height = (b3 / scale_before) - (b1 / scale_before) + T(1.f);
64  const T ctr_x = (b0 / scale_before) + T(0.5f) * width;
65  const T ctr_y = (b1 / scale_before) + T(0.5f) * height;
66  for (size_t j = 0; j < num_classes; ++j)
67  {
68  // Extract deltas
69  const size_t delta_id = id.y() * deltas_width + 4u * j;
70  const T dx = delta_ptr[delta_id] / T(bbinfo.weights()[0]);
71  const T dy = delta_ptr[delta_id + 1] / T(bbinfo.weights()[1]);
72  T dw = delta_ptr[delta_id + 2] / T(bbinfo.weights()[2]);
73  T dh = delta_ptr[delta_id + 3] / T(bbinfo.weights()[3]);
74  // Clip dw and dh
75  dw = std::min(dw, T(bbinfo.bbox_xform_clip()));
76  dh = std::min(dh, T(bbinfo.bbox_xform_clip()));
77  // Determine the predictions
78  const T pred_ctr_x = dx * width + ctr_x;
79  const T pred_ctr_y = dy * height + ctr_y;
80  const T pred_w = std::exp(dw) * width;
81  const T pred_h = std::exp(dh) * height;
82  // Store the prediction into the output tensor
83  pred_ptr[delta_id] = scale_after * utility::clamp<T>(pred_ctr_x - T(0.5f) * pred_w, T(0), T(img_w - 1));
84  pred_ptr[delta_id + 1] =
85  scale_after * utility::clamp<T>(pred_ctr_y - T(0.5f) * pred_h, T(0), T(img_h - 1));
86  pred_ptr[delta_id + 2] =
87  scale_after * utility::clamp<T>(pred_ctr_x + T(0.5f) * pred_w - offset, T(0), T(img_w - 1));
88  pred_ptr[delta_id + 3] =
89  scale_after * utility::clamp<T>(pred_ctr_y + T(0.5f) * pred_h - offset, T(0), T(img_h - 1));
90  }
91  },
92  box_it);
93 }
