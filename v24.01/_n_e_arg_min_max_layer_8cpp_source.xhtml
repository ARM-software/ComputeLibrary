57 void NEArgMinMaxLayer::configure(ITensor *input, int axis, ITensor *output, const ReductionOperation &op)
58 {
59  ARM_COMPUTE_LOG_PARAMS(input, axis, output, op);
60  _impl->reduction_function = std::make_unique<NEReductionOperation>();
61  if (output->info() &&
62  (output->info()->data_type() == DataType::S64 || output->info()->data_type() == DataType::U64))
63  {
64  _impl->memory_group = MemoryGroup(std::move(_impl->memory_manager));
65  _impl->cast_function = std::make_unique<NECast>();
66  _impl->tmp_reduction_result = std::make_unique<Tensor>();
67  _impl->reduction_function->configure(input, _impl->tmp_reduction_result.get(), axis, op, false);
68  _impl->cast_function->configure(_impl->tmp_reduction_result.get(), output, ConvertPolicy::SATURATE);
69  _impl->memory_group.manage(_impl->tmp_reduction_result.get());
70  _impl->tmp_reduction_result->allocator()->allocate();
71  }
72  else
73  {
74  _impl->reduction_function->configure(input, output, axis, op, false);
75  }
76 }
