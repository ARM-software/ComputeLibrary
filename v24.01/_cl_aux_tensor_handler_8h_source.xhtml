38 /* Tensor handler to wrap and handle tensor allocations on workspace buffers */
39 class CLAuxTensorHandler
40 {
41 public:
42  CLAuxTensorHandler(
43  int slot_id, TensorInfo &info, ITensorPack &pack, bool pack_inject = false, bool bypass_alloc = false)
44  : _tensor()
45  {
46  if (info.total_size() == 0)
47  {
48  return;
49  }
50  _tensor.allocator()->soft_init(info);
51 
52  ICLTensor *packed_tensor = utils::cast::polymorphic_downcast<ICLTensor *>(pack.get_tensor(slot_id));
53  if ((packed_tensor == nullptr) || (info.total_size() > packed_tensor->info()->total_size()))
54  {
55  if (!bypass_alloc)
56  {
57  _tensor.allocator()->allocate();
58  ARM_COMPUTE_LOG_INFO_WITH_FUNCNAME_ACL("Allocating auxiliary tensor");
59  }
60 
61  if (pack_inject)
62  {
63  pack.add_tensor(slot_id, &_tensor);
64  _injected_tensor_pack = &pack;
65  _injected_slot_id = slot_id;
66  }
67  }
68  else
69  {
70  _tensor.allocator()->import_memory(packed_tensor->cl_buffer());
71  }
72  }
73 
74  CLAuxTensorHandler(TensorInfo &info, ICLTensor &tensor) : _tensor()
75  {
76  _tensor.allocator()->soft_init(info);
77  if (info.total_size() <= tensor.info()->total_size())
78  {
79  _tensor.allocator()->import_memory(tensor.cl_buffer());
80  }
81  }
82 
83  CLAuxTensorHandler(const CLAuxTensorHandler &) = delete;
84  CLAuxTensorHandler &operator=(const CLAuxTensorHandler) = delete;
85 
86  ~CLAuxTensorHandler()
87  {
88  if (_injected_tensor_pack)
89  {
90  _injected_tensor_pack->remove_tensor(_injected_slot_id);
91  }
92  }
93 
94  ICLTensor *get()
95  {
96  return &_tensor;
97  }
98 
99  ICLTensor *operator()()
100  {
101  return &_tensor;
102  }
103 
104 private:
105  CLTensor _tensor{};
106  ITensorPack *_injected_tensor_pack{nullptr};
107  int _injected_slot_id{TensorType::ACL_UNKNOWN};
108 };
