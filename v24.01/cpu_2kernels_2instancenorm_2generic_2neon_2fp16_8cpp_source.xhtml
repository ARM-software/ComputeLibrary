36 template <typename InputType, typename AccType>
37 void vector_float_sum_fp16(AccType &result, AccType &result_square, const InputType &inputs)
38 {
39  result = wrapper::vadd(result, inputs);
40  result_square = wrapper::vadd(result_square, wrapper::vmul(inputs, inputs));
41 }

43 template <typename InputType, typename AccType>
44 InputType vector_float_norm_fp16(const InputType &inputs,
45  const AccType &vec_mean,
46  const AccType &vec_multip,
47  const AccType &vec_beta)

58 template <>
59 inline float16x8_t vector_float_norm_fp16(const float16x8_t &inputs,
60  const float32x4_t &vec_mean,
61  const float32x4_t &vec_multip,
62  const float32x4_t &vec_beta)
63 {
64  const auto input_low = wrapper::vcvt<float>(wrapper::vgetlow(inputs));
65  const auto input_high = wrapper::vcvt<float>(wrapper::vgethigh(inputs));
66  const auto result_low = wrapper::vcvt<float16_t>(vector_float_norm_fp16(input_low, vec_mean, vec_multip, vec_beta));
67  const auto result_high =
68  wrapper::vcvt<float16_t>(vector_float_norm_fp16(input_high, vec_mean, vec_multip, vec_beta));
69  float16x8_t result = wrapper::vcombine(result_low, result_high);
70 
71  return result;
72 }

74 template <typename AccType>
75 void instance_normalization_nchw_fp16(
76  const ITensor *input, ITensor *output, float gamma, float beta, float epsilon, const Window &window)
77 {
78  /** SIMD vector tag type. */
79  using ExactTagType = typename wrapper::traits::neon_bitvector_tag_t<float16_t, wrapper::traits::BitWidth::W128>;
80 
81  // Clear X/Y dimensions on execution window as we handle the planes manually
82  Window win = window;
83  win.set(Window::DimX, Window::Dimension(0, 1, 1));
84  win.set(Window::DimY, Window::Dimension(0, 1, 1));
85 
86  constexpr int window_step_x = 16 / sizeof(float16_t);
87  const unsigned int elements_plane = input->info()->dimension(0) * output->info()->dimension(1);
88 
89  Iterator input_it(input, win);
90  execute_window_loop(
91  win,
92  [&](const Coordinates &id)
93  {
94  Window win_plane = window;
95  win_plane.set(Window::DimX, Window::Dimension(0, 1, 1));
96  win_plane.set(Window::DimZ, Window::Dimension(id[2], id[2] + 1, 1));
97  win_plane.set(3, Window::Dimension(id[3], id[3] + 1, 1));
98 
99  Iterator input_plane_it(input, win_plane);
100  Iterator output_plane_it(output, win_plane);
101 
102  auto sum_h_w = static_cast<AccType>(0.f);
103  auto sum_squares_h_w = static_cast<AccType>(0.f);
104 
105  execute_window_loop(
106  win_plane,
107  [&](const Coordinates &)
108  {
109  const auto input_ptr = reinterpret_cast<const float16_t *>(input_plane_it.ptr());
110 
111  auto vec_sum_h_w = wrapper::vdup_n(static_cast<AccType>(0.f), ExactTagType{});
112  auto vec_sum_squares_h_w = wrapper::vdup_n(static_cast<AccType>(0.f), ExactTagType{});
113 
114  // Compute S elements per iteration
115  int x = window.x().start();
116  for (; x <= (window.x().end() - window_step_x); x += window_step_x)
117  {
118  auto vec_input_val = wrapper::vloadq(input_ptr + x);
119  vector_float_sum_fp16(vec_sum_h_w, vec_sum_squares_h_w, vec_input_val);
120  }
121 
122  auto vec2_sum_h_w = wrapper::vpadd(wrapper::vgethigh(vec_sum_h_w), wrapper::vgetlow(vec_sum_h_w));
123  auto vec2_sum_squares_h_w =
124  wrapper::vpadd(wrapper::vgethigh(vec_sum_squares_h_w), wrapper::vgetlow(vec_sum_squares_h_w));
125 
126  vec2_sum_h_w = wrapper::vpadd(vec2_sum_h_w, vec2_sum_h_w);
127  vec2_sum_squares_h_w = wrapper::vpadd(vec2_sum_squares_h_w, vec2_sum_squares_h_w);
128 
129  sum_h_w += wrapper::vgetlane(vec2_sum_h_w, 0);
130  sum_squares_h_w += wrapper::vgetlane(vec2_sum_squares_h_w, 0);
131 
132  // Compute left-over elements
133  for (; x < window.x().end(); ++x)
134  {
135  const auto value = static_cast<AccType>(*(input_ptr + x));
136  sum_h_w += value;
137  sum_squares_h_w += value * value;
138  }
139  },
140  input_plane_it, output_plane_it);
141 
142  const auto mean_h_w = sum_h_w / elements_plane;
143  const auto var_h_w = sum_squares_h_w / elements_plane - mean_h_w * mean_h_w;
144 
145  const auto multip_h_w = gamma / std::sqrt(var_h_w + epsilon);
146  const auto vec_mean_h_w = wrapper::vdup_n(static_cast<AccType>(mean_h_w), ExactTagType{});
147  const auto vec_multip_h_w = wrapper::vdup_n(static_cast<AccType>(multip_h_w), ExactTagType{});
148  const auto vec_beta = wrapper::vdup_n(static_cast<AccType>(beta), ExactTagType{});
149 
150  execute_window_loop(
151  win_plane,
152  [&](const Coordinates &)
153  {
154  auto input_ptr = reinterpret_cast<const float16_t *>(input_plane_it.ptr());
155  auto output_ptr = reinterpret_cast<float16_t *>(output_plane_it.ptr());
156 
157  // Compute S elements per iteration
158  int x = window.x().start();
159  for (; x <= (window.x().end() - window_step_x); x += window_step_x)
160  {
161  const auto vec_val = wrapper::vloadq(input_ptr + x);
162  const auto normalized_vec =
163  vector_float_norm_fp16(vec_val, vec_mean_h_w, vec_multip_h_w, vec_beta);
164  wrapper::vstore(output_ptr + x, normalized_vec);
165  }
166 
167  // Compute left-over elements
168  for (; x < window.x().end(); ++x)
169  {
170  const auto val = static_cast<AccType>(*(input_ptr + x));
171  *(output_ptr + x) = static_cast<float16_t>((val - mean_h_w) * multip_h_w + beta);
172  }
173  },
174  input_plane_it, output_plane_it);
175  },
176  input_it);
177 }
