36 void matrix_addition_f16(const ITensor *src, ITensor *dst, const Window &window, float beta)
37 {
38  ARM_COMPUTE_ERROR_ON_NULLPTR(src, dst);
39  const float16x8_t beta_f16 = vdupq_n_f16(beta);
40 
41  constexpr int window_step_x = 16;
42  const auto window_start_x = static_cast<int>(window.x().start());
43  const auto window_end_x = static_cast<int>(window.x().end());
44 
45  Window win = window.collapse_if_possible(window, Window::DimZ);
46  win.set(Window::DimX, Window::Dimension(0, 1, 1));
47 
48  Iterator in(src, win);
49  Iterator out(dst, win);
50 
51  execute_window_loop(
52  win,
53  [&](const Coordinates &)
54  {
55  const auto in_ptr = reinterpret_cast<const float16_t *>(in.ptr());
56  const auto out_ptr = reinterpret_cast<float16_t *>(out.ptr());
57 
58  int x = window_start_x;
59  for (; x < (window_end_x - window_step_x); x += window_step_x)
60  {
61  float16x8x2_t alpha_ab = vld2q_f16(out_ptr + x);
62  const float16x8x2_t c = vld2q_f16(in_ptr + x);
63  // Multiply matrix C by its weight and accumulate
64  alpha_ab.val[0] = vaddq_f16(alpha_ab.val[0], vmulq_f16(c.val[0], beta_f16));
65  alpha_ab.val[1] = vaddq_f16(alpha_ab.val[1], vmulq_f16(c.val[1], beta_f16));
66 
67  vst2q_f16(out_ptr + x, alpha_ab);
68  }
69 
70  // Left-over loop
71  for (; x < window_end_x; ++x)
72  {
73  *(out_ptr + x) += *(in_ptr + x) * static_cast<float16_t>(beta);
74  }
75  },
76  in, out);
77 }
78 } // namespace
79 void neon_fp16_gemm_matrix_add(const ITensor *src, ITensor *dst, const Window &window, float beta)
80 {
81  return matrix_addition_f16(src, dst, window, beta);
82 }
