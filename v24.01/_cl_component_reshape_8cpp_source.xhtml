38 Status ClComponentReshape::validate(const ArgumentPack<ITensorInfo> &tensors)
39 {
40  const auto src = tensors.get_const_tensor(TensorType::ACL_SRC_0);
41  const auto dst = tensors.get_const_tensor(TensorType::ACL_DST_0);
42  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(src, dst);
43  ARM_COMPUTE_RETURN_ERROR_ON_F16_UNSUPPORTED(src);
44  ARM_COMPUTE_RETURN_ERROR_ON(src->data_type() == DataType::UNKNOWN);
45  ARM_COMPUTE_RETURN_ERROR_ON(src->tensor_shape().total_size() == 0);
46  ARM_COMPUTE_RETURN_ERROR_ON(dst->tensor_shape().total_size() == 0);
47  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(src, dst);
48  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_QUANTIZATION_INFO(src, dst);
49  ARM_COMPUTE_RETURN_ERROR_ON(src->tensor_shape().total_size() != dst->tensor_shape().total_size());
50 
51  return Status{};
52 }
