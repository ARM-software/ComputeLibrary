59 template <typename T, const ActFpImplParams &P>
60 void fp_neon_activation_impl(const ITensor *src,
61  ITensor *dst,
62  const ActivationLayerInfo &act_info,
63  const Window &window)
64 {
65  /** SIMD vector tag type. */
66  using ExactTagType =
67  typename arm_compute::wrapper::traits::neon_bitvector_tag_t<T, wrapper::traits::BitWidth::W128>;
68  constexpr int window_step_x = P.step_x;
69  const auto window_start_x = static_cast<int>(window.x().start());
70  const auto window_end_x = static_cast<int>(window.x().end());
71  const ActivationLayerInfo::ActivationFunction act = act_info.activation();
72  Window win_collapsed = window.collapse_if_possible(window, Window::DimZ);
73  win_collapsed.set(Window::DimX, Window::Dimension(0, 1, 1));
74  Iterator input(src, win_collapsed);
75  Iterator output(dst, win_collapsed);
76  // In case of non-aarch64, a small delta value is added to the input
77  // to prevent NAN values caused by zeros in inputs to SQRT.
78  // In case of aarh64, we call vsqrt directly, so we don't use delta.
79 #ifndef __aarch64__
80  const auto delta = wrapper::vdup_n(static_cast<T>(P.delta), ExactTagType{});
81 #else /* #ifndef __aarch64__ */
82  const auto const_inv_2 = wrapper::vdup_n(static_cast<T>(0.5f), ExactTagType{});
83  const auto const_inv_sqrt_2 = wrapper::vdup_n(static_cast<T>(0.70710678118f), ExactTagType{});
84 #endif /* __aarch64__ */
85  const auto const_1 = wrapper::vdup_n(static_cast<T>(1.f), ExactTagType{});
86  const auto const_0 = wrapper::vdup_n(static_cast<T>(0.f), ExactTagType{});
87  const auto const_6 = wrapper::vdup_n(static_cast<T>(6.f), ExactTagType{});
88  const auto const_3 = wrapper::vdup_n(static_cast<T>(3.f), ExactTagType{});
89  const auto const_inv_6 = wrapper::vdup_n(static_cast<T>(0.166666667f), ExactTagType{});
90  constexpr float soft_relu_thresh = 12.f;
91  const auto vsoft_relu_thresh = wrapper::vdup_n(static_cast<T>(soft_relu_thresh), ExactTagType{});
92  const auto va = wrapper::vdup_n(static_cast<T>(act_info.a()), ExactTagType{});
93  const auto vb = wrapper::vdup_n(static_cast<T>(act_info.b()), ExactTagType{});
94  const auto a = static_cast<T>(act_info.a());
95  const auto b = static_cast<T>(act_info.b());
96  execute_window_loop(
97  win_collapsed,
98  [&](const Coordinates &)
99  {
100  const auto input_ptr = reinterpret_cast<const T *>(input.ptr());
101  const auto output_ptr = reinterpret_cast<T *>(output.ptr());
102  wrapper::traits::neon_bitvector_t<T, wrapper::traits::BitWidth::W128> tmp;
103  // Compute S elements per iteration
104  int x = window_start_x;
105  for (; x <= (window_end_x - window_step_x); x += window_step_x)
106  {
107  const auto vin = wrapper::vloadq(input_ptr + x);
108  switch (act)
109  {
110  case ActivationLayerInfo::ActivationFunction::ABS:
111  tmp = wrapper::vabs(vin);
112  break;
113  case ActivationLayerInfo::ActivationFunction::LINEAR:
114  tmp = wrapper::vmla(vb, va, vin);
115  break;
116  case ActivationLayerInfo::ActivationFunction::LOGISTIC:
117  tmp = wrapper::vinv(wrapper::vadd(const_1, wrapper::vexpq(wrapper::vneg(vin))));
118  break;
119  case ActivationLayerInfo::ActivationFunction::RELU:
120  tmp = wrapper::vmax(const_0, vin);
121  break;
122  case ActivationLayerInfo::ActivationFunction::BOUNDED_RELU:
123  tmp = wrapper::vmin(va, wrapper::vmax(const_0, vin));
124  break;
125  case ActivationLayerInfo::ActivationFunction::LU_BOUNDED_RELU:
126  tmp = wrapper::vmin(va, wrapper::vmax(vb, vin));
127  break;
128  case ActivationLayerInfo::ActivationFunction::LEAKY_RELU:
129  tmp = wrapper::vbsl(wrapper::vcgt(vin, const_0), vin, wrapper::vmul(va, vin));
130  break;
131  case ActivationLayerInfo::ActivationFunction::SOFT_RELU:
132  tmp = wrapper::vbsl(wrapper::vcgt(vin, vsoft_relu_thresh), vin,
133  wrapper::vlog(wrapper::vadd(const_1, wrapper::vexpq(vin))));
134  break;
135  case ActivationLayerInfo::ActivationFunction::ELU:
136  tmp = wrapper::vbsl(wrapper::vcge(vin, const_0), vin,
137  wrapper::vmul(va, wrapper::vsub(wrapper::vexpq(vin), const_1)));
138  break;
139  case ActivationLayerInfo::ActivationFunction::SQRT:
140 #ifdef __aarch64__
141  tmp = wrapper::vsqrt(vin);
142 #else /* __aarch64__ */
143  {
144  const auto bitmask = wrapper::vceq(vin, wrapper::vdup_n(0.f, ExactTagType{}));
145  tmp = wrapper::vinv(wrapper::vinvsqrt(wrapper::vadd(vin, mask_float_vector(delta, bitmask))));
146  tmp = mask_float_vector(tmp, wrapper::vnot(bitmask));
147  }
148 #endif /* __aarch64__ */
149  break;
150  case ActivationLayerInfo::ActivationFunction::SQUARE:
151  tmp = wrapper::vmul(vin, vin);
152  break;
153  case ActivationLayerInfo::ActivationFunction::TANH:
154  tmp = wrapper::vmul(va, wrapper::vtanh(wrapper::vmul(vb, vin)));
155  break;
156  case ActivationLayerInfo::ActivationFunction::IDENTITY:
157  tmp = vin;
158  break;
159  case ActivationLayerInfo::ActivationFunction::HARD_SWISH:
160  tmp = wrapper::vmul(
161  vin,
162  wrapper::vmul(const_inv_6,
163  wrapper::vmin(const_6, wrapper::vmax(const_0, wrapper::vadd(vin, const_3)))));
164  break;
165  case ActivationLayerInfo::ActivationFunction::SWISH:
166  tmp = wrapper::vmul(vin, wrapper::vinv(wrapper::vadd(
167  const_1, wrapper::vexpq(wrapper::vneg(wrapper::vmul(va, vin))))));
168  break;
169 #ifdef __aarch64__
170  case ActivationLayerInfo::ActivationFunction::GELU:
171  tmp = wrapper::vmul(
172  vin,
173  wrapper::vmul(const_inv_2,
174  wrapper::vadd(const_1, wrapper::verf(wrapper::vmul(vin, const_inv_sqrt_2)))));
175  break;
176 #endif /* __aarch64__ */
177  default:
178  ARM_COMPUTE_ERROR("Unsupported activation function");
179  }
180  wrapper::vstore(output_ptr + x, tmp);
181  }
182  // Compute left-over elements
183  for (; x < window_end_x; ++x)
184  {
185  const T in = *(reinterpret_cast<const T *>(input_ptr + x));
186  T tmp;
187  switch (act)
188  {
189  case ActivationLayerInfo::ActivationFunction::ABS:
190  tmp = std::abs(in);
191  break;
192  case ActivationLayerInfo::ActivationFunction::LINEAR:
193  tmp = a * in + b;
194  break;
195  case ActivationLayerInfo::ActivationFunction::LOGISTIC:
196  tmp = static_cast<T>(1) / (static_cast<T>(1) + std::exp(-in));
197  break;
198  case ActivationLayerInfo::ActivationFunction::RELU:
199  tmp = std::max<T>(static_cast<T>(0), in);
200  break;
201  case ActivationLayerInfo::ActivationFunction::BOUNDED_RELU:
202  tmp = std::min<T>(a, std::max(static_cast<T>(0), in));
203  break;
204  case ActivationLayerInfo::ActivationFunction::LU_BOUNDED_RELU:
205  tmp = std::min<T>(a, std::max<T>(b, in));
206  break;
207  case ActivationLayerInfo::ActivationFunction::LEAKY_RELU:
208  tmp = (in > 0) ? in : a * in;
209  break;
210  case ActivationLayerInfo::ActivationFunction::SOFT_RELU:
211  tmp = (in > soft_relu_thresh) ? in : std::log(static_cast<T>(1) + std::exp(in));
212  break;
213  case ActivationLayerInfo::ActivationFunction::ELU:
214  tmp = (in >= 0) ? in : a * (std::exp(in) - 1);
215  break;
216  case ActivationLayerInfo::ActivationFunction::SQRT:
217  tmp = std::sqrt(in);
218  break;
219  case ActivationLayerInfo::ActivationFunction::SQUARE:
220  tmp = in * in;
221  break;
222  case ActivationLayerInfo::ActivationFunction::TANH:
223  tmp = a * std::tanh(b * in);
224  break;
225  case ActivationLayerInfo::ActivationFunction::IDENTITY:
226  tmp = in;
227  break;
228  case ActivationLayerInfo::ActivationFunction::HARD_SWISH:
229  tmp = in * ((std::min(std::max((in + 3), 0.0f), 6.0f)) * 0.166666667f);
230  break;
231  case ActivationLayerInfo::ActivationFunction::SWISH:
232  tmp = in / (static_cast<T>(1) + std::exp(-a * in));
233  break;
234  case ActivationLayerInfo::ActivationFunction::GELU:
235  tmp = in * static_cast<T>(0.5f * (1.0f + erff(static_cast<float>(in) / 1.41421356237f)));
236  break;
237  default:
238  ARM_COMPUTE_ERROR("Unsupported activation function");
239  }
240  *(output_ptr + x) = tmp;
241  }
242  },
243  input, output);
244 }
