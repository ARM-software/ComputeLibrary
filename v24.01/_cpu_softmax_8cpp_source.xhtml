56 void CpuSoftmaxGeneric::configure(const ITensorInfo *src, ITensorInfo *dst, float beta, int32_t axis, bool is_log)
57 {
58  // Perform validation step
59  ARM_COMPUTE_ERROR_ON_NULLPTR(src, dst);
60  ARM_COMPUTE_ERROR_THROW_ON(CpuSoftmaxGeneric::validate(src, dst, beta, axis));
61  ARM_COMPUTE_LOG_PARAMS(src, dst, beta, axis);
62 
63  const unsigned int actual_axis =
64  static_cast<unsigned int>(wrap_around(axis, static_cast<int32_t>(src->num_dimensions())));
65 
66  _needs_permute = actual_axis > 0;
67 
68  if (_needs_permute)
69  {
70  _permute_input.configure(src, &_input_permuted,
71  softmax_helpers::get_permutation_vector_from_softmax_axis(actual_axis));
72  }
73 
74  // We want to deal with a 2D input. Either it is the permuted version of the original input (4D case)
75  // or it is the original input case (2D case)
76  const ITensorInfo *tmp_input = (_needs_permute ? &_input_permuted : src);
77 
78  TensorInfo tensor_info_tmp;
79  if (is_data_type_quantized_asymmetric(src->data_type()))
80  {
81  // Create intermediate tensors shapes
82  const TensorInfo input_info = tmp_input->clone()->reset_padding().set_is_resizable(true);
83  tensor_info_tmp = input_info.clone()->set_data_type(DataType::F32);
84  }
85 
86  // Init intermediate tensors
87  _tmp = TensorInfo(tensor_info_tmp);
88 
89  // Configure kernels
90  auto sm = std::make_unique<kernels::CpuSoftmaxKernel>();
91  if (_needs_permute)
92  {
93  // The normalization kernel stores the result in a permuted output tensor
94  sm->configure(tmp_input, &_output_permuted, beta, is_log, &_tmp);
95 
96  // Re-permute the permuted output into the requested (4D) output
97  _permute_output.configure(&_output_permuted, dst,
98  softmax_helpers::get_permutation_vector_from_softmax_axis(actual_axis));
99  }
100  else
101  {
102  // Softmax 2D case
103  sm->configure(tmp_input, dst, beta, is_log, &_tmp);
104  }
105  _softmax_kernel = std::move(sm);
106 
107  if (_tmp.total_size() > 0)
108  {
109  _aux_mem[InternalTensorIdx::TMP] =
110  MemoryInfo(offset_int_vec(InternalTensorIdx::TMP), MemoryLifetime::Temporary, _tmp.total_size());
111  }
112 
113  _aux_mem[InternalTensorIdx::PERMUTED_SRC] = MemoryInfo(offset_int_vec(InternalTensorIdx::PERMUTED_SRC),
114  MemoryLifetime::Temporary, _input_permuted.total_size());
115  _aux_mem[InternalTensorIdx::PERMUTED_DST] = MemoryInfo(offset_int_vec(InternalTensorIdx::PERMUTED_DST),
116  MemoryLifetime::Temporary, _output_permuted.total_size());
117 }

120 CpuSoftmaxGeneric::validate(const ITensorInfo *src, const ITensorInfo *dst, float beta, int32_t axis, bool is_log)
121 {
122  // Perform validation step
123  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(src, dst);
124  ARM_COMPUTE_RETURN_ERROR_ON_MSG(src->num_dimensions() > 4, "Only up to 4 dimensions are supported");
125  ARM_COMPUTE_UNUSED(beta);
126  ARM_COMPUTE_RETURN_ERROR_ON(axis < static_cast<int32_t>(-src->num_dimensions()) ||
127  static_cast<int32_t>(src->num_dimensions()) <= axis);
128 
129  // Create intermediate tensor info
130  TensorInfo tensor_info_tmp;
131 
132  if (is_data_type_quantized_asymmetric(src->data_type()))
133  {
134  tensor_info_tmp = src->clone()->set_data_type(DataType::F32).set_is_resizable(true);
135  }
136 
137  const unsigned int actual_axis =
138  static_cast<unsigned int>(wrap_around(axis, static_cast<int32_t>(src->num_dimensions())));
139 
140  const bool needs_permute = actual_axis > 0;
141 
142  if (needs_permute)
143  {
144  const PermutationVector permutation_vector =
145  softmax_helpers::get_permutation_vector_from_softmax_axis(actual_axis);
146  const TensorShape permuted_shape =
147  misc::shape_calculator::compute_permutation_output_shape(*src, permutation_vector);
148  TensorInfo input_permuted(src->clone()->set_tensor_shape(permuted_shape));
149  ARM_COMPUTE_RETURN_ON_ERROR(CpuPermute::validate(src, &input_permuted, permutation_vector));
150  TensorInfo output_permuted(dst->clone()->set_tensor_shape(permuted_shape));
151  ARM_COMPUTE_RETURN_ON_ERROR(CpuPermute::validate(&output_permuted, dst, permutation_vector));
152  }
153 
154  ARM_COMPUTE_RETURN_ON_ERROR(kernels::CpuSoftmaxKernel::validate(src, dst, beta, is_log, &tensor_info_tmp));
155 
156  return Status{};
157 }

159 void CpuSoftmaxGeneric::run(ITensorPack &tensors)
160 {
161  ARM_COMPUTE_ERROR_ON_MSG(tensors.empty(), "No inputs provided");
162 
163  auto src = tensors.get_const_tensor(TensorType::ACL_SRC);
164  auto dst = tensors.get_tensor(TensorType::ACL_DST);
165 
166  CpuAuxTensorHandler tmp(offset_int_vec(InternalTensorIdx::TMP), _tmp, tensors, true);
167 
168  CpuAuxTensorHandler input_permuted(offset_int_vec(InternalTensorIdx::PERMUTED_SRC), _input_permuted, tensors, true);
169  CpuAuxTensorHandler output_permuted(offset_int_vec(InternalTensorIdx::PERMUTED_DST), _output_permuted, tensors,
170  true);
171 
172  ITensorPack softmax_pack;
173 
174  if (_needs_permute)
175  {
176  ITensorPack permute_in_pack = {{TensorType::ACL_SRC, src}, {TensorType::ACL_DST, input_permuted.get()}};
177  _permute_input.run(permute_in_pack);
178 
179  softmax_pack = {{TensorType::ACL_SRC_0, input_permuted.get()},
180  {TensorType::ACL_DST_0, output_permuted.get()},
181  {TensorType::ACL_DST_1, tmp.get()}};
182  }
183  else
184  {
185  softmax_pack = {{TensorType::ACL_SRC_0, src}, {TensorType::ACL_DST_0, dst}, {TensorType::ACL_DST_1, tmp.get()}};
186  }
187 
188  NEScheduler::get().schedule_op(_softmax_kernel.get(), Window::DimY, _softmax_kernel->window(), softmax_pack);
189 
190  if (_needs_permute)
191  {
192  ITensorPack permute_out_pack;
193  permute_out_pack.add_tensor(TensorType::ACL_SRC, output_permuted.get());
194  permute_out_pack.add_tensor(TensorType::ACL_DST, dst);
195  _permute_output.run(permute_out_pack);
196  }
197 }
