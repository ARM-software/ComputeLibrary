58 void GpuCkwMatMul::write_component_code(const ComponentGroup &comp_group,
59  GpuCkwVariableTable &vtable,
60  GpuCkwScopedKernelWriter writer) const
61 {
62  const auto root_window = comp_group.get_root_component()->ckw_component_driver()->get_window();
63 
64  GpuCkwComponentArgument *lhs =
65  vtable.declare_variable(comp_group, writer, _lhs, TensorStorageType::ClBufferUint8Ptr, "lhs");
66  GpuCkwComponentArgument *rhs =
67  vtable.declare_variable(comp_group, writer, _rhs, TensorStorageType::ClBufferUint8Ptr, "rhs");
68  GpuCkwComponentArgument *dst =
69  vtable.declare_variable(comp_group, writer, _dst, TensorStorageType::ClBufferUint8Ptr, "dst");
70 
71  // Constants
72  const int height_idx = get_data_layout_dimension_index(_lhs->data_layout(), DataLayoutDimension::HEIGHT);
73  const auto &rhs_h = writer->declare_tile("rhs_h", static_cast<int32_t>(_rhs->dimension(height_idx)));
74  const int m = static_cast<int>(_dst->dimension(1));
75  const int n = static_cast<int>(_dst->dimension(0));
76  const int k =
77  _attributes.adj_lhs() ? static_cast<int>(_lhs->tensor_shape().y()) : static_cast<int>(_lhs->tensor_shape().x());
78  const int m0 = root_window.y().step();
79  const int n0 = root_window.x().step();
80  const int k0 = _settings.k0();
81  const int partial_store_m0 = m % m0;
82  const int partial_store_n0 = n % n0;
83 
84  const auto &const_1 = writer->declare_tile("1", 1);
85  auto &const_0 = writer->declare_tile("0", 0);
86  auto &k0_tile = writer->declare_tile("k0", k0);
87  auto &k_tile = writer->declare_tile("k", k);
88 
89  auto &gid_0 = writer->declare_tile("gid_0", ckw::DataType::Int32);
90  auto &gid_1 = writer->declare_tile("gid_1", ckw::DataType::Int32);
91  auto &gid_2 = writer->declare_tile("gid_2", ckw::DataType::Int32);
92 
93  writer->op_get_global_id(gid_0, 0);
94  writer->op_get_global_id(gid_1, 1);
95  writer->op_get_global_id(gid_2, 2);
96 
97  auto &x = writer->declare_tile("x", ckw::DataType::Int32);
98  auto &y = writer->declare_tile("y", ckw::DataType::Int32);
99  auto &z = writer->declare_tile("z", ckw::DataType::Int32);
100 
101  get_coord(writer, x, gid_0, n0, partial_store_n0, "gid_x_", const_0);
102  get_coord(writer, y, gid_1, m0, partial_store_m0, "gid_y_", const_0);
103  get_coord(writer, z, gid_2, 1, 0, "gid_z_", const_0);
104 
105  TensorTileSampler lhs_sampler;
106  lhs_sampler.height(m0);
107  lhs_sampler.width(k0);
108  lhs_sampler.format(TensorSamplerFormat::C_W_H);
109  lhs_sampler.address_mode_x(TensorSamplerAddressModeX::None);
110  lhs_sampler.address_mode_y(TensorSamplerAddressModeY::None);
111  lhs_sampler.address_mode_z(TensorSamplerAddressModeZ::None);
112 
113  TensorTileSampler rhs_sampler;
114  rhs_sampler.height(k0);
115  rhs_sampler.width(n0);
116  rhs_sampler.format(TensorSamplerFormat::C_WH_1);
117  rhs_sampler.address_mode_x(TensorSamplerAddressModeX::None);
118  rhs_sampler.address_mode_y(TensorSamplerAddressModeY::None);
119  rhs_sampler.address_mode_z(TensorSamplerAddressModeZ::None);
120 
121  TensorTileSampler dst_sampler;
122  dst_sampler.width(n0);
123  dst_sampler.height(m0);
124  dst_sampler.format(TensorSamplerFormat::C_W_H);
125  dst_sampler.address_mode_x(TensorSamplerAddressModeX::OverlappingMin);
126  dst_sampler.address_mode_y(TensorSamplerAddressModeY::None);
127  dst_sampler.address_mode_z(TensorSamplerAddressModeZ::None);
128  dst_sampler.x(x);
129  dst_sampler.y(y);
130  dst_sampler.z(z);
131  dst_sampler.b(const_0);
132 
133  if (!dst->has_tile())
134  {
135  auto &dst_tile = writer->declare_tile("dst_tile", ckw::TileInfo(to_ckw(_dst->data_type()), m0, n0));
136  dst->init_virtual_tensor(dst_tile, dst_sampler);
137  }
138  auto &dst_tile = dst->tile();
139 
140  // Initialize the accumulators
141  writer->op_assign(dst_tile, const_0);
142 
143  auto &rhs_z = writer->declare_tile("rhs_z", ckw::DataType::Int32);
144  writer->op_binary_expression(rhs_z, z, BinaryOp::Mul, rhs_h);
145 
146  auto &k_i = writer->declare_tile("k_i", ckw::DataType::Int32);
147  auto &k_limit = writer->declare_tile("k_limit", k - k0);
148 
149  auto &x_i = writer->declare_tile("x_i", ckw::DataType::Int32);
150  writer->op_assign(x_i, const_0);
151 
152  writer->op_assign(k_i, const_0);
153 
154  // *INDENT-OFF*
155  // clang-format off
156  writer->op_for_loop(k_i, BinaryOp::LessEqual, k_limit, k_i, AssignmentOp::Increment, k0_tile,
157  [&]()
158  {
159  //Initialize tiles
160  // lhs_tile
161  auto &a = writer->declare_tile("a", ckw::TileInfo(to_ckw(_lhs->data_type()), m0, k0));
162  // rhs_tile
163  auto &b = writer->declare_tile("b", ckw::TileInfo(to_ckw(_rhs->data_type()), n0, k0));
164  writer->op_assign(a, const_0);
165  writer->op_assign(b, const_0);
166 
167  // Loading the tiles
168  // LHS
169  lhs_sampler.x(x_i);
170  lhs_sampler.y(y);
171  lhs_sampler.z(z);
172  lhs_sampler.b(const_0);
173  writer->op_load(a, lhs->tensor(), lhs_sampler);
174 
175  // RHS
176  auto &y_i = writer->declare_tile("y_i", ckw::DataType::Int32);
177  writer->op_binary_expression(y_i, x, BinaryOp::Add, rhs_z);
178  rhs_sampler.x(k_i);
179  rhs_sampler.y(y_i);
180  rhs_sampler.z(const_0);
181  rhs_sampler.b(const_0);
182  writer->op_load(b, rhs->tensor(), rhs_sampler);
183 
184  // Perform Matmul
185  writer->op_binary_expression(dst_tile, a, BinaryOp::MatMul_Nt_T, b);
186  writer->op_binary_expression(x_i, x_i, BinaryOp::Add, k0_tile);
187  });
188 // *INDENT-ON*
189  // clang-format on
190 
191  // Handling leftovers
192  if (k % k0 != 0)
193  {
194  // *INDENT-OFF*
195  // clang-format off
196  writer->op_for_loop(k_i, BinaryOp::Less, k_tile, k_i, AssignmentOp::Increment, const_1,
197  [&]()
198  {
199  //Initialize tiles
200  // lhs_tile
201  auto &a =
202  writer->declare_tile("a_leftover", ckw::TileInfo(to_ckw(_lhs->data_type()), m0, 1));
203  // rhs_tile
204  auto &b =
205  writer->declare_tile("b_leftover", ckw::TileInfo(to_ckw(_rhs->data_type()), n0, 1));
206  writer->op_assign(a, const_0);
207  writer->op_assign(b, const_0);
208 
209  // Loading the tiles
210  // LHS
211  lhs_sampler.x(x_i);
212  lhs_sampler.y(y);
213  lhs_sampler.z(z);
214  lhs_sampler.b(const_0);
215  writer->op_load(a, lhs->tensor(), lhs_sampler);
216 
217  // RHS
218  auto &y_i = writer->declare_tile("y_i_leftover", ckw::DataType::Int32);
219  writer->op_binary_expression(y_i, x, BinaryOp::Add, rhs_z);
220  rhs_sampler.x(k_i);
221  rhs_sampler.y(y_i);
222  rhs_sampler.z(const_0);
223  rhs_sampler.b(const_0);
224  writer->op_load(b, rhs->tensor(), rhs_sampler);
225 
226  // Perform Matmul
227  writer->op_binary_expression(dst_tile, a, BinaryOp::MatMul_Nt_T, b);
228  writer->op_binary_expression(x_i, x_i, BinaryOp::Add, const_1);
229  });
230 // *INDENT-ON*
231  // clang-format on
232  }
233 }

235 Window GpuCkwMatMul::get_window() const
236 {
237  ARM_COMPUTE_ERROR_ON_MSG(_dst->tensor_shape().total_size() == 0U, "Destination tensor is not initialized");
238 
239  const int m = _dst->dimension(1);
240  const int n = _dst->dimension(0);
241  const bool adj_lhs = _attributes.adj_lhs();
242 
243  int m0 = adj_lhs ? adjust_vec_size(_settings.m0(), m) : std::min(_settings.m0(), m);
244  int n0 = adjust_vec_size(_settings.n0(), n);
245 
246  // Configure kernel window
247  Window win = calculate_max_window(_dst->tensor_shape(), Steps(n0, m0));
248  win = win.collapse(win, Window::DimZ);
249 
250  return win;
251 }
