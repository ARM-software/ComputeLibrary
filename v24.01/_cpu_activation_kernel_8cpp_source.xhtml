130 Status validate_arguments(const ITensorInfo *src, const ITensorInfo *dst, const ActivationLayerInfo &activation_info)
131 {
132  ARM_COMPUTE_RETURN_ERROR_ON_CPU_F16_UNSUPPORTED(src);
133  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(src, 1, DataType::QASYMM8_SIGNED, DataType::QASYMM8,
134  DataType::QSYMM16, DataType::F16, DataType::F32);
135 
136  const auto *uk = CpuActivationKernel::get_implementation(ActivationDataTypeISASelectorData{
137  src->data_type(), CPUInfo::get().get_cpu_model(), CPUInfo::get().get_isa(), activation_info.activation()});
138  ARM_COMPUTE_RETURN_ERROR_ON(uk == nullptr || uk->ukernel == nullptr);
139 
140  const DataType data_type = src->data_type();
141  const QuantizationInfo &oq_info = (dst != nullptr) ? dst->quantization_info() : src->quantization_info();
142  const ActivationLayerInfo::ActivationFunction f_act = activation_info.activation();
143 
144  ARM_COMPUTE_RETURN_ERROR_ON_MSG(
145  is_data_type_quantized_asymmetric(data_type) &&
146  (std::find(std::begin(qasymm8_activations), std::end(qasymm8_activations), f_act) ==
147  std::end(qasymm8_activations)),
148  "For QASYMM8 only hard swish, leaky relu, tanh, logistic, relu and lower/upper bounded relu are supported");
149 
150  ARM_COMPUTE_RETURN_ERROR_ON_MSG(is_data_type_quantized_symmetric(data_type) &&
151  (std::find(std::begin(qsymm16_activations), std::end(qsymm16_activations),
152  f_act) == std::end(qsymm16_activations)),
153  "For QSYMM16 only tanh and logistic are supported");
154  ARM_COMPUTE_RETURN_ERROR_ON((data_type == DataType::QASYMM8 || data_type == DataType::QASYMM16) &&
155  (f_act == ActivationLayerInfo::ActivationFunction::TANH) &&
156  (oq_info != QuantizationInfo(1.f / 128.f, 128)));
157  ARM_COMPUTE_RETURN_ERROR_ON((data_type == DataType::QASYMM8 || data_type == DataType::QASYMM16) &&
158  (f_act == ActivationLayerInfo::ActivationFunction::LOGISTIC) &&
159  (oq_info != QuantizationInfo(1.f / 256.f, 0)));
160 
161  ARM_COMPUTE_RETURN_ERROR_ON(data_type == DataType::QASYMM8_SIGNED &&
162  (f_act == ActivationLayerInfo::ActivationFunction::TANH) &&
163  (oq_info != QuantizationInfo(1.f / 128.f, 0)));
164  ARM_COMPUTE_RETURN_ERROR_ON(data_type == DataType::QASYMM8_SIGNED &&
165  (f_act == ActivationLayerInfo::ActivationFunction::LOGISTIC) &&
166  (oq_info != QuantizationInfo(1.f / 256.f, -128)));
167 
168  ARM_COMPUTE_RETURN_ERROR_ON(is_data_type_quantized_symmetric(data_type) &&
169  (f_act == ActivationLayerInfo::ActivationFunction::TANH) &&
170  (oq_info != QuantizationInfo(1.f / 32768.f, 0)));
171  ARM_COMPUTE_RETURN_ERROR_ON(is_data_type_quantized_symmetric(data_type) &&
172  (f_act == ActivationLayerInfo::ActivationFunction::LOGISTIC) &&
173  (oq_info != QuantizationInfo(1.f / 32768.f, 0)));
174 
175  // Checks performed when dst is configured
176  if ((dst != nullptr) && (dst->total_size() != 0))
177  {
178  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_SHAPES(src, dst);
179  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(src, dst);
180  }
181 
182  return Status{};
183 }

198 #ifdef __aarch64__
199 void init_lut(ActivationLayerInfo::ActivationFunction act_func,
200  DataType data_type,
201  const UniformQuantizationInfo &qi_in,
202  const UniformQuantizationInfo &qi_out,
203  ActivationLayerInfo::LookupTable256 &lut,
204  float a,
205  float b)
206 {
207  for (size_t i = 0; i < lut.size(); ++i)
208  {
209  float tmp_f =
210  (data_type == DataType::QASYMM8) ? dequantize_qasymm8(i, qi_in) : dequantize_qasymm8_signed(i, qi_in);
211  switch (act_func)
212  {
213  case ActivationLayerInfo::ActivationFunction::HARD_SWISH:
214  tmp_f = tmp_f * ((std::min(std::max((tmp_f + 3), 0.0f), 6.0f)) * 0.166666667f);
215  break;
216  case ActivationLayerInfo::ActivationFunction::LEAKY_RELU:
217  tmp_f = tmp_f > 0 ? tmp_f : tmp_f * a;
218  break;
219  case ActivationLayerInfo::ActivationFunction::LOGISTIC:
220  tmp_f = 1.f / (1.f + std::exp(-tmp_f));
221  break;
222  case ActivationLayerInfo::ActivationFunction::ABS:
223  tmp_f = std::abs(tmp_f);
224  break;
225  case ActivationLayerInfo::ActivationFunction::LINEAR:
226  tmp_f = a * tmp_f + b;
227  break;
228  case ActivationLayerInfo::ActivationFunction::BOUNDED_RELU:
229  tmp_f = std::min<>(a, std::max(0.f, tmp_f));
230  break;
231  case ActivationLayerInfo::ActivationFunction::LU_BOUNDED_RELU:
232  tmp_f = std::min<>(a, std::max<>(b, tmp_f));
233  break;
234  case ActivationLayerInfo::ActivationFunction::SOFT_RELU:
235  tmp_f = (tmp_f > 12.f) ? tmp_f : std::log(1.f + std::exp(tmp_f));
236  break;
237  case ActivationLayerInfo::ActivationFunction::ELU:
238  tmp_f = (tmp_f >= 0) ? tmp_f : a * (std::exp(tmp_f) - 1);
239  break;
240  case ActivationLayerInfo::ActivationFunction::SQRT:
241  tmp_f = std::sqrt(tmp_f);
242  break;
243  case ActivationLayerInfo::ActivationFunction::SQUARE:
244  tmp_f = tmp_f * tmp_f;
245  break;
246  case ActivationLayerInfo::ActivationFunction::TANH:
247  tmp_f = a * std::tanh(b * tmp_f);
248  break;
249  case ActivationLayerInfo::ActivationFunction::IDENTITY:
250  break;
251  case ActivationLayerInfo::ActivationFunction::SWISH:
252  tmp_f = tmp_f / (1.f + std::exp(-a * tmp_f));
253  break;
254  case ActivationLayerInfo::ActivationFunction::GELU:
255  tmp_f = tmp_f * (0.5f * (1.0f + erff(tmp_f / 1.41421356237f)));
256  break;
257  default:
258  ARM_COMPUTE_ERROR("Not supported");
259  tmp_f = 0;
260  break;
261  }
262  lut[i] =
263  (data_type == DataType::QASYMM8) ? quantize_qasymm8(tmp_f, qi_out) : quantize_qasymm8_signed(tmp_f, qi_out);
264  }
265 }
266 #endif // __aarch64__

269 void CpuActivationKernel::configure(const ITensorInfo *src, ITensorInfo *dst, ActivationLayerInfo activation_info)
270 {
271  ARM_COMPUTE_UNUSED(dst);
272  ARM_COMPUTE_ERROR_ON_NULLPTR(src);
273  ARM_COMPUTE_ERROR_THROW_ON(validate_arguments(src, dst, activation_info));
274 
275  const auto uk = CpuActivationKernel::get_implementation(ActivationDataTypeISASelectorData{
276  src->data_type(), CPUInfo::get().get_cpu_model(), CPUInfo::get().get_isa(), activation_info.activation()});
277  if (dst != nullptr)
278  {
279  // dst auto inizialitation if not yet initialized
280  auto_init_if_empty(*dst, *src->clone());
281  }
282 
283  ARM_COMPUTE_ERROR_ON_NULLPTR(uk);
284 
285  _run_method = uk->ukernel;
286  _name = std::string("CpuActivationKernel").append("/").append(uk->name);
287 
288 #ifdef __aarch64__
289  // Initialise lut_manager
290  LUTManager &lut_manager = LUTManager::get_instance();
291 
292  if ((src->data_type() == DataType::QASYMM8 || src->data_type() == DataType::QASYMM8_SIGNED) &&
293  activation_info.activation() != ActivationFunction::RELU)
294  {
295  ActivationLayerInfo::LookupTable256 tmp_lut;
296  init_lut(activation_info.activation(), src->data_type(), src->quantization_info().uniform(),
297  (dst) ? dst->quantization_info().uniform() : src->quantization_info().uniform(), tmp_lut,
298  activation_info.a(), activation_info.b());
299  activation_info.setLookupTable256(tmp_lut);
300  }
301 
302  if (src->data_type() == DataType::F16 &&
303  activation_info.activation() == ActivationLayerInfo::ActivationFunction::LOGISTIC)
304  {
305  const LUTInfo info = {activation_info.activation(), src->data_type(), src->quantization_info()};
306  activation_info.setLookupTable65536((lut_manager.get_lut_table(info)));
307  }
308 #endif // __aarch64__
309  _act_info = activation_info;
310 
311  Window win;
312 
313  // Use squashed window
314  std::tie(win, _split_dimension) = calculate_squashed_or_max_window(*src);
315  ICPPKernel::configure(win);
316 }

356  ARM_COMPUTE_ERROR_ON(_run_method == nullptr);

358  const ITensor *src = tensors.get_const_tensor(TensorType::ACL_SRC);
359  ITensor *dst = tensors.get_tensor(TensorType::ACL_DST);
360
