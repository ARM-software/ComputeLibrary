85 inline Window Window::split_window(const size_t dimension, const size_t id, const size_t total) const
86 {
87  ARM_COMPUTE_ERROR_ON(id >= total);
88  ARM_COMPUTE_ERROR_ON(dimension >= Coordinates::num_max_dimensions);
89 
90  Window out;
91 
92  for(size_t d = 0; d < Coordinates::num_max_dimensions; ++d)
93  {
94  if(d == dimension)
95  {
96  int start = _dims[d].start();
97  int end = _dims[d].end();
98  int per_sub_window = (num_iterations(d) / total) * _dims[d].step();
99 
100  start += id * per_sub_window;
101 
102  if(id != total - 1)
103  {
104  end = start + per_sub_window;
105  }
106 
107  out.set(d, Dimension(start, end, _dims[d].step()));
108  }
109  else
110  {
111  out.set(d, _dims[d]);
112  }
113  }
114 
115  return out;
116 }
