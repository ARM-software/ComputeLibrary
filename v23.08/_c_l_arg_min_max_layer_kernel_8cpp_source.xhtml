43 Status validate_arguments(const ITensorInfo *input, const ITensorInfo *output, unsigned int axis, ReductionOperation op)
44 {
45  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(input, output);
46  ARM_COMPUTE_RETURN_ERROR_ON_F16_UNSUPPORTED(input);
47  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(input, 1, DataType::QASYMM8, DataType::QASYMM8_SIGNED, DataType::S32, DataType::F16, DataType::F32);
48  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(output, 1, DataType::S32, DataType::S64);
49 
50  ARM_COMPUTE_RETURN_ERROR_ON_MSG(op != ReductionOperation::ARG_IDX_MAX && op != ReductionOperation::ARG_IDX_MIN, "Only ARG_IDX_MAX and ARG_IDX_MIN are supported");
51  ARM_COMPUTE_RETURN_ERROR_ON_MSG(axis >= TensorShape::num_max_dimensions, "Reduction axis greater than max number of dimensions");
52  ARM_COMPUTE_RETURN_ERROR_ON_MSG(axis > 3, "Unsupported reduction axis");
53 
54  if(output->total_size() != 0)
55  {
56  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(output, 1, DataType::U32, DataType::S32, DataType::S64, DataType::U64);
57  }
58 
59  return Status{};
60 }

74 void CLArgMinMaxLayerKernel::configure(const CLCompileContext &compile_context, const ICLTensor *input, ICLTensor *output, unsigned int axis, ReductionOperation op)
75 {
76  ARM_COMPUTE_ERROR_ON_NULLPTR(input, output);
77 
78  TensorShape output_shape{ input->info()->tensor_shape() };
79  output_shape.set(axis, 1);
80  auto_init_if_empty(*output->info(), input->info()->clone()->set_tensor_shape(output_shape).set_data_type(DataType::S32).reset_padding().set_is_resizable(true));
81 
82  ARM_COMPUTE_ERROR_THROW_ON(validate_arguments(input->info(), output->info(), axis, op));
83 
84  auto padding_info = get_padding_info({ input, output });
85 
86  _input = input;
87  _output = output;
88  _reduction_axis = axis;
89  _op = op;
90 
91  // Set build options
92  const auto adjusted_vector_size = adjust_vec_size(16U, input->info()->dimension(0));
93  const auto vector_size = (adjusted_vector_size == 3U && axis == 0U) ? 2U : adjusted_vector_size; // the opencl kernel only supports sizes 2, 4, 8 and 16.
94 
95  CLBuildOptions build_opts;
96  build_opts.add_option("-DDATA_TYPE=" + get_cl_type_from_data_type(input->info()->data_type()));
97  build_opts.add_option("-DVEC_SIZE_LEFTOVER=" + support::cpp11::to_string(input->info()->dimension(0) % vector_size));
98  build_opts.add_option("-DVEC_SIZE=" + support::cpp11::to_string(vector_size));
99  build_opts.add_option_if(is_data_type_float(input->info()->data_type()), "-DFLOAT_DATA_TYPE");
100  build_opts.add_option_if_else(op == ReductionOperation::ARG_IDX_MAX, "-DARG_MAX", "-DARG_MIN");
101  build_opts.add_option("-DDATA_TYPE_OUTPUT=" + get_cl_type_from_data_type(output->info()->data_type()));
102  build_opts.add_option("-DCOND_DATA_TYPE=" + get_cl_select_type_from_data_type(input->info()->data_type()));
103  build_opts.add_option("-DUNROLL_WITH_PRAGMA=1");
104 
105  // Create kernel
106  std::string kernel_axis_name;
107  switch(axis)
108  {
109  case 0:
110  build_opts.add_option("-DWIDTH=" + support::cpp11::to_string(input->info()->dimension(0)));
111  kernel_axis_name = "x";
112  break;
113  case 1:
114  build_opts.add_option("-DHEIGHT=" + support::cpp11::to_string(input->info()->dimension(1)));
115  kernel_axis_name = "y";
116  break;
117  case 2:
118  build_opts.add_option("-DDEPTH=" + support::cpp11::to_string(input->info()->dimension(2)));
119  kernel_axis_name = "z";
120  break;
121  case 3:
122  build_opts.add_option("-DDEPTH=" + support::cpp11::to_string(input->info()->dimension(2)));
123  build_opts.add_option("-DBATCH=" + support::cpp11::to_string(input->info()->dimension(3)));
124  kernel_axis_name = "w";
125  break;
126  default:
127  ARM_COMPUTE_ERROR("Not supported");
128  }
129  _kernel = create_kernel(compile_context, "arg_min_max_" + kernel_axis_name, build_opts.options());
130 
131  // Configure kernel window
132  Window win = calculate_max_window(*input->info(), Steps(vector_size));
133  ICLKernel::configure_internal(win);
134 
135  ARM_COMPUTE_ERROR_ON(has_padding_changed(padding_info));
136 }

144 void CLArgMinMaxLayerKernel::run(const Window &window, cl::CommandQueue &queue)
145 {
146  ARM_COMPUTE_ERROR_ON_UNCONFIGURED_KERNEL(this);
147  ARM_COMPUTE_ERROR_ON_INVALID_SUBWINDOW(IKernel::window(), window);
148 
149  switch(_reduction_axis)
150  {
151  case 0:
152  {
153  // Set out window
154  Window out_window(window);
155  Window in_window(window);
156  out_window.set(Window::DimX, Window::Dimension(0, 0, 0));
157  in_window.set(Window::DimX, Window::Dimension(0, _input->info()->dimension(0), _input->info()->dimension(0)));
158  in_window.set(Window::DimY, Window::Dimension(0, _input->info()->dimension(1), 1u));
159 
160  // Get first input and output slices
161  Window in_slice = in_window.first_slice_window_2D();
162  Window out_slice = out_window.first_slice_window_2D();
163  do
164  {
165  unsigned int idx = 0;
166  add_2D_tensor_argument(idx, _input, in_slice);
167  add_2D_tensor_argument(idx, _output, out_slice);
168  enqueue(queue, *this, in_slice, lws_hint());
169  }
170  while(in_window.slide_window_slice_2D(in_slice) && out_window.slide_window_slice_2D(out_slice));
171  }
172  break;
173  case 1:
174  {
175  // Get first input and output slices
176  Window window_in{ window };
177  window_in.set(Window::DimY, Window::Dimension(0, _input->info()->dimension(1), _input->info()->dimension(1)));
178  Window in_slice = window_in.first_slice_window_2D();
179  Window out_slice = window.first_slice_window_2D();
180 
181  do
182  {
183  unsigned int idx = 0;
184  add_2D_tensor_argument(idx, _input, in_slice);
185  add_2D_tensor_argument(idx, _output, out_slice);
186  enqueue(queue, *this, in_slice, lws_hint());
187  }
188  while(window_in.slide_window_slice_2D(in_slice) && window.slide_window_slice_2D(out_slice));
189  }
190  break;
191  case 2:
192  {
193  // Get first input and output slices
194  Window window_in{ window };
195  window_in.set(Window::DimZ, Window::Dimension(0, _input->info()->dimension(2), _input->info()->dimension(2)));
196  Window in_slice = window_in.first_slice_window_3D();
197  Window out_slice = window.first_slice_window_3D();
198 
199  do
200  {
201  unsigned int idx = 0;
202  add_3D_tensor_argument(idx, _input, in_slice);
203  add_3D_tensor_argument(idx, _output, out_slice);
204  enqueue(queue, *this, in_slice, lws_hint());
205  }
206  while(window_in.slide_window_slice_3D(in_slice) && window.slide_window_slice_3D(out_slice));
207  }
208  break;
209  case 3:
210  {
211  // Get first input and output slices
212  Window window_in{ window };
213  window_in.set(3, Window::Dimension(0, 1, 1));
214  Window in_slice = window_in.first_slice_window_4D();
215  Window out_slice = window.first_slice_window_4D();
216 
217  do
218  {
219  unsigned int idx = 0;
220  add_4D_tensor_argument(idx, _input, in_slice);
221  add_4D_tensor_argument(idx, _output, out_slice);
222  enqueue(queue, *this, in_slice, lws_hint());
223  }
224  while(window_in.slide_window_slice_4D(in_slice) && window.slide_window_slice_4D(out_slice));
225  }
226  break;
227  default:
228  ARM_COMPUTE_ERROR("Not supported");
229  }
