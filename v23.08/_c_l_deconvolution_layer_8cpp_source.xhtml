67 void CLDeconvolutionLayer::configure(const CLCompileContext &compile_context, ICLTensor *input, ICLTensor *weights, const ICLTensor *bias, ICLTensor *output, const PadStrideInfo &deconv_info,
68  const WeightsInfo &weights_info)
69 {
70  ARM_COMPUTE_ERROR_ON_NULLPTR(input, weights, output);
71  ARM_COMPUTE_LOG_PARAMS(input, weights, bias, output, deconv_info, weights_info);
72 
73  switch(CLDeconvolutionLayer::get_deconvolution_method(input->info(), weights->info(), nullptr, output->info(), deconv_info, weights_info))
74  {
75  case DeconvolutionMethod::DIRECT:
76  {
77  auto op = std::make_unique<opencl::ClTransposedConvolution>();
78  op->configure(compile_context, input->info(), weights->info(), bias != nullptr ? bias->info() : nullptr, output->info(), deconv_info);
79 
80  _impl->src = input;
81  _impl->weights = weights;
82  _impl->biases = bias;
83  _impl->dst = output;
84 
85  _impl->op = std::move(op);
86  break;
87  }
88  case DeconvolutionMethod::UPSCALE_CONV2D:
89  {
90  auto f = std::make_unique<CLDirectDeconvolutionLayer>();
91  f->configure(compile_context, input, weights, bias, output, deconv_info, weights_info);
92  _function = std::move(f);
93  break;
94  }
95  case DeconvolutionMethod::GEMM:
96  {
97  auto f = std::make_unique<CLGEMMDeconvolutionLayer>(_memory_manager);
98  f->configure(compile_context, input, weights, bias, output, deconv_info);
99  _function = std::move(f);
100  break;
101  }
102  default:
103  ARM_COMPUTE_ERROR("Not supported.");
104  break;
105  }
106 }

108 Status CLDeconvolutionLayer::validate(const ITensorInfo *input, const ITensorInfo *weights, const ITensorInfo *bias, ITensorInfo *output, const PadStrideInfo &deconv_info,
109  const WeightsInfo &weights_info)
110 {
111  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(input, weights, output);
112  switch(CLDeconvolutionLayer::get_deconvolution_method(input, weights, bias, output, deconv_info, weights_info))
113  {
114  case DeconvolutionMethod::DIRECT:
115  {
116  // Validate transposed convolution operator
117  ARM_COMPUTE_RETURN_ON_ERROR(opencl::ClTransposedConvolution::validate(input, weights, bias, output, deconv_info));
118  break;
119  }
120  case DeconvolutionMethod::UPSCALE_CONV2D:
121  {
122  // Validate direct convolution layer
123  ARM_COMPUTE_RETURN_ON_ERROR(CLDirectDeconvolutionLayer::validate(input, weights, bias, output, deconv_info, weights_info));
124  break;
125  }
126  case DeconvolutionMethod::GEMM:
127  {
128  // Validate gemm-based convolution layer
129  ARM_COMPUTE_RETURN_ON_ERROR(CLGEMMDeconvolutionLayer::validate(input, weights, bias, output, deconv_info));
130  break;
131  }
132  default:
133  ARM_COMPUTE_ERROR("Not supported.");
134  break;
135  }
136 
137  return Status{};
138 }

140 DeconvolutionMethod CLDeconvolutionLayer::get_deconvolution_method(const ITensorInfo *input, const ITensorInfo *weights, const ITensorInfo *bias, ITensorInfo *output, const PadStrideInfo &deconv_info,
141  const WeightsInfo &weights_info)
142 {
143  ARM_COMPUTE_UNUSED(output, bias, weights_info);
144 
145  if(is_data_type_quantized_per_channel(weights->data_type()))
146  {
147  return DeconvolutionMethod::UPSCALE_CONV2D;
148  }
149 
150  const DataLayout data_layout = input->data_layout();
151 
152  const size_t idx_w = get_data_layout_dimension_index(data_layout, DataLayoutDimension::WIDTH);
153  const size_t idx_h = get_data_layout_dimension_index(data_layout, DataLayoutDimension::HEIGHT);
154  const size_t idx_n = get_data_layout_dimension_index(data_layout, DataLayoutDimension::BATCHES);
155  const size_t ofm = weights->tensor_shape()[idx_n];
156 
157  if(weights->dimension(idx_w) != deconv_info.stride().first || weights->dimension(idx_h) != deconv_info.stride().second)
158  {
159  // We observe better performance for FP32 types only when ofm <= 16.
160  // A better heuristic is required for selecting the method for FP16 data types.
161  if(input->data_layout() == DataLayout::NHWC && !((input->data_type() == DataType::F32) && (ofm > 16)))
162  {
163  return DeconvolutionMethod::DIRECT;
164  }
165  else
166  {
167  return DeconvolutionMethod::UPSCALE_CONV2D;
168  }
169  }
170 
171  return DeconvolutionMethod::GEMM;
172 }

174 void CLDeconvolutionLayer::run()
175 {

177 
178  if(_impl->op != nullptr)
179  {
180  // Optimized Operator will be used
181  ITensorPack pack;
182

184  pack.add_tensor(TensorType::ACL_SRC_1, _impl->weights);
185  pack.add_tensor(TensorType::ACL_SRC_2, _impl->biases);
186  pack.add_tensor(TensorType::ACL_DST, _impl->dst);
187 
188  _impl->op->run(pack);
189  }
190  else
191  {
192  _function->run();
