32 template <typename ScalarType>
33 inline ScalarType elementwise_op_scalar_imp(ElementWiseUnary op, const ScalarType &a)
34 {
35  switch(op)
36  {
37  case ElementWiseUnary::RSQRT:
38  return 1 / sqrt(a);
39  case ElementWiseUnary::EXP:
40  return std::exp(a);
41  case ElementWiseUnary::NEG:
42  return -a;
43  case ElementWiseUnary::LOG:
44  return std::log(a);
45  case ElementWiseUnary::ABS:
46  return std::abs(a);
47  case ElementWiseUnary::ROUND:
48  return support::cpp11::nearbyint(a);
49  case ElementWiseUnary::SIN:
50  return std::sin(a);
51  default:
52  ARM_COMPUTE_ERROR("NOT_SUPPORTED!");
53  }
54 }

58 {
59  switch(op)

62  return wrapper::vinvsqrt(a);
63  case ElementWiseUnary::EXP:
64  return wrapper::vexpq(a);
65  case ElementWiseUnary::NEG:
66  return wrapper::vneg(a);
67  case ElementWiseUnary::LOG:
68  return wrapper::vlog(a);
69  case ElementWiseUnary::ABS:
70  return wrapper::vabs(a);
71  case ElementWiseUnary::ROUND:
72  return wrapper::vround(a);
73  case ElementWiseUnary::SIN:
74  return wrapper::vsin(a);
75  default:
76  ARM_COMPUTE_ERROR("NOT_SUPPORTED!");
77  }

80 template <typename ScalarType>
81 void elementwise_op(const ITensor *in, ITensor *out, const Window &window, ElementWiseUnary op)
82 {
83  const int window_step_x = 16 / sizeof(ScalarType);
84  const auto window_start_x = static_cast<int>(window.x().start());
85  const auto window_end_x = static_cast<int>(window.x().end());
86 
87  Window win = window;
88  win.set(Window::DimX, Window::Dimension(0, 1, 1));
89 
90  Iterator input(in, win);
91  Iterator output(out, win);
92 
93  execute_window_loop(win, [&](const Coordinates &)
94  {
95  auto output_ptr = reinterpret_cast<ScalarType *>(output.ptr());
96  const auto input_ptr = reinterpret_cast<const ScalarType *>(input.ptr());
97 
98  int x = window_start_x;
99  for(; x <= window_end_x - window_step_x; x += window_step_x)
100  {
101  wrapper::vstore(output_ptr + x, elementwise_op_imp<ScalarType>(op, wrapper::vloadq(input_ptr + x)));
102  }
103  for(; x < window_end_x; ++x)
104  {
105  *(output_ptr + x) = elementwise_op_scalar_imp(op, *(input_ptr + x));
106  }
107  },
108  input, output);
109 }

116 template <>
117 void elementwise_op<int8_t>(const ITensor *in, ITensor *out, const Window &window, ElementWiseUnary op)
118 {
119  const int window_step_x = 16;
120  const auto window_start_x = static_cast<int>(window.x().start());
121  const auto window_end_x = static_cast<int>(window.x().end());
122  const UniformQuantizationInfo qi_in = in->info()->quantization_info().uniform();
123  const UniformQuantizationInfo qi_out = out->info()->quantization_info().uniform();
124  const auto min_clamped_value = vdupq_n_f32((-128 - qi_out.offset) * qi_out.scale);
125  const auto max_clamped_value = vdupq_n_f32((127 - qi_out.offset) * qi_out.scale);
126  Window win = window;
127  win.set(Window::DimX, Window::Dimension(0, 1, 1));
128 
129  Iterator input(in, win);
130  Iterator output(out, win);
131 
132  execute_window_loop(win, [&](const Coordinates &)
133  {
134  int8x16_t vout;
135  auto output_ptr = reinterpret_cast<int8_t *>(output.ptr());
136  const auto input_ptr = reinterpret_cast<const int8_t *>(input.ptr());
137  const auto vconst_0_f32 = vdupq_n_f32(0);
138  auto clamped_value = (op == ElementWiseUnary::LOG) ? min_clamped_value : max_clamped_value;
139 
140  int x = window_start_x;
141  for(; x <= window_end_x - window_step_x; x += window_step_x)
142  {
143  const auto vin = wrapper::vloadq(input_ptr + x);
144 
145  // De-quantize
146  const auto vin_deq = vdequantize(vin, qi_in);
147 
148  // Perform activation
149  float32x4x4_t vtmp_deq =
150  {
151  {
152  elementwise_op_imp<float>(op, vin_deq.val[0]),
153  elementwise_op_imp<float>(op, vin_deq.val[1]),
154  elementwise_op_imp<float>(op, vin_deq.val[2]),
155  elementwise_op_imp<float>(op, vin_deq.val[3]),
156  }
157  };
158 
159  if((op == ElementWiseUnary::LOG) || (op == ElementWiseUnary::RSQRT))
160  {
161  vtmp_deq.val[0] = vbslq_f32(vcleq_f32(vin_deq.val[0], vconst_0_f32), clamped_value, vtmp_deq.val[0]);
162  vtmp_deq.val[1] = vbslq_f32(vcleq_f32(vin_deq.val[1], vconst_0_f32), clamped_value, vtmp_deq.val[1]);
163  vtmp_deq.val[2] = vbslq_f32(vcleq_f32(vin_deq.val[2], vconst_0_f32), clamped_value, vtmp_deq.val[2]);
164  vtmp_deq.val[3] = vbslq_f32(vcleq_f32(vin_deq.val[3], vconst_0_f32), clamped_value, vtmp_deq.val[3]);
165  }
166 
167  // Re-quantize to new output space
168  vout = vquantize_signed(vtmp_deq, qi_out);
169  wrapper::vstore(output_ptr + x, vout);
170  }
171  for(; x < window_end_x; ++x)
172  {
173  qasymm8_signed_t in = *(reinterpret_cast<const qasymm8_signed_t *>(input_ptr + x));
174  qasymm8_signed_t tmp = 0;
175  float tmp_f = dequantize_qasymm8_signed(in, qi_in);
176  if(tmp_f <= 0.0)
177  {
178  if(op == ElementWiseUnary::LOG)
179  {
180  tmp_f = (-128 - qi_out.offset) * qi_out.scale;
181  }
182  else if(op == ElementWiseUnary::RSQRT)
183  {
184  tmp_f = (127 - qi_out.offset) * qi_out.scale;
185  }
186  else
187  {
188  tmp_f = elementwise_op_scalar_imp<float>(op, tmp_f);
189  }
190  }
191  else
192  {
193  tmp_f = elementwise_op_scalar_imp<float>(op, tmp_f);
194  }
195  tmp = quantize_qasymm8_signed(tmp_f, qi_out, RoundingPolicy::TO_ZERO); // Set rounding policy TO_ZERO to be compatible with vquantize_signed() used above that follow same policy for armv7a.
196  // For aarch64 LUT is used and rounding to nearest is used
197  *(output_ptr + x) = tmp;
198  }
199  },
200  input, output);
201 }

209  const UniformQuantizationInfo qi_out = out->info()->quantization_info().uniform();
210  const auto vconst_0_f32 = vdupq_n_f32(0);

223  auto output_ptr = reinterpret_cast<uint8_t *>(output.ptr());
224  const auto input_ptr = reinterpret_cast<const uint8_t *>(input.ptr());
225  int x = window_start_x;
226  for(; x <= window_end_x - window_step_x; x += window_step_x)
227  {
228  const auto vin = wrapper::vloadq(input_ptr + x);
229 
230  // De-quantize
231  const auto vin_deq = vdequantize(vin, qi_in);
232

234  float32x4x4_t vtmp_deq =
235  {
236  {
237  elementwise_op_imp<float>(op, vin_deq.val[0]),
238  elementwise_op_imp<float>(op, vin_deq.val[1]),
239  elementwise_op_imp<float>(op, vin_deq.val[2]),
240  elementwise_op_imp<float>(op, vin_deq.val[3]),
241  }
242  };
243  if((op == ElementWiseUnary::LOG) || (op == ElementWiseUnary::RSQRT))
244  {
245  vtmp_deq.val[0] = vbslq_f32(vcleq_f32(vin_deq.val[0], vconst_0_f32), clamped_value, vtmp_deq.val[0]);
246  vtmp_deq.val[1] = vbslq_f32(vcleq_f32(vin_deq.val[1], vconst_0_f32), clamped_value, vtmp_deq.val[1]);
247  vtmp_deq.val[2] = vbslq_f32(vcleq_f32(vin_deq.val[2], vconst_0_f32), clamped_value, vtmp_deq.val[2]);
248  vtmp_deq.val[3] = vbslq_f32(vcleq_f32(vin_deq.val[3], vconst_0_f32), clamped_value, vtmp_deq.val[3]);
249  }
250 
251  // Re-quantize to new output space
252  vout = vquantize(vtmp_deq, qi_out);
253  wrapper::vstore(output_ptr + x, vout);
254  }
255  for(; x < window_end_x; ++x)
256  {
257  qasymm8_t in = *(reinterpret_cast<const qasymm8_t *>(input_ptr + x));
258  qasymm8_t tmp = 0;
259  float tmp_f = dequantize_qasymm8(in, qi_in);
260  if(tmp_f <= 0.0)
261  {
262  if(op == ElementWiseUnary::LOG)
263  {
264  tmp_f = (0 - qi_out.offset) * qi_out.scale;
265  }
266  else if(op == ElementWiseUnary::RSQRT)
267  {
268  tmp_f = (255 - qi_out.offset) * qi_out.scale;
269  }
270  else
271  {
272  tmp_f = elementwise_op_scalar_imp<float>(op, tmp_f);
273  }
274  }
275  else
276  {
277  tmp_f = elementwise_op_scalar_imp<float>(op, tmp_f);
278  }
279  tmp = quantize_qasymm8(tmp_f, qi_out, RoundingPolicy::TO_ZERO);
280  *(output_ptr + x) = tmp;
281  }
282  },
283  input, output);
