68 void GpuCkwElementwiseBinary::write_component_code(const ComponentGroup &comp_group, GpuCkwVariableTable &vtable, GpuCkwScopedKernelWriter writer) const
69 {
70  const auto root_window = comp_group.get_root_component()->ckw_component_driver()->get_window();
71  const auto n0 = static_cast<int32_t>(root_window.x().step());
72  const auto m0 = static_cast<int32_t>(root_window.y().step());
73 
74  GpuCkwComponentArgument *lhs = vtable.declare_variable(comp_group, writer, _lhs, TensorStorageType::ClBufferUint8Ptr, "lhs");
75  GpuCkwComponentArgument *rhs = vtable.declare_variable(comp_group, writer, _rhs, TensorStorageType::ClBufferUint8Ptr, "rhs");
76  GpuCkwComponentArgument *dst = vtable.declare_variable(comp_group, writer, _dst, TensorStorageType::ClBufferUint8Ptr, "dst");
77 
78  auto &gid_0 = writer->declare_tile("gid_0", ckw::DataType::Int32);
79  auto &gid_1 = writer->declare_tile("gid_1", ckw::DataType::Int32);
80  auto &gid_2 = writer->declare_tile("gid_2", ckw::DataType::Int32);
81 
82  writer->op_get_global_id(gid_0, 0);
83  writer->op_get_global_id(gid_1, 1);
84  writer->op_get_global_id(gid_2, 2);
85 
86  auto &const_0 = writer->declare_tile("0", 0);
87 
88  // Load the LHS and RHS tiles
89  if(!lhs->has_tile())
90  {
91  auto sampler = create_boundary_aware_2d_sampler(writer, gid_0, gid_1, _lhs->dimension(0), _lhs->dimension(1), n0, m0, "lhs_", const_0);
92  sampler.format(TensorSamplerFormat::C_WH_1); // 3rd dimension collapsed with 2nd dimension
93  sampler.z(const_0);
94  sampler.b(gid_2);
95  writer->op_load_once(lhs, sampler);
96  }
97  if(!rhs->has_tile())
98  {
99  auto sampler = create_boundary_aware_2d_sampler(writer, gid_0, gid_1, _rhs->dimension(0), _rhs->dimension(1), n0, m0, "rhs_", const_0);
100  sampler.format(TensorSamplerFormat::C_WH_1); // 3rd dimension collapsed with 2nd dimension
101  sampler.z(const_0);
102  sampler.b(gid_2);
103  writer->op_load_once(rhs, sampler);
104  }
105 
106  auto dst_sampler = create_boundary_aware_2d_sampler(writer, gid_0, gid_1, _dst->dimension(0), _dst->dimension(1), n0, m0, "dst_", const_0);
107  dst_sampler.format(TensorSamplerFormat::C_WH_1); // 3rd dimension collapsed with 2nd dimension
108  dst_sampler.z(const_0);
109  dst_sampler.b(gid_2);
110 
111  // Prepare the output tile.
112  if(!dst->has_tile())
113  {
114  auto &tile = writer->declare_tile("dst_tile", ckw::TileInfo(to_ckw(_dst->data_type()), dst_sampler.height(), dst_sampler.width()));
115  dst->init_virtual_tensor(tile, dst_sampler);
116  }
117 
118  auto &lhs_tile = lhs->tile();
119  auto &rhs_tile = rhs->tile();
120  auto &dst_tile = dst->tile();
121 
122  // Perform the operation.
123  writer->op_binary_expression(dst_tile, lhs_tile, to_ckw(_attributes), rhs_tile);
124 }

126 Window GpuCkwElementwiseBinary::get_window() const
127 {
128  ARM_COMPUTE_ERROR_ON_MSG(_dst->tensor_shape().total_size() == 0U, "Destination tensor is not initialized");
129 
130  TensorShape output_shape = _dst->tensor_shape();
131  // Collapse Dim 1 (W) and Dim 2 (H) together, leave Dim 0 (C) unchanged
132  // This is in line with the collapsing convention used by operators like Conv2d
133  output_shape.collapse(2U, 1U);
134  constexpr unsigned int vector_size_byte_opencl = 16;
135  const unsigned int num_elems_processed_per_iteration = adjust_vec_size(vector_size_byte_opencl / _dst->element_size(), _dst->dimension(0));
136  Window win = calculate_max_window(output_shape, Steps(num_elems_processed_per_iteration));
137 
138  return win;
139 }
