72 void GpuCkwDirectConv2d::write_component_code(const ComponentGroup &comp_group, GpuCkwVariableTable &vtable, GpuCkwScopedKernelWriter writer) const
73 {
74  const auto desc = _settings.direct_conv_descriptor();
75  ARM_COMPUTE_ERROR_ON_MSG(desc.export_input_to_cl_image || desc.export_output_to_cl_image,
76  "Only the weights tensor can be exported to cl_image");
77 
78  const unsigned int channel_idx = get_data_layout_dimension_index(_src->data_layout(), DataLayoutDimension::CHANNEL);
79  const unsigned int width_idx = get_data_layout_dimension_index(_wei->data_layout(), DataLayoutDimension::WIDTH);
80  const unsigned int height_idx = get_data_layout_dimension_index(_wei->data_layout(), DataLayoutDimension::HEIGHT);
81 
82  const auto root_window = comp_group.get_root_component()->ckw_component_driver()->get_window();
83 
84  // Tunable parameters
85  const int32_t m0 = root_window.y().step();
86  const int32_t n0 = root_window.x().step();
87  const int32_t k0 = adjust_vec_size(_settings.direct_conv_descriptor().k0, _src->dimension(channel_idx));
88  const int32_t partial_n0 = _dst->dimension(0) % n0;
89 
90  const int32_t K = _src->dimension(channel_idx);
91 
92  // Exporting the weights tensor to an OpenCL image object is currently only supported when:
93  // a) k0 is equal to 4
94  // The current implementation expects to read a vector of 4 float values into the OpenCL image object.
95  // b) K is a multiple of 4
96  // This is a limitation in the current interface due to the variable table being responsible for maintaining
97  // information about the TensorStorageType rather than the TensorTileSampler. As a result, TensorStorageType cannot
98  // be reassigned, and we cannot use a texture object for the weights tensor in cases where we expect to have an
99  // extra loop to compute the left-over elements.
100  const bool use_cl_image_for_weights = desc.export_weights_to_cl_image && (k0 == 4) && (K % 4 == 0);
101 
102  GpuCkwComponentArgument *src = vtable.declare_variable(comp_group, writer, _src, TensorStorageType::ClBufferUint8Ptr, "src");
103  GpuCkwComponentArgument *wei = vtable.declare_variable(
104  comp_group, writer, _wei, use_cl_image_for_weights ? TensorStorageType::ClImage2dReadOnly : TensorStorageType::ClBufferUint8Ptr, "wei");
105  GpuCkwComponentArgument *dst = vtable.declare_variable(comp_group, writer, _dst, TensorStorageType::ClBufferUint8Ptr, "dst");
106  GpuCkwComponentArgument *bia = nullptr;
107 
108  const bool using_bias = _bia != nullptr;
109 
110  if(using_bias)
111  {
112  bia = vtable.declare_variable(comp_group, writer, _bia, TensorStorageType::ClBufferUint8Ptr, "bia");
113  }
114 
115  // Constants
116  const auto kernel_height = static_cast<int32_t>(_wei->dimension(height_idx));
117  const auto kernel_width = static_cast<int32_t>(_wei->dimension(width_idx));
118  const auto src_channels = static_cast<int32_t>(_src->dimension(channel_idx));
119  auto &tile_kernel_w = writer->declare_tile("kernel_w", kernel_width);
120  auto &tile_kernel_size = writer->declare_tile("kernel_size", kernel_width * kernel_height);
121  auto &tile_src_c = writer->declare_tile("src_c", static_cast<int32_t>(_src->dimension(channel_idx)));
122  auto &tile_dst_w = writer->declare_tile("dst_w", static_cast<int32_t>(_dst->dimension(width_idx)));
123  auto &tile_stride_x = writer->declare_tile("stride_x", static_cast<int32_t>(_attributes.stride().x()));
124  auto &tile_stride_y = writer->declare_tile("stride_y", static_cast<int32_t>(_attributes.stride().y()));
125  auto &tile_pad_x = writer->declare_tile("pad_x", static_cast<int32_t>(_attributes.pad().left));
126  auto &tile_pad_y = writer->declare_tile("pad_y", static_cast<int32_t>(_attributes.pad().top));
127  auto &tile_k0 = writer->declare_tile("k0", k0);
128  auto &tile_0 = writer->declare_tile("0", 0);
129  auto &tile_1 = writer->declare_tile("1", 1);
130 
131  auto &tile_gid_0 = writer->declare_tile("gid_0", ckw::DataType::Int32);
132  auto &tile_gid_1 = writer->declare_tile("gid_1", ckw::DataType::Int32);
133  auto &tile_gid_2 = writer->declare_tile("gid_2", ckw::DataType::Int32);
134 
135  writer->op_get_global_id(tile_gid_0, 0);
136  writer->op_get_global_id(tile_gid_1, 1);
137  writer->op_get_global_id(tile_gid_2, 2);
138 
139  auto &tile_cout = writer->declare_tile("cout", ckw::DataType::Int32); // OFM
140  auto &tile_mout = writer->declare_tile("mout", ckw::DataType::Int32); // WIDTH x HEIGHT
141  auto &tile_bout = writer->declare_tile("bout", ckw::DataType::Int32); // BATCH SIZE IDX
142 
143  // Get the boundary aware coordinates at each global dimension index
144  get_coord(writer, tile_cout, tile_gid_0, n0, partial_n0, tile_cout.name() + "_dim0_", tile_0);
145  get_coord(writer, tile_mout, tile_gid_1, m0, 0, tile_mout.name() + "_dim1_", tile_0);
146  get_coord(writer, tile_bout, tile_gid_2, 1, 0, tile_bout.name() + "_dim2_", tile_0);
147 
148  TensorTileSampler src_sampler;
149  src_sampler.width(k0);
150  src_sampler.height(m0);
151  src_sampler.format(TensorSamplerFormat::C_WH_1);
152  // We cannot have out-of-bounds reads in the X dimension (mapped to the IFMs) as we have an extra loop to
153  // compute left-over elements
154  src_sampler.address_mode_x(TensorSamplerAddressModeX::None);
155  // We cannot have out-of-bounds reads when the kernel height is equal to 1. Otherwise, we need to ensure the
156  // indirection buffer mi does not contain negative values representing out-of-bounds reads.
157  src_sampler.address_mode_y(kernel_height == 1 ? TensorSamplerAddressModeY::None : TensorSamplerAddressModeY::SkipMinEdgeOnly);
158  src_sampler.address_mode_z(TensorSamplerAddressModeZ::None);
159 
160  TensorTileSampler wei_sampler;
161  wei_sampler.width(k0);
162  wei_sampler.height(n0);
163  wei_sampler.format(TensorSamplerFormat::C_WH_1);
164  // We cannot have out-of-bounds accesses for the weights
165  wei_sampler.address_mode_x(TensorSamplerAddressModeX::None);
166  wei_sampler.address_mode_y(TensorSamplerAddressModeY::None);
167  wei_sampler.address_mode_z(TensorSamplerAddressModeZ::None);
168 
169  TensorTileSampler dst_sampler;
170  dst_sampler.width(n0);
171  dst_sampler.height(m0);
172  dst_sampler.format(TensorSamplerFormat::C_WH_1);
173  dst_sampler.address_mode_x(TensorSamplerAddressModeX::OverlappingMin);
174  dst_sampler.address_mode_y(TensorSamplerAddressModeY::ClampToMaxEdgeOnly);
175  dst_sampler.address_mode_z(TensorSamplerAddressModeZ::None);
176  dst_sampler.x(tile_cout);
177  dst_sampler.y(tile_mout);
178  dst_sampler.z(tile_0);
179  dst_sampler.b(tile_bout);
180 
181  if(!dst->has_tile())
182  {
183  auto &tile = writer->declare_tile("dst", TileInfo(to_ckw(_dst->data_type()), m0, n0));
184  dst->init_virtual_tensor(tile, dst_sampler);
185  }
186  auto &tile_dst = dst->tile();
187 
188  writer->op_assign(tile_dst, tile_0);
189 
190  // We create a 2d container of size (M0, 1) to store the indices for iteration
191  TileContainer it;
192  for(int m = 0; m < m0; ++m)
193  {
194  std::vector<std::string> idx { std::to_string(m) };
195  it.push_back({ idx });
196  }
197  const auto &tile_it = writer->declare_tile("it", it, ckw::DataType::Int32);
198 
199  auto &tile_xi = writer->declare_tile("xi", TileInfo(ckw::DataType::Int32, m0, 1));
200  auto &tile_yi = writer->declare_tile("yi", TileInfo(ckw::DataType::Int32, m0, 1));
201 
202  // Convert the linear index to coordinate
203  // xi = ((mout + i) % dst_w) * stride_x - pad_x
204  // yi = ((mout + i) / dst_w) * stride_y - pad_y
205  writer->op_binary_expression(tile_xi, tile_mout, BinaryOp::Add, tile_it);
206  writer->op_binary_expression(tile_yi, tile_mout, BinaryOp::Add, tile_it);
207  writer->op_binary_expression(tile_xi, tile_xi, BinaryOp::Mod, tile_dst_w);
208  writer->op_binary_expression(tile_yi, tile_yi, BinaryOp::Div, tile_dst_w);
209  writer->op_binary_expression(tile_xi, tile_xi, BinaryOp::Mul, tile_stride_x);
210  writer->op_binary_expression(tile_yi, tile_yi, BinaryOp::Mul, tile_stride_y);
211  writer->op_binary_expression(tile_xi, tile_xi, BinaryOp::Sub, tile_pad_x);
212  writer->op_binary_expression(tile_yi, tile_yi, BinaryOp::Sub, tile_pad_y);
213 
214  auto &tile_y_b = writer->declare_tile("y_b", ckw::DataType::Int32);
215  writer->op_binary_expression(tile_y_b, tile_cout, BinaryOp::Mul, tile_kernel_size);
216 
217  auto &tile_i = writer->declare_tile("i", ckw::DataType::Int32);
218  writer->op_assign(tile_i, tile_0);
219 
220  // clang-format off
221  writer->op_for_loop(tile_i, BinaryOp::Less, tile_kernel_size, tile_i, AssignmentOp::Increment, tile_1, [&]()
222  {
223  auto &tile_x_k = writer->declare_tile("x_k", ckw::DataType::Int32);
224  auto &tile_y_k = writer->declare_tile("y_k", ckw::DataType::Int32);
225 
226  writer->op_binary_expression(tile_x_k, tile_i, BinaryOp::Mod, tile_kernel_w);
227  writer->op_binary_expression(tile_y_k, tile_i, BinaryOp::Div, tile_kernel_w);
228 
229  auto &tile_ck = writer->declare_tile("ck", ckw::DataType::Int32);
230  writer->op_assign(tile_ck, tile_0);
231 
232  auto &tile_mi = writer->declare_tile("mi", TileInfo(ckw::DataType::Int32, m0, 1));
233  // Construct an indirection buffer containing the precalculated addresses of elements in the source tensor
234  // x_s = xi + x_k
235  // y_s = yi + y_k
236  // mi = x_s + y_s * width;
237  // mi = select(-1, mi, x_s >= 0);
238  // mi = select(-1, mi, x_s < width);
239  // mi = select(-1, mi, y_s >= 0);
240  // mi = select(-1, mi, y_s < height);
241  writer->util_get_indirect_buffer(tile_mi, src->tensor(), src_sampler, tile_xi, tile_yi, tile_x_k, tile_y_k);
242 
243  src_sampler.x(tile_ck);
244  src_sampler.y(tile_mi);
245  src_sampler.z(tile_0);
246  src_sampler.b(tile_bout);
247 
248  wei_sampler.x(tile_ck);
249  wei_sampler.y(tile_y_b);
250  wei_sampler.z(tile_0);
251  wei_sampler.b(tile_0);
252 
253  auto &tile_src_c_minus_k0 = writer->declare_tile("src_c_minus_k0", src_channels - k0);
254 
255  writer->op_for_loop(tile_ck, BinaryOp::LessEqual, tile_src_c_minus_k0, tile_ck, AssignmentOp::Increment, tile_k0, [&]()
256  {
257  auto &tile_lhs = writer->declare_tile("lhs", TileInfo(to_ckw(_src->data_type()), m0, k0));
258  auto &tile_rhs = writer->declare_tile("rhs", TileInfo(to_ckw(_wei->data_type()), n0, k0));
259  writer->op_assign(tile_lhs, tile_0);
260  writer->op_assign(tile_rhs, tile_0);
261 
262  writer->op_load_indirect(tile_lhs, src->tensor(), src_sampler);
263  writer->op_load(tile_rhs, wei->tensor(), wei_sampler, tile_kernel_size);
264 
265  writer->op_binary_expression(tile_dst, tile_lhs, BinaryOp::MatMul_Nt_T, tile_rhs);
266  });
267 
268  // Left-over accumulations for when K is not a multiple of k0
269  if(!(K % k0 == 0))
270  {
271  writer->op_for_loop(tile_ck, BinaryOp::Less, tile_src_c, tile_ck, AssignmentOp::Increment, tile_1, [&]()
272  {
273  auto &tile_lhs = writer->declare_tile("lhs_leftover", TileInfo(to_ckw(_src->data_type()), m0, 1));
274  auto &tile_rhs = writer->declare_tile("rhs_leftover", TileInfo(to_ckw(_wei->data_type()), n0, 1));
275  writer->op_assign(tile_lhs, tile_0);
276  writer->op_assign(tile_rhs, tile_0);
277 
278  writer->op_load_indirect(tile_lhs, src->tensor(), src_sampler);
279  writer->op_load(tile_rhs, wei->tensor(), wei_sampler, tile_kernel_size);
280 
281  writer->op_binary_expression(tile_dst, tile_lhs, BinaryOp::MatMul_Nt_T, tile_rhs);
282  });
283  }
284 
285  writer->op_binary_expression(tile_y_b, tile_y_b, BinaryOp::Add, tile_1);
286  });
287  // clang-format on
288 
289  // Bias addition
290  // NOTE: This operation will be removed from this kernel as the interface is standardized. The intended way of
291  // performing bias addition is to fuse this convolution kernel with a following elementwise addition kernel.
292  if(using_bias)
293  {
294  if(!bia->has_tile())
295  {
296  // Reuse the destination sampler for the bias
297  writer->op_load_once(bia, dst_sampler);
298  }
299  auto &tile_bia = bia->tile();
300 
301  writer->op_binary_expression(tile_dst, tile_dst, BinaryOp::Add, tile_bia);
302  }
303 }

305 Window GpuCkwDirectConv2d::get_window() const
306 {
307  ARM_COMPUTE_ERROR_ON_MSG(_dst->tensor_shape().total_size() == 0U, "Destination tensor is not initialized");
308 
309  const auto dst_shape = _dst->tensor_shape();
310  const auto desc = _settings.direct_conv_descriptor();
311 
312  const unsigned int n0 = adjust_vec_size(desc.n0, dst_shape[0]);
313  const unsigned int m0 = adjust_vec_size(desc.m0, dst_shape[1] * dst_shape[2]);
314 
315  Window win = calculate_max_window(dst_shape, Steps(n0, m0));
316 
317  const size_t dim_y_collapsed = ceil_to_multiple(dst_shape[1] * dst_shape[2], m0);
318  win.set(Window::DimY, Window::Dimension(0, dim_y_collapsed, m0));
319  win.set(Window::DimZ, Window::Dimension(0, dst_shape.total_size_upper(3), 1));
320 
321  return win;
322 }
