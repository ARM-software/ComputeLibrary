50 Status ClComponentElementwiseBinary::validate(const ArgumentPack<ITensorInfo> &tensors, const ElementwiseBinaryCommonAttributes &attributes)
51 {
52  const auto lhs = tensors.get_const_tensor(TensorType::ACL_SRC_0);
53  const auto rhs = tensors.get_const_tensor(TensorType::ACL_SRC_1);
54  const auto dst = tensors.get_const_tensor(TensorType::ACL_DST_0);
55 
56  // Check operator type
57  ARM_COMPUTE_RETURN_ERROR_ON_MSG(supported_ops.find(attributes.operation()) == supported_ops.end(), "Provided Elementwise operation not supported.");
58 
59  // Check validity
60  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(lhs, rhs, dst);
61 
62  //Check data type for different elementwise operators
63  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(lhs, 1, DataType::F32, DataType::F16, DataType::S32, DataType::S16, DataType::U8);
64 
65  // dst shape is correct
66  const TensorShape out_shape = TensorShape::broadcast_shape(lhs->tensor_shape(), rhs->tensor_shape());
67  ARM_COMPUTE_RETURN_ERROR_ON_MSG(out_shape.total_size() == 0, "Inputs are not broadcast compatible");
68  ARM_COMPUTE_RETURN_ERROR_ON_MSG(detail::have_different_dimensions(out_shape, dst->tensor_shape(), 0), "Wrong shape for dst.");
69 
70  const auto &lhs_shape = lhs->tensor_shape();
71  const auto &rhs_shape = rhs->tensor_shape();
72  const auto &dst_shape = dst->tensor_shape();
73 
74  ARM_COMPUTE_RETURN_ERROR_ON_MSG(
75  detail::have_different_dimensions(lhs_shape, dst_shape, 0) && detail::have_different_dimensions(rhs_shape, dst_shape, 0),
76  "Only LHS or RHS can be broadcasting, not both.");
77 
78  // Dimension Y and Z are collapsed together in the current kernel implementation,
79  // hence they cannot be independently broadcast or non-broadcast.
80  // See: ClTemplateElementwiseBinary::get_window
81  ARM_COMPUTE_RETURN_ERROR_ON_MSG(
82  (lhs_shape[1] != dst_shape[1] || rhs_shape[1] != dst_shape[1]) != (lhs_shape[2] != dst_shape[2] || rhs_shape[2] != dst_shape[2]),
83  "Dimension Y and Z must both be either broadcast or non-broadcast.");
84 
85  ARM_COMPUTE_RETURN_ERROR_ON_MSG(
86  detail::have_different_dimensions(lhs_shape, dst_shape, 3),
87  "LHS broadcast in dimension 3 or higher is not supported.");
88 
89  ARM_COMPUTE_RETURN_ERROR_ON_MSG(
90  detail::have_different_dimensions(rhs_shape, dst_shape, 3),
91  "RHS broadcast in dimension 3 or higher is not supported.");
92 
93  // Matching data type
94  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(lhs, rhs);
95  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(lhs, dst);
96 
97  // Matching data layout
98  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_LAYOUT(lhs, rhs);
99  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_LAYOUT(lhs, dst);
100 
101  // All tensor infos are initialized
102  ARM_COMPUTE_RETURN_ERROR_ON(lhs->tensor_shape().total_size() == 0);
103  ARM_COMPUTE_RETURN_ERROR_ON(rhs->tensor_shape().total_size() == 0);
104  ARM_COMPUTE_RETURN_ERROR_ON(dst->tensor_shape().total_size() == 0);
105 
106  // Device requirements are met
107  ARM_COMPUTE_RETURN_ERROR_ON_F16_UNSUPPORTED(lhs);
108 
109  return Status{};
110 }
