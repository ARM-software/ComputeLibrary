65 unsigned int calculate_number_of_stages_only_x_axis(size_t input_x_dimension, unsigned int axis)
66 {
67  // We need only 1 stage for all axis except x-axis
68  if (axis != 0)
69  {
70  return 1;
71  }
72  // Calculate number of WGs. 16 elements per thread, 8 threads per WG
73  const auto num_of_wg = static_cast<unsigned int>(ceil(input_x_dimension / 128.f));
74 
75  // Calculate number of stages. First stage performs op and the rest reduction sum
76  // depending on the size of the input. Last stage should have only 1 WG.
77  const unsigned int num_of_stages = num_of_wg / 128 + 2;
78  return num_of_stages;
79 }
