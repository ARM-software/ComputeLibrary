145 def generate_bp_file(cpp_files, opencl_files):
146  arm_files = [f for f in cpp_files if "a32_" in f]
147  arm64_files = [f for f in cpp_files if any(a64 in f for a64 in ["a64_", "sve_", 'sme_', 'sme2_'])]
148  gen_files = [x for x in cpp_files if x not in arm_files + arm64_files]
149 
150  arm_files.sort()
151  arm64_files.sort()
152  gen_files.sort()
153  opencl_files.sort()
154 
155  bp_file = bp_tm.render(srcs=gen_files,
156  arm_srcs=arm_files,
157  arm64_srcs=arm64_files,
158  cl_srcs=opencl_files)
159  return bp_file

162 def list_all_files(repo_path):
163  """ Gets the list of files to include to the Android.bp
164 
165  :param repo_path: Path of the repository
166  :return: The filtered list of useful filess
167  """
168  if not repo_path.endswith('/'):
169  repo_path = repo_path + "/"
170 
171  # Get cpp files
172  cpp_files = []
173  cl_files = []
174  for path, subdirs, files in os.walk(repo_path):
175  for file in files:
176  if file.endswith(".cpp"):
177  cpp_files.append(os.path.join(path, file))
178  elif file.endswith(".cl"):
179  cl_files.append(os.path.join(path, file))
180  # Include CL headers
181  if "src/core/CL/cl_kernels" in path and file.endswith(".h"):
182  cl_files.append(os.path.join(path, file))
183  # Filter out unused cpp files
184  filtered_cpp_files = []
185  for cpp_file in cpp_files:
186  if any(ep in cpp_file for ep in excluded_paths) or any(ef in cpp_file for ef in excluded_files):
187  continue
188  filtered_cpp_files.append(cpp_file.replace(repo_path, ""))
189  # Filter out unused cl files
190  filtered_cl_files = []
191  for cl_file in cl_files:
192  if any(ep in cl_file for ep in excluded_paths):
193  continue
194  filtered_cl_files.append(cl_file.replace(repo_path, ""))
195 
196  return filtered_cpp_files, filtered_cl_files
