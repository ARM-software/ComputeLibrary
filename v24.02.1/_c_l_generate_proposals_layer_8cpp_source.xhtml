86 void CLGenerateProposalsLayer::configure(const CLCompileContext &compile_context,
87  const ICLTensor *scores,
88  const ICLTensor *deltas,
89  const ICLTensor *anchors,
90  ICLTensor *proposals,
91  ICLTensor *scores_out,
92  ICLTensor *num_valid_proposals,
93  const GenerateProposalsInfo &info)
94 {
95  ARM_COMPUTE_ERROR_ON_NULLPTR(scores, deltas, anchors, proposals, scores_out, num_valid_proposals);
96  ARM_COMPUTE_ERROR_THROW_ON(CLGenerateProposalsLayer::validate(scores->info(), deltas->info(), anchors->info(),
97  proposals->info(), scores_out->info(),
98  num_valid_proposals->info(), info));
99  ARM_COMPUTE_LOG_PARAMS(scores, deltas, anchors, proposals, scores_out, num_valid_proposals, info);
100 
101  _is_nhwc = scores->info()->data_layout() == DataLayout::NHWC;
102  const DataType scores_data_type = scores->info()->data_type();
103  _is_qasymm8 = scores_data_type == DataType::QASYMM8;
104  const int num_anchors = scores->info()->dimension(
105  get_data_layout_dimension_index(scores->info()->data_layout(), DataLayoutDimension::CHANNEL));
106  const int feat_width = scores->info()->dimension(
107  get_data_layout_dimension_index(scores->info()->data_layout(), DataLayoutDimension::WIDTH));
108  const int feat_height = scores->info()->dimension(
109  get_data_layout_dimension_index(scores->info()->data_layout(), DataLayoutDimension::HEIGHT));
110  const int total_num_anchors = num_anchors * feat_width * feat_height;
111  const int pre_nms_topN = info.pre_nms_topN();
112  const int post_nms_topN = info.post_nms_topN();
113  const size_t values_per_roi = info.values_per_roi();
114 
115  const QuantizationInfo scores_qinfo = scores->info()->quantization_info();
116  const DataType rois_data_type = (_is_qasymm8) ? DataType::QASYMM16 : scores_data_type;
117  const QuantizationInfo rois_qinfo =
118  (_is_qasymm8) ? QuantizationInfo(0.125f, 0) : scores->info()->quantization_info();
119 
120  // Compute all the anchors
121  _memory_group.manage(&_all_anchors);
122  _compute_anchors_kernel->configure(compile_context, anchors, &_all_anchors,
123  ComputeAnchorsInfo(feat_width, feat_height, info.spatial_scale()));
124 
125  const TensorShape flatten_shape_deltas(values_per_roi, total_num_anchors);
126  _deltas_flattened.allocator()->init(
127  TensorInfo(flatten_shape_deltas, 1, scores_data_type, deltas->info()->quantization_info()));
128 
129  // Permute and reshape deltas
130  _memory_group.manage(&_deltas_flattened);
131  if (!_is_nhwc)
132  {
133  _memory_group.manage(&_deltas_permuted);
134  _permute_deltas.configure(compile_context, deltas, &_deltas_permuted, PermutationVector{2, 0, 1});
135  _flatten_deltas.configure(compile_context, &_deltas_permuted, &_deltas_flattened);
136  _deltas_permuted.allocator()->allocate();
137  }
138  else
139  {
140  _flatten_deltas.configure(compile_context, deltas, &_deltas_flattened);
141  }
142 
143  const TensorShape flatten_shape_scores(1, total_num_anchors);
144  _scores_flattened.allocator()->init(TensorInfo(flatten_shape_scores, 1, scores_data_type, scores_qinfo));
145 
146  // Permute and reshape scores
147  _memory_group.manage(&_scores_flattened);
148  if (!_is_nhwc)
149  {
150  _memory_group.manage(&_scores_permuted);
151  _permute_scores.configure(compile_context, scores, &_scores_permuted, PermutationVector{2, 0, 1});
152  _flatten_scores.configure(compile_context, &_scores_permuted, &_scores_flattened);
153  _scores_permuted.allocator()->allocate();
154  }
155  else
156  {
157  _flatten_scores.configure(compile_context, scores, &_scores_flattened);
158  }
159 
160  CLTensor *anchors_to_use = &_all_anchors;
161  CLTensor *deltas_to_use = &_deltas_flattened;
162  if (_is_qasymm8)
163  {
164  _all_anchors_f32.allocator()->init(TensorInfo(_all_anchors.info()->tensor_shape(), 1, DataType::F32));
165  _deltas_flattened_f32.allocator()->init(TensorInfo(_deltas_flattened.info()->tensor_shape(), 1, DataType::F32));
166  _memory_group.manage(&_all_anchors_f32);
167  _memory_group.manage(&_deltas_flattened_f32);
168  // Dequantize anchors to float
169  _dequantize_anchors->configure(compile_context, &_all_anchors, &_all_anchors_f32);
170  _all_anchors.allocator()->allocate();
171  anchors_to_use = &_all_anchors_f32;
172  // Dequantize deltas to float
173  _dequantize_deltas->configure(compile_context, &_deltas_flattened, &_deltas_flattened_f32);
174  _deltas_flattened.allocator()->allocate();
175  deltas_to_use = &_deltas_flattened_f32;
176  }
177  // Bounding box transform
178  _memory_group.manage(&_all_proposals);
179  BoundingBoxTransformInfo bbox_info(info.im_width(), info.im_height(), 1.f);
180  _bounding_box_kernel->configure(compile_context, anchors_to_use, &_all_proposals, deltas_to_use, bbox_info);
181  deltas_to_use->allocator()->allocate();
182  anchors_to_use->allocator()->allocate();
183 
184  _all_proposals_to_use = &_all_proposals;
185  if (_is_qasymm8)
186  {
187  _memory_group.manage(&_all_proposals_quantized);
188  // Requantize all_proposals to QASYMM16 with 0.125 scale and 0 offset
189  _all_proposals_quantized.allocator()->init(
190  TensorInfo(_all_proposals.info()->tensor_shape(), 1, DataType::QASYMM16, QuantizationInfo(0.125f, 0)));
191  _quantize_all_proposals->configure(compile_context, &_all_proposals, &_all_proposals_quantized);
192  _all_proposals.allocator()->allocate();
193  _all_proposals_to_use = &_all_proposals_quantized;
194  }
195  // The original layer implementation first selects the best pre_nms_topN anchors (thus having a lightweight sort)
196  // that are then transformed by bbox_transform. The boxes generated are then fed into a non-sorting NMS operation.
197  // Since we are reusing the NMS layer and we don't implement any CL/sort, we let NMS do the sorting (of all the input)
198  // and the filtering
199  const int scores_nms_size = std::min<int>(std::min<int>(post_nms_topN, pre_nms_topN), total_num_anchors);
200  const float min_size_scaled = info.min_size() * info.im_scale();
201  _memory_group.manage(&_classes_nms_unused);
202  _memory_group.manage(&_keeps_nms_unused);
203 
204  // Note that NMS needs outputs preinitialized.
205  auto_init_if_empty(*scores_out->info(), TensorShape(scores_nms_size), 1, scores_data_type, scores_qinfo);
206  auto_init_if_empty(*_proposals_4_roi_values.info(), TensorShape(values_per_roi, scores_nms_size), 1, rois_data_type,
207  rois_qinfo);
208  auto_init_if_empty(*num_valid_proposals->info(), TensorShape(1), 1, DataType::U32);
209 
210  // Initialize temporaries (unused) outputs
211  _classes_nms_unused.allocator()->init(TensorInfo(TensorShape(scores_nms_size), 1, scores_data_type, scores_qinfo));
212  _keeps_nms_unused.allocator()->init(*scores_out->info());
213 
214  // Save the output (to map and unmap them at run)
215  _scores_out = scores_out;
216  _num_valid_proposals = num_valid_proposals;
217 
218  _memory_group.manage(&_proposals_4_roi_values);
219  _cpp_nms.configure(&_scores_flattened, _all_proposals_to_use, nullptr, scores_out, &_proposals_4_roi_values,
220  &_classes_nms_unused, nullptr, &_keeps_nms_unused, num_valid_proposals,
221  BoxNMSLimitInfo(0.0f, info.nms_thres(), scores_nms_size, false, NMSType::LINEAR, 0.5f, 0.001f,
222  true, min_size_scaled, info.im_width(), info.im_height()));
223  _keeps_nms_unused.allocator()->allocate();
224  _classes_nms_unused.allocator()->allocate();
225  _all_proposals_to_use->allocator()->allocate();
226  _scores_flattened.allocator()->allocate();
227 
228  // Add the first column that represents the batch id. This will be all zeros, as we don't support multiple images
229  _pad_kernel->configure(compile_context, &_proposals_4_roi_values, proposals, PaddingList{{1, 0}});
230  _proposals_4_roi_values.allocator()->allocate();
231 }

233 Status CLGenerateProposalsLayer::validate(const ITensorInfo *scores,
234  const ITensorInfo *deltas,
235  const ITensorInfo *anchors,
236  const ITensorInfo *proposals,
237  const ITensorInfo *scores_out,
238  const ITensorInfo *num_valid_proposals,
239  const GenerateProposalsInfo &info)
240 {
241  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(scores, deltas, anchors, proposals, scores_out, num_valid_proposals);
242  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(scores, 1, DataType::QASYMM8, DataType::F16, DataType::F32);
243  ARM_COMPUTE_RETURN_ERROR_ON_DATA_LAYOUT_NOT_IN(scores, DataLayout::NCHW, DataLayout::NHWC);
244  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_LAYOUT(scores, deltas);
245  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(scores, deltas);
246 
247  const int num_anchors =
248  scores->dimension(get_data_layout_dimension_index(scores->data_layout(), DataLayoutDimension::CHANNEL));
249  const int feat_width =
250  scores->dimension(get_data_layout_dimension_index(scores->data_layout(), DataLayoutDimension::WIDTH));
251  const int feat_height =
252  scores->dimension(get_data_layout_dimension_index(scores->data_layout(), DataLayoutDimension::HEIGHT));
253  const int num_images = scores->dimension(3);
254  const int total_num_anchors = num_anchors * feat_width * feat_height;
255  const int values_per_roi = info.values_per_roi();
256 
257  const bool is_qasymm8 = scores->data_type() == DataType::QASYMM8;
258 
259  ARM_COMPUTE_RETURN_ERROR_ON(num_images > 1);
260 
261  if (is_qasymm8)
262  {
263  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(anchors, 1, DataType::QSYMM16);
264  const UniformQuantizationInfo anchors_qinfo = anchors->quantization_info().uniform();
265  ARM_COMPUTE_RETURN_ERROR_ON(anchors_qinfo.scale != 0.125f);
266  }
267 
268  TensorInfo all_anchors_info(
269  anchors->clone()->set_tensor_shape(TensorShape(values_per_roi, total_num_anchors)).set_is_resizable(true));
270  ARM_COMPUTE_RETURN_ON_ERROR(CLComputeAllAnchorsKernel::validate(
271  anchors, &all_anchors_info, ComputeAnchorsInfo(feat_width, feat_height, info.spatial_scale())));
272 
273  TensorInfo deltas_permuted_info =
274  deltas->clone()
275  ->set_tensor_shape(TensorShape(values_per_roi * num_anchors, feat_width, feat_height))
276  .set_is_resizable(true);
277  TensorInfo scores_permuted_info =
278  scores->clone()->set_tensor_shape(TensorShape(num_anchors, feat_width, feat_height)).set_is_resizable(true);
279  if (scores->data_layout() == DataLayout::NHWC)
280  {
281  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_SHAPES(deltas, &deltas_permuted_info);
282  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_SHAPES(scores, &scores_permuted_info);
283  }
284  else
285  {
286  ARM_COMPUTE_RETURN_ON_ERROR(CLPermute::validate(deltas, &deltas_permuted_info, PermutationVector{2, 0, 1}));
287  ARM_COMPUTE_RETURN_ON_ERROR(CLPermute::validate(scores, &scores_permuted_info, PermutationVector{2, 0, 1}));
288  }
289 
290  TensorInfo deltas_flattened_info(
291  deltas->clone()->set_tensor_shape(TensorShape(values_per_roi, total_num_anchors)).set_is_resizable(true));
292  ARM_COMPUTE_RETURN_ON_ERROR(CLReshapeLayer::validate(&deltas_permuted_info, &deltas_flattened_info));
293 
294  TensorInfo scores_flattened_info(
295  scores->clone()->set_tensor_shape(TensorShape(1, total_num_anchors)).set_is_resizable(true));
296  TensorInfo proposals_4_roi_values(
297  deltas->clone()->set_tensor_shape(TensorShape(values_per_roi, total_num_anchors)).set_is_resizable(true));
298 
299  ARM_COMPUTE_RETURN_ON_ERROR(CLReshapeLayer::validate(&scores_permuted_info, &scores_flattened_info));
300 
301  TensorInfo *proposals_4_roi_values_to_use = &proposals_4_roi_values;
302  TensorInfo proposals_4_roi_values_quantized(
303  deltas->clone()->set_tensor_shape(TensorShape(values_per_roi, total_num_anchors)).set_is_resizable(true));
304  proposals_4_roi_values_quantized.set_data_type(DataType::QASYMM16)
305  .set_quantization_info(QuantizationInfo(0.125f, 0));
306  if (is_qasymm8)
307  {
308  TensorInfo all_anchors_f32_info(anchors->clone()
309  ->set_tensor_shape(TensorShape(values_per_roi, total_num_anchors))
310  .set_is_resizable(true)
311  .set_data_type(DataType::F32));
312  ARM_COMPUTE_RETURN_ON_ERROR(CLDequantizationLayer::validate(&all_anchors_info, &all_anchors_f32_info));
313 
314  TensorInfo deltas_flattened_f32_info(deltas->clone()
315  ->set_tensor_shape(TensorShape(values_per_roi, total_num_anchors))
316  .set_is_resizable(true)
317  .set_data_type(DataType::F32));
318  ARM_COMPUTE_RETURN_ON_ERROR(
319  CLDequantizationLayer::validate(&deltas_flattened_info, &deltas_flattened_f32_info));
320 
321  TensorInfo proposals_4_roi_values_f32(deltas->clone()
322  ->set_tensor_shape(TensorShape(values_per_roi, total_num_anchors))
323  .set_is_resizable(true)
324  .set_data_type(DataType::F32));
325  ARM_COMPUTE_RETURN_ON_ERROR(CLBoundingBoxTransformKernel::validate(
326  &all_anchors_f32_info, &proposals_4_roi_values_f32, &deltas_flattened_f32_info,
327  BoundingBoxTransformInfo(info.im_width(), info.im_height(), 1.f)));
328 
329  ARM_COMPUTE_RETURN_ON_ERROR(
330  CLQuantizationLayer::validate(&proposals_4_roi_values_f32, &proposals_4_roi_values_quantized));
331  proposals_4_roi_values_to_use = &proposals_4_roi_values_quantized;
332  }
333  else
334  {
335  ARM_COMPUTE_RETURN_ON_ERROR(
336  CLBoundingBoxTransformKernel::validate(&all_anchors_info, &proposals_4_roi_values, &deltas_flattened_info,
337  BoundingBoxTransformInfo(info.im_width(), info.im_height(), 1.f)));
338  }
339 
340  ARM_COMPUTE_RETURN_ON_ERROR(
341  CLPadLayerKernel::validate(proposals_4_roi_values_to_use, proposals, PaddingList{{1, 0}}));
342 
343  if (num_valid_proposals->total_size() > 0)
344  {
345  ARM_COMPUTE_RETURN_ERROR_ON(num_valid_proposals->num_dimensions() > 1);
346  ARM_COMPUTE_RETURN_ERROR_ON(num_valid_proposals->dimension(0) > 1);
347  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(num_valid_proposals, 1, DataType::U32);
348  }
349 
350  if (proposals->total_size() > 0)
351  {
352  ARM_COMPUTE_RETURN_ERROR_ON(proposals->num_dimensions() > 2);
353  ARM_COMPUTE_RETURN_ERROR_ON(proposals->dimension(0) != size_t(values_per_roi) + 1);
354  ARM_COMPUTE_RETURN_ERROR_ON(proposals->dimension(1) != size_t(total_num_anchors));
355  if (is_qasymm8)
356  {
357  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(proposals, 1, DataType::QASYMM16);
358  const UniformQuantizationInfo proposals_qinfo = proposals->quantization_info().uniform();
359  ARM_COMPUTE_RETURN_ERROR_ON(proposals_qinfo.scale != 0.125f);
360  ARM_COMPUTE_RETURN_ERROR_ON(proposals_qinfo.offset != 0);
361  }
362  else
363  {
364  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(proposals, scores);
365  }
366  }
367 
368  if (scores_out->total_size() > 0)
369  {
370  ARM_COMPUTE_RETURN_ERROR_ON(scores_out->num_dimensions() > 1);
371  ARM_COMPUTE_RETURN_ERROR_ON(scores_out->dimension(0) != size_t(total_num_anchors));
372  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(scores_out, scores);
373  }
374 
375  return Status{};
376 }

378 void CLGenerateProposalsLayer::run_cpp_nms_kernel()
379 {
380  // Map inputs
381  _scores_flattened.map(true);
382  _all_proposals_to_use->map(true);
383 
384  // Map outputs
385  _scores_out->map(CLScheduler::get().queue(), true);
386  _proposals_4_roi_values.map(CLScheduler::get().queue(), true);
387  _num_valid_proposals->map(CLScheduler::get().queue(), true);
388  _keeps_nms_unused.map(true);
389  _classes_nms_unused.map(true);
390 
391  // Run nms
392  _cpp_nms.run();
393 
394  // Unmap outputs
395  _keeps_nms_unused.unmap();
396  _classes_nms_unused.unmap();
397  _scores_out->unmap(CLScheduler::get().queue());
398  _proposals_4_roi_values.unmap(CLScheduler::get().queue());
399  _num_valid_proposals->unmap(CLScheduler::get().queue());
400 
401  // Unmap inputs
402  _scores_flattened.unmap();
403  _all_proposals_to_use->unmap();
404 }

406 void CLGenerateProposalsLayer::run()
407 {
408  // Acquire all the temporaries
409  MemoryGroupResourceScope scope_mg(_memory_group);
410 
411  // Compute all the anchors
412  CLScheduler::get().enqueue(*_compute_anchors_kernel, false);
413 
414  // Transpose and reshape the inputs
415  if (!_is_nhwc)
416  {
417  _permute_deltas.run();
418  _permute_scores.run();
419  }
420  _flatten_deltas.run();
421  _flatten_scores.run();
422 
423  if (_is_qasymm8)
424  {
425  _dequantize_anchors->run();
426  _dequantize_deltas->run();
427  }
428 
429  // Build the boxes
430  CLScheduler::get().enqueue(*_bounding_box_kernel, false);
431 
432  if (_is_qasymm8)
433  {
434  _quantize_all_proposals->run();
435  }
436 
437  // Non maxima suppression
438  run_cpp_nms_kernel();
439  // Add dummy batch indexes
440  CLScheduler::get().enqueue(*_pad_kernel, true);
441 }
