248 Status validate_arguments(const ITensorInfo *input,
249  const ITensorInfo *output,
250  const Size2D &kernel_dims,
251  const PadStrideInfo &conv_info,
252  bool has_bias,
253  const Size2D &dilation,
254  unsigned int num_groups,
255  unsigned int input_pad_right)
256 {
257  ARM_COMPUTE_RETURN_ERROR_ON_CPU_F16_UNSUPPORTED(input);
258  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(output);
259  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(input, 1, DataType::QASYMM8, DataType::QASYMM8_SIGNED,
260  DataType::BFLOAT16, DataType::F16, DataType::F32);
261  ARM_COMPUTE_RETURN_ERROR_ON(is_data_type_quantized(input->data_type()) && has_bias);
262  ARM_COMPUTE_RETURN_ERROR_ON((dilation.x() < 1) || (dilation.y() < 1));
263  ARM_COMPUTE_RETURN_ERROR_ON_MSG(num_groups > 1, "Number of groups greater than one are not supported on Neon");
264 
265  // Since there's no implicit padding added, check the total input spatial dimensions (with conv paddings) are big enough for the kernel dimensions
266  const unsigned int width_idx = get_data_layout_dimension_index(input->data_layout(), DataLayoutDimension::WIDTH);
267  const unsigned int height_idx = get_data_layout_dimension_index(input->data_layout(), DataLayoutDimension::HEIGHT);
268  const unsigned total_width = input->dimension(width_idx) + conv_info.pad_left() + conv_info.pad_right();
269  const unsigned total_height = input->dimension(height_idx) + conv_info.pad_top() + conv_info.pad_bottom();
270  ARM_COMPUTE_RETURN_ERROR_ON((total_width < kernel_dims.width) || (total_height < kernel_dims.height));
271 
272  if (output->total_size() > 0)
273  {
274  TensorInfo expected_output = output->clone()->set_tensor_shape(compute_im2col_conv_shape(
275  input, kernel_dims, conv_info, has_bias, dilation, false, num_groups, input_pad_right));
276  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_SHAPES(&expected_output, output);
277  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input, output);
278  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_QUANTIZATION_INFO(input, output);
279  }
280 
281  return Status{};
282 }

285 void CpuIm2ColKernel::configure(const ITensorInfo *src,
286  ITensorInfo *dst,
287  const Size2D &kernel_dims,
288  const PadStrideInfo &conv_info,
289  bool has_bias,
290  const Size2D &dilation,
291  unsigned int num_groups,
292  unsigned int input_pad_right)
293 {
294  ARM_COMPUTE_ERROR_ON_NULLPTR(src, dst);
295  ARM_COMPUTE_ERROR_THROW_ON(
296  validate_arguments(src, dst, kernel_dims, conv_info, has_bias, dilation, num_groups, input_pad_right));
297  ARM_COMPUTE_UNUSED(num_groups);
298 
299  _data_layout = src->data_layout();
300  const unsigned int width_idx = get_data_layout_dimension_index(_data_layout, DataLayoutDimension::WIDTH);
301  const unsigned int height_idx = get_data_layout_dimension_index(_data_layout, DataLayoutDimension::HEIGHT);
302  const unsigned int channel_idx = get_data_layout_dimension_index(_data_layout, DataLayoutDimension::CHANNEL);
303 
304  _conv_info = conv_info;
305  _kernel_width = kernel_dims.width;
306  _kernel_height = kernel_dims.height;
307  _input_pad_right = input_pad_right;
308  _dilation = dilation;
309  _convolved_dims = scaled_dimensions(src->dimension(width_idx), dst->dimension(height_idx), _kernel_width,
310  _kernel_height, _conv_info, _dilation);
311  _has_bias = has_bias;
312 
313  if (_data_layout == DataLayout::NCHW)
314  {
315  switch (src->data_type())
316  {
317  case DataType::F32:
318  _func = (!conv_info.has_padding()) ? &run_im2col_fp32_nchw_nopad : &run_im2col_fp32_nchw_pad;
319  break;
320  case DataType::F16:
321  _func = (!conv_info.has_padding()) ? &internal_run_im2col_fp16_nchw_nopad
322  : &internal_run_im2col_fp16_nchw_pad;
323  break;
324 #if defined(ARM_COMPUTE_ENABLE_BF16)
325  case DataType::BFLOAT16:
326  _func = (!conv_info.has_padding()) ? &run_im2col_bf16_nchw_nopad : &run_im2col_bf16_nchw_pad;
327  break;
328 #endif /* defined(ARM_COMPUTE_ENABLE_BF16) */
329  case DataType::QASYMM8_SIGNED:
330  case DataType::QASYMM8:
331  _func = (!conv_info.has_padding()) ? &run_im2col_qasymm8_nchw_nopad : &run_im2col_qasymm8_nchw_pad;
332  break;
333  default:
334  ARM_COMPUTE_ERROR("Data type not supported");
335  break;
336  }
337  }
338  else
339  {
340  switch (src->data_type())
341  {
342  case DataType::F32:
343  _func = (!conv_info.has_padding()) ? &run_im2col_fp32_nopad : &run_im2col_fp32_pad;
344  break;
345  case DataType::F16:
346  _func = (!conv_info.has_padding()) ? &internal_run_im2col_fp16_nopad : &internal_run_im2col_fp16_pad;
347  break;
348 #if defined(ARM_COMPUTE_ENABLE_BF16)
349  case DataType::BFLOAT16:
350  _func = (!conv_info.has_padding()) ? &run_im2col_bf16_nopad : &run_im2col_bf16_pad;
351  break;
352 #endif /* defined(ARM_COMPUTE_ENABLE_BF16) */
353  case DataType::QASYMM8:
354  _func = (!conv_info.has_padding()) ? &run_im2col_uint8_nopad_nhwc : &run_im2col_qasymm8_pad_nhwc;
355  break;
356  case DataType::QASYMM8_SIGNED:
357  _func = (!conv_info.has_padding()) ? &run_im2col_int8_nopad_nhwc : &run_im2col_qasymm8_pad_nhwc;
358  break;
359  default:
360  ARM_COMPUTE_ERROR("Data type not supported");
361  break;
362  }
363  }
364 
365  // Output tensor auto initialization if not yet initialized
366  auto_init_if_empty(
367  *dst, src->clone()->set_tensor_shape(compute_im2col_conv_shape(src, kernel_dims, conv_info, has_bias, dilation,
368  false, num_groups, _input_pad_right)));
369 
370  std::pair<unsigned int, unsigned int> convolved_dims =
371  scaled_dimensions(src->dimension(width_idx), src->dimension(height_idx), kernel_dims.width, kernel_dims.height,
372  conv_info, dilation);
373 
374  Window win = calculate_max_window(*src, Steps());
375  win.set(width_idx, Window::Dimension(0, convolved_dims.first, 1));
376  win.set(height_idx, Window::Dimension(0, convolved_dims.second, 1));
377  win.set(channel_idx, Window::Dimension(0, 1, 1));
378  // Configure kernel window
379  ICpuKernel::configure(win);
380 }
