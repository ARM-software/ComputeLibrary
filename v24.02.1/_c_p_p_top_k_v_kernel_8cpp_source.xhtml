50 Status validate_arguments(const ITensorInfo *predictions,
51  const ITensorInfo *targets,
52  ITensorInfo *output,
53  const unsigned int k)
54 {
55  ARM_COMPUTE_UNUSED(k);
56  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(predictions, 1, DataType::QASYMM8, DataType::QASYMM8_SIGNED,
57  DataType::S32, DataType::F16, DataType::F32);
58  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(targets, 1, DataType::U32);
59 
60  ARM_COMPUTE_RETURN_ERROR_ON(predictions->num_dimensions() > 2);
61  ARM_COMPUTE_RETURN_ERROR_ON(targets->num_dimensions() > 1);
62  ARM_COMPUTE_RETURN_ERROR_ON(targets->dimension(0) != predictions->dimension(1));
63  // Validate configured output
64  if (output->total_size() != 0)
65  {
66  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DIMENSIONS(output->tensor_shape(), targets->tensor_shape());
67  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(output, 1, DataType::U8);
68  }
69 
70  return Status{};
71 }

74 template <typename T>
75 void CPPTopKVKernel::run_topkv()
76 {
77  for (unsigned int i = 0; i < _batch_size; ++i)
78  {
79  const auto target_class_id = *reinterpret_cast<uint32_t *>(_targets->ptr_to_element(Coordinates{i}));
80  const auto predicted_value =
81  *reinterpret_cast<T *>(_predictions->ptr_to_element(Coordinates{target_class_id, i}));
82 
83  // The variable rank indicates how many values there are before the target_class_id
84  unsigned int rank = 0;
85  for (unsigned int j = 0; (j < _num_classes) && (rank < _k); ++j)
86  {
87  const auto current_prediction = *reinterpret_cast<T *>(_predictions->ptr_to_element(Coordinates{j, i}));
88  if (greater_than(current_prediction, predicted_value))
89  {
90  rank++;
91  }
92  }
93  *(_output->ptr_to_element(Coordinates{i})) = static_cast<uint8_t>(rank < _k);
94  }
95 }

138 void CPPTopKVKernel::run(const Window &window, const ThreadInfo &info)
139 {
140  ARM_COMPUTE_UNUSED(window, info);
141  switch (_predictions->info()->data_type())
142  {
143  case DataType::F32:
144  run_topkv<float>();
145  break;
146  case DataType::F16:
147  run_topkv<half>();
148  break;
149  case DataType::S32:
150  run_topkv<int>();
151  break;
152  case DataType::QASYMM8:
153  run_topkv<uint8_t>();
154  break;
155  case DataType::QASYMM8_SIGNED:
156  run_topkv<int8_t>();
157  break;
158  default:
159  ARM_COMPUTE_ERROR("Not supported");
160  }
161 }
