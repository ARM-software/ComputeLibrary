53 Status validate_arguments(const ITensorInfo *src,
54  const ITensorInfo *bias,
55  const ITensorInfo *dst,
56  const DirectConvolutionLayerOutputStageKernelInfo &info)
57 {
58  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(src);
59  ARM_COMPUTE_RETURN_ERROR_ON_CPU_F16_UNSUPPORTED(src);
60  ARM_COMPUTE_RETURN_ERROR_ON(src->data_layout() == DataLayout::UNKNOWN);
61  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(src, 1, DataType::F16, DataType::S32, DataType::F32);
62 
63  if (bias != nullptr)
64  {
65  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(src, bias);
66  ARM_COMPUTE_RETURN_ERROR_ON(bias->dimension(0) != src->dimension(get_data_layout_dimension_index(
67  src->data_layout(), DataLayoutDimension::CHANNEL)));
68  ARM_COMPUTE_RETURN_ERROR_ON(bias->num_dimensions() > 1);
69  }
70 
71  if (src->data_type() == DataType::S32)
72  {
73  ARM_COMPUTE_RETURN_ERROR_ON_MSG(dst == nullptr, "In-place computation not allowed for quantized output");
74  }
75 
76  // Checks performed when output is configured
77  if ((dst != nullptr) && (dst->total_size() != 0))
78  {
79  if (is_data_type_float(src->data_type()))
80  {
81  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(src, dst);
82  }
83  else
84  {
85  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(dst, 1, DataType::QASYMM8, DataType::QASYMM8_SIGNED);
86  }
87  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_SHAPES(src, dst);
88  }
89  else if (src->data_type() == DataType::S32)
90  {
91  // In case of quantized computation and unconfigured output, the output data type must be provided through DirectConvolutionLayerOutputStageKernelInfo
92  ARM_COMPUTE_RETURN_ERROR_ON((info.output_data_type != DataType::QASYMM8) &&
93  (info.output_data_type != DataType::QASYMM8_SIGNED));
94  }
95 
96  return Status{};
97 }

99 template <typename T>
100 typename std::enable_if<arm_compute::utils::traits::is_floating_point<T>::value, void>::type
101 output_stage_nchw(ITensor *src,
102  const ITensor *bias,
103  const Window &window,
104  ITensor *dst,
105  int result_fixedpoint_multiplier,
106  int result_shift,
107  int result_offset_after_shift)
108 {
109  const bool has_bias = bias != nullptr;
110  /** SIMD vector tag type. */
111  using ExactTagType = typename wrapper::traits::neon_bitvector_tag_t<T, wrapper::traits::BitWidth::W128>;
112 
113  ARM_COMPUTE_ERROR_ON(src->info()->data_layout() == DataLayout::UNKNOWN);
114  ARM_COMPUTE_UNUSED(result_fixedpoint_multiplier);
115  ARM_COMPUTE_UNUSED(result_shift);
116  ARM_COMPUTE_UNUSED(result_offset_after_shift);
117 
118  const int window_start_x = window.x().start();
119  const int window_end_x = window.x().end();
120  const int window_step_x = 16 / src->info()->element_size();
121  Window win = window;
122  win.set(Window::DimX, Window::Dimension(0, 1, 1));
123 
124  Iterator in(src, win);
125  Iterator out(dst, win);
126  execute_window_loop(
127  win,
128  [&](const Coordinates &id)
129  {
130  int x = window_start_x;
131  for (; x <= (window_end_x - window_step_x); x += window_step_x)
132  {
133  // Get bias and pointer to input
134  const auto in_ptr = reinterpret_cast<const T *>(in.ptr()) + x;
135  auto v_in = wrapper::vloadq(in_ptr);
136 
137  // Accumulate bias
138  if (has_bias)
139  {
140  const auto vb = wrapper::vdup_n(
141  *reinterpret_cast<const T *>(bias->ptr_to_element(Coordinates(id.z()))), ExactTagType{});
142  v_in = wrapper::vadd(v_in, vb);
143  }
144 
145  const auto out_ptr = reinterpret_cast<T *>(out.ptr()) + x;
146  wrapper::vstore(out_ptr, v_in);
147  }
148 
149  // Left-overs loop
150  for (; x < window_end_x; ++x)
151  {
152  // Get bias and pointer to input
153  auto s_in = *(reinterpret_cast<const T *>(in.ptr()) + x);
154 
155  // Accumulate bias
156  if (has_bias)
157  {
158  const auto b = *reinterpret_cast<const T *>(bias->ptr_to_element(Coordinates(id.z())));
159  s_in += b;
160  }
161 
162  *(reinterpret_cast<T *>(out.ptr()) + x) = s_in;
163  }
164  },
165  in, out);
166 }

277  {
278  // Get bias and pointer to input

280  int32x4x4_t v_in = {{wrapper::vloadq(in_ptr), wrapper::vloadq(in_ptr + 4), wrapper::vloadq(in_ptr + 8),

290  }
291

295  }
296 
297  // Left-overs loop
298  for (; x < window_end_x; ++x)
299  {
300  // Get bias and pointer to input
301  int32_t s_in = *(reinterpret_cast<const int32_t *>(in.ptr()) + x);
302 
303  // Accumulate bias
304  if (has_bias)
305  {
306  const auto b = *reinterpret_cast<const int32_t *>(bias->ptr_to_element(Coordinates(id.z())));
307  s_in += b;
308  }
309 
310  const auto out_ptr = reinterpret_cast<TOut *>(out.ptr()) + x;

313  std::numeric_limits<TOut>::lowest(), std::numeric_limits<TOut>::max(), false);

410 void CpuDirectConv2dOutputStageKernel::configure(ITensorInfo *src,
411  const ITensorInfo *bias,
412  ITensorInfo *dst,
413  const DirectConvolutionLayerOutputStageKernelInfo &info)
414 {
415  ARM_COMPUTE_UNUSED(bias);
416  // Perform validation step
417  ARM_COMPUTE_ERROR_ON_NULLPTR(src);
418  ARM_COMPUTE_ERROR_THROW_ON(validate_arguments(src, bias, dst, info));
419 
420  _func = nullptr;
421  _result_fixedpoint_multiplier = info.result_fixedpoint_multiplier;
422  _result_shift = info.result_shift;
423  _result_offset_after_shift = info.result_offset_after_shift;
424 
425  // Auto-initialize output output if required
426  if (dst != nullptr)
427  {
428  // Work out expected output data type
429  const DataType output_dt = (src->data_type() == DataType::S32) ? info.output_data_type : DataType::S32;
430  // Output tensor auto initialization if not yet initialized
431  auto_init_if_empty(*dst, src->clone()->set_data_type(output_dt));
432  }
433 
434  Window win = calculate_max_window(*src, Steps());
435 
436  ICpuKernel::configure(win);
437 
438  const bool is_qasymm8_signed =
439  (dst != nullptr) ? is_data_type_quantized_asymmetric_signed(dst->data_type()) : false;
440 
441  // Set appropriate function
442  if (src->data_layout() == DataLayout::NCHW)
443  {
444  switch (src->data_type())
445  {
446  case DataType::S32:
447  {
448  if (is_qasymm8_signed)
449  {
450  _func = &output_stage_nchw<int8_t>;
451  }
452  else
453  {
454  _func = &output_stage_nchw<uint8_t>;
455  }
456  break;
457  }
458 #ifdef __ARM_FEATURE_FP16_VECTOR_ARITHMETIC
459  case DataType::F16:
460  {
461  _func = &output_stage_nchw<float16_t>;
462  break;
463  }
464 #endif /* __ARM_FEATURE_FP16_VECTOR_ARITHMETIC */
465  case DataType::F32:
466  {
467  _func = &output_stage_nchw<float>;
468  break;
469  }
470  default:
471  {
472  ARM_COMPUTE_ERROR("Unsupported combination of types among the inputs.");
473  }
474  }
475  }
476  else
477  {
478  switch (src->data_type())
479  {
480  case DataType::S32:
481  {
482  if (is_qasymm8_signed)
483  {
484  _func = &output_stage_nhwc<int8_t>;
485  }
486  else
487  {
488  _func = &output_stage_nhwc<uint8_t>;
489  }
490  break;
491  }
492 #ifdef __ARM_FEATURE_FP16_VECTOR_ARITHMETIC
493  case DataType::F16:
494  {
495  _func = &output_stage_nhwc<float16_t>;
496  break;
497  }
498 #endif /* __ARM_FEATURE_FP16_VECTOR_ARITHMETIC */
499  case DataType::F32:
500  {
501  _func = &output_stage_nhwc<float>;
502  break;
503  }
504  default:
505  {
506  ARM_COMPUTE_ERROR("Unsupported combination of types among the inputs.");
507  }
508  }
509  }
