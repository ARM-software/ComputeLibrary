32 void bounding_box_transform_qsymm16(const ITensor *boxes,
33  ITensor *pred_boxes,
34  const ITensor *deltas,
35  BoundingBoxTransformInfo bbinfo,
36  const Window &window)
37 
38 {
39  const size_t num_classes = deltas->info()->tensor_shape()[0] >> 2;
40  const size_t deltas_width = deltas->info()->tensor_shape()[0];
41  const int img_h = std::floor(bbinfo.img_height() / bbinfo.scale() + 0.5f);
42  const int img_w = std::floor(bbinfo.img_width() / bbinfo.scale() + 0.5f);
43 
44  const auto scale_after = (bbinfo.apply_scale() ? bbinfo.scale() : 1.f);
45  const auto scale_before = bbinfo.scale();
46  const auto offset = (bbinfo.correct_transform_coords() ? 1.f : 0.f);
47 
48  auto pred_ptr =
49  reinterpret_cast<uint16_t *>(pred_boxes->buffer() + pred_boxes->info()->offset_first_element_in_bytes());
50  auto delta_ptr = reinterpret_cast<uint8_t *>(deltas->buffer() + deltas->info()->offset_first_element_in_bytes());
51 
52  const auto boxes_qinfo = boxes->info()->quantization_info().uniform();
53  const auto deltas_qinfo = deltas->info()->quantization_info().uniform();
54  const auto pred_qinfo = pred_boxes->info()->quantization_info().uniform();
55 
56  Iterator box_it(boxes, window);
57  execute_window_loop(
58  window,
59  [&](const Coordinates &id)
60  {
61  const auto ptr = reinterpret_cast<uint16_t *>(box_it.ptr());
62  const auto b0 = dequantize_qasymm16(*ptr, boxes_qinfo);
63  const auto b1 = dequantize_qasymm16(*(ptr + 1), boxes_qinfo);
64  const auto b2 = dequantize_qasymm16(*(ptr + 2), boxes_qinfo);
65  const auto b3 = dequantize_qasymm16(*(ptr + 3), boxes_qinfo);
66  const float width = (b2 / scale_before) - (b0 / scale_before) + 1.f;
67  const float height = (b3 / scale_before) - (b1 / scale_before) + 1.f;
68  const float ctr_x = (b0 / scale_before) + 0.5f * width;
69  const float ctr_y = (b1 / scale_before) + 0.5f * height;
70  for (size_t j = 0; j < num_classes; ++j)
71  {
72  // Extract deltas
73  const size_t delta_id = id.y() * deltas_width + 4u * j;
74  const float dx = dequantize_qasymm8(delta_ptr[delta_id], deltas_qinfo) / bbinfo.weights()[0];
75  const float dy = dequantize_qasymm8(delta_ptr[delta_id + 1], deltas_qinfo) / bbinfo.weights()[1];
76  float dw = dequantize_qasymm8(delta_ptr[delta_id + 2], deltas_qinfo) / bbinfo.weights()[2];
77  float dh = dequantize_qasymm8(delta_ptr[delta_id + 3], deltas_qinfo) / bbinfo.weights()[3];
78  // Clip dw and dh
79  dw = std::min(dw, bbinfo.bbox_xform_clip());
80  dh = std::min(dh, bbinfo.bbox_xform_clip());
81  // Determine the predictions
82  const float pred_ctr_x = dx * width + ctr_x;
83  const float pred_ctr_y = dy * height + ctr_y;
84  const float pred_w = std::exp(dw) * width;
85  const float pred_h = std::exp(dh) * height;
86  // Store the prediction into the output tensor
87  pred_ptr[delta_id] = quantize_qasymm16(
88  scale_after * utility::clamp<float>(pred_ctr_x - 0.5f * pred_w, 0.f, img_w - 1.f), pred_qinfo);
89  pred_ptr[delta_id + 1] = quantize_qasymm16(
90  scale_after * utility::clamp<float>(pred_ctr_y - 0.5f * pred_h, 0.f, img_h - 1.f), pred_qinfo);
91  pred_ptr[delta_id + 2] = quantize_qasymm16(
92  scale_after * utility::clamp<float>(pred_ctr_x + 0.5f * pred_w - offset, 0.f, img_w - 1.f),
93  pred_qinfo);
94  pred_ptr[delta_id + 3] = quantize_qasymm16(
95  scale_after * utility::clamp<float>(pred_ctr_y + 0.5f * pred_h - offset, 0.f, img_h - 1.f),
96  pred_qinfo);
97  }
98  },
99  box_it);
100 }
