36  *
37  * @tparam T Type to create a reverse range on

39 template <typename T>
40 class reverse_iterable
41 {
42 public:
43  /** Default constructor
44  *
45  * @param[in] it Value to reverse iterate on
46  */
47  explicit reverse_iterable(T &it) : _it(it)
48  {
49  }
50 
51  /** Get beginning of iterator.
52  *
53  * @return beginning of iterator.
54  */
55  typename T::reverse_iterator begin()
56  {
57  return _it.rbegin();
58  }
59 
60  /** Get end of iterator.
61  *
62  * @return end of iterator.
63  */
64  typename T::reverse_iterator end()
65  {

67  }
68 
69  /** Get beginning of const iterator.
70  *
71  * @return beginning of const iterator.
72  */
73  typename T::const_reverse_iterator cbegin()

77 
78  /** Get end of const iterator.
79  *
80  * @return end of const iterator.
81  */
82  typename T::const_reverse_iterator cend()

85  }
86 
87 private:

89 };
