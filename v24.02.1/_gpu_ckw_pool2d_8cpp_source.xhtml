59 void GpuCkwPool2d::write_component_code(const ComponentGroup &comp_group,
60  GpuCkwVariableTable &vtable,
61  GpuCkwScopedKernelWriter writer) const
62 {
63  const uint32_t width_idx = get_data_layout_dimension_index(_src->data_layout(), DataLayoutDimension::WIDTH);
64  const uint32_t height_idx = get_data_layout_dimension_index(_src->data_layout(), DataLayoutDimension::HEIGHT);
65 
66  /********************************************************************************
67  * 1 - Define tensors
68  ********************************************************************************/
69  GpuCkwComponentArgument *src = vtable.declare_variable(comp_group, writer, _src, "src");
70  GpuCkwComponentArgument *dst = vtable.declare_variable(comp_group, writer, _dst, "dst");
71 
72  /********************************************************************************
73  * 2 - Define CKW constants
74  ********************************************************************************/
75  const auto dst_dt = to_ckw(_dst->data_type());
76  const auto pool_sz_x = static_cast<int32_t>(_attributes.pool_size().x());
77  const auto pool_sz_y = static_cast<int32_t>(_attributes.pool_size().y());
78  const auto pad_x = static_cast<int32_t>(_attributes.pad().left);
79  const auto pad_y = static_cast<int32_t>(_attributes.pad().top);
80  const auto stride_x = static_cast<int32_t>(_attributes.stride().x());
81  const auto stride_y = static_cast<int32_t>(_attributes.stride().y());
82  const auto src_w = static_cast<int32_t>(_src->dimension(width_idx));
83  const auto src_h = static_cast<int32_t>(_src->dimension(height_idx));
84  const auto dst_h = static_cast<int32_t>(_dst->dimension(height_idx));
85 
86  // CKW constants
87  auto const_pool_sz_x_i32 = writer->declare_constant_tile(ckw::ConstantData({{pool_sz_x}}, ckw::DataType::Int32));
88  auto const_pool_sz_y_i32 = writer->declare_constant_tile(ckw::ConstantData({{pool_sz_y}}, ckw::DataType::Int32));
89  auto const_pad_x_i32 = writer->declare_constant_tile(ckw::ConstantData({{pad_x}}, ckw::DataType::Int32));
90  auto const_pad_y_i32 = writer->declare_constant_tile(ckw::ConstantData({{pad_y}}, ckw::DataType::Int32));
91  auto const_stride_x_i32 = writer->declare_constant_tile(ckw::ConstantData({{stride_x}}, ckw::DataType::Int32));
92  auto const_stride_y_i32 = writer->declare_constant_tile(ckw::ConstantData({{stride_y}}, ckw::DataType::Int32));
93  auto const_src_w_i32 = writer->declare_constant_tile(ckw::ConstantData({{src_w}}, ckw::DataType::Int32));
94  auto const_src_h_i32 = writer->declare_constant_tile(ckw::ConstantData({{src_h}}, ckw::DataType::Int32));
95  auto const_dst_h_i32 = writer->declare_constant_tile(ckw::ConstantData({{dst_h}}, ckw::DataType::Int32));
96  auto const_0_i32 = writer->declare_constant_tile(ckw::ConstantData({{0}}, ckw::DataType::Int32));
97  auto const_pos_1_i32 = writer->declare_constant_tile(ckw::ConstantData({{1}}, ckw::DataType::Int32));
98  auto const_0_fp = writer->declare_constant_tile(ckw::ConstantData({{0.0f}}, dst_dt));
99  auto const_lowest_val_fp =
100  writer->declare_constant_tile(ckw::ConstantData({{std::numeric_limits<float>::lowest()}}, ckw::DataType::Fp32));
101  auto const_neg_inf_val_fp = writer->declare_constant_tile(ckw::ConstantData({{-1.0f / 0.0f}}, ckw::DataType::Fp32));
102 
103  /********************************************************************************
104  * 3 - Define the compute block parameters and destination tile (if not root component)
105  * Bind the tile to the tensor to share it among different components and
106  * initialize the compute block parameters
107  ********************************************************************************/
108  // The n0 and m0 parameters from root_window only refers to the output
109  const auto root_window = comp_group.get_root_component()->ckw_component_driver()->get_window();
110 
111  // Destination compute block size
112  const int32_t dst_n0 = root_window.x().step();
113  const int32_t dst_m0 = root_window.y().step();
114 
115  // Destination compute block size left-over
116  const int32_t dst_n0_partial = _dst->dimension(0) % dst_n0;
117  const int32_t dst_m0_partial = _dst->dimension(1) % dst_m0;
118 
119  // Shift-back for the overlapping-min strategy
120  const int32_t dst_shift_back = (dst_n0 - dst_n0_partial) % dst_n0;
121 
122  ckw::TensorSampler sampler_dst;
123  sampler_dst.format(ckw::TensorSamplerFormat::Dim0_Dim1_Dim2);
124  if (dst_n0_partial == 0)
125  {
126  sampler_dst.address_mode_x(ckw::TensorSamplerAddressModeX::None);
127  }
128  else
129  {
130  sampler_dst.address_mode_x(ckw::TensorSamplerAddressModeX::OverlappingMin);
131  }
132 
133  if (dst_m0_partial == 0)
134  {
135  sampler_dst.address_mode_y(ckw::TensorSamplerAddressModeY::None);
136  }
137  else
138  {
139  sampler_dst.address_mode_y(ckw::TensorSamplerAddressModeY::ClampToBorderMaxOnly);
140  }
141 
142  sampler_dst.address_mode_z(ckw::TensorSamplerAddressModeZ::None);
143  sampler_dst.storage(ckw::TensorStorageType::BufferUint8Ptr);
144 
145  // Declare destination tile
146  auto tile_dst = writer->declare_tile("dst", ckw::TileInfo(dst_dt, dst_m0, dst_n0));
147 
148  // Initialize destination tile
149  writer->op_assign(tile_dst, const_0_fp);
150 
151  // Bind tile to the tensor
152  dst->init_virtual_tensor(tile_dst, sampler_dst);
153 
154  /********************************************************************************
155  * 4 - Define the compute block parameters CKW constants
156  ********************************************************************************/
157  // Only now we can declare the N0 and M0 as constant
158  auto const_dst_n0_i32 = writer->declare_constant_tile(ckw::ConstantData({{dst_n0}}, ckw::DataType::Int32));
159  auto const_dst_m0_i32 = writer->declare_constant_tile(ckw::ConstantData({{dst_m0}}, ckw::DataType::Int32));
160  auto const_shift_back_dst_n0_i32 =
161  writer->declare_constant_tile(ckw::ConstantData({{dst_shift_back}}, ckw::DataType::Int32));
162 
163  /********************************************************************************
164  * 5 - Define the sampler for the input tensor
165  ********************************************************************************/
166  ckw::TensorSampler sampler_src;
167  sampler_src.format(ckw::TensorSamplerFormat::Dim0_Dim1_Dim2);
168  sampler_src.address_mode_x(ckw::TensorSamplerAddressModeX::None);
169  sampler_src.address_mode_y(ckw::TensorSamplerAddressModeY::None);
170  sampler_src.address_mode_z(ckw::TensorSamplerAddressModeZ::None);
171 
172  /********************************************************************************
173  * 6 - Extra operations required before writing the main code
174  ********************************************************************************/
175  // Check if it is global pooling
176  const bool is_global_pooling = (pool_sz_x == src_w) && (pool_sz_y == src_h) && (pad_x == 0) && (pad_y == 0);
177 
178  // Accumulate always in F32 if the pool type is not MAX
179  const bool acc_f32 = (dst_dt == ckw::DataType::Fp32) ||
180  ((dst_dt == ckw::DataType::Fp16) && _attributes.pool_type() != PoolingType::MAX);
181 
182  const auto acc_dt = acc_f32 ? ckw::DataType::Fp32 : ckw::DataType::Fp16;
183 
184  const bool is_wider_acc = dst_dt != acc_dt;
185 
186  /********************************************************************************
187  * 7 - Get the coordinates of the destination tile
188  ********************************************************************************/
189  auto tile_gid_0 = writer->declare_tile("gid_0", ckw::TileInfo(ckw::DataType::Int32));
190  auto tile_gid_1 = writer->declare_tile("gid_1", ckw::TileInfo(ckw::DataType::Int32));
191  auto tile_gid_2 = writer->declare_tile("gid_2", ckw::TileInfo(ckw::DataType::Int32));
192 
193  writer->op_get_global_id(tile_gid_0, 0);
194  writer->op_get_global_id(tile_gid_1, 1);
195  writer->op_get_global_id(tile_gid_2, 2);
196 
197  auto tile_cout0 = writer->declare_tile("cout0", ckw::TileInfo(ckw::DataType::Int32)); // OFM
198  auto tile_mout0 = writer->declare_tile("mout0", ckw::TileInfo(ckw::DataType::Int32)); // WIDTH
199  auto tile_mout1 = writer->declare_tile("mout1", ckw::TileInfo(ckw::DataType::Int32)); // HEIGHT
200  auto tile_bout0 = writer->declare_tile("bout0", ckw::TileInfo(ckw::DataType::Int32)); // BATCH SIZE IDX
201 
202  // Calculate coordinates
203  get_coordinate_from_gws_overlapping_min(writer, tile_cout0, tile_gid_0, const_dst_n0_i32,
204  const_shift_back_dst_n0_i32, const_0_i32);
205  get_coordinate_from_gws(writer, tile_mout0, tile_gid_1, const_dst_m0_i32);
206  writer->op_binary(tile_mout1, ckw::BinaryOp::Mod, tile_gid_2, const_dst_h_i32);
207  writer->op_binary(tile_bout0, ckw::BinaryOp::Div, tile_gid_2, const_dst_h_i32);
208 
209  /********************************************************************************
210  * 8 - Write the rest of the code
211  ********************************************************************************/
212  // A tile used to temporarily store results or as an accumulator in case of AVG and L2 pooling.
213  auto tile_res = writer->declare_tile("tile_res", ckw::TileInfo(acc_dt, dst_m0, dst_n0));
214 
215  // Initialise result tile with appropriate value
216  if (_attributes.pool_type() == PoolingType::MAX)
217  {
218  if (_settings.use_inf_as_limit())
219  {
220  writer->op_cast(tile_res, const_neg_inf_val_fp, ckw::ConvertPolicy::None);
221  }
222  else
223  {
224  writer->op_cast(tile_res, const_lowest_val_fp, ckw::ConvertPolicy::None);
225  }
226  }
227  else
228  {
229  writer->op_cast(tile_res, const_0_fp, ckw::ConvertPolicy::None);
230  }
231 
232  // tile_idx_in_w = tile_mout0 * STRIDE_X - PAD_X
233  auto tile_src_coord_x_start = writer->declare_tile("idx_in_w", ckw::DataType::Int32);
234  writer->op_binary(tile_src_coord_x_start, ckw::BinaryOp::Mul, tile_mout0, const_stride_x_i32);
235  writer->op_binary(tile_src_coord_x_start, ckw::BinaryOp::Sub, tile_src_coord_x_start, const_pad_x_i32);
236 
237  // tile_idx_in_h = tile_mout1 * STRIDE_Y - PAD_Y
238  auto tile_src_coord_y_start = writer->declare_tile("idx_in_h", ckw::DataType::Int32);
239  writer->op_binary(tile_src_coord_y_start, ckw::BinaryOp::Mul, tile_mout1, const_stride_y_i32);
240  writer->op_binary(tile_src_coord_y_start, ckw::BinaryOp::Sub, tile_src_coord_y_start, const_pad_y_i32);
241 
242  auto tile_neg_src_coord_x_start = writer->declare_tile("neg_src_coord_x_start", ckw::DataType::Int32);
243  auto tile_neg_src_coord_y_start = writer->declare_tile("neg_src_coord_y_start", ckw::DataType::Int32);
244 
245  writer->op_binary(tile_neg_src_coord_x_start, ckw::BinaryOp::Sub, const_0_i32, tile_src_coord_x_start);
246  writer->op_binary(tile_neg_src_coord_y_start, ckw::BinaryOp::Sub, const_0_i32, tile_src_coord_y_start);
247 
248  // int pool_x_s = max((int)0, -idx_in_w);
249  // int pool_x_e = min((int)POOL_SIZE_X, (int)SRC_WIDTH - idx_in_w);
250  // int pool_y_s = max((int)0, -idx_in_h);
251  // int pool_y_e = min((int)POOL_SIZE_Y, (int)SRC_HEIGHT - idx_in_h);
252  auto tile_pool_x_s = writer->declare_tile("pool_x_s", ckw::DataType::Int32);
253  auto tile_pool_y_s = writer->declare_tile("pool_y_s", ckw::DataType::Int32);
254  auto tile_pool_x_e = writer->declare_tile("pool_x_e", ckw::DataType::Int32);
255  auto tile_pool_y_e = writer->declare_tile("pool_y_e", ckw::DataType::Int32);
256 
257  writer->op_binary(tile_pool_x_s, ckw::BinaryOp::Max, const_0_i32, tile_neg_src_coord_x_start);
258  writer->op_binary(tile_pool_x_e, ckw::BinaryOp::Add, const_src_w_i32, tile_neg_src_coord_x_start);
259  writer->op_binary(tile_pool_x_e, ckw::BinaryOp::Min, const_pool_sz_x_i32, tile_pool_x_e);
260  writer->op_binary(tile_pool_y_s, ckw::BinaryOp::Max, const_0_i32, tile_neg_src_coord_y_start);
261  writer->op_binary(tile_pool_y_e, ckw::BinaryOp::Add, const_src_h_i32, tile_neg_src_coord_y_start);
262  writer->op_binary(tile_pool_y_e, ckw::BinaryOp::Min, const_pool_sz_y_i32, tile_pool_y_e);
263 
264  // #if defined(EXCLUDE_PADDING)
265  // int filter_size = (pool_y_e - pool_y_s) * (pool_x_e - pool_x_s);
266  // #else // defined(EXCLUDE_PADDING)
267  // int filter_size = POOL_SIZE_X * POOL_SIZE_Y;
268  // #endif // defined(EXCLUDE_PADDING)
269  auto tile_filter_size = writer->declare_tile("filter_size", ckw::DataType::Int32);
270  if (_attributes.exclude_padding())
271  {
272  auto tile_x_diff = writer->declare_tile("x_diff", ckw::DataType::Int32);
273  auto tile_y_diff = writer->declare_tile("y_diff", ckw::DataType::Int32);
274 
275  writer->op_binary(tile_x_diff, ckw::BinaryOp::Sub, tile_pool_x_e, tile_pool_x_s);
276  writer->op_binary(tile_y_diff, ckw::BinaryOp::Sub, tile_pool_y_e, tile_pool_y_s);
277  writer->op_binary(tile_filter_size, ckw::BinaryOp::Mul, tile_x_diff, tile_y_diff);
278  }
279  else
280  {
281  writer->op_binary(tile_filter_size, ckw::BinaryOp::Mul, const_pool_sz_x_i32, const_pool_sz_y_i32);
282  }
283 
284  auto tile_x = writer->declare_tile("x", ckw::DataType::Int32);
285  auto tile_y = writer->declare_tile("y", ckw::DataType::Int32);
286 
287  if (is_global_pooling)
288  {
289  writer->op_assign(tile_y, const_0_i32);
290  writer->op_assign(tile_pool_y_e, const_pool_sz_y_i32);
291  }
292  else
293  {
294  writer->op_assign(tile_y, tile_pool_y_s);
295  }
296 
297  // Y dim for-loop
298  writer->op_for_loop(
299  tile_y, ckw::BinaryOp::Less, tile_pool_y_e, tile_y, ckw::AssignmentOp::Increment, const_pos_1_i32,
300  [&]()
301  {
302  // Reset the iterator for the inner loop
303  if (is_global_pooling)
304  {
305  writer->op_assign(tile_x, const_0_i32);
306  writer->op_assign(tile_pool_x_e, const_pool_sz_x_i32);
307  }
308  else
309  {
310  writer->op_assign(tile_x, tile_pool_x_s);
311  }
312 
313  auto tile_src_coord_y = writer->declare_tile("src_coord_y", ckw::DataType::Int32);
314  writer->op_binary(tile_src_coord_y, ckw::BinaryOp::Add, tile_src_coord_y_start, tile_y);
315 
316  // X dim for-loop
317  writer->op_for_loop(
318  tile_x, ckw::BinaryOp::Less, tile_pool_x_e, tile_x, ckw::AssignmentOp::Increment, const_pos_1_i32,
319  [&]()
320  {
321  auto tile_src_coord_x = writer->declare_tile("src_coord_x", ckw::DataType::Int32);
322  writer->op_binary(tile_src_coord_x, ckw::BinaryOp::Add, tile_src_coord_x_start, tile_x);
323 
324  ckw::DataType src_dt = to_ckw(_src->data_type());
325  auto tile_src = writer->declare_tile("tile_src", ckw::TileInfo(acc_dt, dst_m0, dst_n0));
326 
327  // Load src tile
328  if (is_wider_acc)
329  {
330  auto tile_src0 = writer->declare_tile("src_tile0", ckw::TileInfo(src_dt, dst_m0, dst_n0));
331  writer->op_load(tile_src0, src->tensor(), sampler_src, tile_cout0, tile_src_coord_x,
332  tile_src_coord_y, tile_bout0);
333  writer->op_cast(tile_src, tile_src0, ckw::ConvertPolicy::None);
334  }
335  else
336  {
337  writer->op_load(tile_src, src->tensor(), sampler_src, tile_cout0, tile_src_coord_x,
338  tile_src_coord_y, tile_bout0);
339  }
340 
341  // Take the square of the input, for L2 Pooling
342  if (_attributes.pool_type() == PoolingType::L2)
343  {
344  writer->op_binary(tile_src, ckw::BinaryOp::Mul, tile_src, tile_src);
345  }
346 
347  // Perfom Pooling op
348  if (_attributes.pool_type() == PoolingType::MAX)
349  {
350  writer->op_binary(tile_res, ckw::BinaryOp::Max, tile_res, tile_src);
351  }
352  else
353  {
354  writer->op_binary(tile_res, ckw::BinaryOp::Add, tile_res, tile_src);
355  }
356  });
357  });
358 
359  if ((_attributes.pool_type() == PoolingType::AVG) || (_attributes.pool_type() == PoolingType::L2))
360  {
361  // Filter_size is automatically broadcasted in the operation
362  auto tile_filter_size_fp = writer->declare_tile("filter_size_fp", ckw::TileInfo(acc_dt));
363  writer->op_cast(tile_filter_size_fp, tile_filter_size, ckw::ConvertPolicy::None);
364  writer->op_binary(tile_res, ckw::BinaryOp::Div, tile_res, tile_filter_size_fp);
365  }
366 
367  // Take square root of the result in L2 pooling
368  if (_attributes.pool_type() == PoolingType::L2)
369  {
370  writer->op_unary(tile_res, ckw::UnaryOp::Sqrt, tile_res);
371  }
372 
373  // Store the results and do casting if mixed precision
374  if (is_wider_acc)
375  {
376  writer->op_cast(tile_dst, tile_res, ckw::ConvertPolicy::None);
377  }
378  else
379  {
380  writer->op_assign(tile_dst, tile_res);
381  }
382 }

384 Window GpuCkwPool2d::get_window() const
385 {
386  ARM_COMPUTE_ERROR_ON_MSG(_dst->tensor_shape().total_size() == 0U, "Destination tensor is not initialized");
387 
388  TensorShape output_shape = _dst->tensor_shape();
389  const uint32_t vec_size = adjust_vec_size(((_dst->data_type() == DataType::F32) ? 2 : 4), _dst->dimension(0));
390  // Create and configure kernel window
391  auto win = calculate_max_window(output_shape, Steps(vec_size));
392  win = win.collapse_if_possible(win, Window::DimZ); // collapse window on batch size.
393  return win;
394 }
