68 std::string ClTemplateDirectConv2d::get_component_code(const ComponentGroup &comp_group) const
69 {
70  ARM_COMPUTE_UNUSED(comp_group);
71 
72  const auto channel_idx = get_data_layout_dimension_index(_src->data_layout(), DataLayoutDimension::CHANNEL);
73  const auto k0 = adjust_vec_size(_settings.direct_conv_descriptor().k0, _src->dimension(channel_idx));
74  const bool leftover_loop = (_src->dimension(channel_idx) % k0) != 0;
75 
76  std::string code = R"_(
77 //------------------ START KERNEL {{meta_kernel_id}} ---------------------
78 // IN_0(src) {{src}}
79 // IN_1(wei) {{weight}}
80 )_";
81  if (_bias && _bias->has_valid_id())
82  {
83  code += R"_(
84 // IN_1(bia) {{bias}}
85 )_";
86  }
87  code += R"_(
88 // OUT(dst, accum) {{dst}}
89 
90 TILE(uint, M0, 1, g_dst_indirect_y);
91 
92 {
93 #define _IWEI_WIDTH {{WEI_WIDTH}}
94 #define _IWEI_HEIGHT {{WEI_HEIGHT}}
95 #define _ISRC_WIDTH {{SRC_WIDTH}}
96 #define _ISRC_HEIGHT {{SRC_HEIGHT}}
97 #define _ISRC_CHANNELS {{SRC_CHANNELS}}
98 #define _IDST_WIDTH {{DST_WIDTH}}
99 #define _IDST_HEIGHT {{DST_HEIGHT}}
100 #define _IDST_CHANNELS {{DST_CHANNELS}}
101 #define _IY_MULTIPLIER (_IWEI_WIDTH * _IWEI_HEIGHT)
102 
103  TILE(int, M0, 1, xi);
104  TILE(int, M0, 1, yi);
105 
106  // Convert the linear index to coordinate
107  LOOP_UNROLLING(int, i, 0, 1, M0,
108  {
109  xi[0].s[i] = ((g_ind_1 + i) % _IDST_WIDTH) * {{STRIDE_X}};
110  yi[0].s[i] = ((g_ind_1 + i) / _IDST_WIDTH) * {{STRIDE_Y}};
111  xi[0].s[i] -= {{PAD_LEFT}};
112  yi[0].s[i] -= {{PAD_TOP}};
113  })
114 
115  LOOP_UNROLLING(int, i, 0, 1, M0,
116  {
117  {{dst}}[i].v = 0;
118  })
119 
120  for(int i = 0; i < (_IWEI_WIDTH * _IWEI_HEIGHT); ++i)
121  {
122  int xk = i % _IWEI_WIDTH;
123  int yk = i / _IWEI_WIDTH;
124 
125  TILE(int, 1, M0, my);
126 
127  LOOP_UNROLLING(int, i, 0, 1, M0,
128  {
129  int x_s = xi[0].s[i] + xk;
130  int y_s = yi[0].s[i] + yk;
131  my[0].s[i] = x_s + y_s *_ISRC_WIDTH;
132  my[0].s[i] = my[0].s[i] + g_ind_2 * (int)(_ISRC_WIDTH * _ISRC_HEIGHT);
133  my[0].s[i] = select(-1, my[0].s[i], x_s >= 0);
134  my[0].s[i] = select(-1, my[0].s[i], x_s < _ISRC_WIDTH);
135  my[0].s[i] = select(-1, my[0].s[i], y_s >= 0);
136  my[0].s[i] = select(-1, my[0].s[i], y_s < _ISRC_HEIGHT);
137  })
138 
139  int ck = 0;
140  for(; ck <= (_ISRC_CHANNELS - K0); ck += K0)
141  {
142  TILE({{SRC_DATA_TYPE}}, M0, K0, a);
143  TILE({{WEI_DATA_TYPE}}, N0, K0, b);
144 
145  LOOP_UNROLLING(int, i, 0, 1, M0,
146  {
147  a[i].v = {{ZERO_VALUE}};
148  })
149 
150  LOOP_UNROLLING(int, i, 0, 1, N0,
151  {
152  b[i].v = {{ZERO_VALUE}};
153  })
154 
155  T_LOAD2D_INDIRECT({{SRC_DATA_TYPE}}, M0, K0, {{SRC_TENSOR_TYPE}}, {{src}}, ck, {{src}}_stride_y, my, a);
156 
157  T_LOAD({{WEI_DATA_TYPE}}, N0, K0, {{WEI_TENSOR_TYPE}}, {{weight}}, ck, g_ind_0 * _IY_MULTIPLIER + i, _IY_MULTIPLIER, {{weight}}_stride_y, b);
158 
159  T_MMUL({{SRC_DATA_TYPE}}, {{WEI_DATA_TYPE}}, {{ACC_DATA_TYPE}}, M0, N0, K0, NT, T, a, b, {{dst}});
160  }
161 )_";
162 
163  if (leftover_loop)
164  {
165  code += R"_(
166  for(; ck < _ISRC_CHANNELS; ++ck)
167  {
168  TILE({{SRC_DATA_TYPE}}, M0, 1, a);
169  TILE({{WEI_DATA_TYPE}}, N0, 1, b);
170 
171  LOOP_UNROLLING(int, i, 0, 1, M0,
172  {
173  a[i].v = {{ZERO_VALUE}};
174  })
175 
176  LOOP_UNROLLING(int, i, 0, 1, N0,
177  {
178  b[i].v = {{ZERO_VALUE}};
179  })
180 
181  T_LOAD2D_INDIRECT({{SRC_DATA_TYPE}}, M0, 1, {{SRC_TENSOR_TYPE}}, {{src}}, ck, {{src}}_stride_y, my, a);
182 
183  T_LOAD({{WEI_DATA_TYPE}}, N0, 1, BUFFER, {{weight}}, ck, g_ind_0 * _IY_MULTIPLIER + i, _IY_MULTIPLIER, {{weight}}_stride_y, b);
184 
185  T_MMUL({{SRC_DATA_TYPE}}, {{WEI_DATA_TYPE}}, {{ACC_DATA_TYPE}}, M0, N0, 1, NT, T, a, b, {{dst}});
186  }
187  )_";
188  }
189 
190  code += R"_(
191 #undef _I_WEI_WIDTH
192 #undef _I_WEI_HEIGHT
193 #undef _ISRC_WIDTH
194 #undef _ISRC_HEIGHT
195 #undef _ISRC_CHANNELS
196 #undef _IDST_WIDTH
197 #undef _IDST_HEIGHT
198 #undef _IDST_CHANNELS
199 #undef _IY_MULTIPLIER
200 
201  }
202 )_";
203 
204  if (_bias && _bias->has_valid_id())
205  {
206  code += R"_(
207  TILE({{BIA_DATA_TYPE}}, 1, N0, bias0);
208 
209  T_LOAD({{BIA_DATA_TYPE}}, 1, N0, BUFFER, {{bias}}, g_ind_0, 0, 1, 0, bias0);
210 
211  T_ELTWISE_BROADCAST_ADD_X({{ACC_DATA_TYPE}}, M0, N0, {{dst}}, bias0, {{dst}});
212  )_";
213  }
214 
215  code += R"_(
216  LOOP_UNROLLING(int, i, 0, 1, M0,
217  {
218  g_dst_indirect_y[i].v = (uint)min(g_ind_1 + i, (int)({{DST_WIDTH}} * {{DST_HEIGHT}}) - 1);
219  g_dst_indirect_y[i].v += g_ind_2 * (int)({{DST_WIDTH}} * {{DST_HEIGHT}});
220  })
221 }
222 //------------------ END KERNEL {{meta_kernel_id}} ---------------------
223 )_";
224  return code;
225 }

244 TagLUT ClTemplateDirectConv2d::get_tag_lut(const GpuKernelVariableTable &vtable, const ComponentGroup &comp_group) const
245 {
246  TagLUT lut{};
247  // Arguments and global shared variables
248  lut["src"] = vtable.get_variable(_src);
249  lut["weight"] = vtable.get_variable(_weight);
250 
251  if (_bias && _bias->has_valid_id()) // optional bias
252  {
253  lut["bias"] = vtable.get_variable(_bias);
254  lut["BIA_DATA_TYPE"] = get_cl_type_from_data_type(_bias->data_type());
255  }
256  lut["dst"] = vtable.get_variable(_dst);
257 
258  const auto dst_argument = vtable.get_variable(comp_group.get_any_dst_tensor());
259  lut["arg_dst"] = dst_argument.uniq_name;
260 
261  // Local build options
262  lut["meta_kernel_id"] = id();
263  lut["ACC_DATA_TYPE"] = _src->data_type();
264  lut["SRC_DATA_TYPE"] = _src->data_type();
265  lut["WEI_DATA_TYPE"] = _weight->data_type();
266 
267  lut["SRC_TENSOR_TYPE"] = "BUFFER";
268  switch (vtable.get_variable(_weight).kernel_argument_info.type)
269  {
270  case GpuKernelArgumentInfo::Type::Image_Export_To_ClImage2D:
271  case GpuKernelArgumentInfo::Type::Image_3D_Export_To_ClImage2D:
272  case GpuKernelArgumentInfo::Type::Tensor_4D_t_Image:
273  {
274  lut["WEI_TENSOR_TYPE"] = "IMAGE";
275  break;
276  }
277  default:
278  {
279  lut["WEI_TENSOR_TYPE"] = "BUFFER";
280  break;
281  }
282  }
283  const auto width_idx = 1;
284  const auto height_idx = 2;
285  const auto channel_idx = 0;
286 
287  lut["SRC_WIDTH"] = _src->dimension(width_idx);
288  lut["SRC_HEIGHT"] = _src->dimension(height_idx);
289  lut["SRC_CHANNELS"] = _src->dimension(channel_idx);
290 
291  lut["WEI_WIDTH"] = _weight->dimension(width_idx);
292  lut["WEI_HEIGHT"] = _weight->dimension(height_idx);
293 
294  lut["DST_WIDTH"] = _dst->dimension(width_idx);
295  lut["DST_HEIGHT"] = _dst->dimension(height_idx);
296  lut["DST_CHANNELS"] = _dst->dimension(channel_idx);
297 
298  lut["STRIDE_X"] = _attributes.stride().x();
299  lut["STRIDE_Y"] = _attributes.stride().y();
300 
301  lut["PAD_LEFT"] = _attributes.pad().left;
302  lut["PAD_TOP"] = _attributes.pad().top;
303 
304  lut["ZERO_VALUE"] = 0;
305 
306  return lut;
307 }

309 CLBuildOptions ClTemplateDirectConv2d::get_build_options(const ComponentGroup &comp_group) const
310 {
311  const unsigned int channel_idx = get_data_layout_dimension_index(_src->data_layout(), DataLayoutDimension::CHANNEL);
312 
313  const auto root_window = comp_group.get_root_component()->template_writer()->get_window();
314  const unsigned int n0 = root_window.x().step();
315  const unsigned int m0 = root_window.y().step();
316  const unsigned int k0 = adjust_vec_size(_settings.direct_conv_descriptor().k0, _src->dimension(channel_idx));
317  const unsigned int partial_store_n0 = _dst->dimension(0) % n0;
318 
319  CLBuildOptions build_opts{};
320  if (_settings.fast_relaxed_math())
321  {
322  build_opts.add_option("-cl-fast-relaxed-math");
323  }
324  else
325  {
326  // -cl-fast-relaxed-math also sets -cl-finite-math-only and -cl-unsafe-math-optimizations
327  // to disable -cl-finite-math-only, we only include -cl-unsafe-math-optimizations
328  build_opts.add_option("-cl-unsafe-math-optimizations");
329  }
330 
331  build_opts.add_option("-DN0=" + support::cpp11::to_string(n0));
332  build_opts.add_option("-DM0=" + support::cpp11::to_string(m0));
333  build_opts.add_option("-DK0=" + support::cpp11::to_string(k0));
334  build_opts.add_option("-DPARTIAL_N0=" + support::cpp11::to_string(partial_store_n0));
335 
336  return build_opts;
337 }

371 Window ClTemplateDirectConv2d::get_window() const
372 {
373  ARM_COMPUTE_ERROR_ON_MSG(_dst->tensor_shape().total_size() == 0U, "Destination tensor is not initialized");
374 
375  const auto output_shape = _dst->tensor_shape();
376  const auto desc = _settings.direct_conv_descriptor();
377 
378  const unsigned int n0 = adjust_vec_size(desc.n0, output_shape[0]);
379  const unsigned int m0 = adjust_vec_size(desc.m0, output_shape[1] * output_shape[2]);
380 
381  // Create and configure kernel window
382  Window win = calculate_max_window(output_shape, Steps(n0, m0));
383 
384  const size_t dim_y_collapsed = ceil_to_multiple(output_shape[1] * output_shape[2], m0);
385  win.set(Window::DimY, Window::Dimension(0, dim_y_collapsed, m0));
386  win.set(Window::DimZ, Window::Dimension(0, output_shape.total_size_upper(3), 1));
387 
388  return win;
389 }
