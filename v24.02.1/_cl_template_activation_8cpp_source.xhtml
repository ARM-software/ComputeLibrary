55 std::string ClTemplateActivation::get_component_code(const ComponentGroup &comp_group) const
56 {
57  std::string code;
58  const bool is_root = (comp_group.get_root_component()->id() == this->id());
59 
60  code = R"_(
61 //------------------ START KERNEL {{meta_kernel_id}} ---------------------
62 )_";
63  if (is_root)
64  {
65  code += R"_(
66 // IN(src) {{src}}
67 // OUT(dst, accum) {{dst}}
68 
69 TILE({{DATA_TYPE}}, M0, N0, {{src}});
70 TILE(uint, M0, 1, g_dst_indirect_y);
71 {
72  {{src}}_offset_first_element_in_bytes += g_ind_2 * {{src}}_stride_z;
73 
74  T_LOAD({{DATA_TYPE}}, M0, N0, {{TENSOR_TYPE}}, {{src}}, g_ind_0, g_ind_1, 1, {{src}}_stride_y, {{src}});
75 
76  T_ACTIVATION({{DATA_TYPE}}, M0, N0, {{ACT}}, {{A_VAL}}, {{B_VAL}}, {{src}}, {{dst}});
77 }
78 
79 LOOP_UNROLLING(int, i, 0, 1, M0,
80 {
81  g_dst_indirect_y[i].v = (uint)min((int)(g_ind_1 + i), (int)({{arg_dst}}_w) - 1);
82  g_dst_indirect_y[i].v += (int)(g_ind_2 % {{arg_dst}}_h) * (int)({{arg_dst}}_w);
83  g_dst_indirect_y[i].v += (int)(g_ind_2 / {{arg_dst}}_h) * (int)({{arg_dst}}_w * {{arg_dst}}_h);
84 })
85 )_";
86  }
87  else
88  {
89  code += R"_(
90 // IN/OUT(src, accum) {{src}}
91 
92 {
93  T_ACTIVATION({{DATA_TYPE}}, M0, N0, {{ACT}}, {{A_VAL}}, {{B_VAL}}, {{src}}, {{dst}});
94 }
95 )_";
96  }
97  code += R"_(
98 //------------------ END KERNEL {{meta_kernel_id}} ---------------------
99 )_";
100  return code;
101 }

112 TagLUT ClTemplateActivation::get_tag_lut(const GpuKernelVariableTable &vtable, const ComponentGroup &comp_group) const
113 {
114  ARM_COMPUTE_UNUSED(comp_group);
115 
116  TagLUT lut{};
117  // Arguments and global shared variables
118  lut["src"] = vtable.get_variable(_src);
119  lut["dst"] = vtable.get_variable(_dst);
120 
121  const auto dst_argument = vtable.get_variable(comp_group.get_any_dst_tensor());
122  lut["arg_dst"] = dst_argument.uniq_name;
123 
124  // Local build options
125  lut["meta_kernel_id"] = id();
126  lut["DATA_TYPE"] = get_cl_type_from_data_type(_src->data_type());
127  lut["TENSOR_TYPE"] = "BUFFER";
128 
129  const auto f_act = lower_string(string_from_activation_func(_attributes.activation()));
130 
131  lut["ACT"] = f_act;
132  lut["A_VAL"] = float_to_string_with_full_precision(_attributes.a());
133  lut["B_VAL"] = float_to_string_with_full_precision(_attributes.b());
134 
135  return lut;
136 }

138 CLBuildOptions ClTemplateActivation::get_build_options(const ComponentGroup &comp_group) const
139 {
140  /// NOTE: For now tile sizes (n0, m0) are set by the execution window. This may change in the future
141  const auto root_window = comp_group.get_root_component()->template_writer()->get_window();
142  const unsigned int n0 = root_window.x().step();
143  const unsigned int m0 = root_window.y().step();
144  const unsigned int partial_store_n0 = _dst->dimension(0) % n0;
145 
146  CLBuildOptions build_opts;
147  build_opts.add_option("-DN0=" + support::cpp11::to_string(n0));
148  build_opts.add_option("-DM0=" + support::cpp11::to_string(m0));
149  build_opts.add_option("-DPARTIAL_N0=" + support::cpp11::to_string(partial_store_n0));
150 
151  return build_opts;
152 }

171 Window ClTemplateActivation::get_window() const
172 {
173  ARM_COMPUTE_ERROR_ON_MSG(_dst->tensor_shape().total_size() == 0U, "Destination tensor is not initialized");
174  const unsigned int n0 = adjust_vec_size(16 / _dst->element_size(), _dst->dimension(0));
175  Window win = calculate_max_window(*_dst, Steps(n0));
176  return win.collapse(win, Window::DimZ);
177 }
