37 template <typename ScalarType>
38 inline ScalarType elementwise_op_scalar_imp(ElementWiseUnary op, const ScalarType &a)
39 {
40  switch (op)
41  {
42  case ElementWiseUnary::RSQRT:
43  return 1 / sqrt(a);
44  case ElementWiseUnary::EXP:
45  return std::exp(a);
46  case ElementWiseUnary::NEG:
47  return -a;
48  case ElementWiseUnary::LOG:
49  return std::log(a);
50  case ElementWiseUnary::ABS:
51  return std::abs(a);
52  case ElementWiseUnary::ROUND:
53  return support::cpp11::nearbyint(a);
54  case ElementWiseUnary::SIN:
55  return std::sin(a);
56  default:
57  ARM_COMPUTE_ERROR("NOT_SUPPORTED!");
58  }
59 }

61 template <typename ScalarType, typename VectorType>
62 inline VectorType elementwise_op_imp(ElementWiseUnary op, const VectorType &a)
63 {
64  switch (op)
65  {
66  case ElementWiseUnary::RSQRT:
67  return wrapper::vinvsqrt(a);
68  case ElementWiseUnary::EXP:
69  return wrapper::vexpq(a);
70  case ElementWiseUnary::NEG:
71  return wrapper::vneg(a);
72  case ElementWiseUnary::LOG:
73  return wrapper::vlog(a);
74  case ElementWiseUnary::ABS:
75  return wrapper::vabs(a);
76  case ElementWiseUnary::ROUND:
77  return wrapper::vround(a);
78  case ElementWiseUnary::SIN:
79  return wrapper::vsin(a);
80  default:
81  ARM_COMPUTE_ERROR("NOT_SUPPORTED!");
82  }
83 }

85 template <typename ScalarType>
86 inline void elementwise_op(const ITensor *in, ITensor *out, const Window &window, ElementWiseUnary op)
87 {
88  const int window_step_x = 16 / sizeof(ScalarType);
89  const auto window_start_x = static_cast<int>(window.x().start());
90  const auto window_end_x = static_cast<int>(window.x().end());
91 
92  Window win = window;
93  win.set(Window::DimX, Window::Dimension(0, 1, 1));
94 
95  Iterator input(in, win);
96  Iterator output(out, win);
97 
98  execute_window_loop(
99  win,
100  [&](const Coordinates &)
101  {
102  auto output_ptr = reinterpret_cast<ScalarType *>(output.ptr());
103  const auto input_ptr = reinterpret_cast<const ScalarType *>(input.ptr());
104 
105  int x = window_start_x;
106  for (; x <= window_end_x - window_step_x; x += window_step_x)
107  {
108  wrapper::vstore(output_ptr + x, elementwise_op_imp<ScalarType>(op, wrapper::vloadq(input_ptr + x)));
109  }
110  for (; x < window_end_x; ++x)
111  {
112  *(output_ptr + x) = elementwise_op_scalar_imp(op, *(input_ptr + x));
113  }
114  },
115  input, output);
116 }

118 template <>
119 inline void elementwise_op<int8_t>(const ITensor *in, ITensor *out, const Window &window, ElementWiseUnary op)
120 {
121  const int window_step_x = 16;
122  const auto window_start_x = static_cast<int>(window.x().start());
123  const auto window_end_x = static_cast<int>(window.x().end());
124  const UniformQuantizationInfo qi_in = in->info()->quantization_info().uniform();
125  const UniformQuantizationInfo qi_out = out->info()->quantization_info().uniform();
126  const auto min_clamped_value = vdupq_n_f32((-128 - qi_out.offset) * qi_out.scale);
127  const auto max_clamped_value = vdupq_n_f32((127 - qi_out.offset) * qi_out.scale);
128  Window win = window;
129  win.set(Window::DimX, Window::Dimension(0, 1, 1));
130 
131  Iterator input(in, win);
132  Iterator output(out, win);
133 
134  execute_window_loop(
135  win,
136  [&](const Coordinates &)
137  {
138  int8x16_t vout;
139  auto output_ptr = reinterpret_cast<int8_t *>(output.ptr());
140  const auto input_ptr = reinterpret_cast<const int8_t *>(input.ptr());
141  const auto vconst_0_f32 = vdupq_n_f32(0);
142  auto clamped_value = (op == ElementWiseUnary::LOG) ? min_clamped_value : max_clamped_value;
143 
144  int x = window_start_x;
145  for (; x <= window_end_x - window_step_x; x += window_step_x)
146  {
147  const auto vin = wrapper::vloadq(input_ptr + x);
148 
149  // De-quantize
150  const auto vin_deq = vdequantize(vin, qi_in);
151 
152  // Perform activation
153  float32x4x4_t vtmp_deq = {{
154  elementwise_op_imp<float>(op, vin_deq.val[0]),
155  elementwise_op_imp<float>(op, vin_deq.val[1]),
156  elementwise_op_imp<float>(op, vin_deq.val[2]),
157  elementwise_op_imp<float>(op, vin_deq.val[3]),
158  }};
159 
160  if ((op == ElementWiseUnary::LOG) || (op == ElementWiseUnary::RSQRT))
161  {
162  vtmp_deq.val[0] =
163  vbslq_f32(vcleq_f32(vin_deq.val[0], vconst_0_f32), clamped_value, vtmp_deq.val[0]);
164  vtmp_deq.val[1] =
165  vbslq_f32(vcleq_f32(vin_deq.val[1], vconst_0_f32), clamped_value, vtmp_deq.val[1]);
166  vtmp_deq.val[2] =
167  vbslq_f32(vcleq_f32(vin_deq.val[2], vconst_0_f32), clamped_value, vtmp_deq.val[2]);
168  vtmp_deq.val[3] =
169  vbslq_f32(vcleq_f32(vin_deq.val[3], vconst_0_f32), clamped_value, vtmp_deq.val[3]);
170  }
171 
172  // Re-quantize to new output space
173  vout = vquantize_signed(vtmp_deq, qi_out);
174  wrapper::vstore(output_ptr + x, vout);
175  }
176  for (; x < window_end_x; ++x)
177  {
178  qasymm8_signed_t in = *(reinterpret_cast<const qasymm8_signed_t *>(input_ptr + x));
179  qasymm8_signed_t tmp = 0;
180  float tmp_f = dequantize_qasymm8_signed(in, qi_in);
181  if (tmp_f <= 0.0)
182  {
183  if (op == ElementWiseUnary::LOG)
184  {
185  tmp_f = (-128 - qi_out.offset) * qi_out.scale;
186  }
187  else if (op == ElementWiseUnary::RSQRT)
188  {
189  tmp_f = (127 - qi_out.offset) * qi_out.scale;
190  }
191  else
192  {
193  tmp_f = elementwise_op_scalar_imp<float>(op, tmp_f);
194  }
195  }
196  else
197  {
198  tmp_f = elementwise_op_scalar_imp<float>(op, tmp_f);
199  }
200  tmp = quantize_qasymm8_signed(
201  tmp_f, qi_out,
202  RoundingPolicy::
203  TO_ZERO); // Set rounding policy TO_ZERO to be compatible with vquantize_signed() used above that follow same policy for armv7a.
204  // For aarch64 LUT is used and rounding to nearest is used
205  *(output_ptr + x) = tmp;
206  }
207  },
208  input, output);
209 }

261

265  }

267  {
268  qasymm8_t in = *(reinterpret_cast<const qasymm8_t *>(input_ptr + x));
269  qasymm8_t tmp = 0;
270  float tmp_f = dequantize_qasymm8(in, qi_in);
271  if (tmp_f <= 0.0)
272  {
273  if (op == ElementWiseUnary::LOG)
274  {
275  tmp_f = (0 - qi_out.offset) * qi_out.scale;
276  }
277  else if (op == ElementWiseUnary::RSQRT)
278  {
279  tmp_f = (255 - qi_out.offset) * qi_out.scale;

284  }
285  }
286  else
287  {
288  tmp_f = elementwise_op_scalar_imp<float>(op, tmp_f);
289  }
290  tmp = quantize_qasymm8(tmp_f, qi_out, RoundingPolicy::TO_ZERO);
291  *(output_ptr + x) = tmp;
292  }
293  },
294  input, output);
