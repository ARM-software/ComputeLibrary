48 Status ClComponentElementwiseBinary::validate(const ArgumentPack<ITensorInfo> &tensors,
49  const ElementwiseBinaryCommonAttributes &attributes)
50 {
51  const auto lhs = tensors.get_const_tensor(TensorType::ACL_SRC_0);
52  const auto rhs = tensors.get_const_tensor(TensorType::ACL_SRC_1);
53  const auto dst = tensors.get_const_tensor(TensorType::ACL_DST_0);
54 
55  // Check operator type
56  ARM_COMPUTE_RETURN_ERROR_ON_MSG(supported_ops.find(attributes.operation()) == supported_ops.end(),
57  "Provided Elementwise operation not supported.");
58 
59  // Check validity
60  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(lhs, rhs, dst);
61 
62  //Check data type for different elementwise operators
63  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(lhs, 1, DataType::F32, DataType::F16, DataType::S32,
64  DataType::S16, DataType::U8);
65 
66  // dst shape is correct
67  const TensorShape out_shape = TensorShape::broadcast_shape(lhs->tensor_shape(), rhs->tensor_shape());
68  ARM_COMPUTE_RETURN_ERROR_ON_MSG(out_shape.total_size() == 0, "Inputs are not broadcast compatible");
69  ARM_COMPUTE_RETURN_ERROR_ON_MSG(detail::have_different_dimensions(out_shape, dst->tensor_shape(), 0),
70  "Wrong shape for dst.");
71 
72  const auto &lhs_shape = lhs->tensor_shape();
73  const auto &rhs_shape = rhs->tensor_shape();
74  const auto &dst_shape = dst->tensor_shape();
75 
76  ARM_COMPUTE_RETURN_ERROR_ON_MSG(detail::have_different_dimensions(lhs_shape, dst_shape, 0) &&
77  detail::have_different_dimensions(rhs_shape, dst_shape, 0),
78  "Only LHS or RHS can be broadcasting, not both.");
79 
80  // Dimension Y and Z are collapsed together in the current kernel implementation,
81  // hence they cannot be independently broadcast or non-broadcast.
82  // See: ClTemplateElementwiseBinary::get_window
83  ARM_COMPUTE_RETURN_ERROR_ON_MSG((lhs_shape[1] != dst_shape[1] || rhs_shape[1] != dst_shape[1]) !=
84  (lhs_shape[2] != dst_shape[2] || rhs_shape[2] != dst_shape[2]),
85  "Dimension Y and Z must both be either broadcast or non-broadcast.");
86 
87  ARM_COMPUTE_RETURN_ERROR_ON_MSG(detail::have_different_dimensions(lhs_shape, dst_shape, 3),
88  "LHS broadcast in dimension 3 or higher is not supported.");
89 
90  ARM_COMPUTE_RETURN_ERROR_ON_MSG(detail::have_different_dimensions(rhs_shape, dst_shape, 3),
91  "RHS broadcast in dimension 3 or higher is not supported.");
92 
93  // Matching data type
94  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(lhs, rhs);
95  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(lhs, dst);
96 
97  // Matching data layout
98  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_LAYOUT(lhs, rhs);
99  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_LAYOUT(lhs, dst);
100 
101  // All tensor infos are initialized
102  ARM_COMPUTE_RETURN_ERROR_ON(lhs->tensor_shape().total_size() == 0);
103  ARM_COMPUTE_RETURN_ERROR_ON(rhs->tensor_shape().total_size() == 0);
104  ARM_COMPUTE_RETURN_ERROR_ON(dst->tensor_shape().total_size() == 0);
105 
106  // Device requirements are met
107  ARM_COMPUTE_RETURN_ERROR_ON_F16_UNSUPPORTED(lhs);
108 
109  return Status{};
110 }
