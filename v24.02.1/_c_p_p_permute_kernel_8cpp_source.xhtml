58 template <typename T>
59 void CPPPermuteKernel::run_permute(const Window &window)
60 {
61  // Permute strides
62  Strides strides = _output->info()->strides_in_bytes();
63  Strides perm_strides = strides;
64  permute_strides(perm_strides, _perm);
65 
66  // Create output window
67  Window window_out(window);
68  const Window::Dimension zero_window = Window::Dimension(0, 0, 0);
69  for (size_t d = 0; d <= _perm.num_dimensions(); ++d)
70  {
71  window_out.set(d, zero_window);
72  }
73 
74  // Create iterators
75  Iterator in(_input, window);
76  Iterator out(_output, window_out);
77 
78  if (_input->info()->num_dimensions() <= 3)
79  {
80  execute_window_loop(
81  window,
82  [&](const Coordinates &id)
83  {
84  const int idx = id[0] * perm_strides[0] + id[1] * perm_strides[1] + id[2] * perm_strides[2];
85  *(reinterpret_cast<T *>(out.ptr() + idx)) = *(reinterpret_cast<const T *>(in.ptr()));
86  },
87  in, out);
88  }
89  else if (_input->info()->num_dimensions() >= 4)
90  {
91  execute_window_loop(
92  window,
93  [&](const Coordinates &id)
94  {
95  const int idx = id[0] * perm_strides[0] + id[1] * perm_strides[1] + id[2] * perm_strides[2] +
96  id[3] * perm_strides[3];
97  *(reinterpret_cast<T *>(out.ptr() + idx)) = *(reinterpret_cast<const T *>(in.ptr()));
98  },
99  in, out);
100  }
101 }

107 void CPPPermuteKernel::configure(const ITensor *input, ITensor *output, const PermutationVector &perm)
108 {
109  ARM_COMPUTE_ERROR_ON_NULLPTR(input, output);
110  const TensorShape output_shape = misc::shape_calculator::compute_permutation_output_shape(*input->info(), perm);
111  // Output auto inizialitation if not yet initialized
112  auto_init_if_empty(*output->info(), input->info()->clone()->set_tensor_shape(output_shape));
113 
114  // Perform validation step
115  ARM_COMPUTE_ERROR_THROW_ON(validate_arguments(input->info(), output->info(), perm));
116 
117  _input = input;
118  _output = output;
119  _perm = perm;
120 
121  switch (input->info()->element_size())
122  {
123  case 1:
124  _func = &CPPPermuteKernel::run_permute<uint8_t>;
125  break;
126  case 2:
127  _func = &CPPPermuteKernel::run_permute<uint16_t>;
128  break;
129  case 4:
130  _func = &CPPPermuteKernel::run_permute<uint32_t>;
131  break;
132  default:
133  ARM_COMPUTE_ERROR("Element size not supported");
134  break;
135  }
136 
137  // Configure kernel window
138  Window win = calculate_max_window(*input->info(), Steps());
139 
140  // The CPPPermute doesn't need padding so update_window_and_padding() can be skipped
141  Coordinates coord;
142  coord.set_num_dimensions(output->info()->num_dimensions());
143  output->info()->set_valid_region(ValidRegion(coord, output->info()->tensor_shape()));
144 
145  ICPPKernel::configure(win);
146 }
