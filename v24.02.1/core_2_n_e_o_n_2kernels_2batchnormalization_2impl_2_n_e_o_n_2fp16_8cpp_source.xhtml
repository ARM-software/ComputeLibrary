51 template <typename T>
52 void batch_normalization(ITensor *src,
53  ITensor *dst,
54  const ITensor *mean,
55  const ITensor *var,
56  const ITensor *beta,
57  const ITensor *gamma,
58  float epsilon,
59  ActivationLayerInfo &act_info,
60  const Window &window)
61 {
62  /** SIMD vector tag type. */
63  using ExactTagType = typename wrapper::traits::neon_bitvector_tag_t<float16_t, wrapper::traits::BitWidth::W128>;
64 
65  const int window_step_x = 8;
66  const auto window_start_x = static_cast<int>(window.x().start());
67  const auto window_end_x = static_cast<int>(window.x().end());
68 
69  Window win_collapsed = window.collapse_if_possible(window, Window::DimZ);
70  win_collapsed.set(Window::DimX, Window::Dimension(0, 1, 1));
71 
72  Iterator input(src, win_collapsed);
73  Iterator output(dst, win_collapsed);
74 
75  const auto input_mean = reinterpret_cast<const float16_t *>(mean->ptr_to_element(Coordinates(0, 0)));
76  const auto input_var = reinterpret_cast<const float16_t *>(var->ptr_to_element(Coordinates(0, 0)));
77  const auto input_gamma =
78  (gamma != nullptr) ? reinterpret_cast<const float16_t *>(gamma->ptr_to_element(Coordinates(0, 0))) : nullptr;
79  const auto input_beta =
80  (beta != nullptr) ? reinterpret_cast<const float16_t *>(beta->ptr_to_element(Coordinates(0, 0))) : nullptr;
81 
82  T activation_functor(act_info);
83 
84  const auto epsilon_vec = wrapper::vdup_n(static_cast<float16_t>(epsilon), ExactTagType{});
85  execute_window_loop(
86  win_collapsed,
87  [&](const Coordinates &)
88  {
89  const auto input_ptr = reinterpret_cast<const float16_t *>(input.ptr());
90  const auto output_ptr = reinterpret_cast<float16_t *>(output.ptr());
91 
92  // Perform core calculations using vector operations
93  int x = window_start_x;
94  for (; x <= (window_end_x - window_step_x); x += window_step_x)
95  {
96  // Conctruct vectors
97  const auto mean_vec = wrapper::vloadq(input_mean + x);
98  const auto var_vec = wrapper::vloadq(input_var + x);
99  const auto gamma_vec = (input_gamma != nullptr)
100  ? wrapper::vloadq(input_gamma + x)
101  : wrapper::vdup_n(static_cast<float16_t>(1.f), ExactTagType{});
102  const auto beta_vec = (input_beta != nullptr)
103  ? wrapper::vloadq(input_beta + x)
104  : wrapper::vdup_n(static_cast<float16_t>(0.f), ExactTagType{});
105 
106  // Calculate denominator
107  const auto denominator = wrapper::vinvsqrt(wrapper::vadd(var_vec, epsilon_vec));
108 
109  // Calculate x bar
110  const auto numerator = wrapper::vsub(wrapper::vloadq(input_ptr + x), mean_vec);
111  const auto x_bar = wrapper::vmul(numerator, denominator);
112  auto res = wrapper::vmla(beta_vec, x_bar, gamma_vec);
113 
114  // Perform fused activation
115  if (act_info.enabled())
116  {
117  activation_functor(res);
118  }
119 
120  // Store results
121  wrapper::vstore(output_ptr + x, res);
122  }
123 
124  // Compute left-over elements
125  for (; x < window_end_x; ++x)
126  {
127  // Conctruct vectors
128  const float16_t gamma = (input_gamma != nullptr) ? input_gamma[x] : 1.f;
129  const float16_t beta = (input_beta != nullptr) ? input_beta[x] : 0.f;
130 
131  const float16_t denominator = sqrt(input_var[x] + epsilon);
132  const float16_t numerator = input_ptr[x] - input_mean[x];
133  const float16_t x_bar = numerator / denominator;
134  float16_t res = beta + x_bar * gamma;
135 
136  // Perform fused activation
137  if (act_info.enabled())
138  {
139  activation_functor(res);
140  }
141 
142  // Store results
143  *reinterpret_cast<float16_t *>(output_ptr + x) = res;
144  }
145  },
146  input, output);
147 }
