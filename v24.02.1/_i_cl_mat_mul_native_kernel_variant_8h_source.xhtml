56 /** Basic container for the OpenCL MatMul Native variant functions */
57 template <class T>
58 class ClMatMulNativeVariantArray
59 {
60 public:
61  /** Alias for Float index */
62  static constexpr size_t DT_FLOAT = 0;
63  /** Alias for Quantized type index */
64  static constexpr size_t DT_QUANTIZED = 1;
65 
66  /** Constructor
67  *
68  * @param[in] func_float Function to call for matmul native float (F32, F16)
69  * @param[in] func_quantized Function to call for matmul native quantized (QASYMM8, QASYMM8_SIGNED, QSYMM8_PER_CHANNEL)
70  *
71  */
72  ClMatMulNativeVariantArray(T func_float, T func_quantized) : _configs{func_float, func_quantized}
73  {
74  }
75 
76  /** Method to return the matmul native variant function based on data type
77  *
78  * @param[in] data_type Input data type
79  *
80  * @return the valid function otherwise it returns nullptr if the data type is not valid
81  */
82  T get_function(DataType data_type)
83  {
84  switch (data_type)
85  {
86  case DataType::F32:
87  case DataType::F16:
88  return _configs.at(DT_FLOAT);
89  case DataType::QASYMM8:
90  case DataType::QASYMM8_SIGNED:
91  case DataType::QSYMM8_PER_CHANNEL:
92  return _configs.at(DT_QUANTIZED);
93  default:
94  return nullptr;
95  }
96  }
97 
98 private:
99  std::array<T, 2> _configs;
100 };
