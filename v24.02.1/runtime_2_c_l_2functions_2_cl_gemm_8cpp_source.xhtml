75 void CLGEMM::configure(const CLCompileContext &compile_context,
76  const ICLTensor *a,
77  const ICLTensor *b,
78  const ICLTensor *c,
79  ICLTensor *output,
80  float alpha,
81  float beta,
82  const GEMMInfo &gemm_info)
83 {
84  ARM_COMPUTE_ERROR_ON_NULLPTR(a, b, output);
85 
86  _impl->b = b;
87  _impl->op = std::make_unique<OperatorType>();
88  _impl->is_prepared = gemm_info.retain_internal_weights();
89 
90  _impl->op->configure(compile_context, a->info(), b->info(), c != nullptr ? c->info() : nullptr, output->info(),
91  alpha, beta, gemm_info);
92  _impl->aux_mem_req = _impl->op->workspace();
93 
94  // Manage/allocate auxilairy tensors
95  if (_impl->is_prepared)
96  {
97  _impl->run_pack.add_const_tensor(ACL_SRC_0, a);
98  _impl->run_pack.add_tensor(ACL_DST, output);
99  }
100  else
101  {
102  _impl->run_pack = {{ACL_SRC_0, a}, {ACL_SRC_2, c}, {ACL_DST, output}};
103  _impl->prep_pack = {{ACL_SRC_1, _impl->b}};
104 
105  _impl->workspace_tensors =
106  manage_workspace<CLTensor>(_impl->op->workspace(), _impl->memory_group, _impl->run_pack, _impl->prep_pack);
107  }
108 }

130 void CLGEMM::prepare()
131 {
132  if (!_impl->is_prepared)
133  {
134  _impl->op->prepare(_impl->prep_pack);
135 
136  auto has_reshape =
137  std::find_if(_impl->aux_mem_req.begin(), _impl->aux_mem_req.end(),
138  [](const MemoryInfo &m) -> bool { return m.lifetime == MemoryLifetime::Persistent; });
139 
140  if (has_reshape != std::end(_impl->aux_mem_req))
141  {
142  _impl->b->mark_as_unused();
143  }
144  else
145  {
146  // Pack the B matrix to be used as the underlying GEMM performs no reshapes
147  _impl->run_pack.add_const_tensor(ACL_SRC_1, _impl->b);
148  }
149  _impl->is_prepared = true;
150  }
