51 void ClFillKernel::configure(const CLCompileContext &compile_context,
52  ITensorInfo *tensor,
53  const PixelValue &constant_value,
54  Window *window)
55 {
56  ARM_COMPUTE_ERROR_ON_NULLPTR(tensor);
57  ARM_COMPUTE_ERROR_THROW_ON(validate(tensor, constant_value, window));
58 
59  const DataType data_type = tensor->data_type();
60  const int vec_size_x = 16 / tensor->element_size();
61 
62  // Create and update the window (if needed)
63  _full_window = calculate_max_window(*tensor);
64  Window win = _full_window;
65  if (window != nullptr)
66  {
67  ARM_COMPUTE_ERROR_ON_INVALID_SUBWINDOW(win, *window);
68  win = *window;
69  }
70 
71  const int output_width_x = win.num_iterations(0);
72  const bool multi_access_x = output_width_x >= vec_size_x;
73  const bool remainder_x = output_width_x % vec_size_x > 0;
74 
75  if (multi_access_x)
76  {
77  win.set(Window::DimX,
78  Window::Dimension(win.x().start(), ceil_to_multiple(win.x().end(), vec_size_x), vec_size_x));
79  }
80  ICLKernel::configure_internal(win);
81 
82  // Create kernel
83  CLBuildOptions build_opts;
84  build_opts.add_option("-DDATA_TYPE=" + get_cl_type_from_data_type(data_type));
85  build_opts.add_option("-DCONSTANT_VALUE=" + string_from_pixel_value(constant_value, data_type));
86  build_opts.add_option_if(multi_access_x, "-DVEC_SIZE=" + support::cpp11::to_string(vec_size_x));
87  build_opts.add_option_if(multi_access_x && remainder_x,
88  "-DLAST_ACCESSED_X=" +
89  support::cpp11::to_string(std::max<int>(output_width_x - vec_size_x, 0)));
90  _kernel = create_kernel(compile_context, "memset", build_opts.options());
91 }

104 void ClFillKernel::run_op(ITensorPack &tensors, const Window &window, cl::CommandQueue &queue)
105 {
106  ARM_COMPUTE_ERROR_ON_UNCONFIGURED_KERNEL(this);
107  ARM_COMPUTE_ERROR_ON_INVALID_SUBWINDOW(ICLKernel::window(), window);
108 
109  const auto tensor =
110  utils::cast::polymorphic_downcast<const ICLTensor *>(tensors.get_const_tensor(TensorType::ACL_SRC));
111 
112  // Collapse all the batches on the third
113  Window collapsed = window.collapse_if_possible(_full_window, Window::DimZ);
114  Window slice = collapsed.first_slice_window_3D();
115 
116  do
117  {
118  unsigned int idx = 0;
119  add_3D_tensor_argument(idx, tensor, slice);
120  enqueue(queue, *this, slice, lws_hint());
121  } while (collapsed.slide_window_slice_3D(slice));
122 }
