43 Status validate_arguments(const ITensorInfo *input, const ITensorInfo *output, int32_t stride)
44 {
45  //Note: ARM_COMPUTE_RETURN_ERROR_ON_CPU_F16_UNSUPPORTED(input) is not needed here as this kernel doesn't use CPU FP16 instructions.
46  ARM_COMPUTE_RETURN_ERROR_ON(input->data_type() == DataType::UNKNOWN);
47  ARM_COMPUTE_RETURN_ERROR_ON(input->data_layout() == DataLayout::UNKNOWN);
48 
49  const size_t idx_width = get_data_layout_dimension_index(input->data_layout(), DataLayoutDimension::WIDTH);
50  const size_t idx_height = get_data_layout_dimension_index(input->data_layout(), DataLayoutDimension::HEIGHT);
51 
52  ARM_COMPUTE_RETURN_ERROR_ON(stride <= 0);
53  ARM_COMPUTE_RETURN_ERROR_ON_MSG((input->tensor_shape()[idx_width] % stride) != 0, "The width of the input tensor must be a multiple of stride");
54  ARM_COMPUTE_RETURN_ERROR_ON_MSG((input->tensor_shape()[idx_height] % stride) != 0, "The height of the input tensor must be a multiple of stride");
55 
56  // Validate output if initialized
57  if(output->total_size() != 0)
58  {
59  const TensorInfo tensor_info_output = output->clone()->set_tensor_shape(misc::shape_calculator::compute_reorg_output_shape(*input, stride));
60  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_SHAPES(output, &tensor_info_output);
61  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input, output);
62  }
63 
64  return Status{};
65 }

73 void NEReorgLayerKernel::configure(const ITensor *input, ITensor *output, int32_t stride)
74 {
75  ARM_COMPUTE_ERROR_ON_NULLPTR(input, output);
76 
77  // Output auto inizialitation if not yet initialized
78  const TensorShape output_shape = misc::shape_calculator::compute_reorg_output_shape(*input->info(), stride);
79  auto_init_if_empty(*output->info(), input->info()->clone()->set_tensor_shape(output_shape));
80 
81  // Perform validation step
82  ARM_COMPUTE_ERROR_THROW_ON(validate_arguments(input->info(), output->info(), stride));
83 
84  _input = input;
85  _output = output;
86  _stride = stride;
87 
88  // The NEReorgLayerKernel doesn't need padding so update_window_and_padding() can be skipped
89 
90  // Configure kernel window
91  Window win = calculate_max_window(*output->info(), Steps());
92 
93  ICPPKernel::configure(win);
94 }

102 void NEReorgLayerKernel::run(const Window &window, const ThreadInfo &info)
103 {
104  ARM_COMPUTE_UNUSED(info);
105  ARM_COMPUTE_ERROR_ON_UNCONFIGURED_KERNEL(this);
106  ARM_COMPUTE_ERROR_ON_INVALID_SUBWINDOW(ICPPKernel::window(), window);
107 
108  const DataLayout data_layout = _input->info()->data_layout();
109  const size_t idx_w = get_data_layout_dimension_index(data_layout, DataLayoutDimension::WIDTH);
110  const size_t idx_h = get_data_layout_dimension_index(data_layout, DataLayoutDimension::HEIGHT);
111  const size_t idx_c = get_data_layout_dimension_index(data_layout, DataLayoutDimension::CHANNEL);
112 
113  const unsigned int stride = _stride;
114  const unsigned int out_c = _output->info()->tensor_shape()[idx_c] / (stride * stride);
115  const uint8_t *in_ptr = _input->buffer();
116 
117  // Collapse
118  Window collapsed_window = window.collapse_if_possible(window, 4);
119 
120  // Create Iterator
121  Iterator out(_output, collapsed_window);
122 
123  // Perform reorg
124  execute_window_loop(collapsed_window, [&](const Coordinates & id)
125  {
126  // Get spatial coords and channels
127  const unsigned int w = id[idx_w];
128  const unsigned int h = id[idx_h];
129  const unsigned int c = id[idx_c];
130 
131  // Calculate mapping
132  const unsigned int offset = c / out_c;
133  Coordinates map_coords = id;
134  map_coords.set(idx_w, w * stride + offset % stride);
135  map_coords.set(idx_h, h * stride + offset / stride);
136  map_coords.set(idx_c, c % out_c);
137 
138  // Perform mapping
139  std::memcpy(out.ptr(), in_ptr + _input->info()->offset_element_in_bytes(map_coords), _input->info()->element_size());
140  },
141  out);
142 }
