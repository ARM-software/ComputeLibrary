30 /* Internal enum to define the weight format a kernel is expecting.
31  *
32  * This is distinct from the "external" WeightFormat defined in arm_gemm.hpp primarily to allow for SVE, where
33  * internally kernels are defined in terms of multiples of the SVE vector length, but externally they are converted
34  * to a fixed format (based on the VL of the machine we are running on).
35  *
36  * Encoded as a bitfield:
37  * bit 0 : SVE flag
38  * bit 4 : BF16 convert flag (fast mode)
39  * bits 11-8 : block length (bytes)
40  * bits 15-12: vector count
41  */
42 enum class KernelWeightFormat {
43  NON_FIXED = 0,
44  VL128_BL16 = 0x1200,
45  VL128_BL32 = 0x1400,
46  VL128_BL32_BF16 = 0x1410,
47  VL128_BL64 = 0x1800,
48  VL256_BL64 = 0x2800,
49  VL256_BL64_BF16 = 0x2810,
50  VL1VL_BL16 = 0x1201,
51  VL1VL_BL32 = 0x1401,
52  VL1VL_BL32_BF16 = 0x1411,
53  VL1VL_BL64 = 0x1801,
54  VL2VL_BL64 = 0x2801,
55  VL2VL_BL64_BF16 = 0x2811
56 };
