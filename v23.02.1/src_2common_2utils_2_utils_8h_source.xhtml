33 /** Convert a strongly typed enum to an old plain c enum
34  *
35  * @tparam E Plain old C enum
36  * @tparam SE Strongly typed resulting enum
37  *
38  * @param[in] v Value to convert
39  *
40  * @return A corresponding plain old C enumeration
41  */
42 template <typename E, typename SE>
43 constexpr E as_cenum(const SE v) noexcept
44 {
45  return static_cast<E>(static_cast<std::underlying_type_t<SE>>(v));
46 }

48 /** Convert plain old enumeration to a strongly typed enum
49  *
50  * @tparam SE Strongly typed resulting enum
51  * @tparam E Plain old C enum
52  *
53  * @param[in] val Value to convert
54  *
55  * @return A corresponding strongly typed enumeration
56  */
57 template <typename SE, typename E>
58 constexpr SE as_enum(const E val) noexcept
59 {
60  return static_cast<SE>(val);

63 /** Check if the given value is in the given enum value list
64  *
65  * @tparam E The type of the enum
66  *
67  * @param[in] check Value to check
68  * @param[in] list List of enum values to check against
69  *
70  * @return True if the given value is found in the list
71  */
72 template <typename E>
73 bool is_in(E check, std::initializer_list<E> list)
74 {
75  return std::any_of(std::cbegin(list), std::cend(list), [&check](E e)
76  {
77  return check == e;
78  });
79 }
