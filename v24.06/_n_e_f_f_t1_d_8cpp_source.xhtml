53 void NEFFT1D::configure(const ITensor *input, ITensor *output, const FFT1DInfo &config)
54 {
55  ARM_COMPUTE_ERROR_ON_NULLPTR(input, output);
56  ARM_COMPUTE_ERROR_THROW_ON(NEFFT1D::validate(input->info(), output->info(), config));
57  ARM_COMPUTE_LOG_PARAMS(input, output, config);
58 
59  // Decompose size to radix factors
60  const auto supported_radix = NEFFTRadixStageKernel::supported_radix();
61  const unsigned int N = input->info()->tensor_shape()[config.axis];
62  const auto decomposed_vector = arm_compute::helpers::fft::decompose_stages(N, supported_radix);
63  ARM_COMPUTE_ERROR_ON(decomposed_vector.empty());
64 
65  // Flags
66  _run_scale = config.direction == FFTDirection::Inverse;
67 
68  const bool is_c2r = input->info()->num_channels() == 2 && output->info()->num_channels() == 1;
69 
70  // Configure digit reverse
71  FFTDigitReverseKernelInfo digit_reverse_config;
72  digit_reverse_config.axis = config.axis;
73  digit_reverse_config.conjugate = config.direction == FFTDirection::Inverse;
74  TensorInfo digit_reverse_indices_info(TensorShape(input->info()->tensor_shape()[config.axis]), 1, DataType::U32);
75  _digit_reverse_indices.allocator()->init(digit_reverse_indices_info);
76  _memory_group.manage(&_digit_reversed_input);
77  _digit_reverse_kernel = std::make_unique<NEFFTDigitReverseKernel>();
78  _digit_reverse_kernel->configure(input, &_digit_reversed_input, &_digit_reverse_indices, digit_reverse_config);
79 
80  // Create and configure FFT kernels
81  unsigned int Nx = 1;
82  _num_ffts = decomposed_vector.size();
83  _fft_kernels.resize(_num_ffts);
84  _axis = config.axis;
85 
86  for (unsigned int i = 0; i < _num_ffts; ++i)
87  {
88  const unsigned int radix_for_stage = decomposed_vector.at(i);
89 
90  FFTRadixStageKernelInfo fft_kernel_info;
91  fft_kernel_info.axis = config.axis;
92  fft_kernel_info.radix = radix_for_stage;
93  fft_kernel_info.Nx = Nx;
94  fft_kernel_info.is_first_stage = (i == 0);
95  _fft_kernels[i] = std::make_unique<NEFFTRadixStageKernel>();
96  _fft_kernels[i]->configure(&_digit_reversed_input, ((i == (_num_ffts - 1)) && !is_c2r) ? output : nullptr,
97  fft_kernel_info);
98 
99  Nx *= radix_for_stage;
100  }
101 
102  // Configure scale kernel
103  if (_run_scale)
104  {
105  FFTScaleKernelInfo scale_config;
106  scale_config.scale = static_cast<float>(N);
107  scale_config.conjugate = config.direction == FFTDirection::Inverse;
108  _scale_kernel = std::make_unique<NEFFTScaleKernel>();
109  is_c2r ? _scale_kernel->configure(&_digit_reversed_input, output, scale_config)
110  : _scale_kernel->configure(output, nullptr, scale_config);
111  }
112 
113  // Allocate tensors
114  _digit_reversed_input.allocator()->allocate();
115  _digit_reverse_indices.allocator()->allocate();
116 
117  // Init digit reverse indices
118  const auto digit_reverse_cpu = arm_compute::helpers::fft::digit_reverse_indices(N, decomposed_vector);
119  std::copy_n(digit_reverse_cpu.data(), N, reinterpret_cast<unsigned int *>(_digit_reverse_indices.buffer()));
120 }

122 Status NEFFT1D::validate(const ITensorInfo *input, const ITensorInfo *output, const FFT1DInfo &config)
123 {
124  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(input, output);
125  ARM_COMPUTE_RETURN_ERROR_ON(input->data_type() != DataType::F32);
126  ARM_COMPUTE_RETURN_ERROR_ON(input->num_channels() > 2);
127  ARM_COMPUTE_RETURN_ERROR_ON(std::set<unsigned int>({0, 1}).count(config.axis) == 0);
128 
129  // Check if FFT is decomposable
130  const auto supported_radix = NEFFTRadixStageKernel::supported_radix();
131  const unsigned int N = input->tensor_shape()[config.axis];
132  const auto decomposed_vector = arm_compute::helpers::fft::decompose_stages(N, supported_radix);
133  ARM_COMPUTE_RETURN_ERROR_ON(decomposed_vector.empty());
134 
135  // Checks performed when output is configured
136  if ((output != nullptr) && (output->total_size() != 0))
137  {
138  // All combinations are supported except real input with real output (i.e., both input channels set to 1)
139  ARM_COMPUTE_RETURN_ERROR_ON(output->num_channels() == 1 && input->num_channels() == 1);
140  ARM_COMPUTE_RETURN_ERROR_ON(output->num_channels() > 2);
141  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_SHAPES(input, output);
142  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input, output);
143  }
144 
145  return Status{};
146 }

148 void NEFFT1D::run()
149 {
150  MemoryGroupResourceScope scope_mg(_memory_group);
151 
152  NEScheduler::get().schedule(_digit_reverse_kernel.get(), (_axis == 0 ? Window::DimY : Window::DimZ));
153 
154  for (unsigned int i = 0; i < _num_ffts; ++i)
155  {
156  NEScheduler::get().schedule(_fft_kernels[i].get(), (_axis == 0 ? Window::DimY : Window::DimX));
157  }
158 
159  // Run output scaling
160  if (_run_scale)
161  {
162  NEScheduler::get().schedule(_scale_kernel.get(), Window::DimY);
163  }
164 }
