44 NodeID create_grouped_convolution(Graph &g,
45  const NodeParams &params,
46  NodeIdxPair input,
47  NodeID weights,
48  NodeID bias,
49  PadStrideInfo conv_info,
50  ConvolutionMethod method,
51  ActivationLayerInfo fused_act,
52  FastMathHint fast_math_hint,
53  unsigned int num_groups)
54 {
55  bool has_bias = (bias != EmptyNodeID);
56 
57  // Split input
58  const TensorDescriptor input_tensor_desc = get_tensor_descriptor(g, g.node(input.node_id)->outputs()[0]);
59  const unsigned int input_idx = get_dimension_idx(input_tensor_desc.layout, DataLayoutDimension::CHANNEL);
60  NodeID input_split = GraphBuilder::add_split_node(g, params, input, num_groups, input_idx);
61 
62  // Split weights
63  const TensorDescriptor weights_tensor_desc = get_tensor_descriptor(g, g.node(weights)->outputs()[0]);
64  const unsigned int batch_idx = get_dimension_idx(weights_tensor_desc.layout, DataLayoutDimension::BATCHES);
65  NodeID weights_split = GraphBuilder::add_split_node(g, params, {weights, 0}, num_groups, batch_idx);
66 
67  // Split bias
68  NodeID bias_split = EmptyNodeID;
69  if (has_bias)
70  {
71  // Split bias
72  bias_split = GraphBuilder::add_split_node(g, params, {bias, 0}, num_groups, 0);
73  }
74 
75  std::vector<NodeIdxPair> convolution_outputs;
76  for (unsigned int i = 0; i < num_groups; ++i)
77  {
78  NodeParams group_params = params;
79  NodeID conv_nid = g.add_node<ConvolutionLayerNode>(conv_info, 1, method, fast_math_hint);
80  g.add_connection(input_split, i, conv_nid, 0);
81  g.add_connection(weights_split, i, conv_nid, 1);
82  if (has_bias)
83  {
84  g.add_connection(bias_split, i, conv_nid, 2);
85  }
86 
87  // Add group name
88  if (!group_params.name.empty())
89  {
90  group_params.name.append("_g" + arm_compute::support::cpp11::to_string(i));
91  }
92 
93  // Set node parameters
94  INode *node = g.node(conv_nid);
95  ARM_COMPUTE_ERROR_ON(node == nullptr);
96  node->set_common_node_parameters(group_params);
97 
98  // Down-cast node
99  auto *conv_node = arm_compute::utils::cast::polymorphic_downcast<ConvolutionLayerNode *>(node);
100  conv_node->set_fused_activation(fused_act);
101 
102  convolution_outputs.push_back({conv_nid, 0});
103  }
104 
105  // Depth concatenate output
106  return GraphBuilder::add_concatenate_node(g, params, convolution_outputs, DataLayoutDimension::CHANNEL);
107 }

120 void GroupedConvolutionMutator::mutate(Graph &g)
121 {
122  // Early exit if no Convolution layers exist in graph
123  if (g.nodes(NodeType::ConvolutionLayer).empty())
124  {
125  return;
126  }
127 
128  // Total nodes
129  size_t total_nodes = g.nodes().size();
130 
131  // Iterate over convolution nodes
132  for (unsigned int i = 0; i < total_nodes; ++i)
133  {
134  INode *node = g.node(i);
135  if (node != nullptr && node->type() == NodeType::ConvolutionLayer &&
136  arm_compute::utils::cast::polymorphic_downcast<ConvolutionLayerNode *>(node)->num_groups() != 1)
137  {
138  // Validate node
139  backends::IDeviceBackend &backend = backends::BackendRegistry::get().get_backend(node->assigned_target());
140  Status status = backend.validate_node(*node);
141 
142  // If grouped convolution is not supported
143  if (!bool(status))
144  {
145  // Down-cast node
146  auto *conv_node = arm_compute::utils::cast::polymorphic_downcast<ConvolutionLayerNode *>(node);
147 
148  // Get internal convolution info
149  // TODO (geopin01) : Create a descriptor or a clone interface
150  const PadStrideInfo conv_info = conv_node->convolution_info();
151  const ConvolutionMethod conv_method = conv_node->convolution_method();
152  const ActivationLayerInfo fused_act_info = conv_node->fused_activation();
153  const FastMathHint fast_math_hint = conv_node->fast_math_hint();
154  const unsigned int num_groups = conv_node->num_groups();
155  const NodeParams params = conv_node->common_node_params();
156  const Target assigned_target = conv_node->assigned_target();
157 
158  // Extract node ids
159  ARM_COMPUTE_ERROR_ON(conv_node->input_edge(0) == nullptr || conv_node->input_edge(1) == nullptr);
160  const NodeID input_id = conv_node->input_edge(0)->producer()->id();
161  const NodeID weights_id = conv_node->input_edge(1)->producer()->id();
162  const NodeID bias_id =
163  (conv_node->input_edge(2) != nullptr) ? conv_node->input_edge(2)->producer()->id() : EmptyNodeID;
164 
165  // Get driving nodes
166  std::vector<NodeIdxPair> driving_nodes = get_driving_nodes(*node);
167 
168  // Extract activation node accessor if any
169  auto node_accessor = conv_node->output(0)->extract_accessor();
170 
171  // Current max tensor and node id
172  TensorID latest_tid = g.tensors().size();
173  NodeID latest_nid = g.nodes().size();
174 
175  // Create grouped convolution node
176  NodeID grouped_conv_id =
177  create_grouped_convolution(g, params, {input_id, 0}, weights_id, bias_id, conv_info, conv_method,
178  fused_act_info, fast_math_hint, num_groups);
179 
180  // Remove convolution node
181  g.remove_node(node->id());
182 
183  // Update batch normalization node outputs
184  for (auto &driving_node : driving_nodes)
185  {
186  g.add_connection(grouped_conv_id, 0, driving_node.node_id, driving_node.index);
187  }
188 
189  // Update accessor to batch normalization node
190  g.node(grouped_conv_id)->output(0)->set_accessor(std::move(node_accessor));
191 
192  // Configure new tensors and nodes
193  std::for_each(g.tensors().begin() + latest_tid, g.tensors().end(),
194  [](std::unique_ptr<Tensor> &t) { configure_tensor(t.get()); });
195  std::for_each(g.nodes().begin() + latest_nid, g.nodes().end(),
196  [&assigned_target](std::unique_ptr<INode> &n)
197  {
198  if (n != nullptr)
199  {
200  n->set_assigned_target(assigned_target);
201  }
202  });
203  }
204  }
205  }
