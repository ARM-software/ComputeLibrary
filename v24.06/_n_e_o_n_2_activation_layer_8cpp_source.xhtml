56 /** Define relative tolerance of the activation layer.
57  *
58  * @param[in] data_type The data type used.
59  * @param[in] activation The activation function used.
60  *
61  * @return Relative tolerance depending on the activation function.
62  */
63 RelativeTolerance<float> relative_tolerance(DataType data_type, ActivationLayerInfo::ActivationFunction activation)
64 {
65  switch(activation)
66  {
67  case ActivationLayerInfo::ActivationFunction::LOGISTIC:
68  case ActivationLayerInfo::ActivationFunction::ELU:
69  case ActivationLayerInfo::ActivationFunction::SQRT:
70  case ActivationLayerInfo::ActivationFunction::TANH:
71  case ActivationLayerInfo::ActivationFunction::HARD_SWISH:
72  case ActivationLayerInfo::ActivationFunction::SWISH:
73  case ActivationLayerInfo::ActivationFunction::GELU:
74  switch(data_type)
75  {
76  case DataType::F16:
77 #if defined(ENABLE_SVE)
78  return RelativeTolerance<float>(0.25f);
79 #else // !defined(ENABLE_SVE)
80  return RelativeTolerance<float>(0.1f);
81 #endif // defined(ENABLE_SVE)
82  default:
83  return RelativeTolerance<float>(0.05f);
84  }
85  case ActivationLayerInfo::ActivationFunction::SOFT_RELU:
86  switch(data_type)
87  {
88  case DataType::F16:
89 #if defined(ENABLE_SVE)
90  return RelativeTolerance<float>(0.9f);
91 #else // !defined(ENABLE_SVE)
92  return RelativeTolerance<float>(0.01f);
93 #endif // defined(ENABLE_SVE)
94  default:
95  return RelativeTolerance<float>(0.00001f);
96  }
97  default:
98  return RelativeTolerance<float>(0.f);
99  }
100 }

185 template <typename T, ARM_COMPUTE_REQUIRES_TA(arm_compute::utils::traits::is_floating_point<T>::value)>
186 void test_float_sqrt_boundary_value()
187 {
188  constexpr auto vector_size = uint32_t{ 16 };
189 
190  auto data_type = DataType::F32;
191 #ifdef __ARM_FEATURE_FP16_VECTOR_ARITHMETIC
192  data_type = std::is_same<T, half>::value ? DataType::F16 : data_type;
193 #endif /* __ARM_FEATURE_FP16_VECTOR_ARITHMETIC */
194 
195  const auto boundary_value_vector = std::vector<T>
196  {
197  std::numeric_limits<T>::min(),
198  T(0),
199  std::numeric_limits<T>::epsilon(),
200  std::numeric_limits<T>::max(),
201  };
202 
203  // the following size ensures that the whole logic (vector + left-over) to be tested
204  // using all boundary values iff boundary_value_vecotr.size() is smaller than vector_size.
205  auto shape = TensorShape{ vector_size + boundary_value_vector.size() };
206  auto info = ActivationLayerInfo{ ActivationLayerInfo::ActivationFunction::SQRT };
207  auto src = create_tensor<Tensor>(shape, data_type);
208 
209  auto act = NEActivationLayer{};
210  act.configure(&src, nullptr, info);
211  src.allocator()->allocate();
212  library->fill_static_values(Accessor(src), boundary_value_vector);
213  act.run();
214 
215  auto reference_src = SimpleTensor<T> { shape, data_type };
216  library->fill_static_values(reference_src, boundary_value_vector);
217  auto reference_dst = reference::activation_layer<T>(reference_src, info);
218 
219  validate(Accessor(src), reference_dst, tolerance_float_sqrt);
220 }

240 
241  acl::Queue queue(ctx, &err);
242  ARM_COMPUTE_ASSERT(err == acl::StatusCode::Success);
243 
244  // Create activation operator
245  acl::TensorDescriptor src_info({ 2, 3 }, acl::DataType::Float32);
246  acl::TensorDescriptor dst_info({ 2, 3 }, acl::DataType::Float32);
247  acl::ActivationDesc desc{ AclRelu, 6.f, 0.f, false };
248 
249  acl::Activation act(ctx, src_info, dst_info, desc, &err);
250  ARM_COMPUTE_ASSERT(err == acl::StatusCode::Success);
251 
252  // Create tensors and feed
253  acl::Tensor src(ctx, src_info, &err);
254  ARM_COMPUTE_ASSERT(err == acl::StatusCode::Success);
255  acl::Tensor dst(ctx, dst_info, &err);
256  ARM_COMPUTE_ASSERT(err == acl::StatusCode::Success);
257 
258  acl::TensorPack pack(ctx);
259  err = pack.add(src, ACL_SRC);
260  err = pack.add(dst, ACL_DST);
261  ARM_COMPUTE_ASSERT(err == acl::StatusCode::Success);
262 
263  // Execute operator
264  err = act.run(queue, pack);
