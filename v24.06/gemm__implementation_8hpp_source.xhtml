34 /* Structure describing an implementation. For each supported combination
35  * of types, a static list of these structures is built up to describe the
36  * implementations available.

39 struct GemmImplementation {
40  const GemmMethod method;
41  const char * name;
42  const KernelWeightFormat kernel_weight_format = KernelWeightFormat::NON_FIXED;

47  bool do_is_supported(const GemmArgs &args, const OutputStage &os) const {
48  // Check supplied is_supported() function first.
49  if (is_supported != nullptr && !is_supported(args, os)) {
50  return false;
51  }
52 
53  // Check weight format is appropriate.
54  if (args._fixed_format == false) {
55  // Can't return a fixed format kernel if we weren't asked for one.
56  return (kernel_weight_format == KernelWeightFormat::NON_FIXED);
57  } else {
58  // Fixed format kernel requested: if this is a non-fixed format kernel we can't use it.
59  if (kernel_weight_format == KernelWeightFormat::NON_FIXED) {
60  return false;
61  }
62 
63  // If there's no config, or the config says ANY then this one is OK.
64  if (!args._cfg || args._cfg->weight_format == WeightFormat::ANY) {
65  return true;
66  }
67 
68  // If we get here it means there is a config and it specifies a format. Check it matches this kernel.
69  // NOTE: this will execute SVE instructions if it's an SVE kernel, so it's important that is_supported()
70  // was called above first.
71  return (args._cfg->weight_format == get_weight_format(kernel_weight_format, sizeof(Top)));
72  }
73  }

78  } else {
79  return 0;

215 /* Provides the list of implementation descriptors which is processed by the
216  * other functions.
217  *
218  * A specialised version is provided for each supported combination of types.
219  * The end of the list is indicated by a sentinel descriptor with
220  * method==GemmMethod::DEFAULT. */
221 template<typename Top, typename Tret, class OutputStage = Nothing>
222 const GemmImplementation<Top, Tret, OutputStage> *gemm_implementation_list();
223 
224 /*
225  * Select a GEMM implementation for the given arguments.
226  *
227  * The logic here returns the method on the list which supports the
228  * requested problem parameters, matches the provided filters (method and/or
229  * name string match) and offers the lowest cycle estimate. A cycle
230  * estimate of '0' is treated as a special value, causing the corresponding
231  * method to be selected immediately.
232  *
233  * If no method supports the requested parameters and passes the filters,
234  * this function returns false and doesn't touch the provided pointer
235  * reference.
236  */
237 template<typename Top, typename Tret, class OutputStage>
238 bool find_implementation(const GemmArgs &args, const OutputStage &os, const GemmImplementation<Top, Tret, OutputStage> * &impl) {
239  auto gemms = gemm_implementation_list<Top, Tret, OutputStage>();
240  const GemmConfig *cfg = args._cfg;
241 
242  const GemmImplementation<Top, Tret, OutputStage> *saved_impl = nullptr;
243  uint64_t best_estimate = 0;
244 
245  for (const GemmImplementation<Top, Tret, OutputStage> *i = gemms; i->method != GemmMethod::DEFAULT; i++) {
246  /* Skip if this implementation doesn't support these args. */
247  if (!i->do_is_supported(args, os)) {
248  continue;
249  }
250 
251  /* Skip if a specific method is requested and this is a different one. */
252  if (cfg && cfg->method != GemmMethod::DEFAULT && i->method != cfg->method) {
253  continue;
254  }
255 
256  /* Skip if a filter is to be applied and it doesn't match. */
257  if (cfg && cfg->filter != "" && !strstr(i->name, cfg->filter.c_str())) {
258  continue;
259  }
260 
261  /* Test the cycle estimate */
262  uint64_t estimate = i->do_cycle_estimate(args, os);
263 
264  /* Short circuit - if the estimate is zero, return this one immediately. */
265  if (estimate==0) {
266  impl=i;
267  return true;
268  }
269 
270  /* Otherwise, remember this is our best so far if we don't yet have
271  * a valid candidate, or we beat the estimate. */
272  if ((saved_impl == nullptr) || (estimate < best_estimate)) {
273  saved_impl = i;
274  best_estimate = estimate;
275  }
276  }
277 
278  /* Return whichever method gave the best estimate. */
279  if (saved_impl != nullptr) {
280  impl = saved_impl;
281  return true;
282  }
283 
284  return false;
285 }

287 template<typename Top, typename Tret, class OutputStage>
288 std::vector<KernelDescription> get_compatible_kernels(const GemmArgs &args, const OutputStage &os) {
289  std::vector<KernelDescription> res;
290 
291  /* Find out what the default implementation in so we can set the flag accordingly later. */
292  const GemmImplementation<Top, Tret, OutputStage> *default_impl;
293  find_implementation(args, os, default_impl);
294 
295  auto gemms = gemm_implementation_list<Top, Tret, OutputStage>();
296 
297  for (const GemmImplementation<Top, Tret, OutputStage> *i = gemms; i->method != GemmMethod::DEFAULT; i++) {
298  /* Check that this implementation supports the presented problem. */
299 
300  if (!i->do_is_supported(args, os)) {
301  continue;
302  }
303 
304  res.push_back(KernelDescription(i->method, i->name, i==default_impl, i->do_cycle_estimate(args, os)));
305  }
306 
307  return res;
308 }

310 template<typename Top, typename Tret, class OutputStage>
311 bool has_opt_gemm(WeightFormat &wf, const GemmArgs &args, const OutputStage &os) {
312  const GemmImplementation<Top, Tret, OutputStage> *impl;
313  const bool success = find_implementation<Top, Tret, OutputStage>(args, os, impl);
314  if (success)
315  wf = UniqueGemmCommon<Top, Tret>(impl->do_instantiate(args, os))->get_config().weight_format;
316  return success;

320 UniqueGemmCommon<Top, Tret> gemm(const GemmArgs &args, const OutputStage &os) {
321  const GemmImplementation<Top, Tret, OutputStage> *impl;
322 
323  if (find_implementation<Top, Tret, OutputStage>(args, os, impl)) {

331 KernelDescription get_gemm_method(const GemmArgs &args, const OutputStage &os) {
332  const GemmImplementation<Top, Tret, OutputStage> *impl;
333
