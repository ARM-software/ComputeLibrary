37 /** ClDWCNativeKernelConfigurationFactory factory class */
38 class ClDWCNativeKernelConfigurationFactory final
39 {
40 public:
41  /** Static method to call the ClDWCNative kernel configuration class accordingly with the GPU target
42  *
43  * @param[in] gpu GPU target
44  *
45  * @return IClDWCNativeKernelConfig
46  */
47  static std::unique_ptr<IClDWCNativeKernelConfig> create(GPUTarget gpu)
48  {
49  switch (get_arch_from_target(gpu))
50  {
51  case GPUTarget::MIDGARD:
52  // The heuristic for Midgard is the same as the one used for Arm Mali-G71
53  return std::make_unique<ClDWCNativeDefaultConfigBifrost>(GPUTarget::G71);
54  case GPUTarget::BIFROST:
55  return std::make_unique<ClDWCNativeDefaultConfigBifrost>(gpu);
56  case GPUTarget::VALHALL:
57  case GPUTarget::FIFTHGEN:
58  return std::make_unique<ClDWCNativeDefaultConfigValhall>(gpu);
59  default:
60  ARM_COMPUTE_ERROR("Not supported GPU target");
61  }
62  }
63 };
