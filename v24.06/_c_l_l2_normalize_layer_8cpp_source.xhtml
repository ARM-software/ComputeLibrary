60 void CLL2NormalizeLayer::configure(
61  const CLCompileContext &compile_context, ICLTensor *input, ICLTensor *output, int axis, float epsilon)
62 {
63  ARM_COMPUTE_LOG_PARAMS(input, output, axis, epsilon);
64 
65  // Reset auxiliary tensor
66  _sumsq.allocator()->init(TensorInfo());
67 
68  // Manage intermediate buffers
69  _memory_group.manage(&_sumsq);
70 
71  // Configure kernels
72  const uint32_t actual_axis = wrap_around(axis, max_input_tensor_dim);
73  _reduce_func.configure(compile_context, input, &_sumsq, actual_axis, ReductionOperation::SUM_SQUARE);
74  _normalize_kernel->configure(compile_context, input, &_sumsq, output, axis, epsilon);
75 
76  // Allocate intermediate tensor
77  _sumsq.allocator()->allocate();
78 }

80 Status CLL2NormalizeLayer::validate(const ITensorInfo *input, const ITensorInfo *output, int axis, float epsilon)
81 {
82  TensorShape shape(input->tensor_shape());
83 
84  // Create intermediate tensor info
85  TensorInfo sum_sq;
86  sum_sq.set_data_type(input->data_type());
87  sum_sq.set_tensor_shape(shape);
88 
89  const uint32_t actual_axis = wrap_around(axis, max_input_tensor_dim);
90  ARM_COMPUTE_RETURN_ON_ERROR(
91  CLReductionOperation::validate(input, &sum_sq, actual_axis, ReductionOperation::SUM_SQUARE));
92 
93  // Reduce shape on axis
94  shape.set(actual_axis, 1);
95  sum_sq.set_tensor_shape(shape);
96 
97  ARM_COMPUTE_RETURN_ON_ERROR(CLL2NormalizeLayerKernel::validate(input, &sum_sq, output, axis, epsilon));
98 
99  return Status{};
100 }
