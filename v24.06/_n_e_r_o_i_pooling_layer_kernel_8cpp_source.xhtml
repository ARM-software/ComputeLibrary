67 /** Evaluate number needing to be stored in output tensor as quantized format.
68  *
69  * @param[in] input Source tensor. Data types supported: QASYMM8
70  * @param[out] output Destination tensor. Where output value will be stored, same datatype as input
71  * @param[in] region_start_x Beginning region of x coordinate of pooling region
72  * @param[in] region_start_y Beginning region of y coordinate of pooling region
73  * @param[in] region_end_x End of pooling region, x coordinate
74  * @param[in] region_end_y End of pooling region, y coordinate
75  * @param[in] fm Channel index of coordinate in output Tensor to store value
76  * @param[in] px Width index of coodinate in output Tensor to store value
77  * @param[in] py Height index of coordinate in output Tensor to store value
78  * @param[in] roi_batch Index of image to perform Pooling on in input Tensor
79  * @param[in] roi_indx Index of image of coordinate in output Tensor to store value
80  */
81 template <typename T>
82 void template_eval(const ITensor *input,
83  const ITensor *output,
84  int region_start_x,
85  int region_start_y,
86  int region_end_x,
87  int region_end_y,
88  int fm,
89  int px,
90  int py,
91  int roi_batch,
92  int roi_indx)
93 {
94  if ((region_end_x <= region_start_x) || (region_end_y <= region_start_y))
95  {
96  *reinterpret_cast<T *>(output->ptr_to_element(Coordinates(px, py, fm, roi_indx))) = 0;
97  }
98  else
99  {
100  T curr_max = std::numeric_limits<T>::lowest(); // Min value of typename T
101  for (int j = region_start_y; j < region_end_y; ++j)
102  {
103  for (int i = region_start_x; i < region_end_x; ++i)
104  {
105  const auto val = *reinterpret_cast<const T *>(input->ptr_to_element(Coordinates(i, j, fm, roi_batch)));
106  curr_max = std::max(val, curr_max);
107  }
108  }
109 
110  // if quantized datatype, requantize then store in output tensor
111  if (is_data_type_quantized(input->info()->data_type()))
112  {
113  // covert qasymm to new output quantization scale and offset
114  UniformQuantizationInfo uqinfo = compute_requantization_scale_offset(
115  input->info()->quantization_info().uniform(), output->info()->quantization_info().uniform());
116  *reinterpret_cast<T *>(output->ptr_to_element(Coordinates(px, py, fm, roi_indx))) =
117  quantize_qasymm8(curr_max, uqinfo);
118  }
119  else
120  {
121  *reinterpret_cast<T *>(output->ptr_to_element(Coordinates(px, py, fm, roi_indx))) = curr_max;
122  }
123  }
124 }

176 void NEROIPoolingLayerKernel::run(const Window &window, const ThreadInfo &info)
177 {
178  ARM_COMPUTE_UNUSED(info);
179  ARM_COMPUTE_ERROR_ON_UNCONFIGURED_KERNEL(this);
180  ARM_COMPUTE_ERROR_ON_INVALID_SUBWINDOW(INEKernel::window(), window);
181 
182  const size_t values_per_roi = _rois->info()->dimension(0);
183 
184  const int roi_list_start = window.x().start();
185  const int roi_list_end = window.x().end();
186  const int width = _input->info()->dimension(Window::DimX);
187  const int height = _input->info()->dimension(Window::DimY);
188  const int fms = _input->info()->dimension(Window::DimZ);
189  const int pooled_w = _pool_info.pooled_width();
190  const int pooled_h = _pool_info.pooled_height();
191  const float spatial_scale = _pool_info.spatial_scale();
192 
193  const auto *rois_ptr = reinterpret_cast<const uint16_t *>(_rois->buffer());
194  const auto data_type = _input->info()->data_type();
195 
196  for (int roi_indx = roi_list_start; roi_indx < roi_list_end; ++roi_indx)
197  {
198  const unsigned int roi_batch = rois_ptr[values_per_roi * roi_indx];
199  const auto x1 = rois_ptr[values_per_roi * roi_indx + 1];
200  const auto y1 = rois_ptr[values_per_roi * roi_indx + 2];
201  const auto x2 = rois_ptr[values_per_roi * roi_indx + 3];
202  const auto y2 = rois_ptr[values_per_roi * roi_indx + 4];
203 
204  // Scale ROI
205  const int roi_anchor_x = support::cpp11::round(x1 * spatial_scale);
206  const int roi_anchor_y = support::cpp11::round(y1 * spatial_scale);
207  const int roi_width = std::max(support::cpp11::round((x2 - x1) * spatial_scale), 1.f);
208  const int roi_height = std::max(support::cpp11::round((y2 - y1) * spatial_scale), 1.f);
209 
210  // Iterate through all feature maps
211  for (int fm = 0; fm < fms; ++fm)
212  {
213  // Iterate through all output pixels
214  for (int py = 0; py < pooled_h; ++py)
215  {
216  for (int px = 0; px < pooled_w; ++px)
217  {
218  auto region_start_x = static_cast<int>(std::floor((static_cast<float>(px) / pooled_w) * roi_width));
219  auto region_end_x =
220  static_cast<int>(std::floor((static_cast<float>(px + 1) / pooled_w) * roi_width));
221  auto region_start_y =
222  static_cast<int>(std::floor((static_cast<float>(py) / pooled_h) * roi_height));
223  auto region_end_y =
224  static_cast<int>(std::floor((static_cast<float>(py + 1) / pooled_h) * roi_height));
225 
226  region_start_x = std::min(std::max(region_start_x + roi_anchor_x, 0), width);
227  region_end_x = std::min(std::max(region_end_x + roi_anchor_x, 0), width);
228  region_start_y = std::min(std::max(region_start_y + roi_anchor_y, 0), height);
229  region_end_y = std::min(std::max(region_end_y + roi_anchor_y, 0), height);
230 
231  switch (data_type)
232  {
233  case DataType::F32:
234  template_eval<float>(_input, _output, region_start_x, region_start_y, region_end_x,
235  region_end_y, fm, px, py, roi_batch, roi_indx);
236  break;
237  case DataType::QASYMM8:
238  template_eval<qasymm8_t>(_input, _output, region_start_x, region_start_y, region_end_x,
239  region_end_y, fm, px, py, roi_batch, roi_indx);
240  break;
241  default:
242  ARM_COMPUTE_ERROR("DataType not Supported");
243  break;
244  }
245  }
246  }
247  }
248  }
