70 /** Describe how to extract information from a runtime Gpu tensor, and set it as an argument to a gpu kernel at runtime
71  *
72  * A kernel argument is just an argument to the gpu kernel as shown in the argument list below. This contrasts with a "workload argument" which is a tensor (@ref GpuWorkloadArgument)
73  * void kernel(arg0, arg1, ... argN)
74  *
75  * In a kernel generated using dynamic fusion (@ref GpuKernelSourceCode), every kernel argument describes part of a tensor.
76  * A tensor is described as: **storages** followed by **components**
77  *
78  * A storage (@ref TensorStorageType) describes how the tensor runtime memory can be accessed (e.g. via a global uint8 pointer to a CL buffer)
79  * A component (@ref TensorComponentType) describes additional runtime information about the tensor (e.g. the dimensions of the tensor)
80  *
81  * The arguments are arranged in the order of use in the generated kernel code:
82  *
83  * arg0 , arg1 , arg2 , ..., , argN
84  * storage, component0, component1, ..., componentX, storage, component0, component1, ..., componentY
85  * | tensor0 | tensor1 |
86  *
87  * An example argument list:
88  *
89  * void kernel(
90  * image2d_t t0_image, // TensorStorageType::ClImage2dReadOnly
91  * uint8_t* t0_ptr, // TensorStorageType::ClBufferUint8Ptr
92  * uint t0_dim0, // TensorComponentType::Dim0
93  * uint t0_stride1, // TensorComponentType::Stride1
94  * image2d_t t1_ptr, // TensorStorageType::ClImage2dReadOnly
95  * uint t1_dim1xdim2, // TensorComponentType::Dim1xDim2
96  * uint t1_stride1, // TensorComponentType::Stride1
97  * uint t1_stride2, // TensorComponentType:Stride2
98  * )
99  *
100  */
101 class GpuKernelArgumentBinding
102 {
103 public:
104  enum class Type : int32_t
105  {
106  TensorStorage, /** @ref TensorStorageType */
107  TensorComponent /** @ref TensorComponentType */
108  };
109  GpuKernelArgumentBinding(ITensorInfo::Id id, TensorStorageType storage)
110  : _type{Type::TensorStorage}, _id{id}, _value{}
111  {
112  _value.tensor_storage_type = storage;
113  }
114  GpuKernelArgumentBinding(ITensorInfo::Id id, TensorComponentType component)
115  : _type{Type::TensorComponent}, _id{id}, _value{}
116  {
117  _value.tensor_component_type = component;
118  }
119  /** Storage type of the tensor
120  */
121  TensorStorageType tensor_storage_type() const
122  {
123  ARM_COMPUTE_ERROR_ON(_type != Type::TensorStorage);
124  return _value.tensor_storage_type;
125  }
126  /** Component of the tensor
127  */
128  TensorComponentType tensor_component_type() const
129  {
130  ARM_COMPUTE_ERROR_ON(_type != Type::TensorComponent);
131  return _value.tensor_component_type;
132  }
133  /** Id of the tensor this kernel argument belongs to
134  */
135  ITensorInfo::Id id() const
136  {
137  return _id;
138  }
139  /** Type of the kernel argument
140  */
141  Type type() const
142  {
143  return _type;
144  }
145 
146 private:
147  Type _type;
148  ITensorInfo::Id _id;
149  union Value
150  {
151  TensorStorageType tensor_storage_type;
152  TensorComponentType tensor_component_type;
153  };
154  Value _value;
155 };
