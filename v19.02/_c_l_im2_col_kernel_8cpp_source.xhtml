57 Status validate_arguments(const ITensorInfo *input, const ITensorInfo *output, const Size2D &kernel_dims, const PadStrideInfo &conv_info, bool has_bias, const Size2D &dilation,
58  unsigned int num_groups)
59 {
60  const unsigned int channel_idx = get_data_layout_dimension_index(input->data_layout(), DataLayoutDimension::CHANNEL);
61 
62  ARM_COMPUTE_RETURN_ERROR_ON_F16_UNSUPPORTED(input);
63  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(input, 1, DataType::QASYMM8, DataType::F16, DataType::F32);
64  ARM_COMPUTE_RETURN_ERROR_ON(input->data_type() == DataType::QASYMM8 && has_bias);
65  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(output);
66  ARM_COMPUTE_RETURN_ERROR_ON((dilation.x() < 1) || (dilation.y() < 1));
67  ARM_COMPUTE_RETURN_ERROR_ON(input->data_layout() == DataLayout::UNKNOWN);
68  ARM_COMPUTE_RETURN_ERROR_ON(num_groups == 0);
69  ARM_COMPUTE_RETURN_ERROR_ON(input->data_layout() == DataLayout::NHWC && num_groups > 1);
70  ARM_COMPUTE_RETURN_ERROR_ON((input->dimension(channel_idx) % num_groups) != 0);
71 
72  if(output->total_size() > 0)
73  {
74  const TensorInfo tensor_info_output = output->clone()->set_tensor_shape(compute_im2col_conv_shape(input, kernel_dims, conv_info, has_bias, dilation, num_groups == 1, num_groups));
75  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_SHAPES(output, &tensor_info_output);
76  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input, output);
77  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_QUANTIZATION_INFO(input, output);
78  }
79 
80  return Status{};
81 }

83 std::pair<Status, Window> validate_and_configure_window(ITensorInfo *input, ITensorInfo *output, const Size2D &kernel_dims, const PadStrideInfo &conv_info, bool has_bias, const Size2D &dilation,
84  unsigned int num_elems_processed_per_iteration, bool is_padding_required_nchw, unsigned int num_groups)
85 {
86  ARM_COMPUTE_ERROR_ON_NULLPTR(input, output);
87 
88  // Output tensor auto initialization if not yet initialized
89  TensorShape expected_output_shape = compute_im2col_conv_shape(input, kernel_dims, conv_info, has_bias, dilation, num_groups == 1, num_groups);
90 
91  auto_init_if_empty(*output, input->clone()->set_tensor_shape(expected_output_shape));
92 
93  const DataLayout data_layout = input->data_layout();
94  const unsigned int width_idx = get_data_layout_dimension_index(data_layout, DataLayoutDimension::WIDTH);
95  const unsigned int height_idx = get_data_layout_dimension_index(data_layout, DataLayoutDimension::HEIGHT);
96  const unsigned int input_width = input->dimension(width_idx);
97  const unsigned int input_height = input->dimension(height_idx);
98 
99  // Configure the execute window based on the selected optimal OpenCL kernel
100  bool window_changed = false;
101  Window win;
102 
103  if(data_layout == DataLayout::NHWC)
104  {
105  win = calculate_max_window(*input, Steps(num_elems_processed_per_iteration));
106 
107  const int xin_start = 0;
108  const int xin_end = input->dimension(0) < num_elems_processed_per_iteration ? ceil_to_multiple(input->dimension(0), num_elems_processed_per_iteration) : input->dimension(0);
109  const int yin_start = 0;
110  const int yin_end = input->dimension(1);
111 
112  const int xout_start = 0;
113  const int xout_end = input->dimension(0) < num_elems_processed_per_iteration ? output->dimension(0) + (num_elems_processed_per_iteration - input->dimension(0)) : output->dimension(0);
114  const int yout_start = 0;
115  const int yout_end = output->dimension(1);
116 
117  AccessWindowStatic input_access(input, xin_start, yin_start, xin_end, yin_end);
118  AccessWindowStatic output_access(output, xout_start, yout_start, xout_end, yout_end);
119  window_changed = window_changed || update_window_and_padding(win, input_access, output_access);
120  }
121  else
122  {
123  if(is_padding_required_nchw)
124  {
125  const BorderSize border(conv_info.pad_top(), conv_info.pad_right(), conv_info.pad_bottom(), conv_info.pad_left());
126  win = calculate_max_window(*input,
127  Steps(num_elems_processed_per_iteration * conv_info.stride().first, conv_info.stride().second));
128  AccessWindowStatic input_access(input,
129  -border.left,
130  -border.top,
131  ceil_to_multiple(input_width + border.right, kernel_dims.width * num_elems_processed_per_iteration),
132  input_height + border.bottom);
133  window_changed = window_changed || update_window_and_padding(win, input_access);
134  }
135  else
136  {
137  // For the generic case, CLIm2ColKernel doesn't need padding (we do not read out-of-bounds elements) so
138  // update_window_and_padding() can be skipped
139  win = calculate_max_window(*input, Steps());
140  }
141  }
142 
143  output->set_valid_region(ValidRegion(Coordinates(), output->tensor_shape()));
144  // set the Z dimension's step same size as the whole dimension so that one can't split across the Z dimension
145  win.set_dimension_step(Window::DimZ, win[Window::DimZ].end() - win[Window::DimZ].start());
146 
147  Status err = (window_changed) ? ARM_COMPUTE_CREATE_ERROR(ErrorCode::RUNTIME_ERROR, "Insufficient Padding!") : Status{};
148  return std::make_pair(err, win);
149 }

151 Im2ColConfiguration configure_opencl_kernel(const ITensorInfo *input, const Size2D &kernel_dims, const PadStrideInfo &conv_info, bool has_bias, const Size2D &dilation, unsigned int num_groups)
152 {
153  const DataLayout data_layout = input->data_layout();
154  const DataType data_type = input->data_type();
155  const unsigned int width_idx = get_data_layout_dimension_index(data_layout, DataLayoutDimension::WIDTH);
156  const unsigned int height_idx = get_data_layout_dimension_index(data_layout, DataLayoutDimension::HEIGHT);
157  const unsigned int channel_idx = get_data_layout_dimension_index(data_layout, DataLayoutDimension::CHANNEL);
158  const unsigned int input_width = input->dimension(width_idx);
159  const unsigned int input_height = input->dimension(height_idx);
160  const unsigned int input_channel = input->dimension(channel_idx);
161 
162  const std::pair<unsigned int, unsigned int> convolved_dims = scaled_dimensions(input_width, input_height, kernel_dims.width, kernel_dims.height, conv_info, dilation);
163 
164  // Im2Col configuration
165  std::string kernel_name = "im2col_generic_";
166  CLBuildOptions build_opts;
167  unsigned int num_elems_processed_per_iteration = 1;
168  bool is_padding_required_nchw = false;
169 
170  build_opts.add_option("-DDATA_TYPE=" + get_cl_type_from_data_type(data_type));
171  build_opts.add_option("-DELEMENT_SIZE=" + support::cpp11::to_string(input->element_size()));
172  build_opts.add_option("-DKERNEL_WIDTH=" + support::cpp11::to_string(kernel_dims.width));
173  build_opts.add_option("-DKERNEL_HEIGHT=" + support::cpp11::to_string(kernel_dims.height));
174  build_opts.add_option("-DCONVOLVED_WIDTH=" + support::cpp11::to_string(convolved_dims.first));
175  build_opts.add_option("-DCONVOLVED_HEIGHT=" + support::cpp11::to_string(convolved_dims.second));
176  build_opts.add_option("-DSTRIDE_X=" + support::cpp11::to_string(conv_info.stride().first));
177  build_opts.add_option("-DSTRIDE_Y=" + support::cpp11::to_string(conv_info.stride().second));
178  build_opts.add_option("-DPAD_LEFT=" + support::cpp11::to_string(conv_info.pad_left()));
179  build_opts.add_option("-DPAD_TOP=" + support::cpp11::to_string(conv_info.pad_top()));
180  build_opts.add_option("-DPAD_RIGHT=" + support::cpp11::to_string(conv_info.pad_right()));
181  build_opts.add_option("-DPAD_BOTTOM=" + support::cpp11::to_string(conv_info.pad_bottom()));
182  build_opts.add_option("-DSRC_WIDTH=" + support::cpp11::to_string(input_width));
183  build_opts.add_option("-DSRC_HEIGHT=" + support::cpp11::to_string(input_height));
184  build_opts.add_option("-DSRC_DEPTH=" + support::cpp11::to_string(input_channel));
185  build_opts.add_option("-DDILATION_X=" + support::cpp11::to_string(dilation.x()));
186  build_opts.add_option("-DDILATION_Y=" + support::cpp11::to_string(dilation.y()));
187  build_opts.add_option_if(num_groups > 1, "-DNUM_GROUPS=" + support::cpp11::to_string(num_groups));
188  build_opts.add_option_if_else(is_data_type_quantized(data_type), "-DPAD_VALUE=" + support::cpp11::to_string(input->quantization_info().offset), "-DPAD_VALUE=0");
189  build_opts.add_option_if(has_bias, "-DHAS_BIAS");
190 
191  if(data_layout == DataLayout::NHWC)
192  {
193  num_elems_processed_per_iteration = 2;
194  is_padding_required_nchw = false;
195 
196  // Only the 3x3 and 9x9 cases are optimized for NHWC
197  if(kernel_dims == Size2D(3U, 3U))
198  {
199  kernel_name = "im2col3x3_";
200  }
201  else if(kernel_dims == Size2D(9U, 9U))
202  {
203  kernel_name = "im2col9x9_";
204  }
205 
206  build_opts.add_option("-DVECTOR_SIZE=" + support::cpp11::to_string(num_elems_processed_per_iteration));
207  build_opts.add_option("-DLAST_ACCESSED=" + support::cpp11::to_string(std::max(static_cast<int>(input_channel - num_elems_processed_per_iteration), 0)));
208  }
209  else
210  {
211  if(dilation == Size2D(1U, 1U))
212  {
213  const bool squared_im2col = kernel_dims.width == kernel_dims.height;
214  if(squared_im2col)
215  {
216  // Check if we can run an optimized im2col for NCHW
217  switch(kernel_dims.width)
218  {
219  case 1:
220  // Optimized im2col1x1 if stride_x = 1 and conv_info.has_padding() = false
221  if(conv_info.stride().first == 1 && !conv_info.has_padding())
222  {
223  kernel_name = "im2col1x1_stridex1_";
224  num_elems_processed_per_iteration = 4;
225  is_padding_required_nchw = true;
226  }
227  break;
228  case 3:
229  kernel_name = "im2col3x3_";
230  num_elems_processed_per_iteration = 1;
231  is_padding_required_nchw = true;
232  break;
233  case 5:
234  kernel_name = "im2col5x5_";
235  num_elems_processed_per_iteration = 1;
236  is_padding_required_nchw = true;
237  break;
238  case 11:
239  // Optimized im2col11x11 if pad_x = pad_y = 0
240  if(!conv_info.has_padding())
241  {
242  kernel_name = "im2col11x11_padx0_pady0_";
243  num_elems_processed_per_iteration = 1;
244  is_padding_required_nchw = true;
245  }
246  break;
247  default:
248  kernel_name = "im2col_generic_";
249  num_elems_processed_per_iteration = 1;
250  is_padding_required_nchw = false;
251  break;
252  }
253  }
254  else if(kernel_dims.width > 1 && !conv_info.has_padding())
255  {
256  kernel_name = "im2col_generic_padx0_pady0_";
257  num_elems_processed_per_iteration = 1;
258  is_padding_required_nchw = false;
259 
260  // Optimized im2col is performed using one or more vector operations with the specified vector size
261  // and a remainder. For example, for 5x5 convolutions, im2col is performed using vectors of size 4
262  // and scalars; for 7x7 convolutions, using vectors of size 4 and vectors of size 3.
263  // Using the vector size of 4 is always safe since OpenCL supports vectors of size 2 and 3.
264  // Using the vector size of 8, however, may be faster.
265  // For 2x2 convolutions, use vectors of size 2. (For 3x3 convolutions, im2col_kernel3x3_padx0_pady0
266  // is used instead.)
267  const size_t vector_size = std::min(static_cast<size_t>(4), kernel_dims.width);
268  const size_t width_mod_vector_size = kernel_dims.width % vector_size;
269  build_opts.add_option("-DVECTOR_SIZE=" + support::cpp11::to_string(vector_size));
270  build_opts.add_option("-DWIDTH_MOD_VECTOR_SIZE=" + support::cpp11::to_string(width_mod_vector_size));
271  }
272  }
273  }
274 
275  // Append the data layout to the kernel_name
276  kernel_name += lower_string(string_from_data_layout(data_layout));
277 
278  Im2ColConfiguration im2col_config;
279  im2col_config.kernel_name = kernel_name;
280  im2col_config.build_options = build_opts.options();
281  im2col_config.num_elems_processed_per_iteration = num_elems_processed_per_iteration;
282  im2col_config.is_padding_required_nchw = is_padding_required_nchw;
283 
284  return im2col_config;
285 }

293 void CLIm2ColKernel::configure(const ICLTensor *input, ICLTensor *output, const Size2D &kernel_dims, const PadStrideInfo &conv_info, bool has_bias, const Size2D &dilation,
294  unsigned int num_groups)
295 {
296  ARM_COMPUTE_ERROR_ON_NULLPTR(input, output);
297  ARM_COMPUTE_ERROR_THROW_ON(validate_arguments(input->info(), output->info(), kernel_dims, conv_info, has_bias, dilation, num_groups));
298 
299  const DataLayout data_layout = input->info()->data_layout();
300  const unsigned int width_idx = get_data_layout_dimension_index(data_layout, DataLayoutDimension::WIDTH);
301  const unsigned int height_idx = get_data_layout_dimension_index(data_layout, DataLayoutDimension::HEIGHT);
302  const unsigned int input_width = input->info()->dimension(width_idx);
303  const unsigned int input_height = input->info()->dimension(height_idx);
304 
305  // Select and configure the optimal OpenCL kernel to run.
306  // This function returns the OpenCL kernel's name, the arguments to pass at compile time, the number of elements processed per iteration
307  // and the padding requirement flag
308  Im2ColConfiguration im2col_config = configure_opencl_kernel(input->info(), kernel_dims, conv_info, has_bias, dilation, num_groups);
309 
310  // Create kernel
311  _kernel = static_cast<cl::Kernel>(CLKernelLibrary::get().create_kernel(im2col_config.kernel_name, im2col_config.build_options));
312 
313  _input = input;
314  _output = output;
315  _convolved_dims = scaled_dimensions(input_width, input_height, kernel_dims.width, kernel_dims.height, conv_info, dilation);
316  _num_elems_processed_per_iteration = im2col_config.num_elems_processed_per_iteration;
317  _kernel_dims = kernel_dims; // Only needed by the Tuner
318  _conv_info = conv_info; // Only needed by the Tuner
319  _num_groups = num_groups;
320 
321  // Configure kernel window
322  auto win_config = validate_and_configure_window(input->info(), output->info(), kernel_dims, conv_info, has_bias, dilation, im2col_config.num_elems_processed_per_iteration,
323  im2col_config.is_padding_required_nchw, num_groups);
324  ARM_COMPUTE_ERROR_THROW_ON(win_config.first);
325  ICLKernel::configure_internal(win_config.second);
326 
327  // Set config_id for enabling LWS tuning
328  _config_id = im2col_config.kernel_name;
329  _config_id += "_";
330  _config_id += lower_string(string_from_data_type(input->info()->data_type()));
331  _config_id += "_";
332  _config_id += support::cpp11::to_string(num_groups);
333  _config_id += "_";
334  _config_id += support::cpp11::to_string(output->info()->dimension(0));
335  _config_id += "_";
336  _config_id += support::cpp11::to_string(output->info()->dimension(1));
337  _config_id += "_";
338  _config_id += lower_string(string_from_data_layout(input->info()->data_layout()));
339 }

352 void CLIm2ColKernel::run(const Window &window, cl::CommandQueue &queue)
353 {
354  ARM_COMPUTE_ERROR_ON_UNCONFIGURED_KERNEL(this);
355  ARM_COMPUTE_ERROR_ON_MISMATCHING_WINDOWS(ICLKernel::window(), window);
356 
357  // Get initial windows
358  // Collapse in order to have (SRC_DEPTH * BATCH_SIZE) on the 3rd dimension
359  Window window_collapsed = window.collapse_if_possible(ICLKernel::window(), Window::DimZ);
360  window_collapsed.set_dimension_step(Window::DimZ, 1);
361 
362  Window window_output;
363  window_output.use_tensor_dimensions(_output->info()->tensor_shape());
364 
365  const Window first_slice_3d = window_collapsed.first_slice_window_3D();
366 
367  Window slice = first_slice_3d;
368  Window slice_in = first_slice_3d;
369  Window slice_out = window_output.first_slice_window_2D();
370 
371  if(_input->info()->data_layout() == DataLayout::NHWC)
372  {
373  const Window tmp_win = window.collapse_if_possible(ICLKernel::window(), 3);
374  const int num_batches = tmp_win[3].end();
375 
376  slice.set(1, Window::Dimension(0, static_cast<int>(_output->info()->tensor_shape()[1]), 1));
377  slice.set(2, Window::Dimension(0, static_cast<int>(num_batches), 1));
378  }
379  else
380  {
381  slice.set(0, Window::Dimension(0, static_cast<int>(ceil_to_multiple(_convolved_dims.first, _num_elems_processed_per_iteration)), _num_elems_processed_per_iteration));
382  slice.set(1, Window::Dimension(0, static_cast<int>(_convolved_dims.second), 1));
383  // Note: In case of NCHW the 3rd dimension is already set collapsing the input window
384  }
385 
386  // Setup input slice
387  // The dimensions of the input are increased within the OpenCL kernel
388  slice_in.set(Window::DimX, Window::Dimension(0, 0, 0));
389  slice_in.set(Window::DimY, Window::Dimension(0, 0, 0));
390  slice_in.set(Window::DimZ, Window::Dimension(0, 0, 0));
391 
392  // Setup output slice
393  // The dimensions of the output are increased within the OpenCL kernel
394  slice_out.set(Window::DimX, Window::Dimension(0, 0, 0));
395  slice_out.set(Window::DimY, Window::Dimension(0, 0, 0));
396 
397  unsigned int idx = num_arguments_per_3D_tensor() + (_num_groups == 1 ? num_arguments_per_2D_tensor() : num_arguments_per_3D_tensor());
398  _kernel.setArg<cl_uint>(idx++, static_cast<unsigned int>(_input->info()->strides_in_bytes()[3]));
399  _kernel.setArg<cl_uint>(idx++, static_cast<unsigned int>(_output->info()->strides_in_bytes()[((_num_groups == 1) ? 2 : 3)]));
400  do
401  {
402  unsigned int idx = 0;
403  add_3D_tensor_argument(idx, _input, slice_in);
404  if(_num_groups == 1)
405  {
406  add_2D_tensor_argument(idx, _output, slice_out);
407  }
408  else
409  {
410  add_3D_tensor_argument(idx, _output, slice_out);
411  }
412  enqueue(queue, *this, slice, lws_hint());
413  }
414  while(window_collapsed.slide_window_slice_3D(slice) && window_output.slide_window_slice_2D(slice_out) && window_collapsed.slide_window_slice_3D(slice_in));
415 }
