51 inline Status validate_arguments(const ITensorInfo *input0, const ITensorInfo *input1, const ITensorInfo *input2, const ITensorInfo *output, float beta,
52  bool is_interleaved_transposed, const GEMMReshapeInfo &reshape_info, bool fp_mixed_precision)
53 {
54  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(input0, input1, output);
55  ARM_COMPUTE_RETURN_ERROR_ON_F16_UNSUPPORTED(input0);
56  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(input0, 1, DataType::F16, DataType::F32);
57  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input0, input1);
58  ARM_COMPUTE_RETURN_ERROR_ON_MSG((fp_mixed_precision && (input0->data_type() != DataType::F16)), "Mixed precision floating point is supported only for F16 data");
59  ARM_COMPUTE_RETURN_ERROR_ON_MSG(input0->num_dimensions() > 4, "The number of dimensions for the matrix A must be <= 4");
60  ARM_COMPUTE_RETURN_ERROR_ON_MSG(input1->num_dimensions() > 3, "The number of dimensions for the matrix B must be <= 3");
61  ARM_COMPUTE_RETURN_ERROR_ON_MSG(is_interleaved_transposed && reshape_info.reinterpret_input_as_3d(), "The input tensor cannot be reinterpreted as 3D if is_interleaved_transposed is true");
62  ARM_COMPUTE_RETURN_ERROR_ON_MSG(input1->num_dimensions() > 2 && reshape_info.reinterpret_input_as_3d(), "The input1 tensor cannot have more than 2 dimensions if input0 has to be reinterpreted as 3D");
63 
64  const bool is_beta_one = std::abs(1.0f - beta) < 0.00001f;
65  const bool has_vec_c = input2 != nullptr && beta != 0.f;
66  ARM_COMPUTE_RETURN_ERROR_ON_MSG(has_vec_c && !is_beta_one, "Adding input2 is only supported for beta equal to 1");
67 
68  if(!is_interleaved_transposed)
69  {
70  ARM_COMPUTE_RETURN_ERROR_ON(input0->dimension(0) != input1->dimension(1));
71 
72  if(has_vec_c)
73  {
74  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input0, input2);
75  ARM_COMPUTE_RETURN_ERROR_ON_MSG(input2->num_dimensions() > 1, "input2 must be a 1D tensor");
76  ARM_COMPUTE_RETURN_ERROR_ON_MSG(input2->dimension(0) != input1->dimension(0), "Length of Vector C must match the number of columns of matrix B");
77  }
78  }
79  else
80  {
81  GEMMRHSMatrixInfo rhs_info;
82  GEMMLHSMatrixInfo lhs_info;
83  const int m = reshape_info.m();
84  const int n = reshape_info.n();
85  const int k = reshape_info.k();
86  const int mult_transpose1xW_width = reshape_info.mult_transpose1xW_width();
87  const int mult_interleave4x4_height = reshape_info.mult_interleave4x4_height();
88  rhs_info.n0 = 16 / input1->element_size();
89  rhs_info.k0 = 1;
90  rhs_info.h0 = mult_transpose1xW_width;
91  rhs_info.interleave = false;
92  rhs_info.transpose = false;
93  lhs_info.m0 = 4;
94  lhs_info.k0 = 4;
95  lhs_info.v0 = mult_interleave4x4_height;
96  lhs_info.interleave = true;
97  lhs_info.transpose = true;
98 
99  TensorShape tensor_shape0{ input0->tensor_shape() };
100  tensor_shape0.set(0, k);
101  tensor_shape0.set(1, m);
102 
103  TensorShape tensor_shape1{ input1->tensor_shape() };
104  tensor_shape1.set(0, n);
105  tensor_shape1.set(1, k);
106 
107  const TensorInfo tensor_info0 = input0->clone()->set_tensor_shape(tensor_shape0);
108  const TensorInfo tensor_info1 = input1->clone()->set_tensor_shape(tensor_shape1);
109 
110  const TensorInfo tensor_info_reshaped0 = input0->clone()->set_tensor_shape(compute_lhs_reshaped_shape(tensor_info0, lhs_info));
111  const TensorInfo tensor_info_reshaped1 = input1->clone()->set_tensor_shape(compute_rhs_reshaped_shape(tensor_info1, rhs_info));
112 
113  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_SHAPES(input0, &tensor_info_reshaped0);
114  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_SHAPES(input1, &tensor_info_reshaped1);
115 
116  if(has_vec_c)
117  {
118  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input0, input2);
119  ARM_COMPUTE_RETURN_ERROR_ON_MSG(input2->num_dimensions() > 1, "input2 must be a 1D tensor");
120  }
121  }
122 
123  if(output->total_size() != 0)
124  {
125  const TensorInfo tensor_info_output = output->clone()->set_tensor_shape(compute_mm_shape(*input0, *input1, is_interleaved_transposed, reshape_info));
126  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_SHAPES(output, &tensor_info_output);
127  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input0, output);
128  }
129 
130  return Status{};
131 }

133 inline std::pair<Status, Window> validate_and_configure_window(ITensorInfo *input0, ITensorInfo *input1, ITensorInfo *input2, ITensorInfo *output,
134  float beta, bool is_interleaved_transposed, const GEMMReshapeInfo &reshape_info, GPUTarget gpu_target,
135  ElementsProcessed &num_elements_processed)
136 {
137  ARM_COMPUTE_UNUSED(beta);
138  bool window_changed = false;
139  Window win{};
140  Window win_out{};
141 
142  const DataType data_type = input0->data_type();
143  unsigned int &num_elems_processed_per_iteration_x = num_elements_processed[0];
144  unsigned int &num_elems_processed_per_iteration_y = num_elements_processed[1];
145  bool reinterpret_input_as_3d = reshape_info.reinterpret_input_as_3d();
146  bool reinterpret_output_as_3d = (reshape_info.depth_output_gemm3d() != 0);
147  const bool has_vec_c = input2 != nullptr && beta != 0.f;
148 
149  // In case both input and output have to be reinterpreted as 3D tensors,
150  // force reinterpret_input_as_3d and reinterpret_output_as_3d to be false.
151  if(reinterpret_input_as_3d == reinterpret_output_as_3d)
152  {
153  reinterpret_input_as_3d = false;
154  reinterpret_output_as_3d = false;
155  }
156 
157  // Output tensor auto inizialitation if not yet initialized
158  auto_init_if_empty(*output, input0->clone()->set_tensor_shape(compute_mm_shape(*input0, *input1, is_interleaved_transposed, reshape_info)));
159 
160  TensorInfo tmp_info(*output);
161 
162  if(reinterpret_output_as_3d)
163  {
164  // Since the output tensor has to be reinterpreted as 3D and the execute window is based on a 2D GEMM,
165  // the window needs to be constructed on the 2D collapsed version of the tensor
166  TensorShape tmp_shape(output->tensor_shape());
167  tmp_shape.collapse(2U, 1U);
168  tmp_info.set_tensor_shape(tmp_shape);
169  }
170 
171  if(is_interleaved_transposed)
172  {
173  // reinterpret_input_as_3d is not supported if is_interleaved_transposed is set
174  ARM_COMPUTE_ERROR_ON(reshape_info.reinterpret_input_as_3d());
175 
176  // Configure kernel window
177  num_elems_processed_per_iteration_x = max_cl_vector_width / data_size_from_type(data_type);
178  num_elems_processed_per_iteration_y = 4;
179 
180  // Note: bottom paddings are calculated manually as the output can be reinterpreted as 3D tensor
181  // The only way to set properly the paddings, it is to set those explicitly through the AccessWindowStatic
182  const int m = reshape_info.m();
183  const int bottom_pad = (num_elems_processed_per_iteration_y - (m % num_elems_processed_per_iteration_y)) % num_elems_processed_per_iteration_y;
184 
185  win = calculate_max_window(tmp_info, Steps(num_elems_processed_per_iteration_x, num_elems_processed_per_iteration_y));
186  win_out = calculate_max_window(*output, Steps(num_elems_processed_per_iteration_x, num_elems_processed_per_iteration_y));
187 
188  AccessWindowStatic input0_access(input0, 0, 0, input0->dimension(0), input0->dimension(1));
189  AccessWindowStatic input1_access(input1, 0, 0,
190  ceil_to_multiple(input1->dimension(0), num_elems_processed_per_iteration_x),
191  ceil_to_multiple(input1->dimension(1), num_elems_processed_per_iteration_y));
192  AccessWindowStatic output_access(output, 0, 0,
193  ceil_to_multiple(output->dimension(0), num_elems_processed_per_iteration_x),
194  output->dimension(1) + bottom_pad);
195 
196  window_changed = update_window_and_padding(win, input0_access, input1_access) || // window used by the execute_window_loop
197  update_window_and_padding(win_out, output_access); // window used to update the padding requirements of output tensor
198  if(has_vec_c)
199  {
200  AccessWindowHorizontal input2_access(input2, 0, num_elems_processed_per_iteration_x);
201  window_changed = window_changed || update_window_and_padding(win, input2_access);
202  }
203 
204  output_access.set_valid_region(win_out, ValidRegion(Coordinates(0, 0), output->tensor_shape()));
205  }
206  else // The input tensors have not been reshaped
207  {
208  // Special case for 1xN, 2xN, 3xN and 4xN input0 tensor. num_elems_processed_per_iteration_x is set up for the default case.
209  num_elems_processed_per_iteration_x = max_cl_vector_width / data_size_from_type(data_type);
210  num_elems_processed_per_iteration_y = std::min(static_cast<int>(output->dimension(1)), 4);
211 
212  // Note: bottom paddings are calculated manually as the output can be reinterpreted as 3D tensor
213  // The only way to set properly the paddings, it is to set those explicitly through the AccessWindowStatic
214  const int m = reinterpret_input_as_3d ? input0->tensor_shape()[1] * input0->tensor_shape()[2] : input0->tensor_shape()[1];
215  const int bottom_pad = (num_elems_processed_per_iteration_y - (m % num_elems_processed_per_iteration_y)) % num_elems_processed_per_iteration_y;
216 
217  // Create kernels according to the architecture, data type and input size.
218  GPUTarget arch_target = get_arch_from_target(gpu_target);
219  if(arch_target == GPUTarget::BIFROST && data_type == DataType::F32)
220  {
221  num_elems_processed_per_iteration_x = (input1->dimension(0) <= 1000 && input0->num_dimensions() == 1) ? 2 : 4;
222  }
223 
224  // Configure window
225  win = calculate_max_window(tmp_info, Steps(num_elems_processed_per_iteration_x, num_elems_processed_per_iteration_y));
226  win_out = calculate_max_window(*output, Steps(num_elems_processed_per_iteration_x, num_elems_processed_per_iteration_y));
227 
228  AccessWindowStatic input0_access(input0, 0, 0, input0->dimension(0), input0->dimension(1) + bottom_pad);
229  AccessWindowStatic input1_access(input1, 0, 0, ceil_to_multiple(input1->dimension(0), num_elems_processed_per_iteration_x), input1->dimension(1));
230  AccessWindowStatic output_access(output, 0, 0,
231  ceil_to_multiple(output->dimension(0), num_elems_processed_per_iteration_x),
232  output->dimension(1) + bottom_pad);
233 
234  window_changed = update_window_and_padding(win, input0_access, input1_access) || // window used by the execute_window_loop
235  update_window_and_padding(win_out, output_access); // window used to update the padding requirements of output tensor
236  if(has_vec_c)
237  {
238  AccessWindowHorizontal input2_access(input2, 0, num_elems_processed_per_iteration_x);
239  window_changed = window_changed || update_window_and_padding(win, input2_access);
240  }
241 
242  Coordinates coord;
243  coord.set_num_dimensions(output->num_dimensions());
244  output_access.set_valid_region(win_out, ValidRegion(coord, output->tensor_shape()));
245  }
246 
247  // Collapse along the Z direction
248  // This collapse needs to be here in order to tune the Z dimension of LWS
249  Window collapsed = win;
250  const unsigned int dimension_to_collapse = std::min(static_cast<unsigned int>(output->num_dimensions()), 2u);
251  collapsed = win.collapse(win, dimension_to_collapse);
252 
253  Status err = (window_changed) ? ARM_COMPUTE_CREATE_ERROR(ErrorCode::RUNTIME_ERROR, "Insufficient Padding!") : Status{};
254  return std::make_pair(err, collapsed);
255 }

263 void CLGEMMMatrixMultiplyKernel::configure(const ICLTensor *input0, const ICLTensor *input1, const ICLTensor *input2, ICLTensor *output, float alpha, float beta,
264  bool is_interleaved_transposed, const GEMMReshapeInfo &reshape_info, bool fp_mixed_precision)
265 {
266  ARM_COMPUTE_ERROR_ON_NULLPTR(input0, input1, output);
267 
268  // Perform validate step
269  ARM_COMPUTE_ERROR_THROW_ON(validate_arguments(input0->info(), input1->info(), (input2 != nullptr) ? input2->info() : nullptr, output->info(), beta,
270  is_interleaved_transposed, reshape_info, fp_mixed_precision));
271 
272  _input0 = input0;
273  _input1 = input1;
274  _input2 = input2;
275  _output = output;
276  _reinterpret_input_as_3d = reshape_info.reinterpret_input_as_3d();
277  _reinterpret_output_as_3d = (reshape_info.depth_output_gemm3d() != 0);
278 
279  // In case both input and output have to be reinterpreted as 3D tensors,
280  // force reinterpret_input_as_3d and reinterpret_output_as_3d to be false.
281  if(_reinterpret_input_as_3d == _reinterpret_output_as_3d)
282  {
283  _reinterpret_input_as_3d = false;
284  _reinterpret_output_as_3d = false;
285  }
286 
287  // Check if we need to slide the matrix B
288  const unsigned int num_dimensions_input0 = _reinterpret_input_as_3d ? _input0->info()->num_dimensions() - 1 : _input0->info()->num_dimensions();
289 
290  _slide_matrix_b = (_input1->info()->num_dimensions() >= num_dimensions_input0);
291 
292  const DataType data_type = input0->info()->data_type();
293 
294  // Get target architecture
295  GPUTarget gpu_target = get_target();
296 
297  ElementsProcessed num_elements_processed{};
298 
299  // Configure kernel window
300  auto win_config = validate_and_configure_window(input0->info(), input1->info(), (input2 != nullptr) ? input2->info() : nullptr, output->info(), beta, is_interleaved_transposed, reshape_info,
301  gpu_target, num_elements_processed);
302  ARM_COMPUTE_ERROR_THROW_ON(win_config.first);
303  ICLKernel::configure_internal(win_config.second);
304 
305  // Create build options
306  CLBuildOptions build_opts;
307 
308  // Only define ALPHA when alpha is not 1.0f. This avoids performing unnecessary multiplications.
309  if(std::abs(1.0f - alpha) > 0.00001f)
310  {
311  build_opts.add_option("-DALPHA=" + float_to_string_with_full_precision(alpha));
312  }
313  build_opts.add_option_if(_reinterpret_input_as_3d, "-DREINTERPRET_INPUT_AS_3D");
314  build_opts.add_option_if(_reinterpret_output_as_3d, "-DREINTERPRET_OUTPUT_AS_3D");
315  build_opts.add_option_if(_reinterpret_input_as_3d || _reinterpret_output_as_3d, "-DHEIGHT_GEMM3D=" + support::cpp11::to_string(output->info()->dimension(1)));
316  build_opts.add_option_if(_reinterpret_input_as_3d || _reinterpret_output_as_3d, "-DDEPTH_GEMM3D=" + support::cpp11::to_string(output->info()->dimension(2)));
317 
318  // Do not slide matrix B if _slide_matrix_b = false
319  build_opts.add_option_if(!_slide_matrix_b, "-DMATRIX_B_DEPTH=" + support::cpp11::to_string(input1->info()->dimension(2)));
320 
321  const bool is_bifrost = get_arch_from_target(gpu_target) == GPUTarget::BIFROST;
322 
323  _has_vec_c = input2 != nullptr && beta != 0.f;
324 
325  std::string kernel_name;
326  if(is_interleaved_transposed)
327  {
328  const int mult_transpose1xW_width = reshape_info.mult_transpose1xW_width();
329  const int mult_interleave4x4_height = reshape_info.mult_interleave4x4_height();
330 
331  build_opts.add_option("-DCOLS_B=" + support::cpp11::to_string(input1->info()->dimension(0)));
332  build_opts.add_option("-DMULT_TRANSPOSE1XW_WIDTH=" + support::cpp11::to_string(mult_transpose1xW_width));
333  build_opts.add_option("-DMULT_INTERLEAVE4X4_HEIGHT=" + support::cpp11::to_string(mult_interleave4x4_height));
334 
335  if(is_data_type_float(data_type) && is_bifrost)
336  {
337  kernel_name = "gemm_mm_interleaved_transposed_" + lower_string(string_from_data_type(data_type)) + "_bifrost";
338  }
339  else
340  {
341  kernel_name = "gemm_mm_interleaved_transposed_" + lower_string(string_from_data_type(data_type));
342  if(fp_mixed_precision && data_type == DataType::F16)
343  {
344  // currently wider accumulator is only supported for fp16 kernels.
345  kernel_name += "_acc32";
346  }
347  }
348  }
349  else // The input tensors have not been reshaped
350  {
351  build_opts.add_option("-DCOLS_A=" + support::cpp11::to_string(input0->info()->dimension(0)));
352  build_opts.add_option("-DDATA_TYPE=" + get_cl_type_from_data_type(data_type));
353 
354  // Create kernels according to the architecture, data type and input size.
355  if(is_data_type_float(data_type) && is_bifrost)
356  {
357  kernel_name = "gemm_mm_floating_point";
358 
359  if(input0->info()->num_dimensions() != 1)
360  {
361  kernel_name += "_" + lower_string(string_from_data_type(data_type)) + "_bifrost";
362  if(fp_mixed_precision && data_type == DataType::F16)
363  {
364  // currently wider accumulator is only supported for fp16 kernels.
365  kernel_name += "_acc32";
366  }
367  }
368  else if(input1->info()->dimension(0) <= 1000 && data_type == DataType::F32)
369  {
370  // The first kernel is optimized for the case of 1000 or less output elements (e.g. FC8 of AlexNet and VGG-16, and
371  // FC1 of Inception v3). The second kernel is optimized for the case of greater than 1000 output elements (e.g.
372  // FC6 and FC7 of AlexNet and VGG-16).
373  kernel_name += "_" + lower_string(string_from_data_type(data_type)) + "_bifrost_1000";
374  }
375 
376  // The work-group size equal to the Bifrost quad size has been proved to be optimal for these kernels
377  // via exhaustive autotuning over a range of representative layer configurations.
378  set_lws_hint(cl::NDRange(4));
379  }
380  else // (MIDGARD and F32) or (F16)
381  {
382  kernel_name = "gemm_mm_floating_point";
383  }
384  build_opts.add_option("-DNUM_ELEMS_PROCESSED_PER_THREAD_Y=" + support::cpp11::to_string(num_elements_processed.y()));
385  build_opts.add_option("-DNUM_ELEMS_PROCESSED_PER_THREAD_X=" + support::cpp11::to_string(num_elements_processed.x()));
386  }
387 
388  // Configure matrix C addition if necessary
389  build_opts.add_option_if(_has_vec_c, "-DADD_VEC_C");
390 
391  // Create kernel
392  _kernel = static_cast<cl::Kernel>(CLKernelLibrary::get().create_kernel(kernel_name, build_opts.options()));
393 
394  // Set config_id for enabling LWS tuning
395  _config_id = "gemm_";
396  _config_id += (is_interleaved_transposed ? "reshaped_" : "");
397  _config_id += (fp_mixed_precision ? "fp_mixed_" : "");
398  _config_id += (_reinterpret_input_as_3d ? "3di_" : "");
399  _config_id += (_reinterpret_output_as_3d ? "3do_" : "");
400  _config_id += lower_string(string_from_data_type(input0->info()->data_type()));
401  _config_id += "_";
402  _config_id += support::cpp11::to_string(output->info()->dimension(1));
403  _config_id += "_";
404  _config_id += support::cpp11::to_string(output->info()->dimension(0));
405  _config_id += "_";
406  _config_id += support::cpp11::to_string(output->info()->dimension(2));
407  _config_id += "_";
408  _config_id += support::cpp11::to_string(output->info()->dimension(3));
409  _config_id += "_";
410  _config_id += (is_interleaved_transposed ? support::cpp11::to_string(input1->info()->dimension(0)) : support::cpp11::to_string(input1->info()->dimension(1)));
411 }

413 Status CLGEMMMatrixMultiplyKernel::validate(const ITensorInfo *input0, const ITensorInfo *input1, const ITensorInfo *input2, const ITensorInfo *output, float alpha, float beta,
414  bool is_interleaved_transposed, const GEMMReshapeInfo &reshape_info, GPUTarget gpu_target, bool fp_mixed_precision)
415 {

417  ElementsProcessed num_elements_processed{};

420  ARM_COMPUTE_RETURN_ON_ERROR(validate_and_configure_window(input0->clone().get(),
421  input1->clone().get(),
422  (input2 != nullptr) ? input2->clone().get() : nullptr,
423  output->clone().get(),
424  beta,
425  is_interleaved_transposed,
426  reshape_info,
427  gpu_target,
428  num_elements_processed)
429  .first);
430 
431  return Status{};
432 }

434 void CLGEMMMatrixMultiplyKernel::run(const Window &window, cl::CommandQueue &queue)
435 {
436  ARM_COMPUTE_ERROR_ON_UNCONFIGURED_KERNEL(this);
437  ARM_COMPUTE_ERROR_ON_INVALID_SUBWINDOW(ICLKernel::window(), window);
438 
439  if(_input1->info()->num_dimensions() < 3)
440  {
441  // The stride_z for matrix B must be zero if we do not slice
442  ARM_COMPUTE_ERROR_ON(_input1->info()->strides_in_bytes()[3] != 0);
443  }
444 
445  Window slice = window.first_slice_window_3D();
446  Window slice_matrix_b = slice;
447 
448  slice_matrix_b.set(Window::DimX, Window::Dimension(0, 1, 1));
449  slice_matrix_b.set(Window::DimY, Window::Dimension(0, 1, 1));
450 
451  const unsigned int num_arguments_vec_c = (_has_vec_c) ? num_arguments_per_1D_tensor() : 0;
452 
453  if(_reinterpret_input_as_3d)
454  {
455  // Pass bottom paddings to the kernel if the input has to be reinterpreted as 3D tensor
456  const unsigned int idx0 = 3 * num_arguments_per_2D_tensor() + 3 + num_arguments_vec_c;
457  const unsigned int total_cross_plane_pad = _input0->info()->padding().top + _input0->info()->padding().bottom;
458  _kernel.setArg<cl_uint>(idx0, static_cast<unsigned int>(total_cross_plane_pad));
459  }
460 
461  if(_reinterpret_output_as_3d)
462  {
463  // Pass bottom paddings to the kernel if the output has to be reinterpreted as 3D tensor
464  const unsigned int idx0 = 3 * num_arguments_per_2D_tensor() + 3 + (_reinterpret_input_as_3d ? 1 : 0) + num_arguments_vec_c;
465  const unsigned int total_cross_plane_pad = _output->info()->padding().top + _output->info()->padding().bottom;
466  _kernel.setArg<cl_uint>(idx0, static_cast<unsigned int>(total_cross_plane_pad));
467  }
468 
469  do
470  {
471  Window slice_b = slice;
472  // Don't slice matrix B along the z dimension if matrix B has just 2 dimensions and matrix A more than 2
473  // This scenario can happen when the matrix multiplication is used to perform a convolution operation
474  if(!_slide_matrix_b)
475  {
476  slice_b = slice_matrix_b;
477  }
478 
479  unsigned int idx = 0;
480  add_2D_tensor_argument(idx, _input0, slice);
481  add_2D_tensor_argument(idx, _input1, slice_b);
482  if(_has_vec_c)
483  {
484  add_1D_tensor_argument(idx, _input2, slice);
485  }
486  add_2D_tensor_argument(idx, _output, slice);
487  _kernel.setArg<cl_uint>(idx++, static_cast<unsigned int>(_input0->info()->strides_in_bytes()[2]));
488  _kernel.setArg<cl_uint>(idx++, static_cast<unsigned int>(_input1->info()->strides_in_bytes()[2]));
489  _kernel.setArg<cl_uint>(idx++, static_cast<unsigned int>(_output->info()->strides_in_bytes()[2]));
490  enqueue(queue, *this, slice, lws_hint());
491  }
492  while(window.slide_window_slice_3D(slice));
493 }
