44 template <typename T>
45 void CPPFlipWeightsKernel::flip_weights(const Window &window_input)
46 {
47  // Create iterators
48  Iterator in(_input, window_input);
49 
50  const DataLayout data_layout = _input->info()->data_layout();
51  const size_t idx_w = get_data_layout_dimension_index(data_layout, DataLayoutDimension::WIDTH);
52  const size_t idx_h = get_data_layout_dimension_index(data_layout, DataLayoutDimension::HEIGHT);
53 
54  const int kernel_width = _input->info()->dimension(idx_w);
55  const int kernel_height = _input->info()->dimension(idx_h);
56 
57  execute_window_loop(window_input, [&](const Coordinates & id)
58  {
59  const unsigned int x = kernel_width - id[idx_w] - 1;
60  const unsigned int y = kernel_height - id[idx_h] - 1;
61  Coordinates output_coord(id);
62  output_coord.set(idx_w, x);
63  output_coord.set(idx_h, y);
64  *(reinterpret_cast<T *>(_output->ptr_to_element(output_coord))) = *(reinterpret_cast<const T *>(in.ptr()));
65  },
66  in);
67 }

69 void CPPFlipWeightsKernel::configure(const ITensor *input, ITensor *output)
70 {
71  ARM_COMPUTE_ERROR_ON_NULLPTR(input, output);
72  ARM_COMPUTE_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(input, 1, DataType::QASYMM8, DataType::F16, DataType::F32);
73  ARM_COMPUTE_ERROR_ON_MISMATCHING_DATA_LAYOUT(input, output);
74  ARM_COMPUTE_ERROR_ON_MISMATCHING_DATA_TYPES(input, output);
75 
76  _input = input;
77  _output = output;
78 
79  // Configure kernel window
80  Window win = calculate_max_window(*input->info(), Steps());
81 
82  // The CPPFlipWeightsKernel doesn't need padding so update_window_and_padding() can be skipped
83  Coordinates coord;
84  coord.set_num_dimensions(output->info()->num_dimensions());
85  output->info()->set_valid_region(ValidRegion(coord, output->info()->tensor_shape()));
86 
87  ICPPKernel::configure(win);
88 
89  switch(input->info()->data_type())
90  {
91  case DataType::F32:
92  _func = &CPPFlipWeightsKernel::flip_weights<float>;
93  break;
94  case DataType::F16:
95  _func = &CPPFlipWeightsKernel::flip_weights<half>;
96  break;
97  case DataType::QASYMM8:
98  _func = &CPPFlipWeightsKernel::flip_weights<uint8_t>;
99  break;
100  default:
101  ARM_COMPUTE_ERROR("Not supported");
102  }
103 }
