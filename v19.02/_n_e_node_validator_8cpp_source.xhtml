41 Status NENodeValidator::validate(INode *node)
42 {
43  if(node == nullptr)
44  {
45  return Status{};
46  }
47 
48  NodeType type = node->type();
49  switch(type)
50  {
51  case NodeType::BoundingBoxTransformLayer:
52  return ARM_COMPUTE_CREATE_ERROR(arm_compute::ErrorCode::RUNTIME_ERROR, "Unsupported operation : BoundingBoxTransformLayer");
53  case NodeType::ChannelShuffleLayer:
54  return detail::validate_channel_shuffle_layer<NEChannelShuffleLayer>(*polymorphic_downcast<ChannelShuffleLayerNode *>(node));
55  case NodeType::ConvolutionLayer:
56  return detail::validate_convolution_layer<NEConvolutionLayer,
57  NEDirectConvolutionLayer,
58  NEGEMMConvolutionLayer,
59  NEWinogradConvolutionLayer>(*polymorphic_downcast<ConvolutionLayerNode *>(node));
60  case NodeType::DepthwiseConvolutionLayer:
61  return detail::validate_depthwise_convolution_layer<NEDepthwiseConvolutionLayer,
62  NEDepthwiseConvolutionLayer3x3>(*polymorphic_downcast<DepthwiseConvolutionLayerNode *>(node));
63  case NodeType::DetectionOutputLayer:
64  return detail::validate_detection_output_layer<CPPDetectionOutputLayer>(*polymorphic_downcast<DetectionOutputLayerNode *>(node));
65  case NodeType::GenerateProposalsLayer:
66  return ARM_COMPUTE_CREATE_ERROR(arm_compute::ErrorCode::RUNTIME_ERROR, "Unsupported operation : GenerateProposalsLayer");
67  case NodeType::NormalizePlanarYUVLayer:
68  return ARM_COMPUTE_CREATE_ERROR(arm_compute::ErrorCode::RUNTIME_ERROR, "Unsupported operation : NormalizePlanarYUVLayer");
69  case NodeType::PadLayer:
70  return ARM_COMPUTE_CREATE_ERROR(arm_compute::ErrorCode::RUNTIME_ERROR, "Unsupported operation : PadLayer");
71  case NodeType::PermuteLayer:
72  return detail::validate_permute_layer<NEPermute>(*polymorphic_downcast<PermuteLayerNode *>(node));
73  case NodeType::PriorBoxLayer:
74  return detail::validate_priorbox_layer<NEPriorBoxLayer>(*polymorphic_downcast<PriorBoxLayerNode *>(node));
75  case NodeType::ReorgLayer:
76  return detail::validate_reorg_layer<NEReorgLayer>(*polymorphic_downcast<ReorgLayerNode *>(node));
77  case NodeType::ROIAlignLayer:
78  return ARM_COMPUTE_CREATE_ERROR(arm_compute::ErrorCode::RUNTIME_ERROR, "Unsupported operation : ROIAlignLayer");
79  case NodeType::SliceLayer:
80  return ARM_COMPUTE_CREATE_ERROR(arm_compute::ErrorCode::RUNTIME_ERROR, "Unsupported operation : SliceLayer");
81  case NodeType::UpsampleLayer:
82  return detail::validate_upsample_layer<NEUpsampleLayer>(*polymorphic_downcast<UpsampleLayerNode *>(node));
83  case NodeType::YOLOLayer:
84  return detail::validate_yolo_layer<NEYOLOLayer>(*polymorphic_downcast<YOLOLayerNode *>(node));
85  default:
86  return Status{};
87  }
88 }
