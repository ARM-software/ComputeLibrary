41 template <typename T>
42 void quantize_down_int32_to_uint8_scale(const SimpleTensor<T> *in, const SimpleTensor<T> *bias, SimpleTensor<uint8_t> *dst, int32_t result_offset, int32_t result_mult_int, int32_t result_shift,
43  int32_t min, int32_t max)
44 {
45  const int cols_in = in->shape().x();
46 
47  for(int i = 0; i < in->num_elements(); ++i)
48  {
49  int32_t result = ((*in)[i] + result_offset);
50 
51  if(bias != nullptr)
52  {
53  result += (*bias)[i % cols_in];
54  }
55 
56  result *= result_mult_int;
57 
58  result >>= result_shift;
59 
60  // Bounded ReLu
61  if(min != max)
62  {
63  result = std::max(min, std::min(max, result));
64  }
65 
66  (*dst)[i] = static_cast<uint8_t>(std::max(0, std::min(255, result)));
67  }
68 }

128 template <typename T_out, typename T_in>
129 SimpleTensor<T_out> gemmlowp_matrix_multiply_core(const SimpleTensor<T_in> &a, const SimpleTensor<T_in> &b, TensorShape shape_c, int32_t a_offset, int32_t b_offset)
130 {
131  static_assert(std::is_same<typename std::decay<T_out>::type, int32_t>::value, "Only int32_t is allowed for the output");
132 
133  DataType dt = std::is_same<T_out, int32_t>::value ? DataType::S32 : DataType::U32;
134  SimpleTensor<T_out> c(shape_c, dt);
135 
136  const int K = a.shape().x();
137  const int M = a.shape().y();
138  const int N = b.shape().x();
139  const int D = a.shape().z(); // Number of matrices in a batch
140 
141  const int a_stride_z = K * M;
142  // Do not slide the matrix B along the 3rd dimension in case matrix B has less than 3 dimensions
143  const int b_stride_z = b.shape().num_dimensions() > 2 ? N * K : 0;
144  const int c_stride_z = N * M;
145 
146  std::vector<T_out> acc;
147  acc.resize(N);
148 
149  for(int depth = 0; depth < D; ++depth)
150  {
151  const int base_addr_a = depth * a_stride_z;
152  const int base_addr_b = depth * b_stride_z;
153  const int base_addr_c = depth * c_stride_z;
154 
155  for(int i = 0; i < M; ++i)
156  {
157  for(int j = 0; j < N; ++j)
158  {
159  acc[j] = 0;
160  }
161  for(int k = 0; k < K; ++k)
162  {
163  const T_out tmp_a = a_offset + static_cast<T_out>(a[base_addr_a + k + i * K]);
164  for(int j = 0; j < N; ++j)
165  {
166  const T_out tmp_b = b_offset + static_cast<T_out>(b[base_addr_b + j + k * N]);
167  const T_out mult_as_int = tmp_a * tmp_b;
168  acc[j] += mult_as_int;
169  }
170  }
171  for(int j = 0; j < N; ++j)
172  {
173  c[base_addr_c + j + i * N] = acc[j];
174  }
175  }
176  }
177 
178  return c;
179 }
