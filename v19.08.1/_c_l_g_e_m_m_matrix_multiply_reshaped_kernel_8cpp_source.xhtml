59 Status validate_arguments(const ITensorInfo *input0, const ITensorInfo *input1, const ITensorInfo *input2, const ITensorInfo *output, float alpha, float beta, const GEMMLHSMatrixInfo &lhs_info,
60  const GEMMRHSMatrixInfo &rhs_info,
61  const GEMMKernelInfo &gemm_info)
62 {
63  ARM_COMPUTE_UNUSED(alpha);
64  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(input0, input1, output);
65  ARM_COMPUTE_RETURN_ERROR_ON_F16_UNSUPPORTED(input0);
66  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(input0, 1, DataType::F32);
67  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input0, input1);
68  ARM_COMPUTE_RETURN_ERROR_ON_MSG(input0->num_dimensions() > 4, "The number of dimensions for the LHS matrix must be <= 4");
69  ARM_COMPUTE_RETURN_ERROR_ON_MSG(input1->num_dimensions() > 3, "The number of dimensions for the RHS matrix must be <= 3");
70  ARM_COMPUTE_RETURN_ERROR_ON(lhs_info.transpose);
71  ARM_COMPUTE_RETURN_ERROR_ON(!rhs_info.transpose);
72  ARM_COMPUTE_RETURN_ERROR_ON(lhs_info.k0 != rhs_info.k0);
73  ARM_COMPUTE_RETURN_ERROR_ON_MSG(((lhs_info.k0 & (lhs_info.k0 - 1)) && lhs_info.k0 != 3), "Only 2,3,4,8,16 are supported for k0");
74  ARM_COMPUTE_RETURN_ERROR_ON(lhs_info.k0 > 16);
75  ARM_COMPUTE_RETURN_ERROR_ON(lhs_info.m0 < 2 || lhs_info.m0 > 8);
76  ARM_COMPUTE_RETURN_ERROR_ON_MSG(((rhs_info.n0 & (rhs_info.n0 - 1)) && rhs_info.n0 != 3), "Only 2,3,4,8,16 are supported for n0");
77  ARM_COMPUTE_RETURN_ERROR_ON_MSG((gemm_info.reinterpret_input_as_3d || gemm_info.depth_output_gemm3d != 0) && (input2 != nullptr)
78  && (!gemm_info.broadcast_bias),
79  "Bias addition only supported with broadcast mode in case the input or output has to be reinterpreted as 3D");
80 
81  const unsigned int m = gemm_info.m;
82  const unsigned int n = gemm_info.n;
83  const unsigned int k = gemm_info.k;
84 
85  TensorShape tensor_shape0{ input0->tensor_shape() };
86  tensor_shape0.set(0, k);
87  tensor_shape0.set(1, m);
88 
89  TensorShape tensor_shape1{ input1->tensor_shape() };
90  tensor_shape1.set(0, n);
91  tensor_shape1.set(1, k);
92 
93  if(input2 != nullptr && !(helpers::float_ops::is_zero(beta)))
94  {
95  const unsigned int input2_dim0 = input2->dimension(0);
96  const unsigned int input2_dim1 = input2->dimension(1);
97 
98  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input2, input1);
99  if(gemm_info.broadcast_bias)
100  {
101  ARM_COMPUTE_RETURN_ERROR_ON_MSG((input2_dim1 != 1 || input2_dim0 != n), "Incorrect dimension of bias matrix which is to be broadcasted");
102  }
103  else
104  {
105  ARM_COMPUTE_RETURN_ERROR_ON_MSG((input2_dim0 != n || input2_dim1 != m), "Incorrect dimension of bias matrix");
106  }
107  }
108 
109  const TensorInfo tensor_info0 = input0->clone()->set_tensor_shape(tensor_shape0);
110  const TensorInfo tensor_info1 = input1->clone()->set_tensor_shape(tensor_shape1);
111 
112  const TensorInfo tensor_info_reshaped0 = input0->clone()->set_tensor_shape(compute_lhs_reshaped_shape(tensor_info0, lhs_info));
113  const TensorInfo tensor_info_reshaped1 = input1->clone()->set_tensor_shape(compute_rhs_reshaped_shape(tensor_info1, rhs_info));
114 
115  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_SHAPES(input0, &tensor_info_reshaped0);
116  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_SHAPES(input1, &tensor_info_reshaped1);
117 
118  if(output->total_size() != 0)
119  {
120  const TensorInfo tensor_info_output = output->clone()->set_tensor_shape(compute_mm_shape(*input0, *input1, gemm_info));
121  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_SHAPES(output, &tensor_info_output);
122  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input0, output);
123  }
124 
125  return Status{};
126 }

128 std::pair<Status, Window> validate_and_configure_window(ITensorInfo *input0, ITensorInfo *input1, ITensorInfo *input2, ITensorInfo *output, const GEMMLHSMatrixInfo &lhs_info,
129  const GEMMRHSMatrixInfo &rhs_info,
130  const GEMMKernelInfo &gemm_info, ElementsProcessed &num_elements_processed)
131 {
132  unsigned int &num_elems_processed_per_iteration_x = num_elements_processed[0];
133  unsigned int &num_elems_processed_per_iteration_y = num_elements_processed[1];
134  bool reinterpret_output_as_3d = gemm_info.depth_output_gemm3d != 0;
135 
136  Window win{};
137  Window win_out{};
138  bool window_changed = false;
139 
140  // Output tensor auto initialization if not yet initialized
141  auto_init_if_empty(*output, input0->clone()->set_tensor_shape(compute_mm_shape(*input0, *input1, gemm_info)));
142 
143  TensorInfo tmp_info(*output);
144 
145  if(reinterpret_output_as_3d)
146  {
147  // Since the output tensor has to be reinterpreted as 3D and the execute window is based on a 2D GEMM,
148  // the window needs to be constructed on the 2D collapsed version of the tensor
149  TensorShape tmp_shape(output->tensor_shape());
150  tmp_shape.collapse(2U, 1U);
151  tmp_info.set_tensor_shape(tmp_shape);
152  }
153 
154  // Configure kernel window
155  num_elems_processed_per_iteration_x = rhs_info.n0;
156  num_elems_processed_per_iteration_y = lhs_info.m0;
157 
158  // Note: bottom paddings are calculated manually as the output can be reinterpreted as 3D tensor
159  // The only way to set properly the paddings, it is to set those explicitly through the AccessWindowStatic
160  const unsigned int m = gemm_info.m;
161  const unsigned int bottom_pad = (num_elems_processed_per_iteration_y - (m % num_elems_processed_per_iteration_y)) % num_elems_processed_per_iteration_y;
162 
163  win = calculate_max_window(tmp_info, Steps(num_elems_processed_per_iteration_x, num_elems_processed_per_iteration_y));
164  win_out = calculate_max_window(*output, Steps(num_elems_processed_per_iteration_x, num_elems_processed_per_iteration_y));
165 
166  AccessWindowStatic input0_access(input0, 0, 0,
167  ceil_to_multiple(input0->dimension(0), num_elems_processed_per_iteration_y),
168  input0->dimension(1));
169  AccessWindowStatic input1_access(input1, 0, 0,
170  ceil_to_multiple(input1->dimension(0), num_elems_processed_per_iteration_x),
171  input1->dimension(1));
172  AccessWindowStatic output_access(output, 0, 0,
173  ceil_to_multiple(output->dimension(0), num_elems_processed_per_iteration_x),
174  output->dimension(1) + bottom_pad);
175 
176  if(input2 != nullptr)
177  {
178  const int bias_processed_per_iteration_x = num_elems_processed_per_iteration_x;
179 
180  const int bias_processed_per_iteration_y = gemm_info.broadcast_bias ? 1 : num_elems_processed_per_iteration_y;
181 
182  AccessWindowStatic input2_access(input2, 0, 0,
183  ceil_to_multiple(input2->dimension(0), bias_processed_per_iteration_x),
184  ceil_to_multiple(input2->dimension(1), bias_processed_per_iteration_y));
185 
186  window_changed = update_window_and_padding(win, input0_access, input1_access, input2_access) || // window used by the execute_window_loop
187  update_window_and_padding(win_out, output_access); // window used to update the padding requirements of output tensor
188  }
189  else
190  {
191  window_changed = update_window_and_padding(win, input0_access, input1_access) || // window used by the execute_window_loop
192  update_window_and_padding(win_out, output_access); // window used to update the padding requirements of output tensor
193  }
194 
195  output_access.set_valid_region(win_out, ValidRegion(Coordinates(0, 0), output->tensor_shape()));
196 
197  // Collapse along the Z direction
198  // This collapse needs to be here in order to tune the Z dimension of LWS
199  Window collapsed = win;
200  const unsigned int dimension_to_collapse = std::min(static_cast<unsigned int>(output->num_dimensions()), 2u);
201  collapsed = win.collapse(win, dimension_to_collapse);
202 
203  Status err = (window_changed) ? ARM_COMPUTE_CREATE_ERROR(ErrorCode::RUNTIME_ERROR, "Insufficient Padding!") : Status{};
204  return std::make_pair(err, collapsed);
205 }

214 void CLGEMMMatrixMultiplyReshapedKernel::configure(const ICLTensor *input0, const ICLTensor *input1, const ICLTensor *input2, ICLTensor *output, float alpha, float beta,

216  const GEMMRHSMatrixInfo &rhs_info, const GEMMKernelInfo &gemm_info)
217 {
218  ARM_COMPUTE_ERROR_ON_NULLPTR(input0, input1, output);
219 
220  ARM_COMPUTE_ERROR_THROW_ON(validate_arguments(input0->info(), input1->info(), (input2 != nullptr ? input2->info() : nullptr), output->info(), alpha, beta, lhs_info, rhs_info, gemm_info));
221 
222  _input0 = input0;
223  _input1 = input1;
224  _input2 = helpers::float_ops::is_zero(beta) ? nullptr : input2;
225  _output = output;
226  _reinterpret_output_as_3d = gemm_info.depth_output_gemm3d != 0;
227  _k = gemm_info.k;
228  _use_dummy_work_items = preferred_dummy_work_items_support(CLKernelLibrary::get().get_device());
229  _add_bias = _input2 != nullptr;
230  _broadcast_bias = gemm_info.broadcast_bias;
231 
232  // Check if we need to slide the matrix B
233  const unsigned int num_dimensions_input0 = _input0->info()->num_dimensions();
234  _slide_matrix_b = (_input1->info()->num_dimensions() >= num_dimensions_input0);
235 
236  ElementsProcessed num_elements_processed{};
237 
238  // Configure kernel window
239  auto win_config = validate_and_configure_window(input0->info(), input1->info(), input2 != nullptr ? input2->info() : nullptr, output->info(), lhs_info, rhs_info, gemm_info, num_elements_processed);
240  ARM_COMPUTE_ERROR_THROW_ON(win_config.first);
241  ICLKernel::configure_internal(win_config.second);
242 
243  // Create build options
244  CLBuildOptions build_opts;
245  build_opts.add_option("-DDATA_TYPE=" + get_cl_type_from_data_type(input0->info()->data_type()));
246  build_opts.add_option_if(!(helpers::float_ops::is_one(alpha)), "-DALPHA=" + float_to_string_with_full_precision(alpha));
247  build_opts.add_option_if(_input2 != nullptr, "-DBETA=" + float_to_string_with_full_precision(beta));

249  build_opts.add_option_if(_reinterpret_output_as_3d, "-DREINTERPRET_OUTPUT_AS_3D");
250  build_opts.add_option_if(_reinterpret_output_as_3d, "-DHEIGHT_GEMM3D=" + support::cpp11::to_string(output->info()->dimension(1)));
251  build_opts.add_option_if(_reinterpret_output_as_3d, "-DDEPTH_GEMM3D=" + support::cpp11::to_string(output->info()->dimension(2)));
252  build_opts.add_option_if(gemm_info.broadcast_bias, "-DBROADCAST_BIAS");
253  build_opts.add_option_if(!_slide_matrix_b, "-DMATRIX_B_DEPTH=" + support::cpp11::to_string(input1->info()->dimension(2)));
254  build_opts.add_option_if(lhs_info.interleave, "-DLHS_INTERLEAVE");
255  build_opts.add_option_if(rhs_info.interleave, "-DRHS_INTERLEAVE");
256  build_opts.add_option_if(_use_dummy_work_items, "-DDUMMY_WORK_ITEMS");
257  build_opts.add_option("-DM=" + support::cpp11::to_string(gemm_info.m));
258  build_opts.add_option("-DN=" + support::cpp11::to_string(gemm_info.n));
259  build_opts.add_option("-DM0=" + support::cpp11::to_string(lhs_info.m0));
260  build_opts.add_option("-DN0=" + support::cpp11::to_string(rhs_info.n0));
261  build_opts.add_option("-DK0=" + support::cpp11::to_string(lhs_info.k0));
262  build_opts.add_option("-DV0=" + support::cpp11::to_string(lhs_info.v0));
263  build_opts.add_option("-DH0=" + support::cpp11::to_string(rhs_info.h0));
264  build_opts.add_option_if(gemm_info.activation_info.enabled(), "-DACTIVATION_TYPE=" + lower_string(string_from_activation_func(gemm_info.activation_info.activation())));
265  build_opts.add_option_if(gemm_info.activation_info.enabled(), "-DA_VAL=" + float_to_string_with_full_precision(gemm_info.activation_info.a()));
266  build_opts.add_option_if(gemm_info.activation_info.enabled(), "-DB_VAL=" + float_to_string_with_full_precision(gemm_info.activation_info.b()));
267 
268  std::string kernel_name("gemm_mm_reshaped_");
269  kernel_name += lhs_info.transpose ? "lhs_t_" : "lhs_nt_";
270  kernel_name += rhs_info.transpose ? "rhs_t" : "rhs_nt";
271 
272  // Create kernel
273  _kernel = static_cast<cl::Kernel>(CLKernelLibrary::get().create_kernel(kernel_name, build_opts.options()));
274 
275  // Set config_id for enabling LWS tuning
276  _config_id = kernel_name;
277  _config_id += "_";
278  _config_id += (_add_bias ? "add_bias_" : "");
279  _config_id += (_broadcast_bias ? "broadcast_bias_" : "");
280  _config_id += (_reinterpret_output_as_3d ? "3do_" : "");
281  _config_id += (gemm_info.activation_info.enabled() ? "fused_activation_" : "");
282  _config_id += lower_string(string_from_data_type(input0->info()->data_type()));
283  _config_id += "_";
284  _config_id += support::cpp11::to_string(output->info()->dimension(1));
285  _config_id += "_";
286  _config_id += support::cpp11::to_string(output->info()->dimension(0));
287  _config_id += "_";
288  _config_id += support::cpp11::to_string(gemm_info.k);
289  _config_id += "_";

291  _config_id += "_";
292  _config_id += support::cpp11::to_string(lhs_info.m0);
293  _config_id += "_";

326 void CLGEMMMatrixMultiplyReshapedKernel::run(const Window &window, cl::CommandQueue &queue)
327 {
328  ARM_COMPUTE_ERROR_ON_UNCONFIGURED_KERNEL(this);
329  ARM_COMPUTE_ERROR_ON_INVALID_SUBWINDOW(ICLKernel::window(), window);
330 
331  if(_input1->info()->num_dimensions() < 3)
332  {
333  // The stride_z for matrix B must be zero if we do not slice
334  ARM_COMPUTE_ERROR_ON(_input1->info()->strides_in_bytes()[3] != 0);
335  }
336 
337  Window slice = window.first_slice_window_3D();
338  Window slice_matrix_b = slice;
339 
340  slice_matrix_b.set(Window::DimX, Window::Dimension(0, 1, 1));
341  slice_matrix_b.set(Window::DimY, Window::Dimension(0, 1, 1));
342 
343  if(_reinterpret_output_as_3d)
344  {
345  // Pass bottom paddings to the kernel if the output has to be reinterpreted as 3D tensor
346  unsigned int idx0;
347  if(_add_bias)
348  {
349  idx0 = 4 * num_arguments_per_2D_tensor() + 5;
350  }
351  else
352  {
353  idx0 = 3 * num_arguments_per_2D_tensor() + 4;
354  }
355  const unsigned int total_cross_plane_pad = _output->info()->padding().top + _output->info()->padding().bottom;
356  _kernel.setArg<cl_uint>(idx0, static_cast<unsigned int>(total_cross_plane_pad));
357  }
358 
359  do
360  {
361  Window slice_b = slice;
362  // Don't slice matrix B along the z dimension if matrix B has just 2 dimensions and matrix A more than 2
363  // This scenario can happen when the matrix multiplication is used to perform a convolution operation
364  if(!_slide_matrix_b)
365  {
366  slice_b = slice_matrix_b;
367  }
368 
369  unsigned int idx = 0;
370  add_2D_tensor_argument(idx, _input0, slice);
371  add_2D_tensor_argument(idx, _input1, slice_b);
372  add_2D_tensor_argument_if((_add_bias), idx, _input2, slice);
373  add_2D_tensor_argument(idx, _output, slice);
374  _kernel.setArg<cl_uint>(idx++, static_cast<unsigned int>(_k));
375  _kernel.setArg<cl_uint>(idx++, static_cast<unsigned int>(_input0->info()->strides_in_bytes()[2]));
376  _kernel.setArg<cl_uint>(idx++, static_cast<unsigned int>(_input1->info()->strides_in_bytes()[2]));
377  if(_add_bias)
378  {
379  _kernel.setArg<cl_uint>(idx++, static_cast<unsigned int>(_input2->info()->strides_in_bytes()[2]));
380  }
381  _kernel.setArg<cl_uint>(idx++, static_cast<unsigned int>(_output->info()->strides_in_bytes()[2]));
382  enqueue(queue, *this, slice, lws_hint(), _use_dummy_work_items);
383  }
384  while(window.slide_window_slice_3D(slice));
385 }
