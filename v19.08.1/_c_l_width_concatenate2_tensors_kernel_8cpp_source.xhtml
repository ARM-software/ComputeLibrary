47 constexpr unsigned int num_elems_processed_per_iteration = 8;

49 std::pair<Status, Window> validate_and_configure_window(ITensorInfo *input1, ITensorInfo *input2, ITensorInfo *output)
50 {
51  // The window needs to be based on the output
52  Window win = calculate_max_window(*output, Steps(num_elems_processed_per_iteration));
53  AccessWindowStatic input1_access(input1, 0, 0, ceil_to_multiple(input1->dimension(0), num_elems_processed_per_iteration), input1->dimension(1));
54  const unsigned int input2_right_padding = (output->dimension(0) / num_elems_processed_per_iteration) * num_elems_processed_per_iteration - input1->dimension(
55  0) + num_elems_processed_per_iteration - input2->dimension(0);
56  AccessWindowStatic input2_access(input2, -(input1->dimension(0) % num_elems_processed_per_iteration),
57  0, input2->dimension(0) + input2_right_padding, input2->dimension(1));
58  AccessWindowHorizontal output_access(output, 0, num_elems_processed_per_iteration);
59  bool window_changed = update_window_and_padding(win, input1_access, input2_access, output_access);
60 
61  Window win_collapsed = win.collapse(win, Window::DimZ);
62 
63  Status err = (window_changed) ? ARM_COMPUTE_CREATE_ERROR(ErrorCode::RUNTIME_ERROR, "Insufficient Padding!") : Status{};
64  return std::make_pair(err, win_collapsed);
65 }
66 Status validate_arguments(const ITensorInfo *input1, const ITensorInfo *input2, const ITensorInfo *output)
67 {
68  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(input1, input2, output);
69  ARM_COMPUTE_RETURN_ERROR_ON_F16_UNSUPPORTED(input1);
70  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(input1, 1, DataType::U8, DataType::S8, DataType::QASYMM8, DataType::U16, DataType::S16, DataType::F16, DataType::U32,
71  DataType::F32);
72  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input1, input2, output);
73  ARM_COMPUTE_RETURN_ERROR_ON(input1->dimension(0) + input2->dimension(0) > output->dimension(0));
74 
75  for(size_t i = 1; i < Coordinates::num_max_dimensions; ++i)
76  {
77  ARM_COMPUTE_RETURN_ERROR_ON(input1->dimension(i) != output->dimension(i));
78  ARM_COMPUTE_RETURN_ERROR_ON(input2->dimension(i) != output->dimension(i));
79  }
80  ARM_COMPUTE_RETURN_ERROR_ON(input1->num_dimensions() > 4);
81 
82  return Status{};
83 }

98 void CLWidthConcatenate2TensorsKernel::configure(const ICLTensor *input1, const ICLTensor *input2, ICLTensor *output)
99 {
100  ARM_COMPUTE_ERROR_ON_NULLPTR(input1, input2, output);
101  ARM_COMPUTE_ERROR_THROW_ON(validate_arguments(input1->info(), input2->info(), output->info()));
102 
103  _input1 = input1;
104  _input2 = input2;
105  _output = output;
106 
107  // Add build options
108  CLBuildOptions build_opts;
109  build_opts.add_option("-DDATA_TYPE=" + get_underlying_cl_type_from_data_type(input1->info()->data_type()));
110  build_opts.add_option("-DVEC_SIZE=" + support::cpp11::to_string(num_elems_processed_per_iteration));
111  build_opts.add_option("-DDEPTH=" + support::cpp11::to_string(input1->info()->dimension(2)));
112  build_opts.add_option("-DINPUT1_WIDTH=" + support::cpp11::to_string(input1->info()->dimension(0)));
113  build_opts.add_option("-DELEMENT_SIZE=" + support::cpp11::to_string(input1->info()->element_size()));
114 
115  // If input have different quantization info set quantization parameters needed for the re-quantization process
116  const bool have_different_qinfo = helpers::tensor_info::tensors_have_different_quantization_info(output->info(), input1->info(), input2->info());
117  if(is_data_type_quantized_asymmetric(input1->info()->data_type()) && have_different_qinfo)
118  {
119  const UniformQuantizationInfo iq1_info = input1->info()->quantization_info().uniform();
120  const UniformQuantizationInfo iq2_info = input2->info()->quantization_info().uniform();
121  const UniformQuantizationInfo oq_info = output->info()->quantization_info().uniform();
122 
123  build_opts.add_option("-DOFFSET_IN1=" + float_to_string_with_full_precision(iq1_info.offset));
124  build_opts.add_option("-DSCALE_IN1=" + float_to_string_with_full_precision(iq1_info.scale));
125  build_opts.add_option("-DOFFSET_IN2=" + float_to_string_with_full_precision(iq2_info.offset));
126  build_opts.add_option("-DSCALE_IN2=" + float_to_string_with_full_precision(iq2_info.scale));
127  build_opts.add_option("-DOFFSET_OUT=" + float_to_string_with_full_precision(oq_info.offset));
128  build_opts.add_option("-DSCALE_OUT=" + float_to_string_with_full_precision(oq_info.scale));
129  }
130 
131  // Create kernel
132  _kernel = static_cast<cl::Kernel>(CLKernelLibrary::get().create_kernel("concatenate_width_x2", build_opts.options()));
133 
134  // Configure kernel window
135  auto win_config = validate_and_configure_window(input1->info(), input2->info(), output->info());
136  ARM_COMPUTE_ERROR_THROW_ON(std::get<0>(win_config));
137 
138  ICLKernel::configure_internal(std::get<1>(win_config));
139 
140  // Set output valid region
141  output->info()->set_valid_region(ValidRegion(Coordinates(), output->info()->tensor_shape()));
142 
143  // Pass paddings as arguments to the kernel
144  const unsigned int input1_width = input1->info()->dimension(0);
145  const unsigned int input1_right_padding = ceil_to_multiple(input1_width, num_elems_processed_per_iteration) - input1_width;
146  const unsigned int input2_left_padding = input1_width % num_elems_processed_per_iteration;
147  unsigned int idx0 = 3 * num_arguments_per_4D_tensor();
148  _kernel.setArg<cl_uint>(idx0++, input1_right_padding);
149  _kernel.setArg<cl_uint>(idx0++, input2_left_padding);
150 
151  // Set config_id for enabling LWS tuning
152  _config_id = "concatenate_width_x2_";
153  _config_id += lower_string(string_from_data_type(input1->info()->data_type()));
154  _config_id += "_";
155  _config_id += support::cpp11::to_string(input1->info()->dimension(0));
156  _config_id += "_";
157  _config_id += support::cpp11::to_string(input1->info()->dimension(1));
158  _config_id += "_";
159  _config_id += support::cpp11::to_string(input2->info()->dimension(0));
160  _config_id += "_";
161  _config_id += support::cpp11::to_string(input2->info()->dimension(1));
162 }

164 void CLWidthConcatenate2TensorsKernel::run(const Window &window, cl::CommandQueue &queue)
165 {
166  ARM_COMPUTE_ERROR_ON_UNCONFIGURED_KERNEL(this);
167  ARM_COMPUTE_ERROR_ON_INVALID_SUBWINDOW(ICLKernel::window(), window);
168 
169  Window slice = window.first_slice_window_4D();
170 
171  do
172  {
173  unsigned int idx = 0;
174  add_4D_tensor_argument(idx, _input1, slice);
175  add_4D_tensor_argument(idx, _input2, slice);
176  add_4D_tensor_argument(idx, _output, slice);
177  enqueue(queue, *this, window, lws_hint());
178  }
179  while(window.slide_window_slice_4D(slice));
180 }
