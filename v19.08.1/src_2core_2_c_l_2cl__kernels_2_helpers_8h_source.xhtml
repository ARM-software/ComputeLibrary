294 inline Tensor4D update_tensor4D_workitem_ptr(__global uchar *ptr, uint offset_first_element_in_bytes, uint stride_x, uint step_x, uint stride_y, uint step_y, uint stride_z, uint step_z, uint stride_w,
295  uint step_w,
296  uint mod_size)
297 {
298  Tensor4D tensor =
299  {
300  .ptr = ptr,
301  .offset_first_element_in_bytes = offset_first_element_in_bytes,

303  .stride_y = stride_y,
304  .stride_z = stride_z,
305  .stride_w = stride_w
306  };
307 
308  tensor.ptr += tensor.offset_first_element_in_bytes + get_global_id(0) * step_x + get_global_id(1) * step_y + (get_global_id(2) % mod_size) * step_z + (get_global_id(2) / mod_size) * step_w;
309  return tensor;
310 }
