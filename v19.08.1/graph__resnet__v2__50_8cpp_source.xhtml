42  bool do_setup(int argc, char **argv) override
43  {
44  // Parse arguments
45  cmd_parser.parse(argc, argv);
46 
47  // Consume common parameters
48  common_params = consume_common_graph_parameters(common_opts);
49 
50  // Return when help menu is requested
51  if(common_params.help)
52  {
53  cmd_parser.print_help(argv[0]);
54  return false;
55  }
56 
57  // Checks
58  ARM_COMPUTE_EXIT_ON_MSG(arm_compute::is_data_type_quantized_asymmetric(common_params.data_type), "QASYMM8 not supported for this graph");
59 
60  // Print parameter values
61  std::cout << common_params << std::endl;
62 
63  // Get trainable parameters data path
64  std::string data_path = common_params.data_path;
65  std::string model_path = "/cnn_data/resnet_v2_50_model/";
66  if(!data_path.empty())
67  {
68  data_path += model_path;
69  }
70 
71  // Create a preprocessor object
72  std::unique_ptr<IPreprocessor> preprocessor = arm_compute::support::cpp14::make_unique<TFPreproccessor>();
73 
74  // Create input descriptor
75  const TensorShape tensor_shape = permute_shape(TensorShape(224U, 224U, 3U, 1U), DataLayout::NCHW, common_params.data_layout);
76  TensorDescriptor input_descriptor = TensorDescriptor(tensor_shape, common_params.data_type).set_layout(common_params.data_layout);
77 
78  // Set weights trained layout
79  const DataLayout weights_layout = DataLayout::NCHW;
80 
81  graph << common_params.target
82  << common_params.fast_math_hint
83  << InputLayer(input_descriptor, get_input_accessor(common_params, std::move(preprocessor), false /* Do not convert to BGR */))
84  << ConvolutionLayer(
85  7U, 7U, 64U,
86  get_weights_accessor(data_path, "conv1_weights.npy", weights_layout),
87  get_weights_accessor(data_path, "conv1_biases.npy", weights_layout),
88  PadStrideInfo(2, 2, 3, 3))
89  .set_name("conv1/convolution")
90  << PoolingLayer(PoolingLayerInfo(PoolingType::MAX, 3, PadStrideInfo(2, 2, 0, 1, 0, 1, DimensionRoundingType::FLOOR))).set_name("pool1/MaxPool");
91 
92  add_residual_block(data_path, "block1", weights_layout, 64, 3, 2);
93  add_residual_block(data_path, "block2", weights_layout, 128, 4, 2);
94  add_residual_block(data_path, "block3", weights_layout, 256, 6, 2);
95  add_residual_block(data_path, "block4", weights_layout, 512, 3, 1);
96 
97  graph << BatchNormalizationLayer(
98  get_weights_accessor(data_path, "postnorm_moving_mean.npy"),
99  get_weights_accessor(data_path, "postnorm_moving_variance.npy"),
100  get_weights_accessor(data_path, "postnorm_gamma.npy"),
101  get_weights_accessor(data_path, "postnorm_beta.npy"),
102  0.000009999999747378752f)
103  .set_name("postnorm/BatchNorm")
104  << ActivationLayer(ActivationLayerInfo(ActivationLayerInfo::ActivationFunction::RELU)).set_name("postnorm/Relu")
105  << PoolingLayer(PoolingLayerInfo(PoolingType::AVG)).set_name("pool5")
106  << ConvolutionLayer(
107  1U, 1U, 1001U,
108  get_weights_accessor(data_path, "logits_weights.npy", weights_layout),
109  get_weights_accessor(data_path, "logits_biases.npy"),
110  PadStrideInfo(1, 1, 0, 0))
111  .set_name("logits/convolution")
112  << FlattenLayer().set_name("predictions/Reshape")
113  << SoftmaxLayer().set_name("predictions/Softmax")
114  << OutputLayer(get_output_accessor(common_params, 5));
115 
116  // Finalize graph
117  GraphConfig config;
118  config.num_threads = common_params.threads;
119  config.use_tuner = common_params.enable_tuner;
120  config.tuner_mode = common_params.tuner_mode;
121  config.tuner_file = common_params.tuner_file;
122 
123  graph.finalize(common_params.target, config);
124 
125  return true;
126  }

140  void add_residual_block(const std::string &data_path, const std::string &name, DataLayout weights_layout,
141  unsigned int base_depth, unsigned int num_units, unsigned int stride)
142  {
143  for(unsigned int i = 0; i < num_units; ++i)
144  {
145  // Generate unit names
146  std::stringstream unit_path_ss;
147  unit_path_ss << name << "_unit_" << (i + 1) << "_bottleneck_v2_";
148  std::stringstream unit_name_ss;
149  unit_name_ss << name << "/unit" << (i + 1) << "/bottleneck_v2/";
150 
151  std::string unit_path = unit_path_ss.str();
152  std::string unit_name = unit_name_ss.str();
153 
154  const TensorShape last_shape = graph.graph().node(graph.tail_node())->output(0)->desc().shape;
155  unsigned int depth_in = last_shape[arm_compute::get_data_layout_dimension_index(common_params.data_layout, DataLayoutDimension::CHANNEL)];
156  unsigned int depth_out = base_depth * 4;
157 
158  // All units have stride 1 apart from last one
159  unsigned int middle_stride = (i == (num_units - 1)) ? stride : 1;
160 
161  // Preact
162  SubStream preact(graph);
163  preact << BatchNormalizationLayer(
164  get_weights_accessor(data_path, unit_path + "preact_moving_mean.npy"),
165  get_weights_accessor(data_path, unit_path + "preact_moving_variance.npy"),
166  get_weights_accessor(data_path, unit_path + "preact_gamma.npy"),
167  get_weights_accessor(data_path, unit_path + "preact_beta.npy"),
168  0.000009999999747378752f)
169  .set_name(unit_name + "preact/BatchNorm")
170  << ActivationLayer(ActivationLayerInfo(ActivationLayerInfo::ActivationFunction::RELU)).set_name(unit_name + "preact/Relu");
171 
172  // Create bottleneck path
173  SubStream shortcut(graph);
174  if(depth_in == depth_out)
175  {
176  if(middle_stride != 1)
177  {
178  shortcut << PoolingLayer(PoolingLayerInfo(PoolingType::MAX, 1, PadStrideInfo(middle_stride, middle_stride, 0, 0), true)).set_name(unit_name + "shortcut/MaxPool");
179  }
180  }
181  else
182  {
183  shortcut.forward_tail(preact.tail_node());
184  shortcut << ConvolutionLayer(
185  1U, 1U, depth_out,
186  get_weights_accessor(data_path, unit_path + "shortcut_weights.npy", weights_layout),
187  get_weights_accessor(data_path, unit_path + "shortcut_biases.npy", weights_layout),
188  PadStrideInfo(1, 1, 0, 0))
189  .set_name(unit_name + "shortcut/convolution");
190  }
191 
192  // Create residual path
193  SubStream residual(preact);
194  residual << ConvolutionLayer(
195  1U, 1U, base_depth,
196  get_weights_accessor(data_path, unit_path + "conv1_weights.npy", weights_layout),
197  std::unique_ptr<arm_compute::graph::ITensorAccessor>(nullptr),
198  PadStrideInfo(1, 1, 0, 0))
199  .set_name(unit_name + "conv1/convolution")
200  << BatchNormalizationLayer(
201  get_weights_accessor(data_path, unit_path + "conv1_BatchNorm_moving_mean.npy"),
202  get_weights_accessor(data_path, unit_path + "conv1_BatchNorm_moving_variance.npy"),
203  get_weights_accessor(data_path, unit_path + "conv1_BatchNorm_gamma.npy"),
204  get_weights_accessor(data_path, unit_path + "conv1_BatchNorm_beta.npy"),
205  0.000009999999747378752f)
206  .set_name(unit_name + "conv1/BatchNorm")
207  << ActivationLayer(ActivationLayerInfo(ActivationLayerInfo::ActivationFunction::RELU)).set_name(unit_name + "conv1/Relu")
208  << ConvolutionLayer(
209  3U, 3U, base_depth,
210  get_weights_accessor(data_path, unit_path + "conv2_weights.npy", weights_layout),
211  std::unique_ptr<arm_compute::graph::ITensorAccessor>(nullptr),
212  PadStrideInfo(middle_stride, middle_stride, 1, 1))
213  .set_name(unit_name + "conv2/convolution")
214  << BatchNormalizationLayer(
215  get_weights_accessor(data_path, unit_path + "conv2_BatchNorm_moving_mean.npy"),
216  get_weights_accessor(data_path, unit_path + "conv2_BatchNorm_moving_variance.npy"),
217  get_weights_accessor(data_path, unit_path + "conv2_BatchNorm_gamma.npy"),
218  get_weights_accessor(data_path, unit_path + "conv2_BatchNorm_beta.npy"),
219  0.000009999999747378752f)
220  .set_name(unit_name + "conv2/BatchNorm")
221  << ActivationLayer(ActivationLayerInfo(ActivationLayerInfo::ActivationFunction::RELU)).set_name(unit_name + "conv1/Relu")
222  << ConvolutionLayer(
223  1U, 1U, depth_out,
224  get_weights_accessor(data_path, unit_path + "conv3_weights.npy", weights_layout),
225  get_weights_accessor(data_path, unit_path + "conv3_biases.npy", weights_layout),
226  PadStrideInfo(1, 1, 0, 0))
227  .set_name(unit_name + "conv3/convolution");
228 
229  graph << EltwiseLayer(std::move(shortcut), std::move(residual), EltwiseOperation::Add).set_name(unit_name + "add");
230  }
231  }
232 };
