61 void GpuLogicalKernel::add_load_store()
62 {
63  const auto dst_tensors = _comp_group.get_dst_tensors();
64  // Each dst tensor from the component group requires exactly one store component
65  for(const auto &dst_tensor : dst_tensors)
66  {
67  ArgumentPack<ITensorInfo> tensors;
68  // Pass same destination tensor to both source and destination of the store component
69  // In other words, the addition of a store component does not create a new dst tensor
70  // This way we avoid the issue of the dst tensor of the component group differs from that of a logical kernel
71  // This may seem to violate the acyclic-ness of the component graph. But it is fine because at the point of
72  // the construction of the logical kernel, we do not need a graph representation of components anymore
73  // (the graph has been serialized)
74  tensors.add_const_tensor(ACL_SRC_0, dst_tensor);
75  tensors.add_const_tensor(ACL_DST_0, dst_tensor);
76 
77  auto store = _services->component_factory().create<ClComponentStore>(
78  _comp_group.get_root_component()->properties(), // Store component share the same properties as that of the root component
79  tensors);
80  _store_components.push_back(std::move(store));
81  auto success = _comp_group.add_component(_store_components.back().get());
82  ARM_COMPUTE_UNUSED(success);
83  ARM_COMPUTE_ERROR_ON(!success); // It's guaranteed that any load store insertion should be successful
84  }
