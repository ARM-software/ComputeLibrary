57 void CLFlattenLayer::configure(const CLCompileContext &compile_context, const ICLTensor *input, ICLTensor *output)
58 {
59  ARM_COMPUTE_ERROR_ON_NULLPTR(input, output);
60  _impl->src = input;
61  _impl->dst = output;
62  auto_init_if_empty(*output->info(), input->info()->clone()->set_tensor_shape(misc::shape_calculator::compute_flatten_shape(input->info())));
63 
64  _impl->op = std::make_unique<opencl::ClFlatten>();
65  _impl->op->configure(compile_context, _impl->src->info(), _impl->dst->info());
66 }

68 Status CLFlattenLayer::validate(const ITensorInfo *input, const ITensorInfo *output)
69 {
70  // Checks performed when output is configured
71  if(output->total_size() != 0)
72  {
73  const TensorInfo tensor_info_output = input->clone()->set_tensor_shape(misc::shape_calculator::compute_flatten_shape(input));
74  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_SHAPES(output, &tensor_info_output);
75  }
76  return opencl::ClFlatten::validate(input, output);
77 }
