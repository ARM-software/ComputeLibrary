47 inline std::tuple<float, float>
48 calculate_scale_factors(const ITensorInfo *src, const ITensorInfo *dst, DataLayout data_layout, bool align_corners)
49 {
50  const int idx_width = get_data_layout_dimension_index(data_layout, DataLayoutDimension::WIDTH);
51  const int idx_height = get_data_layout_dimension_index(data_layout, DataLayoutDimension::HEIGHT);
52 
53  // Compute the ratio between source width/height and destination width/height
54  const unsigned int src_width = src->dimension(idx_width);
55  const unsigned int src_height = src->dimension(idx_height);
56  const unsigned int dst_width = dst->dimension(idx_width);
57  const unsigned int dst_height = dst->dimension(idx_height);
58 
59  float scale_x = arm_compute::scale_utils::calculate_resize_ratio(src_width, dst_width, align_corners);
60  float scale_y = arm_compute::scale_utils::calculate_resize_ratio(src_height, dst_height, align_corners);
61 
62  return std::make_tuple(scale_x, scale_y);
63 }

65 Status validate_arguments(const ITensorInfo *src, const ITensorInfo *dst, const ScaleKernelInfo &info)
66 {
67  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(src, dst);
68  ARM_COMPUTE_RETURN_ERROR_ON_F16_UNSUPPORTED(src);
69  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(src, 1, DataType::QASYMM8, DataType::QASYMM8_SIGNED,
70  DataType::U8, DataType::S16, DataType::F16, DataType::F32);
71  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(src, dst);
72  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_QUANTIZATION_INFO(src, dst);
73  ARM_COMPUTE_RETURN_ERROR_ON(dst == src);
74  ARM_COMPUTE_RETURN_ERROR_ON(src->num_channels() != 1);
75  ARM_COMPUTE_RETURN_ERROR_ON(
76  info.align_corners &&
77  !arm_compute::scale_utils::is_align_corners_allowed_sampling_policy(info.sampling_policy));
78  ARM_COMPUTE_RETURN_ERROR_ON(is_data_type_quantized(src->data_type()) &&
79  !is_data_type_quantized_asymmetric(src->data_type()));
80 
81  float scale_x = 0.f;
82  float scale_y = 0.f;
83  const DataLayout data_layout = info.data_layout == DataLayout::UNKNOWN ? src->data_layout() : info.data_layout;
84  std::tie(scale_x, scale_y) = calculate_scale_factors(src, dst, data_layout, info.align_corners);
85 
86  ARM_COMPUTE_RETURN_ERROR_ON(info.interpolation_policy == InterpolationPolicy::AREA &&
87  (scale_x > 1.f || scale_y > 1.f));
88 
89  return Status{};
90 }

104 void ClScaleKernel::configure(const CLCompileContext &compile_context,
105  ITensorInfo *src,
106  ITensorInfo *dst,
107  const ScaleKernelInfo &info)
108 {
109  ARM_COMPUTE_ERROR_THROW_ON(validate_arguments(src, dst, info));
110  auto padding_info = get_padding_info({src, dst});
111 
112  // Info required for the static tuning
113  _data_layout = info.data_layout == DataLayout::UNKNOWN ? src->data_layout() : info.data_layout;
114 
115  const bool is_nhwc = _data_layout == DataLayout::NHWC;
116 
117  float scale_x = 0.f;
118  float scale_y = 0.f;
119  std::tie(scale_x, scale_y) = calculate_scale_factors(src, dst, _data_layout, info.align_corners);
120 
121  // Area interpolation behaves as Nearest Neighbour in case of up-sampling
122  auto interpolation_policy_to_use = info.interpolation_policy;
123  if (info.interpolation_policy == InterpolationPolicy::AREA && scale_x <= 1.f && scale_y <= 1.f)
124  {
125  interpolation_policy_to_use = InterpolationPolicy::NEAREST_NEIGHBOR;
126  }
127 
128  // Create kernel
129  const int idx_width = get_data_layout_dimension_index(_data_layout, DataLayoutDimension::WIDTH);
130  const int idx_height = get_data_layout_dimension_index(_data_layout, DataLayoutDimension::HEIGHT);
131  const int idx_channel = get_data_layout_dimension_index(_data_layout, DataLayoutDimension::CHANNEL);
132  const unsigned int src_width = src->dimension(idx_width);
133  const unsigned int src_height = src->dimension(idx_height);
134  const unsigned int dst_width = dst->dimension(idx_width);
135  const unsigned int dst_channels = dst->dimension(idx_channel);
136  unsigned int vec_size = 0;
137  unsigned int vec_size_leftover = 0;
138 
139  CLBuildOptions build_opts;
140  if (_data_layout == DataLayout::NHWC)
141  {
142  vec_size = adjust_vec_size(src->data_type() == DataType::F32 ? 4 : 8, dst_channels);
143  vec_size_leftover = dst_channels % vec_size;
144  build_opts.add_option("-DSRC_TENSOR_TYPE=BUFFER");
145  build_opts.add_option("-DSRC_DATA_TYPE=" + get_cl_type_from_data_type(src->data_type()));
146  build_opts.add_option("-DDST_TENSOR_TYPE=BUFFER");
147  build_opts.add_option("-DDST_DATA_TYPE=" + get_cl_type_from_data_type(dst->data_type()));
148  build_opts.add_option("-DCONSTANT_VALUE=" +
149  string_from_pixel_value(info.constant_border_value, src->data_type()));
150  build_opts.add_option("-DN0=" + support::cpp11::to_string(vec_size));
151  build_opts.add_option("-DPARTIAL_N0=" + support::cpp11::to_string(vec_size_leftover));
152  build_opts.add_option("-DSCALE_" + string_from_interpolation_policy(interpolation_policy_to_use));
153  build_opts.add_option_if(src->num_dimensions() > 3, "-DBATCHED_EXECUTION");
154  build_opts.add_option_if(info.border_mode == BorderMode::REPLICATE, "-DBORDER_MODE_REPLICATE");
155  build_opts.add_option_if(info.border_mode == BorderMode::CONSTANT, "-DBORDER_MODE_CONSTANT");
156  build_opts.add_option_if(info.align_corners, "-DALIGN_CORNERS");
157  build_opts.add_option_if(is_data_type_float(src->data_type()), "-DIS_FLOATING_POINT");
158  build_opts.add_option_if_else(info.sampling_policy == SamplingPolicy::CENTER, "-DSAMPLING_POLICY_CENTER",
159  "-DSAMPLING_POLICY_TOP_LEFT");
160  }
161  else if (_data_layout == DataLayout::NCHW)
162  {
163  vec_size = adjust_vec_size(4, dst_width);
164  vec_size_leftover = dst_width % vec_size;
165  build_opts.add_option("-DDATA_TYPE=" + get_cl_type_from_data_type(src->data_type()));
166  build_opts.add_option("-DCONSTANT_VALUE=" +
167  string_from_pixel_value(info.constant_border_value, src->data_type()));
168  build_opts.add_option("-DSRC_WIDTH=" + support::cpp11::to_string(src_width));
169  build_opts.add_option("-DSRC_HEIGHT=" + support::cpp11::to_string(src_height));
170  build_opts.add_option("-DSCALE_X=" + float_to_string_with_full_precision(scale_x));
171  build_opts.add_option("-DSCALE_Y=" + float_to_string_with_full_precision(scale_y));
172  build_opts.add_option("-DVEC_SIZE=" + support::cpp11::to_string(vec_size));
173  build_opts.add_option("-DVEC_SIZE_LEFTOVER=" + ((vec_size_leftover == 0)
174  ? support::cpp11::to_string(vec_size)
175  : support::cpp11::to_string(vec_size_leftover)));
176  build_opts.add_option_if(info.border_mode == BorderMode::REPLICATE, "-DBORDER_MODE_REPLICATE");
177  build_opts.add_option_if(info.border_mode == BorderMode::CONSTANT, "-DBORDER_MODE_CONSTANT");
178  build_opts.add_option_if(info.align_corners, "-DALIGN_CORNERS");
179  build_opts.add_option_if_else(info.sampling_policy == SamplingPolicy::CENTER, "-DSAMPLING_POLICY_CENTER",
180  "-DSAMPLING_POLICY_TOP_LEFT");
181 
182  const bool is_qasymm_bilinear = is_data_type_quantized_asymmetric(src->data_type()) &&
183  info.interpolation_policy == InterpolationPolicy::BILINEAR;
184  if (is_qasymm_bilinear)
185  {
186  const UniformQuantizationInfo qinfo = src->quantization_info().uniform();
187  build_opts.add_option("-DSCALE=" + support::cpp11::to_string(qinfo.scale));
188  build_opts.add_option("-DOFFSET=" + support::cpp11::to_string(qinfo.offset));
189  }
190  }
191  else
192  {
193  ARM_COMPUTE_ERROR_ON("Unsupported data layout");
194  }
195 
196  std::string interpolation_name = string_from_interpolation_policy(interpolation_policy_to_use);
197  std::transform(interpolation_name.begin(), interpolation_name.end(), interpolation_name.begin(), ::tolower);
198  std::string kernel_name = "scale_" + interpolation_name + "_";
199  kernel_name += lower_string(string_from_data_layout(_data_layout));
200 
201  _kernel = create_kernel(compile_context, kernel_name, build_opts.options());
202 
203  // Configure kernel window
204  Window win = calculate_max_window(*dst, Steps(vec_size));
205  ICLKernel::configure_internal(win);
206 
207  ARM_COMPUTE_ERROR_ON(has_padding_changed(padding_info));
208 
209  // Pass scale kernel arguments
210  if (is_nhwc)
211  {
212  unsigned int idx = 2 * num_arguments_per_4d_tensor_nhwc();
213  _kernel.setArg<cl_float>(idx++, scale_x);
214  _kernel.setArg<cl_float>(idx++, scale_y);
215  }
216  // Set config_id for enabling LWS tuning
217  _config_id = "scale_";
218  _config_id += (info.border_mode == BorderMode::REPLICATE ? "Bord_rep" : "");
219  _config_id += (info.sampling_policy == SamplingPolicy::CENTER ? "center" : "topleft");
220  _config_id += (is_nhwc ? "nhwc" : "nchw");
221  _config_id += "_";
222  _config_id += support::cpp11::to_string(dst->dimension(0));
223  _config_id += "_";
224  _config_id += support::cpp11::to_string(dst->dimension(1));
225  _config_id += "_";
226  _config_id += support::cpp11::to_string(dst->dimension(2));
227  _config_id += "_";
228  _config_id += support::cpp11::to_string(dst->dimension(3));
229 }

231 void ClScaleKernel::run_op(ITensorPack &tensors, const Window &window, cl::CommandQueue &queue)
232 {
233  ARM_COMPUTE_ERROR_ON_UNCONFIGURED_KERNEL(this);
234  ARM_COMPUTE_ERROR_ON_INVALID_SUBWINDOW(ICLKernel::window(), window);
235 
236  auto src = utils::cast::polymorphic_downcast<const ICLTensor *>(tensors.get_const_tensor(TensorType::ACL_SRC));
237  auto dst = utils::cast::polymorphic_downcast<ICLTensor *>(tensors.get_tensor(TensorType::ACL_DST));
238 
239  switch (_data_layout)
240  {
241  case DataLayout::NCHW:
242  {
243  Window slice = window.first_slice_window_2D();
244 
245  do
246  {
247  unsigned int idx = 0;
248  add_2D_tensor_argument(idx, src, slice);
249  add_2D_tensor_argument(idx, dst, slice);
250  enqueue(queue, *this, slice, lws_hint());
251  } while (window.slide_window_slice_2D(slice));
252  break;
253  }
254  case DataLayout::NHWC:
255  {
256  Window collapsed = window.collapse(ICLKernel::window(), Window::DimZ);
257  Window slice = collapsed.first_slice_window_4D();
258 
259  unsigned int idx = 0;
260  add_4d_tensor_nhwc_argument(idx, src);
261  add_4d_tensor_nhwc_argument(idx, dst);
262  enqueue(queue, *this, slice, lws_hint());
263  break;
264  }
265  default:
266  ARM_COMPUTE_ERROR("Data layout not supported");
267  }
268 }
