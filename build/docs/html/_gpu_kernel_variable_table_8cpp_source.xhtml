37 void GpuKernelVariableTable::declare_variable(const GpuKernelComponentGroup &comp_group,
38  const ITensorInfo *tensor,
39  GpuKernelArgumentInfo argument_info,
40  const std::string &alias)
41 {
42  ARM_COMPUTE_ERROR_ON_MSG(!tensor->has_valid_id(), "Tensor info with valid id expected");
43 
44  // Do not re-declare if the variable associated with the tensor has already been declared
45  auto it = _vars.find(tensor->id());
46 
47  if (it != _vars.end())
48  {
49  ARM_COMPUTE_ERROR_ON(!(it->second.kernel_argument_info == argument_info));
50  return;
51  }
52 
53  const auto target = comp_group.get_tile_for_tensor(tensor);
54 
55  if (target != tensor)
56  {
57  // If the tensor uses a shared tile, don't declare another variable.
58  it = _vars.find(target->id());
59 
60  ARM_COMPUTE_ERROR_ON_MSG(it == _vars.end(), "The variable used for this tensor must have been declared.");
61 
62  _vars[tensor->id()] = it->second;
63  }
64  else
65  {
66  // Declare variable associated with the tensor
67  std::stringstream ss;
68  ss << alias << "_t" << abs(tensor->id());
69  const auto uniq_name = ss.str();
70  TensorVariable var{tensor->id(), uniq_name, argument_info};
71 
72  _vars.emplace(tensor->id(), var);
73  }
74 }
