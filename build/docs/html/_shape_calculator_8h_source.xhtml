43 /** Calculate the output tensor shape for the reduce mean operation
44  *
45  * @param[in] input Input tensor shape
46  * @param[in] reduction_axis Reduction axis
47  * @param[in] keep_dims Flag to indicate if dimensions are kept
48  *
49  * @return the calculated shape
50  */
51 inline TensorShape calculate_reduce_mean_shape(ITensorInfo *input, const Coordinates &reduction_axis, bool keep_dims)
52 {
53  const int reduction_ops = reduction_axis.num_dimensions();
54  Coordinates axis_local = reduction_axis;
55  const int input_dims = input->num_dimensions();
56  convert_negative_axis(axis_local, input_dims);
57  TensorShape out_shape = input->tensor_shape();
58  // Configure reshape layer if we want to drop the dimensions
59  if (!keep_dims)
60  {
61  // We have to sort the reduction axis vectors in order for remove_dimension
62  // to work properly
63 
64 // Suppress warning produced by a compiler bug in GCC
65 // https://gcc.gnu.org/bugzilla/show_bug.cgi?id=104165
66 #pragma GCC diagnostic push
67 #pragma GCC diagnostic ignored "-Warray-bounds"
68  std::sort(axis_local.begin(), axis_local.begin() + reduction_ops);
69 #pragma GCC diagnostic pop
70 
71  for (int i = 0; i < reduction_ops; ++i)
72  {
73  out_shape.remove_dimension(axis_local[i] - i, false);
74  }
75  return out_shape;
76  }
77  else
78  {
79  for (int i = 0; i < reduction_ops; ++i)
80  {
81  out_shape.set(axis_local[i], 1);
82  }
83  return out_shape;
84  }
85 }

133 inline TensorShape compute_reorg_output_shape(const ITensorInfo &input, int32_t stride)

148  output_shape.set(idx_height, output_shape[idx_height] / stride);
149  output_shape.set(idx_channel, output_shape[idx_channel] * stride * stride);
150 
151  return output_shape;

154 /** Calculate the reshaped shape of the weights
155  *
156  * @param[in] weights Weights tensor info
157  * @param[in] has_bias (Optional) Set to true if there is bias
158  * @param[in] num_groups (Optional) Number of groups
159  *
160  * @return the calculated shape of the reshaped weights
161  */
162 inline TensorShape
163 compute_weights_reshaped_shape(const ITensorInfo &weights, bool has_bias = false, unsigned int num_groups = 1)
164 {
165  // Number of groups greater than one are only supported for NCHW data layout, and the number of weights must be a multiple of it.
166  ARM_COMPUTE_ERROR_ON(num_groups == 0);
167  ARM_COMPUTE_ERROR_ON(weights.data_layout() == DataLayout::NHWC && num_groups > 1);
168  ARM_COMPUTE_ERROR_ON((weights.dimension(3) % num_groups) != 0);
169 
170  // Calculate output shape
171  TensorShape weights_reshaped{weights.tensor_shape()};
172  weights_reshaped.set(3, weights_reshaped[3] / num_groups);
173 
174  weights_reshaped.collapse(3);
175  const size_t tmp_dim = weights_reshaped[0];
176  weights_reshaped.set(0, weights_reshaped[1]);
177  weights_reshaped.set(1, tmp_dim + (has_bias ? 1 : 0));
178  if (weights.num_dimensions() < 5)
179  {
180  weights_reshaped.set(2, num_groups);
181  }
182 
183  return weights_reshaped;
184 }

186 /** Calculate the Left Hand Side matrix reshaped shape
187  *
188  * @param[in] a Input tensor info
189  * @param[in] lhs_info Left Hand Side matrix information
190  * @param[in] reinterpret_input_as_3d (Optional) Set to true if the input need to be interpreted as 3d
191  *
192  * @return the calculated shape
193  */
194 inline TensorShape compute_lhs_reshaped_shape(const ITensorInfo &a,
195  const GEMMLHSMatrixInfo &lhs_info,
196  bool reinterpret_input_as_3d = false)
197 {
198  ARM_COMPUTE_ERROR_ON(lhs_info.m0 == 0);
199  ARM_COMPUTE_ERROR_ON(lhs_info.k0 == 0);
200  ARM_COMPUTE_ERROR_ON(lhs_info.v0 == 0);
201 
202  // Input width/height
203  const unsigned int input_width = a.dimension(0);
204  const unsigned int input_height = reinterpret_input_as_3d ? a.dimension(1) * a.dimension(2) : a.dimension(1);
205 
206  // Number of horizontal/vertical blocks in the input tensor
207  const unsigned int num_horiz_blocks = std::ceil(input_width / static_cast<float>(lhs_info.k0));
208  const unsigned int num_vert_blocks = std::ceil(input_height / static_cast<float>(lhs_info.m0));
209 
210  // Block size
211  const unsigned int block_size = lhs_info.m0 * lhs_info.k0;
212 
213  // Output width/height
214  const unsigned int output_width = block_size * num_horiz_blocks * lhs_info.v0;
215  const unsigned int output_height = std::ceil(num_vert_blocks / static_cast<float>(lhs_info.v0));
216 
217  TensorShape lhs_shape{a.tensor_shape()};
218  lhs_shape.set(0, output_width);
219  lhs_shape.set(1, output_height);
220 
221  if ((reinterpret_input_as_3d) && (lhs_shape.num_dimensions() > 2))
222  {
223  // When the data format is NHWC and the shapes are Nx1x1
224  // the tensor shape num_dimensions is automatically set to 1 instead of 3.
225  // To avoid failures by removing a dimension that doesn't exist
226  // check if the number of dimensions is greater than 2.
227  lhs_shape.remove_dimension(2);
228  }
229 
230  return lhs_shape;
231 }

268 /** Calculate the interleaved shape of an input tensor
269  *
270  * @param[in] a Input tensor info
271  * @param[in] mult_interleave4x4_height (Optional) Interleave4x4 height
272  * @param[in] reinterpret_input_as_3d (Optional) Set to true if the input need to be interpreted as 3d
273  *
274  * @return the calculated shape
275  */
276 inline TensorShape
277 compute_interleaved_shape(const ITensorInfo &a, int mult_interleave4x4_height = 1, bool reinterpret_input_as_3d = false)
278 {
279  // The interleaved output matrix will have the following shape: [ a_height * W, ceil(a_width / W) ] where W = 4 * mult_interleave4x4_height
280  ARM_COMPUTE_ERROR_ON(mult_interleave4x4_height < 1);
281  const int interleave_width = 4 * mult_interleave4x4_height;
282  TensorShape shape_interleaved_a{a.tensor_shape()};
283  shape_interleaved_a.set(0, a.dimension(0) * interleave_width);
284  if (reinterpret_input_as_3d)
285  {
286  const int M = a.dimension(1) * a.dimension(2);
287  const int height = std::ceil(M / static_cast<float>(interleave_width));
288  shape_interleaved_a.set(1, height);
289 
290  // When the data format is NHWC and the shapes are Nx1x1
291  // the tensor shape num_dimensions is automatically set to 1 instead of 3.
292  // To avoid failures by removing a dimension that doesn't exist
293  // check if the number of dimensions is greater than 2.
294  if (shape_interleaved_a.num_dimensions() > 2)
295  {
296  shape_interleaved_a.remove_dimension(2);
297  }
298  }
299  else
300  {
301  shape_interleaved_a.set(1, std::ceil(a.dimension(1) / static_cast<float>(interleave_width)));
302  }
303 
304  return shape_interleaved_a;
305 }

381  * @param[in] input Input tensor info

386  * @return the calculated shape

389  const Size2D &convolved_dims,
390  bool batch_size_on_z,
391  unsigned int num_groups = 1)
392 {
393  ARM_COMPUTE_ERROR_ON(num_groups == 0);
394  ARM_COMPUTE_ERROR_ON(input.tensor_shape()[1] != (convolved_dims.area()));
395  ARM_COMPUTE_ERROR_ON((num_groups > 1) && input.tensor_shape()[2] != num_groups);

410  col2im_shape.set(height_idx, convolved_dims.height);
411  col2im_shape.set(channel_idx, input.tensor_shape()[0] * num_groups);
412 
413  return col2im_shape;

469 /** Calculate padding required for deconvolution
470  *
471  * @param[in] input Input tensor info
472  * @param[in] weights Weights tensor shape
473  * @param[in] sx Stride on x axis
474  * @param[in] sy Stride on y axis
475  * @param[in] out_dims Output shape dimensions
476  *
477  * @return the padding required
478  */
479 inline std::pair<int32_t, int32_t> compute_deconvolution_padding(const ITensorInfo &input,
480  const ITensorInfo &weights,
481  int32_t sx,
482  int32_t sy,
483  std::pair<uint32_t, uint32_t> out_dims)
484 {
485  const DataLayout data_layout = input.data_layout();
486  const size_t idx_w = get_data_layout_dimension_index(data_layout, DataLayoutDimension::WIDTH);
487  const size_t idx_h = get_data_layout_dimension_index(data_layout, DataLayoutDimension::HEIGHT);
488 
489  // Find the upsampled dimensions
490  int32_t out_x = (static_cast<int32_t>(input.dimension(idx_w)) - 1) * sx + 1;
491  int32_t out_y = (static_cast<int32_t>(input.dimension(idx_h)) - 1) * sy + 1;
492 
493  // Find the padding needed for the convolution with stride 1 in order to match output shape
494  int32_t padx = out_dims.first - (out_x - static_cast<int32_t>(weights.dimension(idx_w)) + 1);
495  int32_t pady = out_dims.second - (out_y - static_cast<int32_t>(weights.dimension(idx_h)) + 1);
496 
497  return std::make_pair(padx, pady);
498 }

500 /** Calculate the upsampled output shape used for deconvolution
501  *
502  * @param[in] input Input tensor info
503  * @param[in] weights Weights tensor shape
504  * @param[in] sx Stride on x axis
505  * @param[in] sy Stride on y axis
506  * @param[in] out_dims Output shape dimensions
507  * @param[in] padx Padding on x axis
508  * @param[in] pady Padding on y axis
509  *
510  * @return the calculated shape
511  */
512 inline TensorShape compute_deconvolution_upsampled_shape(const ITensorInfo &input,
513  const ITensorInfo &weights,
514  unsigned int sx,
515  unsigned int sy,
516  std::pair<unsigned int, unsigned int> &out_dims,
517  uint32_t &padx,
518  uint32_t &pady)
519 {
520  // Find the padding needed for the convolution with stride 1 in order to match output shape
521  const auto padxy =
522  compute_deconvolution_padding(input, weights, static_cast<int32_t>(sx), static_cast<int32_t>(sy), out_dims);
523  padx = static_cast<uint32_t>(padxy.first);
524  pady = static_cast<uint32_t>(padxy.second);
525 
526  const DataLayout data_layout = input.data_layout();
527  const size_t idx_w = get_data_layout_dimension_index(data_layout, DataLayoutDimension::WIDTH);
528  const size_t idx_h = get_data_layout_dimension_index(data_layout, DataLayoutDimension::HEIGHT);
529 
530  // Find the upsampled dimensions
531  uint32_t out_x = (input.dimension(idx_w) - 1) * sx + 1;
532  uint32_t out_y = (input.dimension(idx_h) - 1) * sy + 1;
533 
534  out_x += padx;
535  out_y += pady;
536 
537  TensorShape scale_out_shape(input.tensor_shape());
538  scale_out_shape.set(idx_w, out_x);
539  scale_out_shape.set(idx_h, out_y);
540 
541  return scale_out_shape;
542 }

644  * @param[in] input Input tensor info
645  * @param[in] axis (Optional) Softmax axis
646  *
647  * @return the calculated shape
648  */

650 {
651  // The output shape will be a 2D version of the input. For instance:
652  // - [x,y,z] and axis 1 will return [x, y*z]

663  shape2D.collapse(shape2D.num_dimensions() - 1);

668  shape2D.collapse(shape2D.num_dimensions());

672  {

674  // collapse is an inclusive operation we need to shift

676  }
