203 /** Test case for memory injection in @ref cpu::CpuWinogradConv2d.
204  *
205  * Configure the operator once and inject memory at run-time in multiple executions.
206  *
207  * Checks performed in order:
208  * - Both runs compute the same output
209  */
210 TEST_CASE(MemoryInjection, framework::DatasetMode::ALL)
211 {
212  auto winograd = std::make_unique<cpu::CpuWinogradConv2d>();
213  const auto src_info = TensorInfo(TensorShape(8U, 8U, 32U), 1, DataType::F32);
214  const auto w_info = TensorInfo(TensorShape(1U), 1, DataType::F32);
215  const auto b_info = TensorInfo(TensorShape(1U, 3U, 32U, 1U), 1, DataType::F32);
216  auto dst_info = TensorInfo(TensorShape(8U, 6U, 1U), 1, DataType::F32);
217  const PadStrideInfo pad_info{};
218 
219  winograd->configure(&src_info, &b_info, &w_info, &dst_info, pad_info);
220 
221  // telhs are newly created every call of this lambda function
222  auto a = create_tensor<Tensor>(src_info);
223  auto b = create_tensor<Tensor>(b_info);
224  auto c = create_tensor<Tensor>(w_info);
225  a.allocator()->allocate();
226  b.allocator()->allocate();
227  c.allocator()->allocate();
228 
229  ITensorPack run_pack{ { TensorType::ACL_SRC_0, &a }, { TensorType::ACL_SRC_1, &b }, { TensorType::ACL_SRC_2, &c } };
230  ITensorPack prep_pack{ { TensorType::ACL_SRC_1, &b }, { TensorType::ACL_SRC_2, &c } };
231 
232  auto mg = MemoryGroup{};
233  auto ws = manage_workspace<Tensor>(winograd->workspace(), mg, run_pack, prep_pack);
234  auto run_conv = [&]() -> Tensor
235  {
236  auto dst = create_tensor<Tensor>(dst_info);
237  dst.allocator()->allocate();
238 
239  run_pack.add_tensor(TensorType::ACL_DST, &dst);
240  library->fill_tensor_value(Accessor(a), 1.f);
241  library->fill_tensor_value(Accessor(b), 2.f);
242  library->fill_tensor_value(Accessor(c), 3.f);
243 
244  // This operator is configured once and captured by this lambda.
245  winograd->prepare(prep_pack);
246  winograd->run(run_pack);
247  return dst;
248  };
249 
250  auto result_0 = run_conv();
251  auto result_1 = run_conv();
252 
253  for(size_t i = 0; i < result_0.info()->tensor_shape().total_size(); ++i)
254  {
255  ARM_COMPUTE_EXPECT(((float *)result_0.buffer())[i] == ((float *)result_1.buffer())[i], framework::LogLevel::ERRORS);
256  }
257 }
