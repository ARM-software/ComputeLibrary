37 void u8_neon_scale_nearest(const ITensor *src,
38  ITensor *dst,
39  const ITensor *offsets,
40  float sampling_offset,
41  bool align_corners,
42  const Window &window)
43 {
44  const size_t in_stride_c = src->info()->dimension(0) + src->info()->padding().left + src->info()->padding().right;
45  const size_t in_stride_w = src->info()->dimension(1) + src->info()->padding().top + src->info()->padding().bottom;
46  const size_t in_stride_wc = in_stride_w * in_stride_c;
47  const size_t in_dim_h = src->info()->dimension(2);
48 
49  // Compute the ratio between source height and destination height
50  const auto hr = scale_utils::calculate_resize_ratio(in_dim_h, dst->info()->dimension(2), align_corners);
51  const auto window_start_x = static_cast<int32_t>(window.x().start());
52  const auto window_end_x = static_cast<int32_t>(window.x().end());
53  const int window_step_x = 16;
54 
55  Window win(window);
56  win.set(Window::DimX, Window::Dimension(0, 1, 1));
57  Iterator out(dst, win);
58 
59  const uint8_t *in_ptr_start = src->buffer() + src->info()->offset_first_element_in_bytes();
60  const unsigned int in_stride_bytes_hwc = src->info()->strides_in_bytes()[3];
61 
62  execute_window_loop(
63  win,
64  [&](const Coordinates &id)
65  {
66  const int32_t offset =
67  *reinterpret_cast<const int32_t *>(offsets->ptr_to_element(Coordinates(id.y(), id.z()))) * in_stride_c;
68  const auto in_hi = static_cast<int>(
69  align_corners ? utils::rounding::round_half_away_from_zero((id.z() + sampling_offset) * hr)
70  : std::floor((id.z() + sampling_offset) * hr));
71  const int offset_row = in_hi * in_stride_wc;
72  int32_t x = window_start_x;
73  const uint8_t *in_ptr = reinterpret_cast<const uint8_t *>(in_ptr_start + in_stride_bytes_hwc * id[3]);
74 
75  for (; x <= window_end_x - window_step_x; x += window_step_x)
76  {
77  wrapper::vstore(reinterpret_cast<uint8_t *>(out.ptr()) + x,
78  wrapper::vloadq(in_ptr + offset + offset_row + x));
79  }
80  for (; x < window_end_x; ++x)
81  {
82  *(reinterpret_cast<uint8_t *>(out.ptr()) + x) = *(in_ptr + offset + offset_row + x);
83  }
84  },
85  out);
86 }

88 void u8_neon_scale_bilinear(const ITensor *src,
89  ITensor *dst,
90  const ITensor *offsets,
91  const ITensor *dx,
92  const ITensor *dy,
93  BorderMode border_mode,
94  PixelValue constant_border_value,
95  float sampling_offset,
96  bool align_corners,
97  const Window &window)
98 {
99  // Compute the ratio between source and destination dimensions
100  const float scale_x =
101  scale_utils::calculate_resize_ratio(src->info()->dimension(1), dst->info()->dimension(1), align_corners);
102  const float scale_y =
103  scale_utils::calculate_resize_ratio(src->info()->dimension(2), dst->info()->dimension(2), align_corners);
104 
105  const int input_width = src->info()->dimension(1);
106  const int input_height = src->info()->dimension(2);
107 
108  if (border_mode == BorderMode::CONSTANT)
109  {
110  Iterator out(dst, window);
111  const int in_stride_c = src->info()->dimension(0) + src->info()->padding().left + src->info()->padding().right;
112  const int in_stride_wc =
113  in_stride_c * (input_width + src->info()->padding().top + src->info()->padding().bottom);
114 
115  // Don't increment in Y and Z direction for the input tensor
116  // A pointer to the start of this plane is needed as base for the precomputed offsets
117  Window win_in(window);
118  win_in.set(Window::DimY, Window::Dimension(0, 0, 0));
119  win_in.set(Window::DimZ, Window::Dimension(0, 0, 0));
120  Iterator in(src, win_in);
121 
122  const uint8_t const_border_value = static_cast<uint8_t>(constant_border_value.get<uint8_t>());
123  execute_window_loop(
124  window,
125  [&](const Coordinates &id)
126  {
127  const auto offset =
128  *reinterpret_cast<const int32_t *>(offsets->ptr_to_element(Coordinates(id.y(), id.z())));
129  const auto dx_val = *reinterpret_cast<const float *>(dx->ptr_to_element(Coordinates(id.y(), id.z())));
130  const auto dy_val = *reinterpret_cast<const float *>(dy->ptr_to_element(Coordinates(id.y(), id.z())));
131  const int32_t in_hi = std::floor((id.z() + sampling_offset) * scale_y - sampling_offset);
132  const uint8_t *in_ptr =
133  reinterpret_cast<const uint8_t *>(in.ptr()) + offset * in_stride_c + in_hi * in_stride_wc;
134 
135  const auto a00 = (0 <= offset && offset < input_width && 0 <= in_hi && in_hi < input_height)
136  ? *in_ptr
137  : const_border_value;
138  const auto a01 = (-1 <= offset && offset < input_width - 1 && 0 <= in_hi && in_hi < input_height)
139  ? *(in_ptr + in_stride_c)
140  : const_border_value;
141  const auto a10 = (0 <= offset && offset < input_width && -1 <= in_hi && in_hi < input_height - 1)
142  ? *(in_ptr + in_stride_wc)
143  : const_border_value;
144  const auto a11 = (-1 <= offset && offset < input_width - 1 && -1 <= in_hi && in_hi < input_height - 1)
145  ? *(in_ptr + in_stride_c + in_stride_wc)
146  : const_border_value;
147 
148  *reinterpret_cast<uint8_t *>(out.ptr()) =
149  static_cast<uint8_t>(scale_helpers::delta_bilinear(a00, a01, a10, a11, dx_val, dy_val));
150  },
151  in, out);
152  }
153  else if (border_mode == BorderMode::REPLICATE)
154  {
155  using ExactTagType = typename wrapper::traits::neon_bitvector_tag_t<float, wrapper::traits::BitWidth::W128>;
156 
157  const int in_stride_x = src->info()->strides_in_bytes()[1];
158  const int in_stride_y = src->info()->strides_in_bytes()[2];
159  const int in_stride_b = src->info()->strides_in_bytes()[3];
160  const int out_stride_x = dst->info()->strides_in_bytes()[1];
161  const int out_stride_y = dst->info()->strides_in_bytes()[2];
162  const int out_stride_b = dst->info()->strides_in_bytes()[3];
163 
164  const int out_dim_ch = dst->info()->dimension(0);
165  constexpr int step_cout = 16;
166 
167  Window window_execution = window;
168  window_execution.set(Window::DimX, Window::Dimension(0, 1, 1));
169  Window win_in_out(window);
170  win_in_out.set(Window::DimY, Window::Dimension(0, 0, 0));
171  win_in_out.set(Window::DimZ, Window::Dimension(0, 0, 0));
172  Iterator in(src, win_in_out);
173  Iterator out(dst, win_in_out);
174 
175  const int xo_start = window_execution[1].start();
176  const int xo_end = window_execution[1].end();
177  const int xo_step = window_execution[1].step();
178  const int yo_start = window_execution[2].start();
179  const int yo_end = window_execution[2].end();
180  const int yo_step = window_execution[2].step();
181  const int bo_start = window_execution[3].start();
182  const int bo_end = window_execution[3].end();
183  const int bo_step = window_execution[3].step();
184 
185  const float fp_coord_offset_y = sampling_offset * (scale_y - 1);
186  const float fp_coord_offset_x = sampling_offset * (scale_x - 1);
187 
188  for (int bo = bo_start; bo < bo_end; bo += bo_step)
189  {
190  const uint8_t *in_ptr = in.ptr() + bo * in_stride_b;
191  uint8_t *out_ptr = out.ptr() + bo * out_stride_b;
192 
193  for (int yo = yo_start; yo < yo_end; yo += yo_step)
194  {
195  // Floating-point coordinate
196  const float yi_f = yo * scale_y + fp_coord_offset_y;
197  // Integer coordinate
198  const int yi = static_cast<int>(std::floor(yi_f));
199  // Weight for the y coordinate
200  const float a1 = (yi_f - static_cast<float>(yi));
201  const float b1 = (1.f - a1);
202 
203  const int yi0 = utility::clamp<int>(yi, 0, input_height - 1);
204  const int yi1 = utility::clamp<int>(yi + 1, 0, input_height - 1);
205 
206  const uint8_t *in_ptr_yi0 = in_ptr + yi0 * in_stride_y;
207  const uint8_t *in_ptr_yi1 = in_ptr + yi1 * in_stride_y;
208 
209  uint8_t *out_ptr_yo = out_ptr + yo * out_stride_y;
210  for (int xo = xo_start; xo < xo_end; xo += xo_step)
211  {
212  // Floating-point coordinate
213  const float xi_f = xo * scale_x + fp_coord_offset_x;
214  // Integer coordinate
215  const int xi = static_cast<int>(std::floor(xi_f));
216  // Weight for the x coordinate
217  const float a = (xi_f - static_cast<float>(xi));
218  const float b = (1.f - a);
219 
220  const float s00_s = b * b1;
221  const float s01_s = a * b1;
222  const float s10_s = b * a1;
223  const float s11_s = a * a1;
224 
225  const auto s00 = wrapper::vdup_n(s00_s, ExactTagType{});
226  const auto s01 = wrapper::vdup_n(s01_s, ExactTagType{});
227  const auto s10 = wrapper::vdup_n(s10_s, ExactTagType{});
228  const auto s11 = wrapper::vdup_n(s11_s, ExactTagType{});
229 
230  const int xi0 = utility::clamp<int>(xi, 0, input_width - 1);
231  const int xi1 = utility::clamp<int>(xi + 1, 0, input_width - 1);
232 
233  const auto in_ptr_xi0_yi0 = in_ptr_yi0 + xi0 * in_stride_x;
234  const auto in_ptr_xi1_yi0 = in_ptr_yi0 + xi1 * in_stride_x;
235  const auto in_ptr_xi0_yi1 = in_ptr_yi1 + xi0 * in_stride_x;
236  const auto in_ptr_xi1_yi1 = in_ptr_yi1 + xi1 * in_stride_x;
237 
238  uint8_t *out_ptr_xo_yo = out_ptr_yo + xo * out_stride_x;
239 
240  int cout = 0;
241  for (; cout <= (out_dim_ch - step_cout); cout += step_cout)
242  {
243  const auto in00 = wrapper::vloadq(in_ptr_xi0_yi0 + cout * sizeof(uint8_t));
244  const auto in01 = wrapper::vloadq(in_ptr_xi1_yi0 + cout * sizeof(uint8_t));
245  const auto in10 = wrapper::vloadq(in_ptr_xi0_yi1 + cout * sizeof(uint8_t));
246  const auto in11 = wrapper::vloadq(in_ptr_xi1_yi1 + cout * sizeof(uint8_t));
247 
248  const uint16x8_t in00_low = wrapper::vmovl(wrapper::vgetlow(in00));
249  const uint16x8_t in00_high = wrapper::vmovl(wrapper::vgethigh(in00));
250 
251  const auto in00_0 = wrapper::vcvt<float>(wrapper::vmovl(wrapper::vgetlow(in00_low)));
252  const auto in00_1 = wrapper::vcvt<float>(wrapper::vmovl(wrapper::vgethigh(in00_low)));
253  const auto in00_2 = wrapper::vcvt<float>(wrapper::vmovl(wrapper::vgetlow(in00_high)));
254  const auto in00_3 = wrapper::vcvt<float>(wrapper::vmovl(wrapper::vgethigh(in00_high)));
255 
256  const uint16x8_t in01_low = wrapper::vmovl(wrapper::vgetlow(in01));
257  const uint16x8_t in01_high = wrapper::vmovl(wrapper::vgethigh(in01));
258 
259  const auto in01_0 = wrapper::vcvt<float>(wrapper::vmovl(wrapper::vgetlow(in01_low)));
260  const auto in01_1 = wrapper::vcvt<float>(wrapper::vmovl(wrapper::vgethigh(in01_low)));
261  const auto in01_2 = wrapper::vcvt<float>(wrapper::vmovl(wrapper::vgetlow(in01_high)));
262  const auto in01_3 = wrapper::vcvt<float>(wrapper::vmovl(wrapper::vgethigh(in01_high)));
263 
264  const uint16x8_t in10_low = wrapper::vmovl(wrapper::vgetlow(in10));
265  const uint16x8_t in10_high = wrapper::vmovl(wrapper::vgethigh(in10));
266 
267  const auto in10_0 = wrapper::vcvt<float>(wrapper::vmovl(wrapper::vgetlow(in10_low)));
268  const auto in10_1 = wrapper::vcvt<float>(wrapper::vmovl(wrapper::vgethigh(in10_low)));
269  const auto in10_2 = wrapper::vcvt<float>(wrapper::vmovl(wrapper::vgetlow(in10_high)));
270  const auto in10_3 = wrapper::vcvt<float>(wrapper::vmovl(wrapper::vgethigh(in10_high)));
271 
272  const uint16x8_t in11_low = wrapper::vmovl(wrapper::vgetlow(in11));
273  const uint16x8_t in11_high = wrapper::vmovl(wrapper::vgethigh(in11));
274 
275  const auto in11_0 = wrapper::vcvt<float>(wrapper::vmovl(wrapper::vgetlow(in11_low)));
276  const auto in11_1 = wrapper::vcvt<float>(wrapper::vmovl(wrapper::vgethigh(in11_low)));
277  const auto in11_2 = wrapper::vcvt<float>(wrapper::vmovl(wrapper::vgetlow(in11_high)));
278  const auto in11_3 = wrapper::vcvt<float>(wrapper::vmovl(wrapper::vgethigh(in11_high)));
279 
280  auto out_0 = wrapper::vmul(in00_0, s00);
281  out_0 = wrapper::vmla(out_0, in01_0, s01);
282  out_0 = wrapper::vmla(out_0, in10_0, s10);
283  out_0 = wrapper::vmla(out_0, in11_0, s11);
284 
285  auto out_1 = wrapper::vmul(in00_1, s00);
286  out_1 = wrapper::vmla(out_1, in01_1, s01);
287  out_1 = wrapper::vmla(out_1, in10_1, s10);
288  out_1 = wrapper::vmla(out_1, in11_1, s11);
289 
290  auto out_2 = wrapper::vmul(in00_2, s00);
291  out_2 = wrapper::vmla(out_2, in01_2, s01);
292  out_2 = wrapper::vmla(out_2, in10_2, s10);
293  out_2 = wrapper::vmla(out_2, in11_2, s11);
294 
295  auto out_3 = wrapper::vmul(in00_3, s00);
296  out_3 = wrapper::vmla(out_3, in01_3, s01);
297  out_3 = wrapper::vmla(out_3, in10_3, s10);
298  out_3 = wrapper::vmla(out_3, in11_3, s11);
299 
300 #if defined(__aarch64__) && !defined(BARE_METAL)
301  const auto out_0_int = wrapper::vcvta<uint32_t>(out_0);
302  const auto out_1_int = wrapper::vcvta<uint32_t>(out_1);
303  const auto out_2_int = wrapper::vcvta<uint32_t>(out_2);
304  const auto out_3_int = wrapper::vcvta<uint32_t>(out_3);
305 #else // defined(__aarch64__) && !defined(BARE_METAL)
306  const auto out_0_int = wrapper::vcvt<uint32_t>(out_0);
307  const auto out_1_int = wrapper::vcvt<uint32_t>(out_1);
308  const auto out_2_int = wrapper::vcvt<uint32_t>(out_2);
309  const auto out_3_int = wrapper::vcvt<uint32_t>(out_3);
310 #endif // defined(__aarch64__) && !defined(BARE_METAL)
311  const auto low_part =
312  wrapper::vqmovn(wrapper::vcombine(wrapper::vqmovn(out_0_int), wrapper::vqmovn(out_1_int)));
313  const auto high_part =
314  wrapper::vqmovn(wrapper::vcombine(wrapper::vqmovn(out_2_int), wrapper::vqmovn(out_3_int)));
315  const auto out = wrapper::vcombine(low_part, high_part);
316 
317  wrapper::vstore(out_ptr_xo_yo + cout * sizeof(uint8_t), out);
318  }
319 
320  for (; cout < out_dim_ch; ++cout)
321  {
322  const uint8_t in00 = *(in_ptr_xi0_yi0 + cout * sizeof(uint8_t));
323  const uint8_t in01 = *(in_ptr_xi1_yi0 + cout * sizeof(uint8_t));
324  const uint8_t in10 = *(in_ptr_xi0_yi1 + cout * sizeof(uint8_t));
325  const uint8_t in11 = *(in_ptr_xi1_yi1 + cout * sizeof(uint8_t));
326 
327  float out0 = in00 * s00_s;
328  out0 += in01 * s01_s;
329  out0 += in10 * s10_s;
330  out0 += in11 * s11_s;
331 
332  // Rounding modes of vector and scalar loops should match
333 #if defined(__aarch64__) && !defined(BARE_METAL)
334  *(out_ptr_xo_yo + cout * sizeof(uint8_t)) = static_cast<uint8_t>(std::round(out0));
335 #else // defined(__aarch64__) && !defined(BARE_METAL)
336  *(out_ptr_xo_yo + cout * sizeof(uint8_t)) = static_cast<uint8_t>(out0);
337 #endif // defined(__aarch64__) && !defined(BARE_METAL)
338  }
339  }
340  }
341  }
342  }
343  else
344  {
345  ARM_COMPUTE_ERROR("Not implemented");
346  }
347 }

564 void s16_neon_scale_nearest(const ITensor *src,
565  ITensor *dst,
566  const ITensor *offsets,
567  float sampling_offset,
568  bool align_corners,
569  const Window &window)
570 {
571  const size_t in_stride_c = src->info()->dimension(0) + src->info()->padding().left + src->info()->padding().right;
572  const size_t in_stride_w = src->info()->dimension(1) + src->info()->padding().top + src->info()->padding().bottom;
573  const size_t in_stride_wc = in_stride_w * in_stride_c;
574  const size_t in_dim_h = src->info()->dimension(2);
575 
576  // Compute the ratio between source height and destination height
577  const auto hr = scale_utils::calculate_resize_ratio(in_dim_h, dst->info()->dimension(2), align_corners);
578  const auto window_start_x = static_cast<int32_t>(window.x().start());
579  const auto window_end_x = static_cast<int32_t>(window.x().end());
580  const int window_step_x = 8;
581 
582  Window win(window);
583  win.set(Window::DimX, Window::Dimension(0, 1, 1));
584  Iterator out(dst, win);
585 
586  const uint8_t *in_ptr_start = src->buffer() + src->info()->offset_first_element_in_bytes();
587  const unsigned int in_stride_bytes_hwc = src->info()->strides_in_bytes()[3];
588 
589  execute_window_loop(
590  win,
591  [&](const Coordinates &id)
592  {
593  const int32_t offset =
594  *reinterpret_cast<const int32_t *>(offsets->ptr_to_element(Coordinates(id.y(), id.z()))) * in_stride_c;
595  const auto in_hi = static_cast<int>(
596  align_corners ? utils::rounding::round_half_away_from_zero((id.z() + sampling_offset) * hr)
597  : std::floor((id.z() + sampling_offset) * hr));
598  const int offset_row = in_hi * in_stride_wc;
599  int32_t x = window_start_x;
600  const int16_t *in_ptr = reinterpret_cast<const int16_t *>(in_ptr_start + in_stride_bytes_hwc * id[3]);
601 
602  for (; x <= window_end_x - window_step_x; x += window_step_x)
603  {
604  wrapper::vstore(reinterpret_cast<int16_t *>(out.ptr()) + x,
605  wrapper::vloadq(in_ptr + offset + offset_row + x));
606  }
607  for (; x < window_end_x; ++x)
608  {
609  *(reinterpret_cast<int16_t *>(out.ptr()) + x) = *(in_ptr + offset + offset_row + x);
610  }
611  },
612  out);
613 }

615 void s16_neon_scale_bilinear(const ITensor *src,
616  ITensor *dst,
617  const ITensor *offsets,
618  const ITensor *dx,
619  const ITensor *dy,
620  BorderMode border_mode,
621  PixelValue constant_border_value,
622  float sampling_offset,
623  bool align_corners,
624  const Window &window)
625 {
626  // Compute the ratio between source height and destination height
627  const auto hr =
628  scale_utils::calculate_resize_ratio(src->info()->dimension(2), dst->info()->dimension(2), align_corners);
629 
630  Iterator out(dst, window);
631  const int in_stride_c = src->info()->dimension(0) + src->info()->padding().left + src->info()->padding().right;
632  const int in_dim_w = src->info()->dimension(1);
633  const int in_dim_h = src->info()->dimension(2);
634  const int in_stride_wc = in_stride_c * (in_dim_w + src->info()->padding().top + src->info()->padding().bottom);
635 
636  // Don't increment in Y and Z direction for the input tensor
637  // A pointer to the start of this plane is needed as base for the precomputed offsets
638  Window win_in(window);
639  win_in.set(Window::DimY, Window::Dimension(0, 0, 0));
640  win_in.set(Window::DimZ, Window::Dimension(0, 0, 0));
641  Iterator in(src, win_in);
642 
643  if (border_mode == BorderMode::CONSTANT)
644  {
645  const int16_t const_border_value = static_cast<int16_t>(constant_border_value.get<int16_t>());
646  execute_window_loop(
647  window,
648  [&](const Coordinates &id)
649  {
650  const auto offset =
651  *reinterpret_cast<const int32_t *>(offsets->ptr_to_element(Coordinates(id.y(), id.z())));
652  const auto dx_val = *reinterpret_cast<const float *>(dx->ptr_to_element(Coordinates(id.y(), id.z())));
653  const auto dy_val = *reinterpret_cast<const float *>(dy->ptr_to_element(Coordinates(id.y(), id.z())));
654  const int32_t in_hi = std::floor((id.z() + sampling_offset) * hr - sampling_offset);
655  const int16_t *in_ptr =
656  reinterpret_cast<const int16_t *>(in.ptr()) + offset * in_stride_c + in_hi * in_stride_wc;
657 
658  const auto a00 =
659  (0 <= offset && offset < in_dim_w && 0 <= in_hi && in_hi < in_dim_h) ? *in_ptr : const_border_value;
660  const auto a01 = (-1 <= offset && offset < in_dim_w - 1 && 0 <= in_hi && in_hi < in_dim_h)
661  ? *(in_ptr + in_stride_c)
662  : const_border_value;
663  const auto a10 = (0 <= offset && offset < in_dim_w && -1 <= in_hi && in_hi < in_dim_h - 1)
664  ? *(in_ptr + in_stride_wc)
665  : const_border_value;
666  const auto a11 = (-1 <= offset && offset < in_dim_w - 1 && -1 <= in_hi && in_hi < in_dim_h - 1)
667  ? *(in_ptr + in_stride_c + in_stride_wc)
668  : const_border_value;
669 
670  *reinterpret_cast<int16_t *>(out.ptr()) =
671  static_cast<int16_t>(scale_helpers::delta_bilinear(a00, a01, a10, a11, dx_val, dy_val));
672  },
673  in, out);
674  }
675  else if (border_mode == BorderMode::REPLICATE)
676  {
677  execute_window_loop(
678  window,
679  [&](const Coordinates &id)
680  {
681  const auto offset =
682  *reinterpret_cast<const int32_t *>(offsets->ptr_to_element(Coordinates(id.y(), id.z())));
683  const auto dx_val = *reinterpret_cast<const float *>(dx->ptr_to_element(Coordinates(id.y(), id.z())));
684  const auto dy_val = *reinterpret_cast<const float *>(dy->ptr_to_element(Coordinates(id.y(), id.z())));
685  const int in_hi = std::floor((id.z() + sampling_offset) * hr - sampling_offset);
686 
687  const auto clamped_w = utility::clamp<int>(offset, 0, in_dim_w - 1);
688  const auto clamped_w1 = utility::clamp<int>(offset + 1, 0, in_dim_w - 1);
689  const auto clamped_h = utility::clamp<int>(in_hi, 0, in_dim_h - 1);
690  const auto clamped_h1 = utility::clamp<int>(in_hi + 1, 0, in_dim_h - 1);
691 
692  const auto a00 =
693  *(reinterpret_cast<const int16_t *>(in.ptr()) + clamped_w * in_stride_c + clamped_h * in_stride_wc);
694  const auto a01 = *(reinterpret_cast<const int16_t *>(in.ptr()) + clamped_w1 * in_stride_c +
695  clamped_h * in_stride_wc);
696  const auto a10 = *(reinterpret_cast<const int16_t *>(in.ptr()) + clamped_w * in_stride_c +
697  clamped_h1 * in_stride_wc);
698  const auto a11 = *(reinterpret_cast<const int16_t *>(in.ptr()) + clamped_w1 * in_stride_c +
699  clamped_h1 * in_stride_wc);
700 
701  *reinterpret_cast<int16_t *>(out.ptr()) =
702  static_cast<int16_t>(scale_helpers::delta_bilinear(a00, a01, a10, a11, dx_val, dy_val));
703  },
704  in, out);
705  }
706  else
707  {
708  ARM_COMPUTE_ERROR("Not implemented");
709  }
