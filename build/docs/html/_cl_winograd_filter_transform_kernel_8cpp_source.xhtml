53 Status validate_arguments(const ITensorInfo *input, const ITensorInfo *output, const WinogradInfo &winograd_info)
54 {
55  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(input, 1, DataType::F32, DataType::F16);
56  ARM_COMPUTE_RETURN_ERROR_ON_F16_UNSUPPORTED(input);
57 
58  const Size2D kernel_size = winograd_info.kernel_size;
59  const Size2D output_tile_size = winograd_info.output_tile_size;
60 
61  const size_t idx_w = get_data_layout_dimension_index(input->data_layout(), DataLayoutDimension::WIDTH);
62  const size_t idx_h = get_data_layout_dimension_index(input->data_layout(), DataLayoutDimension::HEIGHT);
63 
64  ARM_COMPUTE_RETURN_ERROR_ON_MSG(
65  !cl_winograd_convolution_layer_supported(output_tile_size, kernel_size, input->data_layout()),
66  "Winograd filter transform not supported");
67  ARM_COMPUTE_RETURN_ERROR_ON(input->dimension(idx_w) != kernel_size.width ||
68  input->dimension(idx_h) != kernel_size.height);
69  ARM_COMPUTE_RETURN_ERROR_ON(input->num_dimensions() > 4);
70 
71  // Checks performed when output is configured
72  if (output->total_size() != 0)
73  {
74  const TensorInfo tensor_info_output =
75  input->clone()->set_tensor_shape(compute_winograd_filter_transform_shape(*input, winograd_info));
76 
77  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_SHAPES(output, &tensor_info_output);
78  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input, output);
79  }
80 
81  return Status{};
82 }

84 std::pair<Status, Window> validate_and_configure_window(ITensorInfo *input, ITensorInfo *output)
85 {
86  ARM_COMPUTE_ERROR_ON_NULLPTR(input, output);
87  ARM_COMPUTE_UNUSED(output);
88 
89  const unsigned int num_elems_processed_per_iteration_x =
90  input->data_layout() == DataLayout::NCHW ? input->dimension(0) : 1;
91  const unsigned int num_elems_processed_per_iteration_y = input->dimension(1);
92  const unsigned int num_elems_read_per_iteration_z =
93  input->data_layout() == DataLayout::NCHW ? 1 : input->dimension(2);
94 
95  Window win =
96  calculate_max_window(*input, Steps(num_elems_processed_per_iteration_x, num_elems_processed_per_iteration_y,
97  num_elems_read_per_iteration_z));
98  Window win_collapsed = win.collapse(win, Window::DimZ);
99  return std::make_pair(Status{}, win_collapsed);
100 }

108 void ClWinogradFilterTransformKernel::configure(const ClCompileContext &compile_context,
109  ITensorInfo *src,
110  ITensorInfo *dst,
111  const WinogradInfo &winograd_info)
112 {
113  ARM_COMPUTE_ERROR_ON_NULLPTR(src, dst);
114 
115  // Output auto initialization if not yet initialized
116  auto_init_if_empty(*dst,
117  src->clone()->set_tensor_shape(compute_winograd_filter_transform_shape(*src, winograd_info)));
118 
119  ARM_COMPUTE_ERROR_THROW_ON(validate_arguments(src, dst, winograd_info));
120  auto padding_info = get_padding_info({src, dst});
121 
122  // Set build options
123  CLBuildOptions build_opts;
124 
125  // For NHWC layouts pass tensor dimesions at runtime
126  if (src->data_layout() == DataLayout::NHWC)
127  {
128  _src_dim_z = src->dimension(2);
129  }
130  else
131  {
132  build_opts.add_option("-DSRC_DIM_Z=" + support::cpp11::to_string(src->dimension(2)));
133  }
134  build_opts.add_option("-DDATA_TYPE=" + get_cl_type_from_data_type(src->data_type()));
135  build_opts.add_option_if(winograd_info.kernel_size.height == 1, "-DWINOGRAD_FILTER_TRANSFORM_HORIZONTAL");
136  build_opts.add_option_if(winograd_info.kernel_size.width == 1, "-DWINOGRAD_FILTER_TRANSFORM_VERTICAL");
137  const Size2D kernel_size = winograd_info.kernel_size;
138  const Size2D output_tile_size = winograd_info.output_tile_size;
139 
140  // Create kernel
141  std::string kernel_name = "winograd_filter_transform_" + output_tile_size.to_string() + "_" +
142  kernel_size.to_string() + "_" + lower_string(string_from_data_layout(src->data_layout()));
143 
144  // A macro guard to compile ONLY the kernel of interest
145  build_opts.add_option("-D" + upper_string(kernel_name));
146  _kernel = create_kernel(compile_context, kernel_name, build_opts.options());
147 
148  // Configure kernel window
149  auto win_config = validate_and_configure_window(src, dst);
150  ARM_COMPUTE_ERROR_THROW_ON(win_config.first);
151  IClKernel::configure_internal(win_config.second);
152  ARM_COMPUTE_ERROR_ON(has_padding_changed(padding_info));
153 }

156  const ITensorInfo *dst,

165 void ClWinogradFilterTransformKernel::run_op(ITensorPack &tensors, const Window &window, cl::CommandQueue &queue)
166 {
167  ARM_COMPUTE_ERROR_ON_UNCONFIGURED_KERNEL(this);
168  ARM_COMPUTE_ERROR_ON_INVALID_SUBWINDOW(IClKernel::window(), window);
169 
170  auto src = utils::cast::polymorphic_downcast<const ICLTensor *>(tensors.get_const_tensor(TensorType::ACL_SRC));
171  auto dst = utils::cast::polymorphic_downcast<ICLTensor *>(tensors.get_tensor(TensorType::ACL_DST));
172 
173  // Setup output window
174  Window window_out;
175  window_out.use_tensor_dimensions(dst->info()->tensor_shape(), 0);
176 
177  unsigned int idx = 0;
178  add_4D_tensor_argument(idx, src, window);
179  add_3D_tensor_argument(idx, dst, window_out);
180  if (src->info()->data_layout() == DataLayout::NHWC)
181  {
182  _kernel.setArg<cl_uint>(idx++, _src_dim_z);
183  }
184  enqueue(queue, *this, window, lws_hint());
185 }
