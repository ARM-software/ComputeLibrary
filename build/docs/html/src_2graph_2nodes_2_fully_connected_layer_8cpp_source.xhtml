60 TensorDescriptor FullyConnectedLayerNode::compute_weights_descriptor(const TensorDescriptor &input_descriptor,
61  unsigned int num_outputs,
62  FullyConnectedLayerInfo fc_info,
63  const QuantizationInfo &weights_quant_info)
64 {
65  unsigned int num_weights = 1;
66  unsigned int num_dimensions = input_descriptor.shape.num_dimensions();
67  // Ignore the batch dimension if there is one:
68  if (num_dimensions == 2 || num_dimensions == 4)
69  {
70  num_dimensions--;
71  }
72  for (unsigned int i = 0; i < num_dimensions; i++)
73  {
74  num_weights *= input_descriptor.shape[i];
75  }
76 
77  TensorDescriptor weights_descriptor = input_descriptor;
78  weights_descriptor.shape = TensorShape(num_weights, num_outputs);
79 
80  // If weights are tranposed, use tranposed shape
81  if (!fc_info.transpose_weights)
82  {
83  weights_descriptor.shape = TensorShape(num_outputs, num_weights);
84  }
85 
86  // Set quantization info if present
87  if (!weights_quant_info.empty())
88  {
89  weights_descriptor.quant_info = weights_quant_info;
90  }
91 
92  return weights_descriptor;
93 }

95 TensorDescriptor FullyConnectedLayerNode::compute_output_descriptor(const TensorDescriptor &input_descriptor,
96  unsigned int num_outputs,
97  const QuantizationInfo &out_quant_info)
98 {
99  // Note: Only 1D batch space is supported at the moment
100  unsigned int batches = input_descriptor.shape[1];
101  if (input_descriptor.shape.num_dimensions() > 2)
102  {
103  batches = input_descriptor.shape[3];
104  }
105 
106  // Set descriptor shape
107  TensorDescriptor output_descriptor = input_descriptor;
108  output_descriptor.shape = TensorShape(num_outputs, batches);
109 
110  // Set quantization info if present
111  if (!out_quant_info.empty())
112  {
113  output_descriptor.quant_info = out_quant_info;
114  }
115 
116  return output_descriptor;
117 }
