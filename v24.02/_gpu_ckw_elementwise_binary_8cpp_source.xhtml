65 void GpuCkwElementwiseBinary::write_component_code(const ComponentGroup &comp_group,
66  GpuCkwVariableTable &vtable,
67  GpuCkwScopedKernelWriter writer) const
68 {
69  /********************************************************************************
70  * 1 - Define tensors
71  ********************************************************************************/
72  GpuCkwComponentArgument *lhs = vtable.declare_variable(comp_group, writer, _lhs, "lhs");
73  GpuCkwComponentArgument *rhs = vtable.declare_variable(comp_group, writer, _rhs, "rhs");
74  GpuCkwComponentArgument *dst = vtable.declare_variable(comp_group, writer, _dst, "dst");
75 
76  /********************************************************************************
77  * 2 - Define CKW constants
78  ********************************************************************************/
79  const auto dst_h = static_cast<int32_t>(_dst->dimension(1));
80 
81  // CKW constants
82  auto const_dst_h_i32 = writer->declare_constant_tile(ckw::ConstantData({{dst_h}}, ckw::DataType::Int32));
83  auto const_pos_1_i32 = writer->declare_constant_tile(ckw::ConstantData({{1}}, ckw::DataType::Int32));
84  auto const_0_i32 = writer->declare_constant_tile(ckw::ConstantData({{0}}, ckw::DataType::Int32));
85 
86  /********************************************************************************
87  * 3 - Define the compute block parameters and destination tile (if not root component)
88  * Bind the tile to the tensor to share it among different components and
89  * initialize the compute block parameters
90  ********************************************************************************/
91  // The compute block parameters depend on the employed tensor format
92 
93  // Destination compute block size
94  int32_t dst_n0 = -1;
95  int32_t dst_m0 = -1;
96 
97  // Destination compute block size left-over
98  int32_t dst_n0_partial = -1;
99  int32_t dst_m0_partial = -1;
100 
101  if (!dst->has_tile())
102  {
103  // If ROOT component, we use ckw::TensorSamplerFormat::Dim0_Dim1xDim2_1
104  // as tensor format
105  const auto root_window = comp_group.get_root_component()->ckw_component_driver()->get_window();
106 
107  dst_n0 = root_window.x().step();
108  dst_m0 = root_window.y().step();
109  dst_n0_partial = _dst->dimension(0) % dst_n0;
110  dst_m0_partial = (_dst->dimension(1) * _dst->dimension(2)) % dst_m0;
111 
112  ckw::TensorSampler sampler_dst;
113  sampler_dst.format(ckw::TensorSamplerFormat::Dim0_Dim1xDim2_1);
114  if (dst_n0_partial == 0)
115  {
116  sampler_dst.address_mode_x(ckw::TensorSamplerAddressModeX::None);
117  }
118  else
119  {
120  sampler_dst.address_mode_x(ckw::TensorSamplerAddressModeX::OverlappingMin);
121  }
122 
123  if (dst_m0_partial == 0)
124  {
125  sampler_dst.address_mode_y(ckw::TensorSamplerAddressModeY::None);
126  }
127  else
128  {
129  sampler_dst.address_mode_y(ckw::TensorSamplerAddressModeY::ClampToBorderMaxOnly);
130  }
131  sampler_dst.address_mode_z(ckw::TensorSamplerAddressModeZ::None);
132  sampler_dst.storage(ckw::TensorStorageType::BufferUint8Ptr);
133 
134  // Declare destination tile
135  ckw::DataType dst_dt = to_ckw(_dst->data_type());
136  auto tile_dst = writer->declare_tile("dst", ckw::TileInfo(dst_dt, dst_m0, dst_n0));
137 
138  // Bind tile to the tensor
139  dst->init_virtual_tensor(tile_dst, sampler_dst);
140  }
141  else
142  {
143  // Change dst_n0 and dst_m0 if NOT root component!
144  dst_n0 = dst->tile().tile_info().width();
145  dst_m0 = dst->tile().tile_info().height();
146 
147  // Here, it is not required the calculation of dst_n0_partial and dst_m0_partial
148  // because if we enter this condition it means that the element-wise op is not the
149  // root component and the address modes have been already set.
150  }
151 
152  const auto &tile_dst = dst->tile();
153 
154  /********************************************************************************
155  * 4 - Define the compute block parameters CKW constants
156  ********************************************************************************/
157  // ...
158 
159  /********************************************************************************
160  * 5 - Define the samplers for the input tensors
161  ********************************************************************************/
162  // Check whether the lhs tensor is a tile or tensor
163  // If it is a tile, create a sampler and load the content in a tile
164  if (!lhs->has_tile())
165  {
166  // Sampler
167  ckw::TensorSampler sampler_lhs = dst->tensor_sampler();
168 
169  bool broadcast_x = false;
170  bool broadcast_y = false;
171 
172  int32_t lhs_n0 = dst_n0;
173  int32_t lhs_m0 = dst_m0;
174 
175  // Check whether we have broadcasting
176  // In case of broadcast, lhs can only be a vector or scalar.
177  // Broadcasting in other dimensions is not supported
178  if (_dst->dimension(0) != _lhs->dimension(0))
179  {
180  broadcast_x = true;
181  lhs_n0 = 1;
182  }
183 
184  if (sampler_lhs.format() == ckw::TensorSamplerFormat::Dim0_Dim1xDim2_1)
185  {
186  if (_dst->dimension(1) * _dst->dimension(2) != _lhs->dimension(1) * _lhs->dimension(2))
187  {
188  broadcast_y = true;
189  lhs_m0 = 1;
190  }
191  }
192  else if (sampler_lhs.format() == ckw::TensorSamplerFormat::Dim0_Dim1_Dim2)
193  {
194  if (_dst->dimension(1) != _lhs->dimension(1))
195  {
196  broadcast_y = true;
197  lhs_m0 = 1;
198  }
199  }
200 
201  const int32_t lhs_partial_n0 = _lhs->dimension(0) % lhs_n0;
202  const int32_t lhs_shift_back = (lhs_n0 - lhs_partial_n0) % lhs_n0;
203 
204  // Constants
205  auto const_lhs_n0_i32 = writer->declare_constant_tile(ckw::ConstantData({{lhs_n0}}, ckw::DataType::Int32));
206  auto const_lhs_m0_i32 = writer->declare_constant_tile(ckw::ConstantData({{lhs_m0}}, ckw::DataType::Int32));
207  auto const_lhs_shift_back_n0_i32 =
208  writer->declare_constant_tile(ckw::ConstantData({{lhs_shift_back}}, ckw::DataType::Int32));
209 
210  auto tile_gid_0 = writer->declare_tile("gid_0_lhs", ckw::TileInfo(ckw::DataType::Int32));
211  auto tile_gid_1 = writer->declare_tile("gid_1_lhs", ckw::TileInfo(ckw::DataType::Int32));
212  auto tile_gid_2 = writer->declare_tile("gid_2_lhs", ckw::TileInfo(ckw::DataType::Int32));
213 
214  writer->op_get_global_id(tile_gid_0, 0);
215  writer->op_get_global_id(tile_gid_1, 1);
216  writer->op_get_global_id(tile_gid_2, 2);
217 
218  auto tile_cout0 = writer->declare_tile("cout0_lhs", ckw::TileInfo(ckw::DataType::Int32)); // OFM
219  auto tile_mout0 =
220  writer->declare_tile("mout0_lhs", ckw::TileInfo(ckw::DataType::Int32)); // WIDTH or WIDTH x HEIGHT
221  auto tile_mout1 = writer->declare_tile("mout1_lhs", ckw::TileInfo(ckw::DataType::Int32)); // HEIGHT or 0
222  auto tile_bout0 = writer->declare_tile("bout0_lhs", ckw::TileInfo(ckw::DataType::Int32)); // BATCH SIZE IDX
223 
224  // Calculate coordinates
225  if (!broadcast_x)
226  {
227  get_coordinate_from_gws_overlapping_min(writer, tile_cout0, tile_gid_0, const_lhs_n0_i32,
228  const_lhs_shift_back_n0_i32, const_0_i32);
229  }
230  else
231  {
232  writer->op_assign(tile_cout0, const_0_i32);
233  }
234 
235  if (!broadcast_y)
236  {
237  get_coordinate_from_gws(writer, tile_mout0, tile_gid_1, const_lhs_m0_i32);
238  }
239  else
240  {
241  writer->op_assign(tile_mout0, const_0_i32);
242  }
243 
244  // Get the boundary aware coordinates at each global dimension index
245  if (sampler_lhs.format() == ckw::TensorSamplerFormat::Dim0_Dim1xDim2_1)
246  {
247  writer->op_assign(tile_mout1, const_0_i32);
248  get_coordinate_from_gws(writer, tile_bout0, tile_gid_2, const_pos_1_i32);
249  }
250  else if (sampler_lhs.format() == ckw::TensorSamplerFormat::Dim0_Dim1_Dim2)
251  {
252  // For tile_mout1 and tile_bout0 the step can only be 1
253  if (!broadcast_y)
254  {
255  writer->op_binary(tile_mout1, ckw::BinaryOp::Mod, tile_gid_2, const_dst_h_i32);
256  }
257  else
258  {
259  // If broadcast_y == true, it means that we have either a scalar or vector
260  // because broadcasting in other dimensions is not supported
261  writer->op_assign(tile_mout1, const_0_i32);
262  }
263 
264  writer->op_binary(tile_bout0, ckw::BinaryOp::Div, tile_gid_2, const_dst_h_i32);
265  }
266 
267  ckw::DataType lhs_dt = to_ckw(_lhs->data_type());
268  auto tile_lhs = writer->declare_tile("lhs", ckw::TileInfo(lhs_dt, lhs_m0, lhs_n0));
269 
270  writer->op_load(tile_lhs, lhs->tensor(), sampler_lhs, tile_cout0, tile_mout0, tile_mout1, tile_bout0);
271 
272  // Here, init_virtual_tensor() is used to bring the tile_lhs outside the compound statement
273  lhs->init_virtual_tensor(tile_lhs, sampler_lhs);
274  }
275 
276  // Check whether the rhs tensor is a tile or tensor
277  // If it is a tile, create a sampler and load the content in a tile
278  if (!rhs->has_tile())
279  {
280  // Sampler
281  ckw::TensorSampler sampler_rhs = dst->tensor_sampler();
282 
283  bool broadcast_x = false;
284  bool broadcast_y = false;
285 
286  int32_t rhs_n0 = dst_n0;
287  int32_t rhs_m0 = dst_m0;
288 
289  // Check whether we have broadcasting
290  // In case of broadcast, rhs can only be a vector or scalar.
291  // Broadcasting in other dimensions is not supported
292  if (_dst->dimension(0) != _rhs->dimension(0))
293  {
294  broadcast_x = true;
295  rhs_n0 = 1;
296  }
297 
298  if (sampler_rhs.format() == ckw::TensorSamplerFormat::Dim0_Dim1xDim2_1)
299  {
300  if (_dst->dimension(1) * _dst->dimension(2) != _rhs->dimension(1) * _rhs->dimension(2))
301  {
302  broadcast_y = true;
303  rhs_m0 = 1;
304  }
305  }
306  else if (sampler_rhs.format() == ckw::TensorSamplerFormat::Dim0_Dim1_Dim2)
307  {
308  if (_dst->dimension(1) != _rhs->dimension(1))
309  {
310  broadcast_y = true;
311  rhs_m0 = 1;
312  }
313  }
314 
315  const int32_t rhs_partial_n0 = _rhs->dimension(0) % rhs_n0;
316  const int32_t rhs_shift_back = (rhs_n0 - rhs_partial_n0) % rhs_n0;
317 
318  // Constants
319  auto const_rhs_n0_i32 = writer->declare_constant_tile(ckw::ConstantData({{rhs_n0}}, ckw::DataType::Int32));
320  auto const_rhs_m0_i32 = writer->declare_constant_tile(ckw::ConstantData({{rhs_m0}}, ckw::DataType::Int32));
321  auto const_rhs_shift_back_n0_i32 =
322  writer->declare_constant_tile(ckw::ConstantData({{rhs_shift_back}}, ckw::DataType::Int32));
323 
324  auto tile_gid_0 = writer->declare_tile("gid_0_rhs", ckw::TileInfo(ckw::DataType::Int32));
325  auto tile_gid_1 = writer->declare_tile("gid_1_rhs", ckw::TileInfo(ckw::DataType::Int32));
326  auto tile_gid_2 = writer->declare_tile("gid_2_rhs", ckw::TileInfo(ckw::DataType::Int32));
327 
328  writer->op_get_global_id(tile_gid_0, 0);
329  writer->op_get_global_id(tile_gid_1, 1);
330  writer->op_get_global_id(tile_gid_2, 2);
331 
332  auto tile_cout0 = writer->declare_tile("cout0_rhs", ckw::TileInfo(ckw::DataType::Int32)); // OFM
333  auto tile_mout0 =
334  writer->declare_tile("mout0_rhs", ckw::TileInfo(ckw::DataType::Int32)); // WIDTH or WIDTH x HEIGHT
335  auto tile_mout1 = writer->declare_tile("mout1_rhs", ckw::TileInfo(ckw::DataType::Int32)); // HEIGHT or 0
336  auto tile_bout0 = writer->declare_tile("bout0_rhs", ckw::TileInfo(ckw::DataType::Int32)); // BATCH SIZE IDX
337 
338  // Calculate coordinates
339  if (!broadcast_x)
340  {
341  get_coordinate_from_gws_overlapping_min(writer, tile_cout0, tile_gid_0, const_rhs_n0_i32,
342  const_rhs_shift_back_n0_i32, const_0_i32);
343  }
344  else
345  {
346  writer->op_assign(tile_cout0, const_0_i32);
347  }
348 
349  if (!broadcast_y)
350  {
351  get_coordinate_from_gws(writer, tile_mout0, tile_gid_1, const_rhs_m0_i32);
352  }
353  else
354  {
355  writer->op_assign(tile_mout0, const_0_i32);
356  }
357 
358  // Get the boundary aware coordinates at each global dimension index
359  if (sampler_rhs.format() == ckw::TensorSamplerFormat::Dim0_Dim1xDim2_1)
360  {
361  writer->op_assign(tile_mout1, const_0_i32);
362  get_coordinate_from_gws(writer, tile_bout0, tile_gid_2, const_pos_1_i32);
363  }
364  else if (sampler_rhs.format() == ckw::TensorSamplerFormat::Dim0_Dim1_Dim2)
365  {
366  // For tile_mout1 and tile_bout0 the step can only be 1
367  const auto src_w = static_cast<int32_t>(_rhs->dimension(1));
368  auto const_src_w = writer->declare_constant_tile(ckw::ConstantData({{src_w}}, ckw::DataType::Int32));
369  if (!broadcast_y)
370  {
371  writer->op_binary(tile_mout1, ckw::BinaryOp::Mod, tile_mout1, const_src_w);
372  }
373  else
374  {
375  // If broadcast_y == true, it means that we have either a scalar or vector
376  // because broadcasting in other dimensions is not supported
377  writer->op_assign(tile_mout1, const_0_i32);
378  }
379 
380  writer->op_binary(tile_bout0, ckw::BinaryOp::Div, tile_mout1, const_src_w);
381  }
382 
383  ckw::DataType rhs_dt = to_ckw(_rhs->data_type());
384  auto tile_rhs = writer->declare_tile("rhs", ckw::TileInfo(rhs_dt, rhs_m0, rhs_n0));
385 
386  writer->op_load(tile_rhs, rhs->tensor(), sampler_rhs, tile_cout0, tile_mout0, tile_mout1, tile_bout0);
387 
388  // Here, init_virtual_tensor() is used to bring the tile_rhs outside the compound statement
389  rhs->init_virtual_tensor(tile_rhs, sampler_rhs);
390  }
391 
392  const auto &tile_lhs = lhs->tile();
393  const auto &tile_rhs = rhs->tile();
394 
395  /********************************************************************************
396  * 7 - Write the rest of the code
397  ********************************************************************************/
398  // Perform the element-wise operation
399  writer->op_binary(tile_dst, to_ckw(_attributes), tile_lhs, tile_rhs);
400 
401  ARM_COMPUTE_ERROR_ON_MSG(dst->has_tile() == false, "You must bind a tile before appending another component");
402 }

404 Window GpuCkwElementwiseBinary::get_window() const
405 {
406  ARM_COMPUTE_ERROR_ON_MSG(_dst->tensor_shape().total_size() == 0U, "Destination tensor is not initialized");
407 
408  TensorShape output_shape = _dst->tensor_shape();
409  // Collapse Dim 1 (W) and Dim 2 (H) together, leave Dim 0 (C) unchanged
410  // This is in line with the collapsing convention used by operators like Conv2d
411  output_shape.collapse(2U, 1U);
412  constexpr uint32_t vector_size_byte_opencl = 16;
413  const uint32_t num_elems_processed_per_iteration =
414  adjust_vec_size(vector_size_byte_opencl / _dst->element_size(), _dst->dimension(0));
415  Window win = calculate_max_window(output_shape, Steps(num_elems_processed_per_iteration));
416 
417  return win;
418 }
