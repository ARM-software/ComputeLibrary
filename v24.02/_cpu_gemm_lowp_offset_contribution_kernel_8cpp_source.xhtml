48 Status validate_arguments(const ITensorInfo *mm_result,
49  const ITensorInfo *vector_sum_col,
50  const ITensorInfo *vector_sum_row,
51  int32_t a_offset,
52  int32_t b_offset)
53 {
54  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(mm_result, 1, DataType::S32);
55 
56  // If a_offset == 0, vector_sum_col can be a nullptr
57  if (a_offset != 0)
58  {
59  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(vector_sum_col, 1, DataType::S32);
60  ARM_COMPUTE_RETURN_ERROR_ON(vector_sum_col->dimension(0) != mm_result->dimension(0));
61  }
62 
63  // If b_offset == 0, vector_sum_row can be a nullptr
64  if (b_offset != 0)
65  {
66  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(vector_sum_row, 1, DataType::S32);
67 
68  // Check if input is a 3D reinterpretation
69  const bool reinterpret_as_3d =
70  mm_result->num_dimensions() > 1 && mm_result->tensor_shape().y() != vector_sum_row->tensor_shape().x();
71 
72  // Validate input
73  ARM_COMPUTE_RETURN_ERROR_ON(reinterpret_as_3d && vector_sum_row->dimension(0) !=
74  (mm_result->dimension(1) * mm_result->dimension(2)));
75  ARM_COMPUTE_RETURN_ERROR_ON(!reinterpret_as_3d && vector_sum_row->dimension(0) != mm_result->dimension(1));
76 
77  TensorShape output_shape = mm_result->tensor_shape();
78  if (output_shape.num_dimensions() > 1)
79  {
80  const unsigned int output_batch_idx = reinterpret_as_3d ? 3 : 2;
81 
82  TensorShape vector_sum_row_shape = vector_sum_row->tensor_shape();
83  vector_sum_row_shape.collapse_from(1);
84  output_shape.collapse_from(output_batch_idx);
85 
86  ARM_COMPUTE_RETURN_ERROR_ON_MSG(vector_sum_row_shape[1] != output_shape[output_batch_idx],
87  "mm_result tensor must have the same number of batches of output tensor");
88 
89  if (a_offset != 0)
90  {
91  TensorShape vector_sum_col_shape = vector_sum_col->tensor_shape();
92  vector_sum_col_shape.collapse_from(1);
93 
94  ARM_COMPUTE_RETURN_ERROR_ON_MSG(vector_sum_col_shape[1] != 1 &&
95  vector_sum_col_shape[1] != vector_sum_row_shape[1],
96  "vector_sum_col tensor must have the same number of batches of "
97  "vector_sum_row_shape or the number of batches must be set to 1");
98  }
99  }
100  }
101 
102  return Status{};
103 }

105 void run_offset_contribution(const Window &window,
106  ITensor *mm_result,
107  const ITensor *vector_sum_col,
108  const ITensor *vector_sum_row,
109  int32_t a_offset,
110  int32_t b_offset,
111  int32_t k_offset,
112  bool slide_vector_sum_col,
113  bool is_gemm3d)
114 {
115  Window collapsed_window = window.collapse_if_possible(window, Window::DimZ);
116  collapsed_window.set(Window::DimX, Window::Dimension(0, 1, 1));
117 
118  const int height_input = is_gemm3d ? mm_result->info()->dimension(1) : 0;
119  const int depth_input = is_gemm3d ? mm_result->info()->dimension(2) : 1;
120 
121  const int window_start_x = window.x().start();
122  const int window_end_x = window.x().end();
123  const int window_step_x = 16;
124 
125  // if vector_sum_col is nullptr then stride_y is 0, else get stride_y
126  const size_t sum_col_stride_y = (vector_sum_col != nullptr) ? (vector_sum_col->info()->strides_in_bytes().y()) : 0;
127  Iterator mm_result_it(mm_result, collapsed_window);
128 
129  if ((a_offset != 0) && (b_offset != 0) && (vector_sum_col != nullptr) && (vector_sum_row != nullptr)) // true, true
130  {
131  // Set window for vector_sum_col
132  Window win_vector_sum_col(collapsed_window);
133  win_vector_sum_col.set(Window::DimY, Window::Dimension(0, 0, 0));
134  win_vector_sum_col.set(Window::DimZ, Window::Dimension(0, 0, 0));
135 
136  // Set window for vector_sum_row
137  Window win_vector_sum_row(collapsed_window);
138  win_vector_sum_row.set(Window::DimX, Window::Dimension(0, 0, 0));
139  win_vector_sum_row.set(Window::DimY, Window::Dimension(0, 0, 0));
140  win_vector_sum_row.set(Window::DimZ, Window::Dimension(0, 0, 0));
141 
142  Iterator vector_sum_col_it(vector_sum_col, win_vector_sum_col);
143  Iterator vector_sum_row_it(vector_sum_row, win_vector_sum_row);
144 
145  const size_t sum_row_stride_y = vector_sum_row->info()->strides_in_bytes().y();
146 
147  // Offset in case vector_sum_col is batched
148  const int vector_sum_col_batch_offset =
149  slide_vector_sum_col ? vector_sum_col->info()->strides_in_bytes().z() : 0;
150 
151  execute_window_loop(
152  collapsed_window,
153  [&](const Coordinates &id)
154  {
155  const int batch_id = id.z() / depth_input;
156  const size_t batch_offset_col = batch_id * (sum_col_stride_y);
157  auto vector_sum_col_ptr = reinterpret_cast<const int32_t *>(vector_sum_col_it.ptr() + batch_offset_col +
158  batch_id * vector_sum_col_batch_offset);
159  auto mm_result_ptr = reinterpret_cast<int32_t *>(mm_result_it.ptr());
160 
161  // Compute the leftover term due to b_offset.
162  int32_t b_offset_term_s32 =
163  *(reinterpret_cast<const int32_t *>(vector_sum_row_it.ptr() + batch_id * sum_row_stride_y) +
164  id.y() + (id.z() % depth_input) * height_input);
165  b_offset_term_s32 *= b_offset;
166 
167  const int32x4_t b_offset_term_s32_vec = vdupq_n_s32(b_offset_term_s32);
168 
169  int x = window_start_x;
170  for (; x <= (window_end_x - window_step_x); x += window_step_x)
171  {
172  // Compute the leftover term due to a_offset.
173  int32x4x4_t a_offset_term_s32 = {
174  {vld1q_s32(vector_sum_col_ptr + x + 0), vld1q_s32(vector_sum_col_ptr + x + 4),
175  vld1q_s32(vector_sum_col_ptr + x + 8), vld1q_s32(vector_sum_col_ptr + x + 12)}};
176 
177  a_offset_term_s32.val[0] = vmulq_n_s32(a_offset_term_s32.val[0], a_offset);
178  a_offset_term_s32.val[1] = vmulq_n_s32(a_offset_term_s32.val[1], a_offset);
179  a_offset_term_s32.val[2] = vmulq_n_s32(a_offset_term_s32.val[2], a_offset);
180  a_offset_term_s32.val[3] = vmulq_n_s32(a_offset_term_s32.val[3], a_offset);
181 
182  // Add a_offset_term_s32 and b_offset_term_s32
183  int32x4x4_t offset_term_s32 = {
184  {vdupq_n_s32(k_offset), vdupq_n_s32(k_offset), vdupq_n_s32(k_offset), vdupq_n_s32(k_offset)}};
185 
186  offset_term_s32.val[0] =
187  vaddq_s32(offset_term_s32.val[0], vaddq_s32(a_offset_term_s32.val[0], b_offset_term_s32_vec));
188  offset_term_s32.val[1] =
189  vaddq_s32(offset_term_s32.val[1], vaddq_s32(a_offset_term_s32.val[1], b_offset_term_s32_vec));
190  offset_term_s32.val[2] =
191  vaddq_s32(offset_term_s32.val[2], vaddq_s32(a_offset_term_s32.val[2], b_offset_term_s32_vec));
192  offset_term_s32.val[3] =
193  vaddq_s32(offset_term_s32.val[3], vaddq_s32(a_offset_term_s32.val[3], b_offset_term_s32_vec));
194 
195  int32x4x4_t in_s32 = {{vld1q_s32(mm_result_ptr + x + 0), vld1q_s32(mm_result_ptr + x + 4),
196  vld1q_s32(mm_result_ptr + x + 8), vld1q_s32(mm_result_ptr + x + 12)}};
197 
198  // Add the offset terms to GEMM's result
199  in_s32.val[0] = vaddq_s32(in_s32.val[0], offset_term_s32.val[0]);
200  in_s32.val[1] = vaddq_s32(in_s32.val[1], offset_term_s32.val[1]);
201  in_s32.val[2] = vaddq_s32(in_s32.val[2], offset_term_s32.val[2]);
202  in_s32.val[3] = vaddq_s32(in_s32.val[3], offset_term_s32.val[3]);
203 
204  // Store the result with the offset contribution
205  vst1q_s32(mm_result_ptr + x + 0, in_s32.val[0]);
206  vst1q_s32(mm_result_ptr + x + 4, in_s32.val[1]);
207  vst1q_s32(mm_result_ptr + x + 8, in_s32.val[2]);
208  vst1q_s32(mm_result_ptr + x + 12, in_s32.val[3]);
209  }
210 
211  // Left-overs loop
212  for (; x < window_end_x; ++x)
213  {
214  // Compute the leftover term due to a_offset.
215  int32_t a_offset_term_s32 = *(vector_sum_col_ptr + x);
216 
217  a_offset_term_s32 *= a_offset;
218 
219  // Add the offset terms to GEMM's result
220  // Store the result with the offset contribution
221  mm_result_ptr[x] += k_offset + a_offset_term_s32 + b_offset_term_s32;
222  }
223  },
224  vector_sum_col_it, vector_sum_row_it, mm_result_it);
225  }
226  else if ((a_offset == 0) && (b_offset != 0) && (vector_sum_row != nullptr)) // false, true
227  {
228  ARM_COMPUTE_ERROR_ON_NULLPTR(vector_sum_row);
229 
230  // Set window for vector_sum_row
231  Window win_vector_sum_row(collapsed_window);
232  win_vector_sum_row.set(Window::DimX, Window::Dimension(0, 0, 0));
233  win_vector_sum_row.set(Window::DimY, Window::Dimension(0, 0, 0));
234  win_vector_sum_row.set(Window::DimZ, Window::Dimension(0, 0, 0));
235 
236  Iterator vector_sum_row_it(vector_sum_row, win_vector_sum_row);
237 
238  const size_t sum_row_stride_y = vector_sum_row->info()->strides_in_bytes().y();
239 
240  execute_window_loop(
241  collapsed_window,
242  [&](const Coordinates &id)
243  {
244  const int batch_id = id.z() / depth_input;
245  auto mm_result_ptr = reinterpret_cast<int32_t *>(mm_result_it.ptr());
246 
247  // Compute the leftover term due to b_offset.
248  int32_t b_offset_term_s32 =
249  *(reinterpret_cast<const int32_t *>(vector_sum_row_it.ptr() + batch_id * sum_row_stride_y) +
250  id.y() + (id.z() % depth_input) * height_input);
251  b_offset_term_s32 *= b_offset;
252 
253  const int32x4_t b_offset_term_s32_vec = vdupq_n_s32(b_offset_term_s32);
254 
255  int x = window_start_x;
256  for (; x <= (window_end_x - window_step_x); x += window_step_x)
257  {
258  int32x4x4_t in_s32 = {{vld1q_s32(mm_result_ptr + x + 0), vld1q_s32(mm_result_ptr + x + 4),
259  vld1q_s32(mm_result_ptr + x + 8), vld1q_s32(mm_result_ptr + x + 12)}};
260 
261  // Add the offset terms to GEMM's result
262  in_s32.val[0] = vaddq_s32(in_s32.val[0], b_offset_term_s32_vec);
263  in_s32.val[1] = vaddq_s32(in_s32.val[1], b_offset_term_s32_vec);
264  in_s32.val[2] = vaddq_s32(in_s32.val[2], b_offset_term_s32_vec);
265  in_s32.val[3] = vaddq_s32(in_s32.val[3], b_offset_term_s32_vec);
266 
267  // Store the result with the offset contribution
268  vst1q_s32(mm_result_ptr + x + 0, in_s32.val[0]);
269  vst1q_s32(mm_result_ptr + x + 4, in_s32.val[1]);
270  vst1q_s32(mm_result_ptr + x + 8, in_s32.val[2]);
271  vst1q_s32(mm_result_ptr + x + 12, in_s32.val[3]);
272  }
273 
274  // Left-overs loop
275  for (; x < window_end_x; ++x)
276  {
277  // Add the offset terms to GEMM's result
278  // Store the result with the offset contribution
279  mm_result_ptr[x] += b_offset_term_s32;
280  }
281  },
282  vector_sum_row_it, mm_result_it);
283  }
284  else if ((a_offset != 0) && (b_offset == 0) && (vector_sum_col != nullptr)) // true, false
285  {
286  // Set window for vector_sum_col
287  Window win_vector_sum_col(collapsed_window);
288  win_vector_sum_col.set(Window::DimY, Window::Dimension(0, 0, 0));
289  win_vector_sum_col.set(Window::DimZ, Window::Dimension(0, 0, 0));
290 
291  Iterator vector_sum_col_it(vector_sum_col, win_vector_sum_col);
292 
293  // Offset in case vector_sum_col is batched
294  const int vector_sum_col_batch_offset =
295  slide_vector_sum_col ? vector_sum_col->info()->strides_in_bytes().z() : 0;
296 
297  execute_window_loop(
298  collapsed_window,
299  [&](const Coordinates &id)
300  {
301  const int batch_id = id.z() / depth_input;
302  const size_t batch_offset_col =
303  batch_id *
304  (sum_col_stride_y); // Value to offset vector_sum_col_ptr to allow for iteration of y values in tensor
305  auto vector_sum_col_ptr = reinterpret_cast<const int32_t *>(vector_sum_col_it.ptr() + batch_offset_col +
306  batch_id * vector_sum_col_batch_offset);
307  auto mm_result_ptr = reinterpret_cast<int32_t *>(mm_result_it.ptr());
308 
309  int x = window_start_x;
310  for (; x <= (window_end_x - window_step_x); x += window_step_x)
311  {
312  // Compute the leftover term due to a_offset.
313  int32x4x4_t a_offset_term_s32 = {
314  {vld1q_s32(vector_sum_col_ptr + x + 0), vld1q_s32(vector_sum_col_ptr + x + 4),
315  vld1q_s32(vector_sum_col_ptr + x + 8), vld1q_s32(vector_sum_col_ptr + x + 12)}};
316 
317  a_offset_term_s32.val[0] = vmulq_n_s32(a_offset_term_s32.val[0], a_offset);
318  a_offset_term_s32.val[1] = vmulq_n_s32(a_offset_term_s32.val[1], a_offset);
319  a_offset_term_s32.val[2] = vmulq_n_s32(a_offset_term_s32.val[2], a_offset);
320  a_offset_term_s32.val[3] = vmulq_n_s32(a_offset_term_s32.val[3], a_offset);
321 
322  int32x4x4_t in_s32 = {{vld1q_s32(mm_result_ptr + x + 0), vld1q_s32(mm_result_ptr + x + 4),
323  vld1q_s32(mm_result_ptr + x + 8), vld1q_s32(mm_result_ptr + x + 12)}};
324 
325  // Add the offset terms to GEMM's result
326  in_s32.val[0] = vaddq_s32(in_s32.val[0], a_offset_term_s32.val[0]);
327  in_s32.val[1] = vaddq_s32(in_s32.val[1], a_offset_term_s32.val[1]);
328  in_s32.val[2] = vaddq_s32(in_s32.val[2], a_offset_term_s32.val[2]);
329  in_s32.val[3] = vaddq_s32(in_s32.val[3], a_offset_term_s32.val[3]);
330 
331  // Store the result with the offset contribution
332  vst1q_s32(mm_result_ptr + x + 0, in_s32.val[0]);
333  vst1q_s32(mm_result_ptr + x + 4, in_s32.val[1]);
334  vst1q_s32(mm_result_ptr + x + 8, in_s32.val[2]);
335  vst1q_s32(mm_result_ptr + x + 12, in_s32.val[3]);
336  }
337 
338  // Left-overs loop
339  for (; x < window_end_x; ++x)
340  {
341  // Compute the leftover term due to a_offset.
342  const int32_t a_offset_term_s32 = *(vector_sum_col_ptr + x);
343 
344  // Add the offset terms to GEMM's result
345  // Store the result with the offset contribution
346  mm_result_ptr[x] += a_offset_term_s32 * a_offset;
347  }
348  },
349  vector_sum_col_it, mm_result_it);
350  }
351  else // false, false

353  // No offset contribution from matrix A and matrix B

359 void CpuGemmLowpOffsetContributionKernel::configure(ITensorInfo *mm_result,
360  ITensorInfo *vector_sum_col,
361  ITensorInfo *vector_sum_row,
362  int32_t k,
363  int32_t a_offset,
364  int32_t b_offset)
365 {
366  // Perform validate step
367  ARM_COMPUTE_UNUSED(vector_sum_row);
368  ARM_COMPUTE_ERROR_ON_NULLPTR(mm_result);
369  ARM_COMPUTE_ERROR_THROW_ON(validate_arguments(mm_result, vector_sum_col, vector_sum_row, a_offset, b_offset));
370 
371  _a_offset = a_offset;
372  _b_offset = b_offset;
373  _k_offset = a_offset * b_offset * k;
374 
375  // If a_offset == 0, vector_sum_col can be a nullptr
376  if (a_offset != 0)
377  {
378  // Check if vector_sum_col_shape should be slidden or not
379  // Don't slide vector_sum_col_shape along the y dimension if vector_sum_col_shape has just 1 dimension and vector_sum_row_shape more than 1
380  // This scenario can happen when the the matrix multiplication is used to perform a convolution operation
381  _slide_vector_sum_col = vector_sum_col->tensor_shape().num_dimensions() > 1;
382  }
383 
384  // Configure kernel window
385  Window win = calculate_max_window(*mm_result, Steps());
386  ICpuKernel::configure(win);
387 }

399 void CpuGemmLowpOffsetContributionKernel::run_op(ITensorPack &tensors, const Window &window, const ThreadInfo &info)
400 {
401  ARM_COMPUTE_UNUSED(info);
402  ARM_COMPUTE_ERROR_ON_UNCONFIGURED_KERNEL(this);
403  ARM_COMPUTE_ERROR_ON_INVALID_SUBWINDOW(ICpuKernel::window(), window);
404 
405  auto vector_sum_col = tensors.get_const_tensor(TensorType::ACL_SRC_0);
406  auto vector_sum_row = tensors.get_const_tensor(TensorType::ACL_SRC_1);
407  auto mm_result = tensors.get_tensor(TensorType::ACL_DST);

413  run_offset_contribution(window, mm_result, vector_sum_col, vector_sum_row, _a_offset, _b_offset, _k_offset,
414  _slide_vector_sum_col, reinterpret_as_3d);
