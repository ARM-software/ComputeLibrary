53 void CLSpaceToBatchLayer::configure(const CLCompileContext &compile_context,
54  const ICLTensor *input,

56  const ICLTensor *paddings,
57  ICLTensor *output)

59  ARM_COMPUTE_ERROR_ON_NULLPTR(input, block_shape, paddings, output);

61 
62  if (input->info()->tensor_shape().total_size() != output->info()->tensor_shape().total_size())
