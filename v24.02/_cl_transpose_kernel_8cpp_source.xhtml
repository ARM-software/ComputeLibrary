53 void ClTransposeKernel::configure(const CLCompileContext &compile_context, const ITensorInfo *src, ITensorInfo *dst)
54 {
55  ARM_COMPUTE_ERROR_ON_NULLPTR(src, dst);
56 
57  // Output auto initialization if not yet initialized
58  const TensorShape dst_shape = misc::shape_calculator::compute_transposed_shape(*src);
59  auto_init_if_empty(*dst, src->clone()->set_tensor_shape(dst_shape));
60 
61  // Explicitly set the tensor shape to preserve dimensions
62  dst->set_tensor_shape(dst_shape);
63 
64  ARM_COMPUTE_ERROR_THROW_ON(ClTransposeKernel::validate(src, dst));
65  auto padding_info = get_padding_info({src, dst});
66 
67  unsigned int vec_size_x;
68  unsigned int vec_size_y;
69 
70  // Set the optimal tile size for each data type without register spilling
71  switch (src->element_size())
72  {
73  case 1:
74  vec_size_x = adjust_vec_size(8, src->dimension(0));
75  vec_size_y = adjust_vec_size(16, src->dimension(1));
76  break;
77  case 2:
78  vec_size_x = adjust_vec_size(8, src->dimension(0));
79  vec_size_y = adjust_vec_size(8, src->dimension(1));
80  break;
81  case 4:
82  vec_size_x = adjust_vec_size(4, src->dimension(0));
83  vec_size_y = adjust_vec_size(8, src->dimension(1));
84  break;
85  default:
86  ARM_COMPUTE_ERROR("Unsupported data type");
87  break;
88  }
89 
90  const int vec_size_x_leftovers = src->dimension(0) % vec_size_x;
91  const int vec_size_y_leftovers = src->dimension(1) % vec_size_y;
92 
93  CLBuildOptions build_opts;
94  build_opts.add_option("-DDATA_TYPE_IN_BYTES=" + support::cpp11::to_string(src->element_size()));
95  build_opts.add_option("-DVEC_SIZE_X=" + support::cpp11::to_string(vec_size_x));
96  build_opts.add_option("-DVEC_SIZE_LEFTOVER_X=" + support::cpp11::to_string(vec_size_x_leftovers));
97  build_opts.add_option("-DVEC_SIZE_Y=" + support::cpp11::to_string(vec_size_y));
98  build_opts.add_option("-DVEC_SIZE_LEFTOVER_Y=" + support::cpp11::to_string(vec_size_y_leftovers));
99 
100  _kernel = create_kernel(compile_context, "transpose", build_opts.options());
101 
102  // Configure kernel window
103  Window win = calculate_max_window(*src, Steps(vec_size_x, vec_size_y));
104  ICLKernel::configure_internal(win);
105  ARM_COMPUTE_ERROR_ON(has_padding_changed(padding_info));
106 }

108 Status ClTransposeKernel::validate(const ITensorInfo *src, const ITensorInfo *dst)
109 {
110  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(src, dst);
111  ARM_COMPUTE_RETURN_ERROR_ON_F16_UNSUPPORTED(src);
112  ARM_COMPUTE_RETURN_ERROR_ON(src->data_type() == DataType::UNKNOWN);
113 
114  // Validate configured dst
115  if (dst->total_size() != 0)
116  {
117  const TensorInfo dst_info =
118  src->clone()->set_tensor_shape(misc::shape_calculator::compute_transposed_shape(*src));
119  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_SHAPES(dst, &dst_info);
120 
121  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_QUANTIZATION_INFO(src, dst);
122  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(src, dst);
123  }
124 
125  return Status{};
126 }

128 void ClTransposeKernel::run_op(ITensorPack &tensors, const Window &window, cl::CommandQueue &queue)
129 {
130  ARM_COMPUTE_ERROR_ON_UNCONFIGURED_KERNEL(this);
131  ARM_COMPUTE_ERROR_ON_MISMATCHING_WINDOWS(ICLKernel::window(), window);
132 
133  const auto src =
134  utils::cast::polymorphic_downcast<const ICLTensor *>(tensors.get_const_tensor(TensorType::ACL_SRC));
135  auto dst = utils::cast::polymorphic_downcast<ICLTensor *>(tensors.get_tensor(TensorType::ACL_DST));
136 
137  // Collapse dimensions higher than width and height into the batch dimension
138  Window collapsed = window.collapse_if_possible(ICLKernel::window(), Window::DimZ);
139  Window slice = collapsed.first_slice_window_3D();
140 
141  do
142  {
143  unsigned int idx = 0;
144  add_3D_tensor_argument(idx, src, slice);
145  add_3D_tensor_argument(idx, dst, slice);
146  enqueue(queue, *this, slice, lws_hint());
147  } while (collapsed.slide_window_slice_3D(slice));
148 }
