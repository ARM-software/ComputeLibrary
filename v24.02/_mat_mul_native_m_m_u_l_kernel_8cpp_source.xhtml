149 TEST_CASE(ValidateInputShapes, framework::DatasetMode::ALL)
150 {
151  if(arm_matrix_multiply_supported(CLKernelLibrary::get().get_device()))
152  {
153  // Configurations are assumed to be Nt/Nt, but will be transposed inside the test to test other configurations
154  using ShapeConfigurationTuple = std::tuple<TensorShape, TensorShape, TensorShape, bool>; // lhs, rhs, bias, result
155  const std::vector<ShapeConfigurationTuple> shape_configurations =
156  {
157  { TensorShape(4U, 1U), TensorShape(3U, 4U), TensorShape(3U), true },
158  { TensorShape(12U, 12U), TensorShape(3U, 12U), TensorShape(3U), true },
159  { TensorShape(8U, 4U), TensorShape(2U, 8U), TensorShape(2U), true },
160  { TensorShape(8U, 4U), TensorShape(2U, 4U), TensorShape(2U), false }, // Mismatch in the K dimension
161  { TensorShape(5U, 0U), TensorShape(2U, 5U), TensorShape(2U), false }, // Invalid dimension
162  { TensorShape(5U, 7U), TensorShape(2U, 5U), TensorShape(2U), false }, // K not a multiple of 4 (MMUL_K0)
163  { TensorShape(8U, 4U, 3U, 4U, 5U, 6U), TensorShape(2U, 8U, 3U, 4U, 5U, 6U), TensorShape(2U), true },
164  { TensorShape(5U, 4U, 3U, 4U, 5U, 1U), TensorShape(2U, 5U, 3U, 4U, 5U, 6U), TensorShape(2U), false }, // No batch broadcasting
165  { TensorShape(5U, 4U, 3U, 4U, 9U, 6U), TensorShape(2U, 5U, 3U, 4U, 5U, 6U), TensorShape(2U), false }, // Mismatch in batch dimension
166  { TensorShape(4U, 1U), TensorShape(3U, 4U), TensorShape(1U), false }, // Bias first dimensions != dst first dimension.
167  { TensorShape(4U, 1U), TensorShape(3U, 4U), TensorShape(5U, 6U), false }, // Bias is 2d which is invalid.
168  };
169 
170  for(auto &tuple : shape_configurations)
171  {
172  const bool expected = std::get<3>(tuple);
173 
174  for(bool adj_lhs :
175  {
176  false, true
177  })
178  {
179  for(bool adj_rhs :
180  {
181  false, true
182  })
183  {
184  TensorShape lhs_shape = std::get<0>(tuple);
185  TensorShape rhs_shape = std::get<1>(tuple);
186  TensorShape bia_shape = std::get<2>(tuple);
187 
188  if(adj_lhs)
189  {
190  permute(lhs_shape, PermutationVector(1U, 0U));
191  }
192 
193  if(adj_rhs)
194  {
195  permute(rhs_shape, PermutationVector(1U, 0U));
196  }
197 
198  const TensorInfo lhs_info = TensorInfo(lhs_shape, 1, DataType::F32);
199  const TensorInfo rhs_info = TensorInfo(rhs_shape, 1, DataType::F32);
200  const TensorInfo bia_info = TensorInfo(bia_shape, 1, DataType::F32);
201  TensorInfo output_info;
202 
203  MatMulKernelInfo matmul_kernel_info{ adj_lhs, adj_rhs, 1, 1, 1, false /* export_rhs_to_cl_image */ };
204 
205  Status status = ClMatMulNativeMMULKernel::validate(&lhs_info, &rhs_info, &bia_info, &output_info, matmul_kernel_info);
206  ARM_COMPUTE_EXPECT(bool(status) == expected, framework::LogLevel::ERRORS);
207  }
208  }
209  }
210  }
211  else
212  {
213  ARM_COMPUTE_TEST_INFO("cl_arm_matrix_multiply not supported. TEST skipped");
214  framework::ARM_COMPUTE_PRINT_INFO();
215  }
216 }

235  { DataType::QASYMM16, DataType::QASYMM16, DataType::S32, DataType::QASYMM16, false }, // no quantized types
236  { DataType::QSYMM16, DataType::QSYMM16, DataType::S32, DataType::QSYMM16, false }, // no quantized types
237  { DataType::QSYMM8, DataType::QSYMM8, DataType::S32, DataType::QSYMM8, false }, // no quantized types
238  { DataType::S64, DataType::S64, DataType::S64, DataType::S64, false }, // no integral types
239  { DataType::S32, DataType::S32, DataType::S32, DataType::S32, false }, // no integral types
240  { DataType::S16, DataType::S16, DataType::S16, DataType::S16, false }, // no integral types
241  { DataType::S8, DataType::S8, DataType::S8, DataType::S8, false }, // no integral types

243  { DataType::U32, DataType::U32, DataType::U32, DataType::U32, false }, // no integral types
244  { DataType::U16, DataType::U16, DataType::U16, DataType::U16, false }, // no integral types

264  else

266  ARM_COMPUTE_TEST_INFO("cl_arm_matrix_multiply not supported. TEST skipped");
