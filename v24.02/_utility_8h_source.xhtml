153 /** Perform an index sort of a given vector.
154  *
155  * @param[in] v Vector to sort
156  *
157  * @return Sorted index vector.
158  */
159 template <typename T>
160 std::vector<size_t> sort_indices(const std::vector<T> &v)
161 {
162  std::vector<size_t> idx(v.size());
163  std::iota(idx.begin(), idx.end(), 0);
164 
165  std::sort(idx.begin(), idx.end(), [&v](size_t i1, size_t i2) { return v[i1] < v[i2]; });
166 
167  return idx;
168 }
