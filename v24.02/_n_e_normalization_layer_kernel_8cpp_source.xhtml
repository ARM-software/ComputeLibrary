48 Status validate_arguments(const ITensorInfo *input,
49  const ITensorInfo *input_squared,
50  const ITensorInfo *output,
51  const NormalizationLayerInfo &norm_info)
52 {
53  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(input, input_squared, output);
54  ARM_COMPUTE_RETURN_ERROR_ON_CPU_F16_UNSUPPORTED(input);
55  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(input, 1, DataType::F16, DataType::F32);
56 
57  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input, input_squared);
58  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_SHAPES(input, input_squared);
59  ARM_COMPUTE_RETURN_ERROR_ON_MSG(!(norm_info.norm_size() % 2), "Normalization size should be odd");
60 
61  // Checks performed when output is configured
62  if (output->total_size() != 0)
63  {
64  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input, output);
65  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_SHAPES(input, output);
66  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_LAYOUT(input, output);
67  }
68 
69  return Status{};
70 }

79 void NENormalizationLayerKernel::configure(const ITensor *input,
80  const ITensor *input_squared,
81  ITensor *output,
82  NormalizationLayerInfo norm_info)
83 {
84  ARM_COMPUTE_ERROR_ON_NULLPTR(input, input_squared, output);
85  // Output tensor auto initialization if not yet initialized
86  auto_init_if_empty(*output->info(), *input->info());
87 
88  // Perform validation step
89  ARM_COMPUTE_ERROR_THROW_ON(validate_arguments(input->info(), input_squared->info(), output->info(), norm_info));
90 
91  const unsigned int norm_idx = get_normalization_dimension_index(input->info()->data_layout(), norm_info);
92 
93  _input = input;
94  _input_squared = input_squared;
95  _output = output;
96  _norm_info = norm_info;
97  switch (_input->info()->data_type())
98  {
99  case DataType::F32:
100  {
101  switch (norm_idx)
102  {
103  case 0:
104  {
105  if (norm_info.type() == NormType::IN_MAP_2D)
106  {
107  _func = REGISTER_FP32_NEON(cpu::neon_normalize_float32_4_0_2D);
108  }
109  else
110  {
111  _func = REGISTER_FP32_NEON(cpu::neon_normalize_float32_4_0);
112  }
113  break;
114  }
115  case 1:
116  if (norm_info.type() == NormType::IN_MAP_2D)
117  {
118  _func = REGISTER_FP32_NEON(cpu::neon_normalize_float32_4_1_2D);
119  }
120  else
121  {
122  _func = REGISTER_FP32_NEON(cpu::neon_normalize_float32_4_1);
123  }
124  break;
125  case 2:
126  _func = REGISTER_FP32_NEON(cpu::neon_normalize_float32_4_2);
127  break;
128  default:
129  break;
130  }
131  break;
132  }
133 #ifdef ARM_COMPUTE_ENABLE_FP16
134  case DataType::F16:
135  {
136  switch (norm_idx)
137  {
138  case 0:
139  {
140  if (norm_info.type() == NormType::IN_MAP_2D)
141  {
142  _func = REGISTER_FP16_NEON(cpu::neon_normalize_float16_8_0_2D);
143  }
144  else
145  {
146  _func = REGISTER_FP16_NEON(cpu::neon_normalize_float16_8_0);
147  }
148  break;
149  }
150  case 1:
151  if (norm_info.type() == NormType::IN_MAP_2D)
152  {
153  _func = REGISTER_FP16_NEON(cpu::neon_normalize_float16_8_1_2D);
154  }
155  else
156  {
157  _func = REGISTER_FP16_NEON(cpu::neon_normalize_float16_8_1);
158  }
159  break;
160  case 2:
161  _func = REGISTER_FP16_NEON(cpu::neon_normalize_float16_8_2);
162  break;
163  default:
164  break;
165  }
166  break;
167  }
168 #endif /* ARM_COMPUTE_ENABLE_FP16 */
169  default:
170  ARM_COMPUTE_ERROR("NOT SUPPORTED!");
171  }
172 
173  // Configure kernel window
174  Window win = calculate_max_window(*input->info(), Steps());
175  INEKernel::configure(win);
176 }
