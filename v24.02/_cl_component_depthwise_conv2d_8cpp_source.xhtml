111 Status ClComponentDepthwiseConv2d::validate(const Properties &properties,
112  const ArgumentPack<ITensorInfo> &tensors,
113  const Attributes &attributes,
114  const Settings &settings)
115 {
116  ARM_COMPUTE_UNUSED(properties, settings);
117  const auto src = tensors.get_const_tensor(TensorType::ACL_SRC_0);
118  const auto wei = tensors.get_const_tensor(TensorType::ACL_SRC_1);
119  const auto bia = tensors.get_const_tensor(TensorType::ACL_SRC_2);
120  const auto dst = tensors.get_const_tensor(TensorType::ACL_DST_0);
121 
122  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(src, wei, dst);
123 
124  // 1. Check validity
125  // Matching data type
126  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(src, wei);
127  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(src, dst);
128  if (bia != nullptr)
129  {
130  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(src, bia);
131  }
132 
133  // Matching data layout
134  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_LAYOUT(src, wei);
135  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_LAYOUT(src, dst);
136  if (bia != nullptr)
137  {
138  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_LAYOUT(src, bia);
139  }
140 
141  // All tensor infos are initialized
142  ARM_COMPUTE_RETURN_ERROR_ON(src->tensor_shape().total_size() == 0);
143  ARM_COMPUTE_RETURN_ERROR_ON(wei->tensor_shape().total_size() == 0);
144  ARM_COMPUTE_RETURN_ERROR_ON(dst->tensor_shape().total_size() == 0);
145  if (bia != nullptr)
146  {
147  ARM_COMPUTE_RETURN_ERROR_ON(bia->tensor_shape().total_size() == 0);
148  }
149  // Device requirements are met
150  ARM_COMPUTE_RETURN_ERROR_ON_F16_UNSUPPORTED(src);
151  // wei shape is correct
152  const DataLayout data_layout = src->data_layout();
153  const size_t channel_idx = get_data_layout_dimension_index(data_layout, DataLayoutDimension::CHANNEL);
154 
155  ARM_COMPUTE_RETURN_ERROR_ON(wei->dimension(channel_idx) !=
156  (src->dimension(channel_idx) * attributes.depth_multiplier()));
157  ARM_COMPUTE_RETURN_ERROR_ON_MSG(wei->num_dimensions() > 3, "Weights can be at most 3 dimensional");
158 
159  // dst shape is correct
160  const PadStrideInfo pad_stride_info =
161  PadStrideInfo(attributes.stride().x(), attributes.stride().y(), attributes.pad().left, attributes.pad().right,
162  attributes.pad().top, attributes.pad().bottom, attributes.dimension_rounding_type());
163  const ConvolutionInfo conv_info{pad_stride_info, attributes.depth_multiplier(), ActivationLayerInfo(),
164  attributes.dilation()};
165  const TensorShape output_shape = misc::shape_calculator::compute_depthwise_convolution_shape(*src, *wei, conv_info);
166 
167  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DIMENSIONS(dst->tensor_shape(), output_shape);
168 
169  // Check strides and dilation
170  ARM_COMPUTE_RETURN_ERROR_ON(conv_info.pad_stride_info.stride().first < 1);
171  ARM_COMPUTE_RETURN_ERROR_ON(conv_info.pad_stride_info.stride().second < 1);
172  ARM_COMPUTE_RETURN_ERROR_ON((conv_info.dilation.x() < 1) || (conv_info.dilation.y() < 1));
173  ARM_COMPUTE_RETURN_ERROR_ON(conv_info.pad_stride_info.stride().first > 1 && settings.m0() != 1);
174  ARM_COMPUTE_RETURN_ERROR_ON(conv_info.dilation.x() > 1 && settings.m0() != 1);
175 
176  if (conv_info.depth_multiplier > 1 && settings.n0() > 1)
177  {
178  ARM_COMPUTE_RETURN_ERROR_ON((conv_info.depth_multiplier % settings.n0()) != 0);
179  }
180 
181  // Check export weights to cl image
182  ARM_COMPUTE_RETURN_ERROR_ON_MSG((settings.export_weights_to_cl_image() == true) &&
183  (export_to_cl_image(wei) == false),
184  "Weights cannot be exported to cl_image!");
185  ARM_COMPUTE_RETURN_ERROR_ON((settings.export_weights_to_cl_image() == true) && ((settings.n0() % 4) != 0));
186 
187  ARM_COMPUTE_RETURN_ERROR_ON(wei->dimension(channel_idx) !=
188  (src->dimension(channel_idx) * conv_info.depth_multiplier));
189 
190  // bia shape is correct
191  if (bia != nullptr)
192  {
193  ARM_COMPUTE_RETURN_ERROR_ON_MSG(bia->dimension(0) != output_shape[channel_idx],
194  "Biases size and number of dst feature maps should match");
195  ARM_COMPUTE_RETURN_ERROR_ON_MSG(bia->num_dimensions() > 1, "Biases should be one dimensional");
196  }
197 
198  // 2. Check support level
199  // Data type
200  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(src, 1, DataType::F16, DataType::F32);
201  // Data layout
202  ARM_COMPUTE_RETURN_ERROR_ON_DATA_LAYOUT_NOT_IN(src, DataLayout::NHWC);
203  // Texture in the input tensor
204  ARM_COMPUTE_RETURN_ERROR_ON((settings.export_input_to_cl_image() == true));
205 
206  return Status{};
207 }
