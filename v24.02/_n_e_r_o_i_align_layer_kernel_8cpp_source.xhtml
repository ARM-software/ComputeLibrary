100 Status validate_arguments(const ITensorInfo *input,
101  const ITensorInfo *rois,
102  ITensorInfo *output,
103  const ROIPoolingLayerInfo &pool_info)
104 {
105  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(input, rois, output);
106  ARM_COMPUTE_RETURN_ERROR_ON(rois->dimension(0) != 5);
107  ARM_COMPUTE_RETURN_ERROR_ON(rois->num_dimensions() > 2);
108  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(input, 1, DataType::QASYMM8, DataType::QASYMM8_SIGNED,
109  DataType::F32, DataType::F16);
110  ARM_COMPUTE_RETURN_ERROR_ON_DATA_LAYOUT_NOT_IN(input, DataLayout::NHWC, DataLayout::NCHW);
111  ARM_COMPUTE_RETURN_ERROR_ON((pool_info.pooled_width() == 0) || (pool_info.pooled_height() == 0));
112  ARM_COMPUTE_RETURN_ERROR_ON_CPU_F16_UNSUPPORTED(input);
113 
114  if (output->total_size() != 0)
115  {
116  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input, output);
117  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_LAYOUT(input, output);
118  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DIMENSIONS(compute_roi_align_shape(*input, *rois, pool_info),
119  output->tensor_shape());
120  }
121 
122  if (input->data_type() == DataType::QASYMM8 || input->data_type() == DataType::QASYMM8_SIGNED)
123  {
124  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(rois, 1, DataType::QASYMM16);
125 
126  const UniformQuantizationInfo rois_qinfo = rois->quantization_info().uniform();
127  ARM_COMPUTE_RETURN_ERROR_ON(rois_qinfo.scale != 0.125f);
128  ARM_COMPUTE_RETURN_ERROR_ON(rois_qinfo.offset != 0);
129  }
130  else
131  {
132  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input, rois);
133  }
134 
135  return Status{};
136 }

144 void NEROIAlignLayerKernel::configure(const ITensor *input,
145  const ITensor *rois,
146  ITensor *output,
147  const ROIPoolingLayerInfo &pool_info)
148 {
149  ARM_COMPUTE_ERROR_ON_NULLPTR(input, output, rois);
150  ARM_COMPUTE_ERROR_THROW_ON(validate_arguments(input->info(), rois->info(), output->info(), pool_info));
151  // Output auto inizialitation if not yet initialized
152  const TensorShape output_shape = compute_roi_align_shape(*input->info(), *rois->info(), pool_info);
153  auto_init_if_empty((*output->info()), output_shape, 1, input->info()->data_type(),
154  input->info()->quantization_info());
155  output->info()->set_data_layout(input->info()->data_layout());
156 
157  // Configure kernel window
158  const unsigned int num_rois = rois->info()->dimension(1);
159  Window window;
160  window.set(Window::DimX, Window::Dimension(0, num_rois));
161  window.set(Window::DimY, Window::Dimension(0, 1));
162 
163  // Set instance variables
164  _input = input;
165  _rois = rois;
166  _output = output;
167  _pool_info = pool_info;
168 
169  INEKernel::configure(window);
170 }

181 void NEROIAlignLayerKernel::run(const Window &window, const ThreadInfo &info)
182 {
183  const DataLayout data_layout = _input->info()->data_layout();
184  if (data_layout == DataLayout::NCHW || data_layout == DataLayout::NHWC)
185  {
186  const auto *uk = get_implementation(ROIAlignSelectorData{_input->info()->data_type()});
187  ARM_COMPUTE_ERROR_ON(uk == nullptr || uk->ukernel == nullptr);
188 
189  uk->ukernel(_input, _output, _rois, _pool_info, window, info);
190  }
191  else
192  {
193  ARM_COMPUTE_ERROR("Invalid layout");
194  }
195 }
