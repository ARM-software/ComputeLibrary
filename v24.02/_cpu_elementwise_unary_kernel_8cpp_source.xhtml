49 std::unique_ptr<uint8_t[]> q8_prepare_lut(ElementWiseUnary op, const ITensorInfo *src, const ITensorInfo *dst)
50 {
51  ARM_COMPUTE_ERROR_ON(src->data_type() != dst->data_type());
52  ARM_COMPUTE_ERROR_ON(!is_data_type_quantized(src->data_type()));
53  ARM_COMPUTE_ERROR_ON(src->element_size() != 1);
54 
55  auto lut = std::unique_ptr<uint8_t[]>(new uint8_t[256]);
56  const auto is_signed = src->data_type() == DataType::QASYMM8_SIGNED;
57  const auto src_qi = src->quantization_info().uniform();
58  const auto dst_qi = dst->quantization_info().uniform();
59 
60  const auto dst_min_fp = (((is_signed) ? -128 : 0) - dst_qi.offset) * dst_qi.scale;
61  const auto dst_max_fp = (((is_signed) ? 127 : 255) - dst_qi.offset) * dst_qi.scale;
62 
63  for (int i = 0; i < 256; ++i)
64  {
65  const auto in =
66  (is_signed) ? dequantize_qasymm8_signed(static_cast<int8_t>(i), src_qi) : dequantize_qasymm8(i, src_qi);
67  float result = 0;
68 
69  switch (op)
70  {
71  case ElementWiseUnary::RSQRT:
72  result = 1 / sqrt(in);
73  break;
74 
75  case ElementWiseUnary::EXP:
76  result = std::exp(in);
77  break;
78 
79  case ElementWiseUnary::NEG:
80  result = -in;
81  break;
82 
83  case ElementWiseUnary::LOG:
84  result = std::log(in);
85  break;
86 
87  case ElementWiseUnary::ABS:
88  result = std::abs(in);
89  break;
90 
91  case ElementWiseUnary::ROUND:
92  result = support::cpp11::nearbyint(in);
93  break;
94 
95  case ElementWiseUnary::SIN:
96  result = std::sin(in);
97  break;
98 
99  default:
100  ARM_COMPUTE_ERROR("NOT_SUPPORTED!");
101  }
102 
103  result = utility::clamp(result, dst_min_fp, dst_max_fp);
104 
105  const auto out = (is_signed) ? static_cast<uint8_t>(quantize_qasymm8_signed(result, dst_qi))
106  : quantize_qasymm8(result, dst_qi);
107  lut[i] = out;
108  }
109 
110  return lut;
111 }

185 void CpuElementwiseUnaryKernel::configure(ElementWiseUnary op, const ITensorInfo &src, ITensorInfo &dst)
186 {
187  ARM_COMPUTE_ERROR_THROW_ON(validate(op, src, dst));
188  const auto uk = CpuElementwiseUnaryKernel::get_implementation(
189  DataTypeISASelectorData{src.data_type(), CPUInfo::get().get_isa()});
190  ARM_COMPUTE_ERROR_ON(uk == nullptr || uk->ukernel == nullptr);
191 
192  _op = op;
193  _run_method = uk->ukernel;
194  _name = std::string("CpuElementwiseUnaryKernel").append("/").append(uk->name);
195 
196  // If input shape is dynamic, expect a configured window and dst at run-time.
197  if (src.is_dynamic())
198  {
199  return;
200  }
201 
202  if (uk->prepare_func != nullptr)
203  {
204  _lut = uk->prepare_func(op, &src, &dst);
205  }
206 
207  auto shape_and_window = compute_output_shape_and_window(src.tensor_shape());
208  auto_init_if_empty(dst, shape_and_window.first, 1, src.data_type());
209  ICpuKernel::configure(shape_and_window.second);
210 }

212 Status CpuElementwiseUnaryKernel::validate(ElementWiseUnary op, const ITensorInfo &src, const ITensorInfo &dst)
213 {
214  ARM_COMPUTE_RETURN_ERROR_ON_CPU_F16_UNSUPPORTED(&src);
215 
216  const auto *uk = CpuElementwiseUnaryKernel::get_implementation(
217  DataTypeISASelectorData{src.data_type(), CPUInfo::get().get_isa()});
218 
219  ARM_COMPUTE_RETURN_ERROR_ON(uk == nullptr || uk->ukernel == nullptr);
220 
221  switch (op)
222  {
223  case ElementWiseUnary::EXP:
224  case ElementWiseUnary::RSQRT:
225  case ElementWiseUnary::LOG:
226  case ElementWiseUnary::ROUND:
227  case ElementWiseUnary::SIN:
228  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(&src, 1, DataType::F16, DataType::F32,
229  DataType::QASYMM8, DataType::QASYMM8_SIGNED);
230  break;
231  case ElementWiseUnary::NEG:
232  case ElementWiseUnary::ABS:
233  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(&src, 1, DataType::F16, DataType::F32, DataType::S32,
234  DataType::QASYMM8, DataType::QASYMM8_SIGNED);
235  break;
236  default:
237  ARM_COMPUTE_ERROR("ElementWiseUnary operation not supported");
238  }
239  // Validate in case of configured dst
240  if (dst.total_size() > 0)
241  {
242  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(&src, &dst);
243  }
244 
245  return Status{};
246 }
