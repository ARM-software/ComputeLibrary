69 void ClDirectConv2d::configure(const CLCompileContext &compile_context,
70  ITensorInfo *src,
71  ITensorInfo *weights,
72  ITensorInfo *biases,
73  ITensorInfo *dst,
74  const PadStrideInfo &conv_info,
75  const ActivationLayerInfo &act_info)
76 {
77  ARM_COMPUTE_ERROR_ON_NULLPTR(src);
78  ARM_COMPUTE_LOG_PARAMS(src, weights, biases, dst, conv_info, act_info);
79 
80  // Initialize the direct convolution descriptor
81  const DirectConvComputeKernelInfo desc = config_direct_convolution_nhwc(src, weights, conv_info);
82 
83  // Configure direct convolution kernel
84  const ActivationLayerInfo conv2d_act_info =
85  (src->data_layout() == DataLayout::NHWC && is_data_type_float(src->data_type())) ? act_info
86  : ActivationLayerInfo();
87  auto k = std::make_unique<kernels::ClDirectConv2dKernel>();
88  k->set_target(CLScheduler::get().target());
89  k->configure(compile_context, src, weights, biases, dst, conv_info, conv2d_act_info, desc);
90  _direct_conv_kernel = std::move(k);
91 
92  // Configure border handler
93  PixelValue zero_value(0.f);
94  if (is_data_type_quantized_asymmetric(src->data_type()))
95  {
96  zero_value = PixelValue(0, src->data_type(), src->quantization_info());
97  }
98  auto b = std::make_unique<CLFillBorderKernel>();
99  b->configure(compile_context, src, _direct_conv_kernel->border_size(), BorderMode::CONSTANT, zero_value);
100  _src_border_handler = std::move(b);
101 
102  // Fused activation is currently supported for NHWC and floating point types
103  if (act_info.enabled() && !conv2d_act_info.enabled())
104  {
105  auto a = std::make_unique<kernels::ClActivationKernel>();
106  a->configure(compile_context, dst, dst, act_info);
107  _activation_kernel = std::move(a);
108  }
109 
110  // Tune kernels
111  CLScheduler::get().tune_kernel_static(*_direct_conv_kernel);
112 }

115  const ITensorInfo *weights,
116  const ITensorInfo *biases,
117  const ITensorInfo *dst,

119  const ActivationLayerInfo &act_info)

123 
124  ARM_COMPUTE_RETURN_ON_ERROR(

126  if (act_info.enabled())

129  }

133 void ClDirectConv2d::run(ITensorPack &tensors)
134 {
135  // Run border handler
136  CLScheduler::get().enqueue_op(*_src_border_handler.get(), tensors, false);
137  // Run direct convolution
138  CLScheduler::get().enqueue_op(*_direct_conv_kernel.get(), tensors, false);
139  // Run activation kernel
140  if (_activation_kernel)
141  {
142  auto act_pack = select_activation_src_dst(tensors);
143  CLScheduler::get().enqueue_op(*_activation_kernel.get(), act_pack, false);
144  }
145 }
