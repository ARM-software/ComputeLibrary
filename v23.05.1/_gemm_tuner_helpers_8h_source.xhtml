34 bool update_padding_for_cl_image(arm_compute::ITensorInfo *tensor)
35 {
36  constexpr unsigned int num_floats_per_pixel = 4;
37 
38  const unsigned int stride_y_in_elements = tensor->strides_in_bytes()[1] / tensor->element_size();
39  const unsigned int pixel_aligment = arm_compute::get_cl_image_pitch_alignment(
40  arm_compute::CLKernelLibrary::get().get_device());
41  if(pixel_aligment == 0)
42  {
43  return false;
44  }
45  const unsigned int row_pitch_alignment = pixel_aligment * num_floats_per_pixel;
46  const unsigned int round_up_width =
47  ((stride_y_in_elements + row_pitch_alignment - 1) / row_pitch_alignment) * row_pitch_alignment;
48  const unsigned int padding = round_up_width - stride_y_in_elements;
49 
50  tensor->extend_padding(arm_compute::PaddingSize(0, padding, 0, 0));
51  return true;
52 }
