43 void fuse_convolution_with_batch_normalization(Graph &g, const Edge *output_edge)
44 {
45  ARM_COMPUTE_ERROR_ON(output_edge == nullptr);
46 
47  auto *conv_node = arm_compute::utils::cast::polymorphic_downcast<ConvolutionLayerNode *>(output_edge->producer());
48  auto *bn_node = arm_compute::utils::cast::polymorphic_downcast<BatchNormalizationLayerNode *>(output_edge->consumer());
49 
50  // Not fusing if number of groups is greater than 1
51  if(conv_node->num_groups() > 1)
52  {
53  return;
54  }
55 
56  ARM_COMPUTE_LOG_GRAPH_VERBOSE("Fusing convolution node with ID : " << output_edge->producer_id()
57  << " with BatchNormalization Layer node with ID : " << output_edge->consumer_id() << std::endl);
58 
59  // Prevent fusion if fused node has an output accessor
60  if(conv_node->output(0)->accessor() == nullptr)
61  {
62  const Target assigned_target = conv_node->assigned_target();
63 
64  // Extract conv inputs
65  const auto conv_input_id = conv_node->input_edge(0)->producer_id();
66  const auto conv_weights_id = conv_node->input_edge(1)->producer_id();
67  const auto conv_info = conv_node->convolution_info();
68  const auto conv_method = conv_node->convolution_method();
69  const auto num_groups = conv_node->num_groups();
70  const auto act_info = bn_node->fused_activation();
71  FastMathHint fast_math_hint = conv_node->fast_math_hint();
72 
73  // Extract bn inputs
74  const auto bn_mean_id = bn_node->input_edge(1)->producer_id();
75  const auto bn_var_id = bn_node->input_edge(2)->producer_id();
76  const auto bn_beta_id = bn_node->input_edge(3)->producer_id();
77  const auto bn_gamma_id = bn_node->input_edge(4)->producer_id();
78  const auto epsilon = bn_node->epsilon();
79 
80  // Create the fused node
81  const NodeID fused_id = g.add_node<FusedConvolutionBatchNormalizationNode>(epsilon, conv_info, num_groups, conv_method, fast_math_hint, act_info);
82 
83  if(conv_node->input_edge(2) != nullptr)
84  {
85  auto conv_bias_id = conv_node->input_edge(2)->producer_id();
86  g.add_connection(conv_bias_id, 0, fused_id, 2);
87  }
88 
89  // Add connections from the conv/batch_norm inputs to the fused node
90  g.add_connection(conv_input_id, 0, fused_id, 0);
91  g.add_connection(conv_weights_id, 0, fused_id, 1);
92  g.add_connection(bn_mean_id, 0, fused_id, 3);
93  g.add_connection(bn_var_id, 0, fused_id, 4);
94  g.add_connection(bn_beta_id, 0, fused_id, 5);
95  g.add_connection(bn_gamma_id, 0, fused_id, 6);
96 
97  auto fused_node = g.node(fused_id);
98  std::vector<NodeIdxPair> bn_driving_nodes = get_driving_nodes(*bn_node);
99 
100  // Extract batch normalization node accessor if any
101  auto bn_node_accessor = bn_node->output(0)->extract_accessor();
102  auto bn_node_name = bn_node->name();
103 
104  // Remove batch normalization node
105  g.remove_node(bn_node->id());
106 
107  // Get driving nodes of batch normalization node
108  for(auto &driving_node : bn_driving_nodes)
109  {
110  g.add_connection(fused_id, 0, driving_node.node_id, driving_node.index);
111  configure_tensor(fused_node->output(0));
112  }
113  // Update fused node outputs
114  fused_node->output(0)->set_accessor(std::move(bn_node_accessor));
115  fused_node->set_assigned_target(assigned_target);
116  fused_node->set_common_node_parameters(NodeParams{ conv_node->name() + "+" + bn_node_name, assigned_target });
117 
118  // Remove convolution node
119  g.remove_node(conv_node->id());
120  }
121  else
122  {
123  ARM_COMPUTE_LOG_GRAPH_VERBOSE("Prevented fusion of convolution with batch normalization due to the presence of an output accessor\n");
124  }

204 template <typename N>
205 void fuse_node_with_activation(Graph &g, const Edge *output_edge, const std::set<Activation> &supported_fused_activations)
206 {
207  ARM_COMPUTE_ERROR_ON(output_edge == nullptr);
208 
209  auto *n_node = arm_compute::utils::cast::polymorphic_downcast<N *>(output_edge->producer());
210  auto *act_node = arm_compute::utils::cast::polymorphic_downcast<ActivationLayerNode *>(output_edge->consumer());
211 
212  ARM_COMPUTE_ERROR_ON(act_node->output(0) == nullptr || n_node->output(0) == nullptr);
213 
214  // Check if activation is supported for fusion
215  if(supported_fused_activations.count(act_node->activation_info().activation()) == 0)
216  {
217  return;
218  }
219 
220  ARM_COMPUTE_LOG_GRAPH_VERBOSE("Fusing node with ID : " << output_edge->producer_id()
221  << " with Activation Layer node with ID : " << output_edge->consumer_id() << std::endl);
222 
223  // Prevent fusion if fused node has an output accessor
224  if(n_node->output(0)->accessor() == nullptr)
225  {
226  // Get driving nodes of activation node
227  std::vector<NodeIdxPair> act_driving_nodes = get_driving_nodes(*act_node);
228 
229  // Set activation info to fused node
230  n_node->set_fused_activation(act_node->activation_info());
231 
232  // Extract activation node accessor if any
233  auto act_node_accessor = act_node->output(0)->extract_accessor();
234 
235  // Remove activation node
236  g.remove_node(act_node->id());
237 
238  // Update fused node outputs
239  for(auto &driving_node : act_driving_nodes)
240  {
241  g.add_connection(n_node->id(), 0, driving_node.node_id, driving_node.index);
242  }
243 
244  // Update accessor to fused node
245  n_node->output(0)->set_accessor(std::move(act_node_accessor));
246  }
247  else
248  {
249  ARM_COMPUTE_LOG_GRAPH_VERBOSE("Prevented fusion of node with activation due to the presence of an output accessor\n");
250  }

253 template <typename N1, typename N2, typename F, typename... Args>
254 void fuse_layer(Graph &g, std::function<bool(INode &)> const &prec, const F fuse_fcn, Args &&... optional_arguments)
255 {
256  // Not interested in the order of nodes
257  for(auto &node : g.nodes())
258  {
259  // Check if the node is of type N and not a branching node
260  if(node && node->type() == N1::node_type && node->output_edges().size() == 1)
261  {
262  const auto output_edge_id = *node->output_edges().begin();
263  const auto output_edge = g.edge(output_edge_id);
264 
265  // Check if following node is an activation layer node
266  if((output_edge != nullptr) && (output_edge->consumer() != nullptr) && (output_edge->consumer()->type() == N2::node_type) && prec(*output_edge->producer()))
267  {
268  fuse_fcn(g, output_edge, optional_arguments...);
269  }
270  }
271  }

280 void NodeFusionMutator::mutate(Graph &g)
281 {
282  // Supported activations when fusing
283  const std::set<Activation> supported_fused_activations = { Activation::RELU, Activation::BOUNDED_RELU, Activation::LU_BOUNDED_RELU };
284 
285  // Preconditions
286  auto empty_prec = [](INode &)
287  {
288  return true;
289  };
290  auto qs8_prec = [&g](INode & n)
291  {
292  ARM_COMPUTE_ERROR_ON(n.output(0) == nullptr);
293 
294  const auto output_edge_id = *n.output_edges().begin();
295  const auto output_edge = g.edge(output_edge_id);
296  // To perform fusion the two nodes must have same output quantization information
297  const bool same_qinfo = n.output(0)->desc().quant_info == output_edge->producer()->output(0)->desc().quant_info;
298  const bool output_qasymm8 = n.output(0)->desc().data_type == DataType::QASYMM8;
299 
300  return (output_qasymm8 && same_qinfo) || !output_qasymm8;
301  };
302 
303  Target target = g.nodes()[0].get()->output(0)->desc().target;
304 
305  // Fusion mutations
306  detail::fuse_layer<BatchNormalizationLayerNode, ActivationLayerNode>(g, empty_prec, detail::fuse_node_with_activation<BatchNormalizationLayerNode>, supported_fused_activations);
307  detail::fuse_layer<ConvolutionLayerNode, ActivationLayerNode>(g, empty_prec, detail::fuse_node_with_activation<ConvolutionLayerNode>, supported_fused_activations);
308  detail::fuse_layer<DepthwiseConvolutionLayerNode, ActivationLayerNode>(g, qs8_prec, detail::fuse_node_with_activation<DepthwiseConvolutionLayerNode>, supported_fused_activations);
309 
310  // Currently fuse batch normalization brings performance uplift only on OpenCL with FP32 data type
311  // TODO (COMPMID-2524): Fuse batch normalization with convolution and depthwise convolution at graph level for NEON - FP32
312  // TODO (COMPMID-2581): Fuse batch normalization with convolution and depthwise convolution at graph level for OpenCL - FP16
313  if(target == Target::CL && (g.nodes()[0].get()->output(0)->desc().data_type == DataType::F32))
314  {
315  //Depthwise Convolution and Batch Normalization Fusion active only for CL
316  detail::fuse_layer<ConvolutionLayerNode, BatchNormalizationLayerNode>(g, empty_prec, detail::fuse_convolution_with_batch_normalization);
317  detail::fuse_layer<DepthwiseConvolutionLayerNode, BatchNormalizationLayerNode>(g, empty_prec, detail::fuse_depthwise_convolution_with_batch_normalization);
318  }
319 }
