73 TensorDescriptor DepthwiseConvolutionLayerNode::compute_output_descriptor(const TensorDescriptor &input_descriptor,
74  const TensorDescriptor &weights_descriptor,
75  const PadStrideInfo &info,
76  int depth_multiplier)
77 {
78  unsigned int output_width = 0;
79  unsigned int output_height = 0;
80 
81  const unsigned int input_width = get_dimension_size(input_descriptor, DataLayoutDimension::WIDTH);
82  const unsigned int input_height = get_dimension_size(input_descriptor, DataLayoutDimension::HEIGHT);
83  const unsigned int input_channels = get_dimension_size(input_descriptor, DataLayoutDimension::CHANNEL);
84  const unsigned int kernel_width = get_dimension_size(weights_descriptor, DataLayoutDimension::WIDTH);
85  const unsigned int kernel_height = get_dimension_size(weights_descriptor, DataLayoutDimension::HEIGHT);
86 
87  std::tie(output_width, output_height) = scaled_dimensions(input_width, input_height, kernel_width, kernel_height, info);
88 
89  const DataLayout data_layout = input_descriptor.layout;
90  TensorDescriptor output_descriptor = input_descriptor;
91  output_descriptor.shape.set(get_dimension_idx(data_layout, DataLayoutDimension::WIDTH), output_width);
92  output_descriptor.shape.set(get_dimension_idx(data_layout, DataLayoutDimension::HEIGHT), output_height);
93  output_descriptor.shape.set(get_dimension_idx(data_layout, DataLayoutDimension::CHANNEL), input_channels * depth_multiplier);
94 
95  return output_descriptor;
96 }

110 TensorDescriptor DepthwiseConvolutionLayerNode::configure_output(size_t idx) const
111 {
112  ARM_COMPUTE_UNUSED(idx);

115 
116  ARM_COMPUTE_ERROR_ON(src == nullptr || weights == nullptr);
117 
118  TensorDescriptor output_info = compute_output_descriptor(src->desc(), weights->desc(), _info, _depth_multiplier);
119  if(!_out_quant_info.empty())

122  }
123
