103 void validate_configuration(TensorShape shape, DataType dt1, DataType dt2, float scale, ConvertPolicy convert_policy, RoundingPolicy rounding_policy)
104 {
105  Tensor src1 = create_tensor<Tensor>(shape, dt1);
106  Tensor src2 = create_tensor<Tensor>(shape, dt2);
107  Tensor dst = create_tensor<Tensor>(shape, dt2);
108 
109  ARM_COMPUTE_EXPECT(src1.info()->is_resizable(), framework::LogLevel::ERRORS);
110  ARM_COMPUTE_EXPECT(src2.info()->is_resizable(), framework::LogLevel::ERRORS);
111  ARM_COMPUTE_EXPECT(dst.info()->is_resizable(), framework::LogLevel::ERRORS);
112 
113  // Create and configure function
114  NEPixelWiseMultiplication multiply;
115  multiply.configure(&src1, &src2, &dst, scale, convert_policy, rounding_policy);
116 
117  // Validate valid region
118  const ValidRegion valid_region = shape_to_valid_region(shape);
119  validate(src1.info()->valid_region(), valid_region);
120  validate(src2.info()->valid_region(), valid_region);
121  validate(dst.info()->valid_region(), valid_region);
122 
123  // Validate padding
124  const PaddingSize padding = PaddingCalculator(shape.x(), 16).required_padding();
125  validate(src1.info()->padding(), padding);
126  validate(src2.info()->padding(), padding);
127  validate(dst.info()->padding(), padding);
128 }
