57 Status validate_arguments(const ITensorInfo *input, const ITensorInfo *output, const PoolingLayerInfo &pool_info)
58 {
59  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(input, output);
60  DataLayout data_layout = input->data_layout();
61  ARM_COMPUTE_RETURN_ERROR_ON_F16_UNSUPPORTED(input);
62  switch(data_layout)
63  {
64  case DataLayout::NCHW:
65  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(input, 1, DataType::QASYMM8, DataType::F16, DataType::F32);
66  break;
67  case DataLayout::NHWC:
68  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(input, 1, DataType::QASYMM8, DataType::F16, DataType::F32);
69  break;
70  default:
71  ARM_COMPUTE_ERROR("Data layout not supported");
72  }
73  ARM_COMPUTE_RETURN_ERROR_ON_MSG((is_data_type_quantized_asymmetric(input->data_type()) && pool_info.pool_type() == PoolingType::L2),
74  "Unsupported combination of parameters!");
75 
76  // Checks performed when output is configured
77  if(output->total_size() != 0)
78  {
79  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input, output);
80  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_LAYOUT(input, output);
81  TensorInfo out_info(TensorInfo(compute_pool_shape(*input, pool_info), 1, output->data_type()));
82  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_SHAPES(output, &out_info);
83  }
84 
85  return Status{};
86 }

88 std::tuple<Status, Window, CLPoolingConfig> validate_and_configure_window(ITensorInfo *input, ITensorInfo *output, const PoolingLayerInfo &pool_info)
89 {
90  ARM_COMPUTE_ERROR_ON_NULLPTR(input, output);
91 
92  // Get data layout
93  const DataLayout data_layout = input->data_layout();
94  const int idx_width = get_data_layout_dimension_index(data_layout, DataLayoutDimension::WIDTH);
95  const int idx_height = get_data_layout_dimension_index(data_layout, DataLayoutDimension::HEIGHT);
96 
97  int pool_stride_x = 0;
98  int pool_stride_y = 0;
99  unsigned int pooled_w = 0;
100  unsigned int pooled_h = 0;
101  int pool_size_x = pool_info.is_global_pooling() ? input->dimension(idx_width) : pool_info.pool_size().width;
102  int pool_size_y = pool_info.is_global_pooling() ? input->dimension(idx_height) : pool_info.pool_size().height;
103  const PadStrideInfo pad_stride_info = pool_info.pad_stride_info();
104  std::tie(pool_stride_x, pool_stride_y) = pad_stride_info.stride();
105  const int pool_pad_right = pad_stride_info.pad_right();
106  const int pool_pad_top = pad_stride_info.pad_top();
107  const int pool_pad_left = pad_stride_info.pad_left();
108  const int pool_pad_bottom = pad_stride_info.pad_bottom();
109  BorderSize border_size = BorderSize(pool_pad_top, pool_pad_right, pool_pad_bottom, pool_pad_left);
110 
111  auto_init(input, output, pool_info);
112  pooled_w = output->tensor_shape()[idx_width];
113  pooled_h = output->tensor_shape()[idx_height];
114 
115  const DataType data_type = input->data_type();
116 
117  const int input_width = input->dimension(idx_width);
118  const int input_height = input->dimension(idx_height);
119 
120  unsigned int num_elems_processed_per_iteration = 0;
121  bool window_changed = false;
122  Window win{};
123  switch(data_layout)
124  {
125  case DataLayout::NCHW:
126  {
127  // Change the number of elements processed per iteration
128  // for pooling 3x3 with stride less equal than 3
129  const bool can_optimize = (pool_size_x == 3) && (pool_size_y == 3) && (pool_stride_x <= 3) && !is_data_type_quantized(data_type);
130  num_elems_processed_per_iteration = can_optimize ? 4 : 1;
131  const unsigned int num_elems_read_per_iteration = (num_elems_processed_per_iteration - 1) * pool_stride_x + pool_size_x;
132 
133  // Number of iterations in X dimension
134  const int num_iterations_x = (pooled_w + num_elems_processed_per_iteration - 1) / num_elems_processed_per_iteration;
135 
136  // Upper limit for the number of right/bottom border elements that are accessed
137  const int upper_bound_w = ((num_iterations_x - 1) * num_elems_processed_per_iteration * pool_stride_x - pool_pad_left + num_elems_read_per_iteration) - input_width;
138  const int upper_bound_h = ((pooled_h - 1) * pool_stride_y - pool_pad_top + pool_size_y) - input_height;
139 
140  border_size.right = std::max(upper_bound_w, pool_pad_right);
141  border_size.bottom = std::max(upper_bound_h, pool_pad_bottom);
142 
143  win = calculate_max_window(*output, Steps(num_elems_processed_per_iteration));
144 
145  AccessWindowRectangle input_access(input, -pool_pad_left, -pool_pad_top, num_elems_read_per_iteration, pool_size_y,
146  pool_stride_x, pool_stride_y);
147  AccessWindowHorizontal output_access(output, 0, num_elems_processed_per_iteration);
148  window_changed = update_window_and_padding(win, input_access, output_access);
149  output_access.set_valid_region(win, ValidRegion(Coordinates(), output->tensor_shape()));
150  break;
151  }
152  case DataLayout::NHWC:
153  {
154  num_elems_processed_per_iteration = 8;
155  win = calculate_max_window(*output, Steps(num_elems_processed_per_iteration));
156 
157  AccessWindowStatic input_access(input,
158  0, -1,
159  ceil_to_multiple(input->dimension(0), num_elems_processed_per_iteration), input->dimension(1));
160  AccessWindowHorizontal output_access(output, 0, num_elems_processed_per_iteration);
161  window_changed = update_window_and_padding(win, input_access, output_access);
162  output_access.set_valid_region(win, ValidRegion(Coordinates(), output->tensor_shape()));
163  break;
164  }
165  default:
166  ARM_COMPUTE_ERROR("Not implemented");
167  }
168 
169  Status err = (window_changed) ? ARM_COMPUTE_CREATE_ERROR(ErrorCode::RUNTIME_ERROR, "Insufficient Padding!") : Status{};
170  return std::make_tuple(err, win, CLPoolingConfig(num_elems_processed_per_iteration, border_size));
171 }

184 void CLPoolingLayerKernel::configure(const ICLTensor *input, ICLTensor *output, const PoolingLayerInfo &pool_info)
185 {
186  ARM_COMPUTE_ERROR_ON_NULLPTR(input, output);
187 
188  int pool_stride_x = 0;
189  int pool_stride_y = 0;
190  const PoolingType pool_type = pool_info.pool_type();
191  DataLayout data_layout = input->info()->data_layout();
192  const int idx_width = get_data_layout_dimension_index(data_layout, DataLayoutDimension::WIDTH);
193  const int idx_height = get_data_layout_dimension_index(data_layout, DataLayoutDimension::HEIGHT);
194  const int idx_channel = get_data_layout_dimension_index(data_layout, DataLayoutDimension::CHANNEL);
195  const int pool_size_x = pool_info.is_global_pooling() ? input->info()->dimension(idx_width) : pool_info.pool_size().width;
196  const int pool_size_y = pool_info.is_global_pooling() ? input->info()->dimension(idx_height) : pool_info.pool_size().height;
197  const PadStrideInfo pad_stride_info = pool_info.pad_stride_info();
198  const bool exclude_padding = pool_info.exclude_padding();
199  std::tie(pool_stride_x, pool_stride_y) = pad_stride_info.stride();
200  const int pool_pad_top = pad_stride_info.pad_top();
201  const int pool_pad_left = pad_stride_info.pad_left();
202 
203  // Set build options
204  CLBuildOptions build_opts;
205 
206  if(is_data_type_quantized_asymmetric(input->info()->data_type()) && input->info()->quantization_info() != output->info()->quantization_info())
207  {
208  const UniformQuantizationInfo iq_info = input->info()->quantization_info().uniform();
209  const UniformQuantizationInfo oq_info = output->info()->quantization_info().uniform();
210 
211  build_opts.add_option("-DOFFSET_IN1=" + float_to_string_with_full_precision(iq_info.offset));
212  build_opts.add_option("-DOFFSET_OUT=" + float_to_string_with_full_precision(oq_info.offset));
213  build_opts.add_option("-DSCALE_IN1=" + float_to_string_with_full_precision(iq_info.scale));
214  build_opts.add_option("-DSCALE_OUT=" + float_to_string_with_full_precision(oq_info.scale));
215  }
216 
217  // Check output dimensions
218  auto_init(input->info(), output->info(), pool_info);
219  ARM_COMPUTE_ERROR_THROW_ON(validate_arguments(input->info(), output->info(), pool_info));
220 
221  // Set instance variables
222  _input = input;
223  _output = output;
224  _pool_info = pool_info;
225 
226  const DataType data_type = input->info()->data_type();
227 
228  build_opts.add_option("-DDATA_TYPE=" + get_cl_type_from_data_type(data_type));
229  build_opts.add_option("-DPOOL_" + string_from_pooling_type(pool_type));
230  build_opts.add_option("-DSTRIDE_X=" + support::cpp11::to_string(pool_stride_x));
231  build_opts.add_option("-DSTRIDE_Y=" + support::cpp11::to_string(pool_stride_y));
232  build_opts.add_option("-DPAD_X=" + support::cpp11::to_string(pool_pad_left));
233  build_opts.add_option("-DPAD_Y=" + support::cpp11::to_string(pool_pad_top));
234  build_opts.add_option("-DPOOL_SIZE_X=" + support::cpp11::to_string(pool_size_x));
235  build_opts.add_option("-DPOOL_SIZE_Y=" + support::cpp11::to_string(pool_size_y));
236 
237  build_opts.add_option_if(data_type == DataType::F16, "-DFP16");
238 
239  // Create kernel
240  switch(data_layout)
241  {
242  case DataLayout::NCHW:
243  {
244  build_opts.add_option("-DMAX_WIDTH=" + support::cpp11::to_string(input->info()->dimension(idx_width) + (exclude_padding ? 0 : pool_pad_left)));
245  build_opts.add_option("-DMAX_HEIGHT=" + support::cpp11::to_string(input->info()->dimension(idx_height) + (exclude_padding ? 0 : pool_pad_top)));
246  if(pool_type != PoolingType::MAX)
247  {
248  build_opts.add_option_if(exclude_padding, "-DEXCLUDE_PADDING");
249  }
250 
251  if((pool_size_x == 3) && (pool_size_y == 3) && !is_data_type_quantized_asymmetric(data_type))
252  {
253  // Check if we have pool3x3 with stride_x less equal than 3. In these cases, run an optimized OpenCL kernel where
254  // each thread computes 4 output elements
255  const bool is_pool3x3_stride_le3 = (pool_size_x == 3) && (pool_size_y == 3) && (pool_stride_x <= 3);
256 
257  std::string kernel_name = ((is_pool3x3_stride_le3) ? "pooling_layer_optimized_" : "pooling_layer_")
258  + support::cpp11::to_string(pool_size_x);
259  _kernel = static_cast<cl::Kernel>(CLKernelLibrary::get().create_kernel(kernel_name, build_opts.options()));
260  }
261  else // Run general case
262  {
263  std::string kernel_name = is_data_type_quantized_asymmetric(data_type) ? "pooling_layer_MxN_quantized_nchw" : "pooling_layer_MxN_nchw";
264  _kernel = static_cast<cl::Kernel>(CLKernelLibrary::get().create_kernel(kernel_name, build_opts.options()));
265  }
266  break;
267  }
268  case DataLayout::NHWC:
269  {
270  build_opts.add_option_if(exclude_padding, "-DEXCLUDE_PADDING");
271  build_opts.add_option("-DMAX_WIDTH=" + support::cpp11::to_string(input->info()->dimension(idx_width)));
272  build_opts.add_option("-DMAX_HEIGHT=" + support::cpp11::to_string(input->info()->dimension(idx_height)));
273  build_opts.add_option_if(output->info()->tensor_shape().total_size_upper(3) > 1,
274  "-DDST_DEPTH=" + support::cpp11::to_string(output->info()->dimension(idx_height)));
275  std::string kernel_name = is_data_type_quantized_asymmetric(data_type) ? "pooling_layer_MxN_quantized_nhwc" : "pooling_layer_MxN_nhwc";
276  _kernel = static_cast<cl::Kernel>(CLKernelLibrary::get().create_kernel(kernel_name, build_opts.options()));
277  break;
278  }
279  default:
280  ARM_COMPUTE_ERROR("Not implemented");
281  }
282 
283  // Configure kernel window
284  auto win_config = validate_and_configure_window(input->info(), output->info(), pool_info);
285 
286  ARM_COMPUTE_ERROR_THROW_ON(std::get<0>(win_config));
287  ICLKernel::configure_internal(std::get<1>(win_config));
288 
289  if(data_layout == DataLayout::NCHW)
290  {
291  CLPoolingConfig pooling_config = std::get<2>(win_config);
292  _num_elems_processed_per_iteration = pooling_config.first;
293  _border_size = pooling_config.second;
294  }
295  else
296  {
297  _border_size = BorderSize(1, 0, 0, 0);
298  _num_elems_processed_per_iteration = 8;
299  }
300 
301  // Set config_id for enabling LWS tuning
302  _config_id = "pooling_layer_";
303  _config_id += lower_string(string_from_data_type(data_type));
304  _config_id += "_";
305  _config_id += lower_string(string_from_data_layout(data_layout));
306  _config_id += "_";
307  _config_id += support::cpp11::to_string(output->info()->dimension(idx_width));
308  _config_id += "_";
309  _config_id += support::cpp11::to_string(output->info()->dimension(idx_height));
310  _config_id += "_";
311  _config_id += support::cpp11::to_string(output->info()->dimension(idx_channel));
312  _config_id += "_";
313  _config_id += lower_string(string_from_data_layout(input->info()->data_layout()));
314 }

324 void CLPoolingLayerKernel::run(const Window &window, cl::CommandQueue &queue)
325 {
326  ARM_COMPUTE_ERROR_ON_UNCONFIGURED_KERNEL(this);
327  ARM_COMPUTE_ERROR_ON_INVALID_SUBWINDOW(IKernel::window(), window);
328 
329  unsigned int pool_stride_x = 0;
330  unsigned int pool_stride_y = 0;
331  std::tie(pool_stride_x, pool_stride_y) = _pool_info.pad_stride_info().stride();
332 
333  // Collapse window
334  Window window_collapsed = window.collapse_if_possible(ICLKernel::window(), Window::DimZ);
335 
336  switch(_input->info()->data_layout())
337  {
338  case DataLayout::NCHW:
339  {
340  Window slice = window_collapsed.first_slice_window_3D();
341  do
342  {
343  // Upsample input by pool size
344  Window in_slice(slice);
345  in_slice.set(Window::DimX, Window::Dimension(in_slice.x().start() - _pool_info.pad_stride_info().pad_left(),
346  (in_slice.x().end() - _pool_info.pad_stride_info().pad_left()) * pool_stride_x,
347  pool_stride_x * _num_elems_processed_per_iteration));
348  in_slice.set(Window::DimY, Window::Dimension(in_slice.y().start() - _pool_info.pad_stride_info().pad_top(),
349  (in_slice.y().end() - _pool_info.pad_stride_info().pad_top()) * pool_stride_y,
350  pool_stride_y));
351 
352  // Set inputs
353  unsigned int idx = 0;
354  add_3D_tensor_argument(idx, _input, in_slice);
355  add_3D_tensor_argument(idx, _output, slice);
356  enqueue(queue, *this, slice, lws_hint());
357  }
358  while(window_collapsed.slide_window_slice_3D(slice));
359  break;
360  }
361  case DataLayout::NHWC:
362  {
363  const size_t total_batches = _output->info()->tensor_shape().total_size_upper(3);
364 
365  Window slice = window_collapsed.first_slice_window_4D();
366  Window in_slice = window_collapsed.first_slice_window_4D();
367  in_slice.set(Window::DimX, Window::Dimension(0, _input->info()->dimension(0), _num_elems_processed_per_iteration));
368  in_slice.set(Window::DimY, Window::Dimension(0, _input->info()->dimension(1), pool_stride_x));
369  in_slice.set(Window::DimZ, Window::Dimension(0, _input->info()->dimension(2), pool_stride_y));
370  in_slice.set(3, Window::Dimension(0, total_batches, 1));
371  do
372  {
373  // Set inputs
374  unsigned int idx = 0;
375  add_4D_tensor_argument(idx, _input, in_slice);
376  add_4D_tensor_argument(idx, _output, slice);
377  enqueue(queue, *this, slice, lws_hint());
378  }
379  while(window.slide_window_slice_4D(slice) && window.slide_window_slice_4D(in_slice));
380  break;
381  }
382  default:
383  ARM_COMPUTE_ERROR("Not implemented");
384  }
