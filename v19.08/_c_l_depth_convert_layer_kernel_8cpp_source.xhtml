44 Status validate_arguments(const ITensorInfo *input, const ITensorInfo *output, ConvertPolicy policy, uint32_t shift)
45 {
46  ARM_COMPUTE_UNUSED(policy);
47  ARM_COMPUTE_RETURN_ERROR_ON_F16_UNSUPPORTED(input);
48  ARM_COMPUTE_RETURN_ERROR_ON(input == output);
49  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(input,
50  1,
51  DataType::U8, DataType::S8, DataType::S16,
52  DataType::U16, DataType::U32, DataType::S32, DataType::F16,
53  DataType::F32);
54  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(output,
55  1,
56  DataType::U8, DataType::S8, DataType::S16,
57  DataType::U16, DataType::U32, DataType::S32, DataType::F16,
58  DataType::F32);
59  ARM_COMPUTE_RETURN_ERROR_ON_MSG(input->data_type() == output->data_type(), "Input and output data types must be different");
60  ARM_COMPUTE_RETURN_ERROR_ON_MSG(is_data_type_float(input->data_type()) && shift != 0, "Shift is used only with integer inputs");
61  ARM_COMPUTE_RETURN_ERROR_ON(shift >= 8);
62 
63  // Validate in case of configured output
64  if(output->total_size() > 0)
65  {
66  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_SHAPES(input, output);
67  }
68 
69  return Status{};
70 }

73 void CLDepthConvertLayerKernel::configure(const ICLTensor *input, ICLTensor *output, ConvertPolicy policy, uint32_t shift)
74 {
75  ARM_COMPUTE_ERROR_ON_NULLPTR(input, output);
76 
77  // Auto initialize output shape if not initialized (We can only auto-configure the shape, datatype must be given)
78  set_shape_if_empty(*output->info(), input->info()->tensor_shape());
79 
80  ARM_COMPUTE_ERROR_THROW_ON(validate_arguments(input->info(), output->info(), policy, shift));
81 
82  // Get data sizes
83  const size_t input_size = data_size_from_type(input->info()->data_type());
84  const size_t output_size = data_size_from_type(output->info()->data_type());
85 
86  // Get number of elements to process per iterations
87  constexpr unsigned int num_elems_processed_per_iteration = 16;
88 
89  // Set build options
90  CLBuildOptions build_opts;
91  build_opts.add_option("-DVEC_SIZE=" + support::cpp11::to_string(num_elems_processed_per_iteration));
92  build_opts.add_option("-DDATA_TYPE_IN=" + get_cl_type_from_data_type(input->info()->data_type()));
93  build_opts.add_option("-DDATA_TYPE_OUT=" + get_cl_type_from_data_type(output->info()->data_type()));
94  // Conversions from float always SATURATE as out-of-bounds conversion from float->integer is implementation defined
95  build_opts.add_option_if(is_data_type_float(input->info()->data_type()) || policy == ConvertPolicy::SATURATE, "-DSATURATE");
96  build_opts.add_option_if(is_data_type_float(input->info()->data_type()) || is_data_type_float(output->info()->data_type()), "-DIS_DATA_TYPE_FLOAT");
97 
98  // Create kernel
99  const std::string kernel_name = (input_size >= output_size) ? "convert_depth_down" : "convert_depth_up";
100  _kernel = static_cast<cl::Kernel>(CLKernelLibrary::get().create_kernel(kernel_name, build_opts.options()));
101 
102  // Set shift arg
103  unsigned int idx = 2 * num_arguments_per_3D_tensor(); //Skip the input and output parameters
104  _kernel.setArg(idx++, shift);
105 
106  // Configure kernel
107  ICLSimple3DKernel::configure(input, output, num_elems_processed_per_iteration);
108 
109  // Collapse window
110  const Window &full_window = window();
111  Window collapsed_window = full_window.collapse_if_possible(full_window, Window::DimZ);
112  ICLKernel::configure_internal(collapsed_window);
113 
114  // Set config_id for enabling LWS tuning
115  _config_id = kernel_name;
116  _config_id += "_";
117  _config_id += lower_string(string_from_data_type(input->info()->data_type()));
118  _config_id += "_";
119  _config_id += support::cpp11::to_string(input->info()->dimension(0));
120  _config_id += "_";
121  _config_id += support::cpp11::to_string(input->info()->dimension(1));
122  _config_id += "_";
123  _config_id += support::cpp11::to_string(output->info()->dimension(0));
124  _config_id += "_";
125  _config_id += support::cpp11::to_string(output->info()->dimension(1));
126 }
