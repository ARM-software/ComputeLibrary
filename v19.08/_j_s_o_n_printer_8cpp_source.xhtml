50 void JSONPrinter::print_strings(T &&first, T &&last)
51 {
52  bool first_entry = true;
53  std::stringstream log;
54 
55  while(first != last)

57  print_separator(first_entry);

67 
68  *_stream << R"(")";

172 void JSONPrinter::print_measurements(const Profiler::MeasurementsMap &measurements)
173 {
174  print_separator(_first_test_entry);
175 
176  *_stream << R"("measurements" : {)";
177 
178  for(auto i_it = measurements.cbegin(), i_end = measurements.cend(); i_it != i_end;)
179  {
180  *_stream << R"(")" << i_it->first << R"(" : {)";
181 
182  auto measurement_to_string = [](const Measurement & measurement)
183  {
184  if(measurement.raw_data().size() == 1)
185  {
186  return measurement.raw_data().front();
187  }
188  else
189  {
190  std::stringstream str;
191  str << R"([")";
192  str << join(measurement.raw_data().begin(), measurement.raw_data().end(), R"(",")");
193  str << R"("])";
194  return str.str();
195  }
196  };
197  *_stream << R"("raw" : [)" << join(i_it->second.begin(), i_it->second.end(), ",", measurement_to_string) << "],";
198  *_stream << R"("unit" : ")" << i_it->second.begin()->unit() << R"(")";
199  *_stream << "}";
200 
201  if(++i_it != i_end)
202  {
203  *_stream << ",";
204  }
205  }
206 
207  *_stream << "}";
208 }
