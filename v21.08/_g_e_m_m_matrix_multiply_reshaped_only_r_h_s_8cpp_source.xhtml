160 /** Configuration test */
161 bool validate_configuration(unsigned int m_value, unsigned int n_value, unsigned int k_value, unsigned int b_value,
162  unsigned int m0_value, unsigned int n0_value, unsigned int k0_value, unsigned int h0_value,
163  bool i_value_rhs, bool t_value_rhs, bool export_to_cl_image, bool broadcast_bias, bool input_as_3d, unsigned int depth_output_gemm3d, const ActivationLayerInfo &act_info,
164  DataType dt_input0, DataType dt_input1, DataType dt_input2, DataType dt_output, float alpha, float beta)
165 {
166  const unsigned int M = m_value;
167  const unsigned int N = n_value;
168  const unsigned int K = k_value;
169 
170  GEMMLHSMatrixInfo lhs_info;
171  lhs_info.m0 = m0_value;
172  lhs_info.k0 = k0_value;
173 
174  GEMMRHSMatrixInfo rhs_info;
175  rhs_info.n0 = n0_value;
176  rhs_info.k0 = k0_value;
177  rhs_info.h0 = h0_value;
178  rhs_info.interleave = i_value_rhs;
179  rhs_info.transpose = t_value_rhs;
180  rhs_info.export_to_cl_image = export_to_cl_image;
181 
182  GEMMKernelInfo kernel_info;
183  kernel_info.m = M;
184  kernel_info.n = N;
185  kernel_info.k = K;
186  kernel_info.depth_output_gemm3d = depth_output_gemm3d;
187  kernel_info.reinterpret_input_as_3d = input_as_3d;
188  kernel_info.broadcast_bias = broadcast_bias;
189  kernel_info.activation_info = act_info;
190 
191  const TensorShape lhs_shape(K, M, b_value);
192  const TensorShape rhs_shape(N, K, b_value);
193  const TensorShape rhs_shape_reshaped = compute_rhs_reshaped_shape(TensorInfo(rhs_shape, 1, dt_input1),
194  rhs_info);
195 
196  const TensorShape dst_shape = compute_mm_shape(TensorInfo(lhs_shape, 1, dt_input0),
197  TensorInfo(rhs_shape_reshaped, 1, dt_input1),
198  kernel_info);
199 
200  const TensorShape bias_shape(N,
201  M, // Correct calculation should be: broadcast_bias? 1 : M, it's wrong here on purpose just for validation test
202  broadcast_bias? 1 : b_value);
203 
204  // Create tensor info
205  TensorInfo lhs = TensorInfo(lhs_shape, 1, dt_input0);
206  TensorInfo rhs_reshaped = TensorInfo(rhs_shape_reshaped, 1, dt_input1);
207  TensorInfo bias = TensorInfo(bias_shape, 1, dt_input2);
208  TensorInfo dst = TensorInfo(dst_shape, 1, dt_output);
209 
210  // Create and configure function
211  CLGEMMMatrixMultiplyReshapedOnlyRHS gemm;
212  return bool(gemm.validate(&lhs, &rhs_reshaped, &bias, &dst, alpha, beta, lhs_info, rhs_info, kernel_info));
213 }

227  * - Unsupported N0: NO can only be 2,3,4,8,16

229  * - Unsupported bias addition: bias broadcast mode is 0 if the input or output has to be reinterpreted as 3D

231  * - Incorrect input0 dimension when input is reinterpreted as 3D: input0->dimension(1) * input0->dimension(2) != m
