62 Status validate_arguments(const ITensorInfo *src, const ITensorInfo *dst, const PoolingLayerInfo &pool_info, const ITensorInfo *indices)
63 {
64  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(src, dst);
65  ARM_COMPUTE_RETURN_ERROR_ON_F16_UNSUPPORTED(src);
66  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(src, 1, DataType::QASYMM8, DataType::QASYMM8_SIGNED, DataType::F16, DataType::F32);
67  ARM_COMPUTE_RETURN_ERROR_ON_MSG((is_data_type_quantized_asymmetric(src->data_type()) && pool_info.pool_type == PoolingType::L2),
68  "Unsupported combination of parameters!");
69 
70  const auto data_layout = pool_info.data_layout == DataLayout::UNKNOWN ? src->data_layout() : pool_info.data_layout;
71  const int idx_width = get_data_layout_dimension_index(data_layout, DataLayoutDimension::WIDTH);
72  const int idx_height = get_data_layout_dimension_index(data_layout, DataLayoutDimension::HEIGHT);
73  const bool is_global_pooling = pool_info.is_global_pooling;
74  unsigned int pool_size_x = is_global_pooling ? src->dimension(idx_width) : pool_info.pool_size.width;
75  unsigned int pool_size_y = is_global_pooling ? src->dimension(idx_height) : pool_info.pool_size.height;
76  int output_width = 0;
77  int output_height = 0;
78  std::tie(output_width, output_height) = scaled_dimensions_signed(src->tensor_shape()[idx_width], src->tensor_shape()[idx_height],
79  pool_size_x, pool_size_y, pool_info.pad_stride_info);
80  ARM_COMPUTE_RETURN_ERROR_ON_MSG((output_width < 1 || output_height < 1), "Calculated output dimension size is invalid");
81 
82  // Check indices
83  if(indices)
84  {
85  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(src, 1, DataType::F16, DataType::F32);
86  ARM_COMPUTE_RETURN_ERROR_ON_MSG(pool_info.pool_type != PoolingType::MAX, "Pooling indices only supported for MAX pooling method");
87  ARM_COMPUTE_RETURN_ERROR_ON_MSG((pool_info.pool_size != Size2D(2, 2)), "Pooling indices only supported for pool size 2x2");
88 
89  if(indices->total_size() != 0)
90  {
91  TensorInfo idx_info(TensorInfo(compute_pool_shape(*src, pool_info), 1, DataType::U32));
92  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_SHAPES(indices, &idx_info);
93  }
94  }
95 
96  // Checks performed when dst is configured
97  if(dst->total_size() != 0)
98  {
99  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(src, dst);
100  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_LAYOUT(src, dst);
101  TensorInfo out_info(TensorInfo(compute_pool_shape(*src, pool_info), 1, dst->data_type()));
102  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_SHAPES(dst, &out_info);
103  }
104 
105  return Status{};
106 }

108 std::tuple<Status, Window, ClPoolingConfig> validate_and_configure_window(ITensorInfo *src, ITensorInfo *dst, const PoolingLayerInfo &pool_info, ITensorInfo *indices = nullptr)
109 {
110  ARM_COMPUTE_ERROR_ON_NULLPTR(src, dst);
111 
112  // Get data layout
113  const DataLayout data_layout = pool_info.data_layout == DataLayout::UNKNOWN ? src->data_layout() : pool_info.data_layout;
114  const int idx_width = get_data_layout_dimension_index(data_layout, DataLayoutDimension::WIDTH);
115  const int idx_height = get_data_layout_dimension_index(data_layout, DataLayoutDimension::HEIGHT);
116 
117  int pool_stride_x = 0;
118  int pool_stride_y = 0;
119  unsigned int pooled_w = 0;
120  unsigned int pooled_h = 0;
121  int pool_size_x = pool_info.is_global_pooling ? src->dimension(idx_width) : pool_info.pool_size.width;
122  int pool_size_y = pool_info.is_global_pooling ? src->dimension(idx_height) : pool_info.pool_size.height;
123  const PadStrideInfo pad_stride_info = pool_info.pad_stride_info;
124  std::tie(pool_stride_x, pool_stride_y) = pad_stride_info.stride();
125  const int pool_pad_right = pad_stride_info.pad_right();
126  const int pool_pad_top = pad_stride_info.pad_top();
127  const int pool_pad_left = pad_stride_info.pad_left();
128  const int pool_pad_bottom = pad_stride_info.pad_bottom();
129  BorderSize border_size = BorderSize();
130 
131  auto_init(src, dst, indices, pool_info);
132  pooled_w = dst->tensor_shape()[idx_width];
133  pooled_h = dst->tensor_shape()[idx_height];
134 
135  const DataType data_type = src->data_type();
136 
137  const int src_width = src->dimension(idx_width);
138  const int src_height = src->dimension(idx_height);
139 
140  unsigned int num_elems_processed_per_iteration = 0;
141  bool window_changed = false;
142  Window win{};
143  switch(data_layout)
144  {
145  case DataLayout::NCHW:
146  {
147  // Initialize border size
148  border_size = BorderSize(pool_pad_top, pool_pad_right, pool_pad_bottom, pool_pad_left);
149  // Change the number of elements processed per iteration
150  // for pooling 3x3 with stride less equal than 3
151  const bool can_optimize = (pool_size_x == 3) && (pool_size_y == 3) && (pool_stride_x <= 3) && !is_data_type_quantized(data_type);
152  num_elems_processed_per_iteration = can_optimize ? 4 : 1;
153  const unsigned int num_elems_read_per_iteration = (num_elems_processed_per_iteration - 1) * pool_stride_x + pool_size_x;
154 
155  // Number of iterations in X dimension
156  const int num_iterations_x = (pooled_w + num_elems_processed_per_iteration - 1) / num_elems_processed_per_iteration;
157 
158  // Upper limit for the number of right/bottom border elements that are accessed
159  const int upper_bound_w = ((num_iterations_x - 1) * num_elems_processed_per_iteration * pool_stride_x - pool_pad_left + num_elems_read_per_iteration) - src_width;
160  const int upper_bound_h = ((pooled_h - 1) * pool_stride_y - pool_pad_top + pool_size_y) - src_height;
161 
162  border_size.right = std::max(upper_bound_w, pool_pad_right);
163  border_size.bottom = std::max(upper_bound_h, pool_pad_bottom);
164 
165  win = calculate_max_window(*dst, Steps(num_elems_processed_per_iteration));
166 
167  AccessWindowRectangle src_access(src, -pool_pad_left, -pool_pad_top, num_elems_read_per_iteration, pool_size_y,
168  pool_stride_x, pool_stride_y);
169  AccessWindowHorizontal dst_access(dst, 0, num_elems_processed_per_iteration);
170 
171  // Update indices window
172  if(indices)
173  {
174  AccessWindowHorizontal indices_access(indices, 0, num_elems_processed_per_iteration);
175  window_changed = update_window_and_padding(win, src_access, dst_access, indices_access);
176  indices_access.set_valid_region(win, ValidRegion(Coordinates(), indices->tensor_shape()));
177  }
178  else
179  {
180  window_changed = update_window_and_padding(win, src_access, dst_access);
181  }
182 
183  dst_access.set_valid_region(win, ValidRegion(Coordinates(), dst->tensor_shape()));
184  break;
185  }
186  case DataLayout::NHWC:
187  {
188  const size_t vec_size = dst->data_type() == DataType::F32 ? 2 : 4;
189 
190  // Initialize border size
191  border_size = BorderSize();
192  num_elems_processed_per_iteration = adjust_vec_size(vec_size, dst->dimension(0));
193  win = calculate_max_window(*dst, Steps(num_elems_processed_per_iteration));
194  break;
195  }
196  default:
197  ARM_COMPUTE_ERROR("Not implemented");
198  }
199 
200  Status err = (window_changed) ? ARM_COMPUTE_CREATE_ERROR(ErrorCode::RUNTIME_ERROR, "Insufficient Padding!") : Status{};
201  return std::make_tuple(err, win, ClPoolingConfig(num_elems_processed_per_iteration, border_size));
202 }

215 void ClPool2dKernel::configure(const ClCompileContext &compile_context, ITensorInfo *src, ITensorInfo *dst, const PoolingLayerInfo &pool_info, ITensorInfo *indices)
216 {
217  ARM_COMPUTE_ERROR_ON_NULLPTR(src, dst);
218 
219  auto padding_info = get_padding_info({ src, dst, indices });
220 
221  // Set instance variables
222  _pool_info = pool_info;
223  _data_layout = pool_info.data_layout == DataLayout::UNKNOWN ? src->data_layout() : pool_info.data_layout;
224  int pool_stride_x = 0;
225  int pool_stride_y = 0;
226  const PoolingType pool_type = pool_info.pool_type;
227  const int idx_width = get_data_layout_dimension_index(_data_layout, DataLayoutDimension::WIDTH);
228  const int idx_height = get_data_layout_dimension_index(_data_layout, DataLayoutDimension::HEIGHT);
229  const int idx_channel = get_data_layout_dimension_index(_data_layout, DataLayoutDimension::CHANNEL);
230  const int idx_batch_size = get_data_layout_dimension_index(_data_layout, DataLayoutDimension::BATCHES);
231  const int pool_size_x = pool_info.is_global_pooling ? src->dimension(idx_width) : pool_info.pool_size.width;
232  const int pool_size_y = pool_info.is_global_pooling ? src->dimension(idx_height) : pool_info.pool_size.height;
233  const PadStrideInfo pad_stride_info = pool_info.pad_stride_info;
234  const bool exclude_padding = pool_info.exclude_padding;
235  std::tie(pool_stride_x, pool_stride_y) = pad_stride_info.stride();
236  const int pool_pad_top = pad_stride_info.pad_top();
237  const int pool_pad_left = pad_stride_info.pad_left();
238 
239  // Set build options
240  CLBuildOptions build_opts;
241  const DataType data_type = src->data_type();
242 
243  // Configure kernel window
244  auto win_config = validate_and_configure_window(src, dst, pool_info, indices);
245 
246  ARM_COMPUTE_ERROR_THROW_ON(std::get<0>(win_config));
247  ICLKernel::configure_internal(std::get<1>(win_config));
248 
249  ClPoolingConfig pooling_config = std::get<2>(win_config);
250  _num_elems_processed_per_iteration = pooling_config.first;
251  _border_size = pooling_config.second;
252 
253  build_opts.add_option("-DVEC_SIZE=" + support::cpp11::to_string(_num_elems_processed_per_iteration));
254 
255  // Tensor paddings are used to calculate the indicies for MAX pooling
256  if(pool_info.pool_size == Size2D(2, 2) && pool_type == PoolingType::MAX && indices && is_data_type_float(data_type))
257  {
258  build_opts.add_option("-DPAD_TENSOR_LEFT=" + support::cpp11::to_string(src->padding().left));
259  build_opts.add_option("-DPAD_TENSOR_RIGHT=" + support::cpp11::to_string(src->padding().right));
260  build_opts.add_option("-DPAD_TENSOR_TOP=" + support::cpp11::to_string(src->padding().top));
261  build_opts.add_option("-DPAD_TENSOR_BOTTOM=" + support::cpp11::to_string(src->padding().bottom));
262  build_opts.add_option("-DTENSOR_CHANNEL=" + support::cpp11::to_string(src->dimension(idx_channel)));
263  build_opts.add_option("-DTENSOR_WIDTH=" + support::cpp11::to_string(src->dimension(idx_width)));
264  build_opts.add_option("-DTENSOR_HEIGHT=" + support::cpp11::to_string(src->dimension(idx_height)));
265  }
266 
267  if(is_data_type_quantized_asymmetric(data_type) && src->quantization_info() != dst->quantization_info())
268  {
269  const UniformQuantizationInfo iq_info = src->quantization_info().uniform();
270  const UniformQuantizationInfo oq_info = dst->quantization_info().uniform();
271 
272  build_opts.add_option("-DOFFSET_IN1=" + float_to_string_with_full_precision(iq_info.offset));
273  build_opts.add_option("-DOFFSET_OUT=" + float_to_string_with_full_precision(oq_info.offset));
274  build_opts.add_option("-DSCALE_IN1=" + float_to_string_with_full_precision(iq_info.scale));
275  build_opts.add_option("-DSCALE_OUT=" + float_to_string_with_full_precision(oq_info.scale));
276  }
277 
278  // Check dst dimensions
279  auto_init(src, dst, indices, pool_info);
280 
281  ARM_COMPUTE_ERROR_THROW_ON(validate_arguments(src, dst, pool_info, indices));
282 
283  build_opts.add_option("-DDATA_TYPE=" + get_cl_type_from_data_type(data_type));
284  build_opts.add_option("-DPOOL_" + string_from_pooling_type(pool_type));
285  build_opts.add_option("-DSTRIDE_X=" + support::cpp11::to_string(pool_stride_x));
286  build_opts.add_option("-DSTRIDE_Y=" + support::cpp11::to_string(pool_stride_y));
287  build_opts.add_option("-DPAD_X=" + support::cpp11::to_string(pool_pad_left));
288  build_opts.add_option("-DPAD_Y=" + support::cpp11::to_string(pool_pad_top));
289  build_opts.add_option("-DPOOL_SIZE_X=" + support::cpp11::to_string(pool_size_x));
290  build_opts.add_option("-DPOOL_SIZE_Y=" + support::cpp11::to_string(pool_size_y));
291 
292  // Set the initial value for the pooling operation accordingly with the data type
293  if(pool_type == PoolingType::MAX)
294  {
295  if(is_data_type_quantized(data_type))
296  {
297  PixelValue type_min{};
298  std::tie(type_min, std::ignore) = get_min_max(data_type);
299  build_opts.add_option("-DINITIAL_VALUE=" + support::cpp11::to_string(type_min.get<int32_t>()));
300  }
301  else
302  {
303  build_opts.add_option("-DINITIAL_VALUE=" + float_to_string_with_full_precision(std::numeric_limits<float>::lowest()));
304  }
305  }
306  else
307  {
308  // Pool AVG and Pool L2 initial value
309  build_opts.add_option("-DINITIAL_VALUE=0");
310  }
311 
312  build_opts.add_option("-DMAX_WIDTH=" + support::cpp11::to_string(src->dimension(idx_width) + (exclude_padding ? 0 : pool_pad_left)));
313  build_opts.add_option("-DMAX_HEIGHT=" + support::cpp11::to_string(src->dimension(idx_height) + (exclude_padding ? 0 : pool_pad_top)));
314 
315  // Create kernel
316  switch(_data_layout)
317  {
318  case DataLayout::NCHW:
319  {
320  const auto use_fp_mixed_precision = (data_type == DataType::F16) && pool_info.fp_mixed_precision;
321  const auto use_wider_accumulator = use_fp_mixed_precision && (pool_type != PoolingType::MAX);
322  const auto acc_data_type = get_cl_type_from_data_type(use_wider_accumulator ? DataType::F32 : data_type);
323  build_opts.add_option("-DACC_DATA_TYPE=" + acc_data_type);
324  build_opts.add_option_if(use_wider_accumulator, "-DFP_MIXED_PRECISION");
325 
326  if(pool_type != PoolingType::MAX)
327  {
328  build_opts.add_option_if(exclude_padding, "-DEXCLUDE_PADDING");
329  }
330 
331  if((pool_size_x == 3) && (pool_size_y == 3) && !is_data_type_quantized_asymmetric(data_type))
332  {
333  // Check if we have pool3x3 with stride_x less equal than 3. In these cases, run an optimized OpenCL kernel where
334  // each thread computes 4 dst elements
335  const bool is_pool3x3_stride_le3 = (pool_size_x == 3) && (pool_size_y == 3) && (pool_stride_x <= 3);
336 
337  std::string kernel_name = ((is_pool3x3_stride_le3) ? "pooling_layer_optimized_" : "pooling_layer_")
338  + support::cpp11::to_string(pool_size_x);
339  _kernel = create_kernel(compile_context, kernel_name, build_opts.options());
340  }
341  else if(pool_info.pool_size == Size2D(2, 2) && pool_type == PoolingType::MAX && indices && is_data_type_float(data_type))
342  {
343  // For max pooling with pool2x2, store indicies which will be used in max unpooling
344  if(data_type == DataType::F32)
345  {
346  std::string kernel_name = "pooling_layer_2_nchw_indices_fp32";
347  _kernel = create_kernel(compile_context, kernel_name, build_opts.options());
348  }
349  else if(data_type == DataType::F16)
350  {
351  std::string kernel_name = "pooling_layer_2_nchw_indices_fp16";
352  _kernel = create_kernel(compile_context, kernel_name, build_opts.options());
353  }
354  }
355  else // Run general case
356  {
357  std::string kernel_name = is_data_type_quantized_asymmetric(data_type) ? "pooling_layer_MxN_quantized_nchw" : "pooling_layer_MxN_nchw";
358  _kernel = create_kernel(compile_context, kernel_name, build_opts.options());
359  }
360  break;
361  }
362  case DataLayout::NHWC:
363  {
364  // Floating point mixed precision is support on F16 only
365  const auto use_fp_mixed_precision = (data_type == DataType::F16) && pool_info.fp_mixed_precision && pool_type != PoolingType::MAX;
366 
367  // Wider accumulation is required to avoid accuracy loss
368  // Case 1: Floating point mixed precision (fp16 src data and fp32 accumulation)
369  // Cast 2: Quantized (int8/uint8 src data and int32 accumulation )
370  DataType acc_data_type = data_type;
371 
372  if(use_fp_mixed_precision)
373  {
374  acc_data_type = DataType::F32;
375  }
376  else if(is_data_type_quantized(data_type) && pool_type != PoolingType::MAX)
377  {
378  acc_data_type = DataType::S32;
379  }
380 
381  build_opts.add_option("-DACC_DATA_TYPE=" + get_cl_type_from_data_type(acc_data_type));
382  build_opts.add_option_if(use_fp_mixed_precision, "-DFP_MIXED_PRECISION");
383  build_opts.add_option_if(exclude_padding, "-DEXCLUDE_PADDING");
384  build_opts.add_option("-DSRC_WIDTH=" + support::cpp11::to_string(src->dimension(idx_width)));
385  build_opts.add_option("-DSRC_HEIGHT=" + support::cpp11::to_string(src->dimension(idx_height)));
386  build_opts.add_option("-DDST_HEIGHT=" + support::cpp11::to_string(dst->dimension(idx_height)));
387  build_opts.add_option("-DDST_CHANNELS=" + support::cpp11::to_string(dst->dimension(idx_channel)));
388  build_opts.add_option("-DDST_BATCH_SIZE=" + support::cpp11::to_string(dst->dimension(idx_batch_size)));
389  build_opts.add_option("-DVEC_SIZE_LEFTOVER=" + support::cpp11::to_string(src->dimension(0) % _num_elems_processed_per_iteration));
390  if(pool_info.pool_size == Size2D(2, 2) && is_data_type_float(data_type))
391  {
392  build_opts.add_option_if(indices != nullptr && pool_type == PoolingType::MAX, "-DEXTRACT_MAX_INDEX");
393 
394  std::string kernel_name = "pooling_layer_2x2_nhwc";
395  _kernel = create_kernel(compile_context, kernel_name, build_opts.options());
396  }
397  else
398  {
399  std::string kernel_name = is_data_type_quantized_asymmetric(data_type) ? "pooling_layer_MxN_quantized_nhwc" : "pooling_layer_MxN_nhwc";
400  _kernel = create_kernel(compile_context, kernel_name, build_opts.options());
401  }
402  break;
403  }
404  default:
405  ARM_COMPUTE_ERROR("Not implemented");
406  }
407 
408  // Set config_id for enabling LWS tuning
409  _config_id = "pooling_layer_";
410  _config_id += lower_string(string_from_data_type(data_type));
411  _config_id += "_";
412  _config_id += lower_string(string_from_data_layout(_data_layout));
413  _config_id += "_";
414  _config_id += support::cpp11::to_string(dst->dimension(idx_width));
415  _config_id += "_";
416  _config_id += support::cpp11::to_string(dst->dimension(idx_height));
417  _config_id += "_";
418  _config_id += support::cpp11::to_string(dst->dimension(idx_channel));
419  _config_id += "_";
420  _config_id += lower_string(string_from_data_layout(src->data_layout()));
421 
422  ARM_COMPUTE_ERROR_ON(src->data_layout() == DataLayout::NHWC && has_padding_changed(padding_info));
423 }

433 void ClPool2dKernel::run_op(ITensorPack &tensors, const Window &window, cl::CommandQueue &queue)
434 {
435  ARM_COMPUTE_ERROR_ON_UNCONFIGURED_KERNEL(this);
436  ARM_COMPUTE_ERROR_ON_INVALID_SUBWINDOW(ICLKernel::window(), window);
437 
438  unsigned int pool_stride_x = 0;
439  unsigned int pool_stride_y = 0;
440  std::tie(pool_stride_x, pool_stride_y) = _pool_info.pad_stride_info.stride();
441 
442  const auto src = utils::cast::polymorphic_downcast<const ICLTensor *>(tensors.get_const_tensor(TensorType::ACL_SRC));
443  auto dst = utils::cast::polymorphic_downcast<ICLTensor *>(tensors.get_tensor(TensorType::ACL_DST_0));
444  auto indices = utils::cast::polymorphic_downcast<ICLTensor *>(tensors.get_tensor(TensorType::ACL_DST_1));
445 
446  // Collapse window
447  Window window_collapsed = window.collapse_if_possible(ICLKernel::window(), Window::DimZ);
448 
449  switch(_data_layout)
450  {
451  case DataLayout::NCHW:
452  {
453  Window slice = window_collapsed.first_slice_window_3D();
454  do
455  {
456  // Upsample src by pool size
457  Window in_slice(slice);
458  in_slice.set(Window::DimX, Window::Dimension(in_slice.x().start() - _pool_info.pad_stride_info.pad_left(),
459  (in_slice.x().end() - _pool_info.pad_stride_info.pad_left()) * pool_stride_x,
460  pool_stride_x * _num_elems_processed_per_iteration));
461  in_slice.set(Window::DimY, Window::Dimension(in_slice.y().start() - _pool_info.pad_stride_info.pad_top(),
462  (in_slice.y().end() - _pool_info.pad_stride_info.pad_top()) * pool_stride_y,
463  pool_stride_y));
464 
465  // Set srcs
466  unsigned int idx = 0;
467  add_3D_tensor_argument(idx, src, in_slice);
468  add_3D_tensor_argument(idx, dst, slice);
469  if(indices && is_data_type_float(src->info()->data_type()) && (_pool_info.pool_size == Size2D(2, 2)))
470  {
471  add_3D_tensor_argument(idx, indices, slice);
472  }
473  enqueue(queue, *this, slice, lws_hint());
474  }
475  while(window_collapsed.slide_window_slice_3D(slice));
476  break;
477  }
478  case DataLayout::NHWC:
479  {
480  const size_t batch_size = dst->info()->tensor_shape().total_size_upper(3);
481 
482  Window slice = window_collapsed.first_slice_window_4D();
483  Window in_slice = window_collapsed.first_slice_window_4D();
484  in_slice.set(Window::DimX, Window::Dimension(0, src->info()->dimension(0), _num_elems_processed_per_iteration));
485  in_slice.set(Window::DimY, Window::Dimension(0, src->info()->dimension(1), pool_stride_x));
486  in_slice.set(Window::DimZ, Window::Dimension(0, src->info()->dimension(2), pool_stride_y));
487  in_slice.set(3, Window::Dimension(0, batch_size, 1));
488  do
489  {
490  // Set srcs
491  unsigned int idx = 0;
492  add_4D_tensor_argument(idx, src, in_slice);
493  add_4D_tensor_argument(idx, dst, slice);
494  if(indices && is_data_type_float(src->info()->data_type()) && (_pool_info.pool_type == PoolingType::MAX) && (_pool_info.pool_size == Size2D(2, 2)))
495  {
496  add_4D_tensor_argument(idx, indices, slice);
497  }
498  enqueue(queue, *this, slice, lws_hint());
499  }
500  while(window.slide_window_slice_4D(slice) && window.slide_window_slice_4D(in_slice));
501  break;
502  }
503  default:
504  ARM_COMPUTE_ERROR("Not implemented");
505  }
