39 Status validate_arguments_optimized(const ITensorInfo *src, const ITensorInfo *weights, const ITensorInfo *biases, const ITensorInfo *dst, const ConvolutionInfo &info)
40 {
41  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(src, weights, dst);
42  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(src, 1, DataType::QASYMM8, DataType::QASYMM8_SIGNED, DataType::F16, DataType::F32);
43  if(!is_data_type_quantized_per_channel(weights->data_type()))
44  {
45  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(src, weights);
46  }
47  ARM_COMPUTE_RETURN_ERROR_ON(src->data_layout() == DataLayout::UNKNOWN);
48  ARM_COMPUTE_RETURN_ERROR_ON(info.dilation.x() < 1 || info.dilation.y() < 1);
49  const size_t idx_w = get_data_layout_dimension_index(src->data_layout(), DataLayoutDimension::WIDTH);
50  const size_t idx_h = get_data_layout_dimension_index(src->data_layout(), DataLayoutDimension::HEIGHT);
51  ARM_COMPUTE_RETURN_ERROR_ON(weights->dimension(idx_w) + (weights->dimension(idx_w) - 1) * (info.dilation.x() - 1) > src->dimension(idx_w) + info.pad_stride_info.pad_left() +
52  info.pad_stride_info.pad_right());
53  ARM_COMPUTE_RETURN_ERROR_ON(weights->dimension(idx_h) + (weights->dimension(idx_h) - 1) * (info.dilation.y() - 1) > src->dimension(idx_h) + info.pad_stride_info.pad_top() +
54  info.pad_stride_info.pad_bottom());
55 
56  if(biases != nullptr)
57  {
58  const unsigned int channel_idx = get_data_layout_dimension_index(src->data_layout(), DataLayoutDimension::CHANNEL);
59  ARM_COMPUTE_RETURN_ERROR_ON(biases->num_dimensions() > 1);
60  ARM_COMPUTE_RETURN_ERROR_ON(biases->dimension(0) != weights->dimension(channel_idx));
61  }
62 
63  ARM_COMPUTE_RETURN_ON_ERROR(CpuDepthwiseConv2dAssemblyDispatch::validate(src, weights, biases, dst, info));
64 
65  // Validate Activation Layer
66  if(info.act_info.enabled() && !CpuDepthwiseConv2dAssemblyDispatch::is_activation_supported(info.act_info))
67  {
68  ARM_COMPUTE_RETURN_ON_ERROR(CpuActivation::validate(dst, nullptr, info.act_info));
69  }
70  return Status{};
71 }

74 void CpuDepthwiseConv2d::CpuDepthwiseConv2dOptimizedInternal::configure(ITensorInfo *src,
75  const ITensorInfo *weights,
76  const ITensorInfo *biases,
77  ITensorInfo *dst,
78  const ConvolutionInfo &info)
79 {
80  ARM_COMPUTE_ERROR_ON_NULLPTR(src, weights, dst);
81  // Perform validation step
82  ARM_COMPUTE_ERROR_THROW_ON(CpuDepthwiseConv2dOptimizedInternal::validate(src, weights, (biases == nullptr) ? nullptr : biases,
83  dst, info));
84 
85  _is_quantized = is_data_type_quantized_asymmetric(src->data_type());
86  _has_bias = biases != nullptr;
87  _is_nchw = src->data_layout() == DataLayout::NCHW;
88  _permute = _is_nchw;
89  _is_prepared = false;
90 
91  // Configure pipeline
92  _is_activationlayer_enabled = info.act_info.enabled() && !CpuDepthwiseConv2dAssemblyDispatch::is_activation_supported(info.act_info);
93 
94  _dwc_optimized_func = std::make_unique<CpuDepthwiseConv2dAssemblyDispatch>();
95  if(_is_nchw)
96  {
97  _permute_input = std::make_unique<cpu::CpuPermute>();
98  _permute_weights = std::make_unique<cpu::CpuPermute>();
99  _permute_output = std::make_unique<cpu::CpuPermute>();
100 
101  auto input_perm = std::make_unique<TensorInfo>();
102  auto weights_perm = std::make_unique<TensorInfo>();
103  auto output_perm = std::make_unique<TensorInfo>();
104 
105  // Configure the function to transform the input tensor from NCHW -> NHWC
106  _permute_input->configure(src, input_perm.get(), PermutationVector(2U, 0U, 1U));
107  input_perm->set_data_layout(DataLayout::NHWC);
108 
109  // Configure the function to transform the weights tensor from IHW -> HWI
110  _permute_weights->configure(weights, weights_perm.get(), PermutationVector(2U, 0U, 1U));
111  weights_perm->set_data_layout(DataLayout::NHWC);
112 
113  output_perm->set_data_layout(DataLayout::NHWC);
114  output_perm->set_quantization_info(dst->quantization_info());
115 
116  // Configure optimized depthwise
117  _dwc_optimized_func->configure(input_perm.get(), weights_perm.get(), biases, output_perm.get(), info);
118 
119  // Configure the function to transform the convoluted output to ACL's native ordering format NCHW
120  output_perm->set_data_layout(DataLayout::NHWC);
121  _permute_output->configure(output_perm.get(), dst, PermutationVector(1U, 2U, 0U));
122  }
123  else
124  {
125  _dwc_optimized_func->configure(src, weights, biases, dst, info);
126  }
127 
128  // Configure activation
129  if(_is_activationlayer_enabled)
130  {
131  _activationlayer_function = std::make_unique<cpu::CpuActivation>();
132  _activationlayer_function->configure(dst, nullptr, info.act_info);
133  }
134 }

145 void CpuDepthwiseConv2d::CpuDepthwiseConv2dOptimizedInternal::run(ITensorPack &tensors)
146 {
147  ARM_COMPUTE_ERROR_ON_MSG(tensors.empty(), "No inputs provided");
148  prepare(tensors);
149 
150  auto bias = tensors.get_const_tensor(TensorType::ACL_SRC_2);
151  auto dst = tensors.get_tensor(TensorType::ACL_DST_0);
152  auto workspace = tensors.get_tensor(TensorType::ACL_INT_3);
153  auto packed_weights = tensors.get_tensor(TensorType::ACL_INT_4);
154 
155  // Permute input
156  if(_permute)
157  {
158  ITensorPack pack;
159  auto src = tensors.get_const_tensor(TensorType::ACL_SRC_0);
160  auto src_perm = tensors.get_tensor(TensorType::ACL_INT_0);
161  pack.add_tensor(TensorType::ACL_SRC, src);
162  pack.add_tensor(TensorType::ACL_DST, src_perm);
163  _permute_input->run(pack);
164  }
165 
166  // Run assembly function
167  if(_is_nchw)
168  {
169  auto src_perm = tensors.get_tensor(TensorType::ACL_INT_0);
170  auto weights_perm = tensors.get_tensor(TensorType::ACL_INT_1);
171  auto dst_perm = tensors.get_tensor(TensorType::ACL_INT_2);
172 
173  ITensorPack pack;
174  pack.add_tensor(TensorType::ACL_SRC_0, src_perm);
175  pack.add_tensor(TensorType::ACL_SRC_1, weights_perm);
176  pack.add_tensor(TensorType::ACL_SRC_2, bias);
177  pack.add_tensor(TensorType::ACL_INT_0, workspace);
178  pack.add_tensor(TensorType::ACL_INT_1, packed_weights);
179  pack.add_tensor(TensorType::ACL_DST, dst_perm);
180  _dwc_optimized_func->run(pack);
181  }
182  else
183  {
184  auto src = tensors.get_tensor(TensorType::ACL_SRC_0);
185  auto weights = tensors.get_tensor(TensorType::ACL_SRC_1);
186  auto dst = tensors.get_tensor(TensorType::ACL_DST);
187 
188  ITensorPack pack;
189  pack.add_tensor(TensorType::ACL_SRC_0, src);
190  pack.add_tensor(TensorType::ACL_SRC_1, weights);
191  pack.add_tensor(TensorType::ACL_SRC_2, bias);
192  pack.add_tensor(TensorType::ACL_INT_0, workspace);
193  pack.add_tensor(TensorType::ACL_INT_1, packed_weights);
194  pack.add_tensor(TensorType::ACL_DST, dst);
195  _dwc_optimized_func->run(pack);
196  }
197 
198  // Permute output
199  if(_is_nchw)
200  {
201  ITensorPack pack;
202  auto dst_perm = tensors.get_tensor(TensorType::ACL_INT_2);
203  pack.add_tensor(TensorType::ACL_SRC, dst_perm);
204  pack.add_tensor(TensorType::ACL_DST, dst);
205  _permute_output->run(pack);
206  }
207 
208  // Run activation
209  if(_is_activationlayer_enabled)
210  {
211  ITensorPack pack;
212  pack.add_tensor(TensorType::ACL_SRC, dst);
213  pack.add_tensor(TensorType::ACL_DST, dst);
214  _activationlayer_function->run(pack);
215  }
216 }

218 void CpuDepthwiseConv2d::CpuDepthwiseConv2dOptimizedInternal::prepare(ITensorPack &tensors)
219 {
220  if(!_is_prepared)
221  {
222  auto weights = tensors.get_const_tensor(TensorType::ACL_SRC_1);
223  auto bias = tensors.get_const_tensor(TensorType::ACL_SRC_2);
224  auto packed_weights = tensors.get_tensor(TensorType::ACL_INT_4);
225 
226  // Permute weights
227  if(_permute)
228  {
229  auto permuted_weights = tensors.get_tensor(TensorType::ACL_INT_1);
230 
231  ITensorPack pack;
232  pack.add_tensor(TensorType::ACL_SRC, weights);
233  pack.add_tensor(TensorType::ACL_DST, permuted_weights);
234  _permute_weights->run(pack);
235 
236  weights->mark_as_unused();
237 
238  ITensorPack pack_opt;
239  pack_opt.add_const_tensor(TensorType::ACL_SRC_1, permuted_weights);
240  pack_opt.add_tensor(TensorType::ACL_SRC_2, bias);
241  pack_opt.add_tensor(TensorType::ACL_INT_1, packed_weights);
242 
243  // Prepare optimized function
244  _dwc_optimized_func->prepare(pack_opt);
245  }
246  else
247  {
248  ITensorPack pack_opt;
249  pack_opt.add_tensor(TensorType::ACL_SRC_1, weights);
250  pack_opt.add_tensor(TensorType::ACL_SRC_2, bias);
251  pack_opt.add_tensor(TensorType::ACL_INT_1, packed_weights);
252 
253  // Prepare optimized function
254  _dwc_optimized_func->prepare(pack_opt);
255  }
256 
257  _is_prepared = true;
258  }

261 void CpuDepthwiseConv2d::CpuDepthwiseConv2dGeneric::configure(ITensorInfo *src, const ITensorInfo *weights, const ITensorInfo *biases, ITensorInfo *dst, const ConvolutionInfo &info)
262 {
263  ARM_COMPUTE_ERROR_ON_NULLPTR(src, weights, dst);
264  ARM_COMPUTE_ERROR_THROW_ON(CpuDepthwiseConv2d::validate(src, weights, (biases == nullptr) ? nullptr : biases,
265  dst, info));
266 
267  _is_nchw = src->data_layout() == DataLayout::NCHW;
268  _is_prepared = !_is_nchw;
269 
270  ITensorInfo *input_to_use = src;
271  const ITensorInfo *weights_to_use = weights;
272  ITensorInfo *output_to_use = dst;
273 
274  auto input_perm = std::make_unique<TensorInfo>();
275  auto weights_perm = std::make_unique<TensorInfo>();
276  auto output_perm = std::make_unique<TensorInfo>(dst->clone()->set_is_resizable(true).reset_padding().set_tensor_shape(TensorShape()));
277 
278  if(_is_nchw)
279  {
280  _permute_input = std::make_unique<cpu::CpuPermute>();
281  _permute_weights = std::make_unique<cpu::CpuPermute>();
282 
283  _permute_input->configure(src, input_perm.get(), PermutationVector(2U, 0U, 1U));
284  input_perm->set_data_layout(DataLayout::NHWC);
285  input_to_use = input_perm.get();
286 
287  _permute_weights->configure(weights, weights_perm.get(), PermutationVector(2U, 0U, 1U));
288  weights_perm->set_data_layout(DataLayout::NHWC);
289  weights_to_use = weights_perm.get();
290 
291  output_to_use = output_perm.get();
292  }
293 
294  _depthwise_conv_kernel = std::make_unique<cpu::kernels::CpuDepthwiseConv2dNativeKernel>();
295  _depthwise_conv_kernel->configure(input_to_use, weights_to_use, biases, output_to_use, info);
296 
297  if(_is_nchw)
298  {
299  _permute_output = std::make_unique<cpu::CpuPermute>();
300  _permute_output->configure(output_perm.get(), dst, PermutationVector(1U, 2U, 0U));
301  output_perm->set_data_layout(DataLayout::NHWC);
302  }
303 
304  //Configure Activation Layer
305  _is_activationlayer_enabled = info.act_info.enabled();
306  if(_is_activationlayer_enabled)
307  {
308  _activationlayer_function = std::make_unique<cpu::CpuActivation>();
309  _activationlayer_function->configure(dst, nullptr, info.act_info);
310  }
311 }

313 Status CpuDepthwiseConv2d::CpuDepthwiseConv2dGeneric::validate(const ITensorInfo *src, const ITensorInfo *weights, const ITensorInfo *biases, const ITensorInfo *dst,
314  const ConvolutionInfo &info)
315 {
316  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(src, weights, dst);
317  if(src->data_layout() == DataLayout::NCHW)
318  {
319  TensorShape permuted_input_shape = src->tensor_shape();
320  TensorShape permuted_weights_shape = weights->tensor_shape();
321  TensorShape permuted_output_shape = misc::shape_calculator::compute_depthwise_convolution_shape(*src, *weights, info);
322  permute(permuted_input_shape, PermutationVector(2U, 0U, 1U));
323  permute(permuted_weights_shape, PermutationVector(2U, 0U, 1U));
324  permute(permuted_output_shape, PermutationVector(2U, 0U, 1U));
325 
326  const TensorInfo permuted_input = TensorInfo(src->clone()->set_is_resizable(true).reset_padding().set_tensor_shape(permuted_input_shape).set_data_layout(DataLayout::NHWC));
327  const TensorInfo permuted_weights = TensorInfo(weights->clone()->set_is_resizable(true).reset_padding().set_tensor_shape(permuted_weights_shape).set_data_layout(DataLayout::NHWC));
328  const TensorInfo permuted_output = TensorInfo(dst->clone()->set_is_resizable(true).reset_padding().set_tensor_shape(permuted_output_shape).set_data_layout(DataLayout::NCHW));
329 
330  ARM_COMPUTE_RETURN_ON_ERROR(CpuPermute::validate(src, &permuted_input, PermutationVector(2U, 0U, 1U)));
331  ARM_COMPUTE_RETURN_ON_ERROR(CpuPermute::validate(weights, &permuted_weights, PermutationVector(2U, 0U, 1U)));
332  ARM_COMPUTE_RETURN_ON_ERROR(CpuPermute::validate(&permuted_output, dst, PermutationVector(1U, 2U, 0U)));
333 
334  ARM_COMPUTE_RETURN_ON_ERROR(cpu::kernels::CpuDepthwiseConv2dNativeKernel::validate(&permuted_input, &permuted_weights, biases, &permuted_output, info));
335  }
336  else
337  {
338  ARM_COMPUTE_RETURN_ON_ERROR(cpu::kernels::CpuDepthwiseConv2dNativeKernel::validate(src, weights, biases, dst, info));
339  }
340 
341  // Validate Activation Layer
342  if(info.act_info.enabled() && !CpuDepthwiseConv2dAssemblyDispatch::is_activation_supported(info.act_info))
343  {
344  ARM_COMPUTE_RETURN_ON_ERROR(CpuActivation::validate(dst, nullptr, info.act_info));
345  }
346 
347  return Status{};
348 }

350 void CpuDepthwiseConv2d::CpuDepthwiseConv2dGeneric::run(ITensorPack &tensors)
351 {
352  auto src = tensors.get_const_tensor(TensorType::ACL_SRC_0);
353  auto weights = tensors.get_const_tensor(TensorType::ACL_SRC_1);
354  auto biases = tensors.get_const_tensor(TensorType::ACL_SRC_2);
355  auto dst = tensors.get_tensor(TensorType::ACL_DST_0);
356 
357  if(_is_nchw)
358  {
359  prepare(tensors);
360  auto src_perm = tensors.get_tensor(TensorType::ACL_INT_0);
361  auto weights_perm = tensors.get_tensor(TensorType::ACL_INT_1);
362  auto dst_perm = tensors.get_tensor(TensorType::ACL_INT_2);
363 
364  ITensorPack pack;
365  pack.add_tensor(TensorType::ACL_SRC, src);
366  pack.add_tensor(TensorType::ACL_DST, src_perm);
367  _permute_input->run(pack);
368 
369  ITensorPack pack_depth;
370  pack_depth.add_const_tensor(TensorType::ACL_SRC_0, src_perm);
371  pack_depth.add_const_tensor(TensorType::ACL_SRC_1, weights_perm);
372  pack_depth.add_tensor(TensorType::ACL_SRC_2, biases);
373  pack_depth.add_tensor(TensorType::ACL_DST, dst_perm);
374  NEScheduler::get().schedule_op(_depthwise_conv_kernel.get(), Window::DimY, _depthwise_conv_kernel->window(), pack_depth);
375  }
376  else
377  {
378  ITensorPack pack_depth;
379  pack_depth.add_tensor(TensorType::ACL_SRC_0, src);
380  pack_depth.add_tensor(TensorType::ACL_SRC_1, weights);
381  pack_depth.add_tensor(TensorType::ACL_SRC_2, biases);
382  pack_depth.add_tensor(TensorType::ACL_DST, dst);
383  NEScheduler::get().schedule_op(_depthwise_conv_kernel.get(), Window::DimY, _depthwise_conv_kernel->window(), pack_depth);
384  }
385 
386  if(_is_nchw)
387  {
388  ITensorPack pack;
389  auto dst_perm = tensors.get_tensor(TensorType::ACL_INT_2);
390  pack.add_tensor(TensorType::ACL_SRC, dst_perm);
391  pack.add_tensor(TensorType::ACL_DST, dst);
392  _permute_output->run(pack);
393  }
394 
395  if(_is_activationlayer_enabled)
396  {
397  ITensorPack pack;
398  pack.add_tensor(TensorType::ACL_SRC, dst);
399  pack.add_tensor(TensorType::ACL_DST, dst);
400  _activationlayer_function->run(pack);
401  }
402 }
