43 bool export_weights_to_cl_image_heuristic(const ITensorInfo *weights, unsigned int depth_multiplier, GPUTarget gpu_target)
44 {
45  if(!export_weights_to_cl_image(weights))
46  {
47  return false;
48  }
49 
50  const size_t idx_w = get_data_layout_dimension_index(weights->data_layout(), DataLayoutDimension::WIDTH);
51  const size_t idx_h = get_data_layout_dimension_index(weights->data_layout(), DataLayoutDimension::HEIGHT);
52  const size_t kernel_w = weights->tensor_shape()[idx_w];
53  const size_t kernel_h = weights->tensor_shape()[idx_h];
54 
55  if((kernel_w == 1) && (kernel_h == 1))
56  {
57  return false;
58  }
59 
60  if(depth_multiplier > 1)
61  {
62  return false;
63  }
64 
65  if(gpu_target == GPUTarget::G71 || get_arch_from_target(gpu_target) == GPUTarget::MIDGARD)
66  {
67  return false;
68  }
69 
70  return true;
71 }

73 void initialize_dwc_native_compute_info(DWCComputeKernelInfo &dwc_compute_info, const ITensorInfo *weights, const PadStrideInfo &conv_info, const Size2D &dilation, unsigned int depth_multiplier,
74  GPUTarget gpu_target)
75 {
76  if(!is_data_type_float(weights->data_type()))
77  {
78  dwc_compute_info.export_weights_to_cl_image = false;
79  dwc_compute_info.n0 = (depth_multiplier == 1) ? 4 : 1;
80  if(conv_info.stride().first == 1 && dilation.x() == 1 && depth_multiplier == 1)
81  {
82  dwc_compute_info.m0 = 2;
83  }
84  else
85  {
86  dwc_compute_info.m0 = 1;
87  }
88 
89  return;
90  }
91 
92  // Floating point path
93 
94  // First check if we can export to cl_image.
95  dwc_compute_info.export_weights_to_cl_image = export_weights_to_cl_image_heuristic(weights, depth_multiplier, gpu_target);
96 
97  // Set n0
98  if(depth_multiplier == 1)
99  {
100  if(dwc_compute_info.export_weights_to_cl_image == false && weights->data_type() == DataType::F16)
101  {
102  dwc_compute_info.n0 = 8;
103  }
104  else
105  {
106  dwc_compute_info.n0 = 4;
107  }
108  }
109  else
110  {
111  dwc_compute_info.n0 = 1;
112  }
113 
114  dwc_compute_info.n0 = adjust_vec_size(dwc_compute_info.n0, weights->dimension(0));
115 
116  // Set m0 only if stride_x == 1 and dilation_x == 1
117  if(conv_info.stride().first == 1 && dilation.x() == 1)
118  {
119  const size_t idx_w = get_data_layout_dimension_index(weights->data_layout(), DataLayoutDimension::WIDTH);
120  const size_t kernel_w = weights->tensor_shape()[idx_w];
121 
122  dwc_compute_info.m0 = (kernel_w >= 9) || (kernel_w == 1) ? 1 : 2;
123  }
124  else
125  {
126  dwc_compute_info.m0 = 1;
127  }
128  return;
129 }

161 void CLDepthwiseConvolutionLayer::configure(const CLCompileContext &compile_context, ICLTensor *input, const ICLTensor *weights, const ICLTensor *biases,
162  ICLTensor *output, const PadStrideInfo &conv_info,
163  unsigned int depth_multiplier, ActivationLayerInfo act_info, const Size2D &dilation)
164 {
165  ARM_COMPUTE_ERROR_ON_NULLPTR(input, weights);
166  ARM_COMPUTE_ERROR_THROW_ON(CLDepthwiseConvolutionLayer::validate(input->info(),
167  weights->info(),
168  biases != nullptr ? biases->info() : nullptr,
169  output != nullptr ? output->info() : input->info(),
170  conv_info,
171  depth_multiplier,
172  act_info,
173  dilation));
174 
175  _is_quantized = is_data_type_quantized(input->info()->data_type());
176  _is_prepared = false;
177  _original_weights = weights;
178  _input = input;
179  _output = output;
180  _needs_permute = input->info()->data_layout() == DataLayout::NCHW;
181 
182  const GPUTarget gpu_target = CLScheduler::get().target();
183 
184  ICLTensor *input_to_use = input;
185  const ICLTensor *weights_to_use = weights;
186  ICLTensor *output_to_use = output;
187  if(_needs_permute)
188  {
189  _memory_group.manage(&_permuted_input);
190  _memory_group.manage(&_permuted_output);
191 
192  // Configure the function to transform the input tensor from NCHW -> NHWC
193  _permute_input_to_nhwc.configure(compile_context, input, &_permuted_input, PermutationVector(2U, 0U, 1U));
194  _permuted_input.info()->set_data_layout(DataLayout::NHWC);
195 
196  // Configure the function to transform the weights tensor from IHW -> HWI
197  _permute_weights_to_nhwc.configure(compile_context, weights, &_permuted_weights, PermutationVector(2U, 0U, 1U));
198  _permuted_weights.info()->set_data_layout(DataLayout::NHWC);
199 
200  // Set output quantization info before dwc kernel configure
201  _permuted_output.info()->set_quantization_info(output->info()->quantization_info());
202 
203  input_to_use = &_permuted_input;
204  weights_to_use = &_permuted_weights;
205  output_to_use = &_permuted_output;
206  }
207 
208  CLTensor *output_multipliers_to_use = nullptr;
209  CLTensor *output_shifts_to_use = nullptr;
210  if(_is_quantized)
211  {
212  const size_t idx_c = get_data_layout_dimension_index(weights->info()->data_layout(), DataLayoutDimension::CHANNEL);
213  const size_t num_filters = (is_data_type_quantized_per_channel(weights->info()->data_type())) ? weights->info()->dimension(idx_c) : 1;
214 
215  _output_multipliers.allocator()->init(TensorInfo(TensorShape(num_filters), 1, DataType::S32));
216  _output_shifts.allocator()->init(TensorInfo(TensorShape(num_filters), 1, DataType::S32));
217 
218  output_multipliers_to_use = &_output_multipliers;
219  output_shifts_to_use = &_output_shifts;
220  }
221 
222  DWCComputeKernelInfo dwc_native_compute_info;
223  initialize_dwc_native_compute_info(dwc_native_compute_info, weights_to_use->info(), conv_info, dilation, depth_multiplier, gpu_target);
224 
225  const ConvolutionInfo conv_kernel_info{ conv_info, depth_multiplier, act_info, dilation };
226 
227  _dwc_native_kernel->configure(compile_context, input_to_use, weights_to_use, biases, output_to_use,
228  dwc_native_compute_info, conv_kernel_info, output_multipliers_to_use, output_shifts_to_use);
229 
230  if(_needs_permute)
231  {
232  _permuted_input.allocator()->allocate();
233 
234  // Configure the function to transform the convoluted output to NCHW format
235  _permuted_output.info()->set_data_layout(DataLayout::NCHW);
236  _permute_output_to_nchw.configure(compile_context, &_permuted_output, output, PermutationVector(1U, 2U, 0U));
237  _permuted_output.allocator()->allocate();
238  }
239 
240  if(_is_quantized)
241  {
242  _output_multipliers.allocator()->allocate();
243  _output_shifts.allocator()->allocate();
244  }
245 }

247 Status CLDepthwiseConvolutionLayer::validate(const ITensorInfo *input, const ITensorInfo *weights, const ITensorInfo *biases, const ITensorInfo *output,
248  const PadStrideInfo &conv_info,
249  unsigned int depth_multiplier, ActivationLayerInfo act_info, const Size2D &dilation)
250 {
251  const bool in_place = input == output || output == nullptr;
252  if(in_place)
253  {
254  output = input;
255  }
256  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_LAYOUT(input, output);
257  const size_t idx_w = get_data_layout_dimension_index(input->data_layout(), DataLayoutDimension::WIDTH);
258  const size_t idx_h = get_data_layout_dimension_index(input->data_layout(), DataLayoutDimension::HEIGHT);
259 
260  ARM_COMPUTE_RETURN_ERROR_ON(weights->dimension(idx_w) + (weights->dimension(idx_w) - 1) * (dilation.x() - 1) > input->dimension(idx_w) + conv_info.pad_left() + conv_info.pad_right());
261  ARM_COMPUTE_RETURN_ERROR_ON(weights->dimension(idx_h) + (weights->dimension(idx_h) - 1) * (dilation.y() - 1) > input->dimension(idx_h) + conv_info.pad_top() + conv_info.pad_bottom());
262 
263  const GPUTarget gpu_target = CLScheduler::get().target();
264 
265  const ConvolutionInfo conv_kernel_info{ conv_info, depth_multiplier, act_info, dilation };
266 
267  const bool needs_permute = input->data_layout() == DataLayout::NCHW;
268 
269  const bool is_quantized = is_data_type_quantized(input->data_type());
270 
271  TensorInfo output_multipliers_shifts_info(TensorInfo(TensorShape(1U), 1, DataType::S32));
272  if(is_quantized)
273  {
274  if(is_data_type_quantized_per_channel(weights->data_type()))
275  {
276  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(weights, 1, DataType::QSYMM8_PER_CHANNEL);
277 
278  const size_t idx_c = get_data_layout_dimension_index(weights->data_layout(), DataLayoutDimension::CHANNEL);
279  output_multipliers_shifts_info.set_tensor_shape(TensorShape(weights->dimension(idx_c)));
280  }
281  else
282  {
283  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input, weights);
284  }
285  }
286 
287  if(needs_permute)
288  {
289  ARM_COMPUTE_RETURN_ERROR_ON_MSG(in_place, "In-place is supported only with NHWC data layout");
290  TensorShape permuted_input_shape = input->tensor_shape();
291  TensorShape permuted_weights_shape = weights->tensor_shape();
292  const ConvolutionInfo info{ conv_info, depth_multiplier, ActivationLayerInfo(), dilation };
293  TensorShape permuted_output_shape = shape_calculator::compute_depthwise_convolution_shape(*input, *weights, info);
294 
295  permute(permuted_input_shape, PermutationVector(2U, 0U, 1U));
296  permute(permuted_weights_shape, PermutationVector(2U, 0U, 1U));
297  permute(permuted_output_shape, PermutationVector(2U, 0U, 1U));
298 
299  const TensorInfo permuted_input = input->clone()->set_is_resizable(true).reset_padding().set_tensor_shape(permuted_input_shape).set_data_layout(DataLayout::NHWC);
300  const TensorInfo permuted_weights = weights->clone()->set_is_resizable(true).reset_padding().set_tensor_shape(permuted_weights_shape).set_data_layout(DataLayout::NHWC);
301  const TensorInfo permuted_output = output->clone()->set_is_resizable(true).reset_padding().set_tensor_shape(permuted_output_shape).set_data_layout(DataLayout::NHWC);
302 
303  ARM_COMPUTE_RETURN_ON_ERROR(CLPermute::validate(input, &permuted_input, PermutationVector(2U, 0U, 1U)));
304  ARM_COMPUTE_RETURN_ON_ERROR(CLPermute::validate(weights, &permuted_weights, PermutationVector(2U, 0U, 1U)));
305 
306  DWCComputeKernelInfo dwc_native_compute_info;
307  initialize_dwc_native_compute_info(dwc_native_compute_info, &permuted_weights, conv_info, dilation, depth_multiplier, gpu_target);
308 
309  ARM_COMPUTE_RETURN_ON_ERROR(CLDepthwiseConvolutionLayerNativeKernel::validate(&permuted_input, &permuted_weights, biases, &permuted_output,
310  dwc_native_compute_info, conv_kernel_info, &output_multipliers_shifts_info, &output_multipliers_shifts_info));
311  ARM_COMPUTE_RETURN_ON_ERROR(CLPermute::validate(&permuted_output, output, PermutationVector(1U, 2U, 0U)));
312  }
313  else
314  {
315  DWCComputeKernelInfo dwc_native_compute_info;
316  initialize_dwc_native_compute_info(dwc_native_compute_info, weights, conv_info, dilation, depth_multiplier, gpu_target);
317  ARM_COMPUTE_RETURN_ON_ERROR(CLDepthwiseConvolutionLayerNativeKernel::validate(input, weights, biases, output, dwc_native_compute_info, conv_kernel_info, &output_multipliers_shifts_info,
318  &output_multipliers_shifts_info));
319  }
320  return Status{};
321 }

323 void CLDepthwiseConvolutionLayer::run()
324 {
325  prepare();
326 
327  MemoryGroupResourceScope scope_mg(_memory_group);
328 
329  if(_needs_permute)
330  {
331  _permute_input_to_nhwc.run();

340 void CLDepthwiseConvolutionLayer::prepare()
341 {
342  if(!_is_prepared)
343  {
344  if(_is_quantized)
345  {
346  _output_multipliers.map();
347  _output_shifts.map();
348  quantization::compute_quantized_multipliers_and_shifts(_input->info(),
349  _original_weights->info(),
350  _output != nullptr ? _output->info() : _input->info(),
351  reinterpret_cast<int32_t *>(_output_multipliers.ptr_to_element(Coordinates(0))),
352  reinterpret_cast<int32_t *>(_output_shifts.ptr_to_element(Coordinates(0))));
353  _output_multipliers.unmap();
354  _output_shifts.unmap();
355  }
356 
357  if(_needs_permute)
358  {
359  ARM_COMPUTE_ERROR_ON(!_original_weights->is_used());
360 
361  _permuted_weights.allocator()->allocate();
362  _permute_weights_to_nhwc.run();
363  _original_weights->mark_as_unused();
364  }
365  _is_prepared = true;
366  }
