214 std::unique_ptr<depthwise::IDepthwiseConvolution> create_convolver(const ITensorInfo *input,
215  const ITensorInfo *weights,
216  ITensorInfo *output,
217  const ConvolutionInfo &info)
218 {
219  const DataType data_type = input->data_type();
220  const TensorShape shape = input->tensor_shape();
221 
222  const int n_batches = shape[3];
223  const int in_rows = shape.z();
224  const int in_cols = shape.y();
225  const int n_channels = shape.x();
226  const int dilation_factor = info.dilation.x();
227  const int padding_top = info.pad_stride_info.pad_top();
228  const int padding_left = info.pad_stride_info.pad_left();
229  const int padding_bottom = info.pad_stride_info.pad_bottom();
230  const int padding_right = info.pad_stride_info.pad_right();
231 
232  const bool is_uniform_quantized = (data_type == DataType::QASYMM8) && (weights->data_type() == DataType::QASYMM8);
233  const bool is_perchannel_quantized = (data_type == DataType::QASYMM8) && (weights->data_type() == DataType::QSYMM8_PER_CHANNEL);
234 
235  const unsigned int stride_x = info.pad_stride_info.stride().first;
236  const unsigned int kernel_size = weights->tensor_shape().y();
237 
238  // Map activation function
239  neon_convolution_kernels::ActivationFunction activation = neon_convolution_kernels::ActivationFunction::None;
240  if(arm_compute::utils::info_helpers::is_relu(info.act_info))
241  {
242  activation = neon_convolution_kernels::ActivationFunction::ReLU;
243  }
244  else if(arm_compute::utils::info_helpers::is_relu6(info.act_info))
245  {
246  activation = neon_convolution_kernels::ActivationFunction::ReLU6;
247  }
248 
249  // Create quantized convolver
250  if(is_uniform_quantized)
251  {
252  const UniformQuantizationInfo input_qinfo = input->quantization_info().uniform();
253  const UniformQuantizationInfo weights_qinfo = weights->quantization_info().uniform();
254  const UniformQuantizationInfo output_qinfo = output->quantization_info().uniform();
255 
256  // Check that quantization info are in the range [0, 255]
257  ARM_COMPUTE_ERROR_ON(input_qinfo.offset < 0 || input_qinfo.offset > 255);
258  ARM_COMPUTE_ERROR_ON(weights_qinfo.offset < 0 || weights_qinfo.offset > 255);
259  ARM_COMPUTE_ERROR_ON(output_qinfo.offset < 0 || output_qinfo.offset > 255);
260  const qasymm8::QAsymm8Params iqinfo{ static_cast<uint8_t>(input_qinfo.offset), input_qinfo.scale };
261  const qasymm8::QAsymm8Params wqinfo{ static_cast<uint8_t>(weights_qinfo.offset), weights_qinfo.scale };
262  const qasymm8::QAsymm8Params oqinfo{ static_cast<uint8_t>(output_qinfo.offset), output_qinfo.scale };
263 
264  // Calculate rescale parameters
265  const float fmultipler = iqinfo.scale * wqinfo.scale / oqinfo.scale;
266  int32_t qmultiplier = 0;
267  int32_t qshift = 0;
268  quantization::calculate_quantized_multiplier_less_than_one(fmultipler, &qmultiplier, &qshift);
269  qasymm8::QAsymm8RescaleParams rescale_params(qshift, qmultiplier, fmultipler);
270 
271  return get_qasymm8_convolver(kernel_size, stride_x, n_batches, in_rows, in_cols, n_channels, dilation_factor, activation,
272  wqinfo, iqinfo, oqinfo, rescale_params, padding_top, padding_left, padding_bottom, padding_right);
273  }
274  else if(is_perchannel_quantized)
275  {
276  const UniformQuantizationInfo input_qinfo = input->quantization_info().uniform();
277  const QuantizationInfo weights_qinfo = weights->quantization_info();
278  const UniformQuantizationInfo output_qinfo = output->quantization_info().uniform();
279 
280  // Check that quantization info are in the range [0, 255]
281  ARM_COMPUTE_ERROR_ON(input_qinfo.offset < 0 || input_qinfo.offset > 255);
282  ARM_COMPUTE_ERROR_ON(output_qinfo.offset < 0 || output_qinfo.offset > 255);
283  const qasymm8::QAsymm8Params iqinfo{ static_cast<uint8_t>(input_qinfo.offset), input_qinfo.scale };
284  const qsymm8::QSymm8PerChannelParams wqinfo{ weights_qinfo.scale() };
285  const qasymm8::QAsymm8Params oqinfo{ static_cast<uint8_t>(output_qinfo.offset), output_qinfo.scale };
286 
287  // Calculate rescale parameters
288  std::vector<float> fmultipliers;
289  std::vector<int32_t> qmultipliers;
290  std::vector<int32_t> qshifts;
291 
292  for(auto const s : wqinfo.scales)
293  {
294  const float fmultipler = iqinfo.scale * s / oqinfo.scale;
295  int32_t qmultiplier = 0;
296  int32_t qshift = 0;
297  quantization::calculate_quantized_multiplier_less_than_one(fmultipler, &qmultiplier, &qshift);
298  fmultipliers.push_back(fmultipler);
299  qmultipliers.push_back(qmultiplier);
300  qshifts.push_back(qshift);
301  }
302 
303  qsymm8::QSymm8PerChannelRescaleParams rescale_params(qshifts, qmultipliers, fmultipliers);
304 
305  return get_qsymm8_perchannel_convolver(kernel_size, stride_x, n_batches, in_rows, in_cols, n_channels, activation,
306  wqinfo, iqinfo, oqinfo, rescale_params, padding_top, padding_left, padding_bottom, padding_right);
307  }
308  else
309  {
310  // Create float convolver
311  switch(data_type)
312  {
313 #ifdef __ARM_FEATURE_FP16_VECTOR_ARITHMETIC
314  case DataType::F16:
315  {
316  return get_fp16_convolver(kernel_size, stride_x, n_batches, in_rows, in_cols, n_channels, dilation_factor, activation, padding_top, padding_left, padding_bottom, padding_right);
317  }
318 #endif // __ARM_FEATURE_FP16_VECTOR_ARITHMETIC
319  case DataType::F32:
320  {
321  return get_fp32_convolver(kernel_size, stride_x, n_batches, in_rows, in_cols, n_channels, dilation_factor, activation, padding_top, padding_left, padding_bottom, padding_right);
322  }
323  default:
324  return nullptr;
325  }
326  }
327 }

347 void CpuDepthwiseConvolutionAssemblyDispatch::configure(const ITensorInfo *input,
348  const ITensorInfo *weights,
349  const ITensorInfo *bias,
350  ITensorInfo *output,
351  const ConvolutionInfo &info)
352 {
353  ARM_COMPUTE_ERROR_ON_NULLPTR(input, weights, output);
354  ARM_COMPUTE_UNUSED(bias);
355  ARM_COMPUTE_ERROR_THROW_ON(CpuDepthwiseConvolutionAssemblyDispatch::validate(input,
356  weights,
357  bias != nullptr ? bias : nullptr,
358  output,
359  info));
360 
361  // Output auto inizialitation if not yet initialized
362  const TensorShape output_shape = misc::shape_calculator::compute_depthwise_convolution_shape(*input, *weights, info);
363  auto_init_if_empty(*output, input->clone()->set_is_resizable(true).reset_padding().set_tensor_shape(output_shape).set_quantization_info(output->quantization_info()));
364 
365  _pImpl->is_prepared = false;
366 
367  // Create convolver
368  _pImpl->dwc_assembly_kernel = create_convolver(input, weights, output, info);
369  ARM_COMPUTE_ERROR_ON(_pImpl->dwc_assembly_kernel == nullptr);
370 
371  // Create assembly kernel wrapper
372  _pImpl->dwc_acl_kernel.configure(_pImpl->dwc_assembly_kernel.get());
373 
374  constexpr size_t alignment = 128;
375 
376  // Create workspace
377  const unsigned int num_threads = NEScheduler::get().num_threads();
378  const size_t workspace_size = _pImpl->dwc_assembly_kernel->get_working_space_size(num_threads);
379  ARM_COMPUTE_ERROR_ON_MSG(workspace_size == 0, "Workspace size cannot be 0 !");
380  _pImpl->mem_req.push_back({ TensorType::ACL_INT_0, workspace_size, alignment });
381 
382  // Create packing tensor
383  const size_t pack_tensor_size = _pImpl->dwc_assembly_kernel->get_packed_params_size();
384  ARM_COMPUTE_ERROR_ON_MSG(pack_tensor_size == 0, "Pack tensor size cannot be 0 !");
385 
386  _pImpl->mem_req.push_back({ TensorType::ACL_INT_1, pack_tensor_size, alignment });
387 }

394 Status CpuDepthwiseConvolutionAssemblyDispatch::validate(const ITensorInfo *input,
395  const ITensorInfo *weights,
396  const ITensorInfo *bias,
397  const ITensorInfo *output,
398  const ConvolutionInfo &info)
399 {
400  ARM_COMPUTE_RETURN_ERROR_ON_CPU_F16_UNSUPPORTED(input);
401  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(input, 1, DataType::QASYMM8, DataType::F16, DataType::F32);
402  if(weights->data_type() != DataType::QSYMM8_PER_CHANNEL)
403  {
404  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input, weights);
405  }
406  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_LAYOUT(input, weights);
407 
408  // Validate convolver
409  ARM_COMPUTE_RETURN_ERROR_ON(!is_optimized_supported(input, weights, info));
410 
411  // Validate activation
412  const bool is_relu = arm_compute::utils::info_helpers::is_relu(info.act_info);
413  const bool is_relu6 = arm_compute::utils::info_helpers::is_relu6(info.act_info);
414  ARM_COMPUTE_RETURN_ERROR_ON(info.act_info.enabled() && !(is_relu || is_relu6));
415 
416  // Check bias
417  if(bias != nullptr)
418  {
419  unsigned int channel_idx = get_data_layout_dimension_index(input->data_layout(), DataLayoutDimension::CHANNEL);
420  ARM_COMPUTE_RETURN_ERROR_ON(bias->num_dimensions() > 1);
421  ARM_COMPUTE_RETURN_ERROR_ON(bias->dimension(0) != weights->dimension(channel_idx));
422  }
423 
424  // Check output
425  if(output->total_size() != 0)
426  {
427  const TensorShape output_shape = misc::shape_calculator::compute_depthwise_convolution_shape(*input, *weights, info);
428  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DIMENSIONS(output->tensor_shape(), output_shape);
429  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input, output);
430  }
431 
432  // The uniform quantization case will only have 1 scale value in the weights quantization info
433  const UniformQuantizationInfo input_qinfo = input->quantization_info().uniform();
434  const QuantizationInfo weights_qinfo = weights->quantization_info();
435  const UniformQuantizationInfo output_qinfo = output->quantization_info().uniform();
436  for(auto const s : weights_qinfo.scale())
437  {
438  const float fmultipler = input_qinfo.scale * s / output_qinfo.scale;
439  ARM_COMPUTE_RETURN_ERROR_ON(fmultipler > 1.f);
440  }
441 
442  return Status{};
443 }

445 bool CpuDepthwiseConvolutionAssemblyDispatch::is_optimized_supported(const ITensorInfo *input,
446  const ITensorInfo *weights,
447  const ConvolutionInfo &info)
448 {
449  ARM_COMPUTE_ERROR_ON_NULLPTR(input, weights);
450 
451  // Reshape input shape if in NHWC format
452  const DataLayout data_layout = input->data_layout();
453  TensorShape in_shape{ input->tensor_shape() };
454  if(data_layout == DataLayout::NHWC)
455  {
456  in_shape.set(Window::DimX, input->tensor_shape().y());
457  in_shape.set(Window::DimY, input->tensor_shape().z());
458  in_shape.set(Window::DimZ, input->tensor_shape().x());
459  }
460 
461  // Check data type
462  const DataType input_type = input->data_type();
463  const bool is_input_type_valid = is_data_type_float(input_type) || input_type == DataType::QASYMM8;
464  const DataType weights_type = weights->data_type();
465  const bool is_weights_type_valid = is_data_type_float(weights_type) || weights_type == DataType::QASYMM8 || weights_type == DataType::QASYMM8_SIGNED
466  || weights_type == DataType::QSYMM8_PER_CHANNEL;
467 
468  // Check weighs size
469  std::set<unsigned int> supported_kernel_sizes = { 3, 5 };
470  const unsigned int width_idx = get_data_layout_dimension_index(data_layout, DataLayoutDimension::WIDTH);
471  const unsigned int height_idx = get_data_layout_dimension_index(data_layout, DataLayoutDimension::HEIGHT);
472  const unsigned int kernel_w = weights->dimension(width_idx);
473  const unsigned int kernel_h = weights->dimension(height_idx);
474  bool weights_supported = (kernel_w == kernel_h) && (supported_kernel_sizes.count(kernel_w) != 0);
475 
476  // Check for supported strides
477  const auto &strides = info.pad_stride_info.stride();
478  bool supported_strides = (strides.first == strides.second) && ((strides.first == 1) || (strides.first == 2));
479 
480  // Check for supported padding
481  const auto pad_top = info.pad_stride_info.pad_top();
482  const auto pad_right = info.pad_stride_info.pad_right();
483  const auto pad_bottom = info.pad_stride_info.pad_bottom();
484  const auto pad_left = info.pad_stride_info.pad_left();
485  PadStrideInfo same_pad = calculate_same_pad(in_shape, TensorShape(kernel_w, kernel_h), info.pad_stride_info, DataLayout::NCHW, info.dilation);
486  bool is_same_padding = (pad_top == same_pad.pad_top()) && (pad_right == same_pad.pad_right()) && (pad_bottom == same_pad.pad_bottom()) && (pad_left == same_pad.pad_left());
487  bool is_valid_padding = (pad_top == 0) && (pad_right == 0) && (pad_bottom == 0) && (pad_left == 0);
488  bool supported_padding = is_same_padding || is_valid_padding;
489  // TODO(COMPMID-2464): Enable once dilated conv with stride 2 is supported
490  bool is_dilation_supported = ((info.dilation == Size2D(1U, 1U)) || ((info.dilation.x() == info.dilation.y()) && strides.first == 1));
491 
492  if(weights_type == DataType::QSYMM8_PER_CHANNEL)
493  {
494  is_dilation_supported = is_dilation_supported && (info.dilation == Size2D(1U, 1U));
495  }
496 
497  return is_input_type_valid && is_weights_type_valid && weights_supported && supported_strides && supported_padding && (info.depth_multiplier == 1) && is_dilation_supported;
498 }
