64 void ClFloorKernel::configure(const ClCompileContext &compile_context, const ITensorInfo *src, ITensorInfo *dst)
65 {
66  ARM_COMPUTE_ERROR_ON_NULLPTR(src, dst);
67 
68  // Auto initialize output
69  auto_init_if_empty(*dst, src->tensor_shape(), 1, src->data_type());
70 
71  // Validate
72  ARM_COMPUTE_ERROR_THROW_ON(validate_arguments(src, dst));
73  auto padding_info = get_padding_info({ src, dst });
74 
75  const unsigned int vec_size_x = adjust_vec_size(max_cl_vector_width / src->element_size(), src->dimension(0));
76  const int vec_size_x_leftovers = src->dimension(0) % vec_size_x;
77  CLBuildOptions build_opts;
78  build_opts.add_option("-DDATA_TYPE=" + get_cl_type_from_data_type(src->data_type()));
79  build_opts.add_option("-DVEC_SIZE=" + support::cpp11::to_string(vec_size_x));
80  build_opts.add_option("-DVEC_SIZE_LEFTOVER=" + support::cpp11::to_string(vec_size_x_leftovers));
81 
82  // Create kernel
83  _kernel = create_kernel(compile_context, "floor_layer", build_opts.options());
84 
85  // Configure kernel window
86  Window win = calculate_max_window(*src, Steps(vec_size_x));
87  IClKernel::configure_internal(win);
88  ARM_COMPUTE_ERROR_ON(has_padding_changed(padding_info));
89 }

97 void ClFloorKernel::run_op(ITensorPack &tensors, const Window &window, cl::CommandQueue &queue)
98 {
99  ARM_COMPUTE_ERROR_ON_UNCONFIGURED_KERNEL(this);
100  ARM_COMPUTE_ERROR_ON_MISMATCHING_WINDOWS(IClKernel::window(), window);
101 
102  const auto src = utils::cast::polymorphic_downcast<const ICLTensor *>(tensors.get_const_tensor(TensorType::ACL_SRC));
103  auto dst = utils::cast::polymorphic_downcast<ICLTensor *>(tensors.get_tensor(TensorType::ACL_DST));
104 
105  Window collapsed = window.collapse_if_possible(IClKernel::window(), Window::DimZ);
106  Window slice = collapsed.first_slice_window_3D();
107 
108  do
109  {
110  unsigned int idx = 0;
111  add_3D_tensor_argument(idx, src, slice);
112  add_3D_tensor_argument(idx, dst, slice);
113  enqueue(queue, *this, slice, lws_hint());
114  }
115  while(collapsed.slide_window_slice_3D(slice));
116 }
