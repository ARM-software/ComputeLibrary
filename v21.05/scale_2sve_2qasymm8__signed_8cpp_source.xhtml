44 void qasymm8_signed_sve_scale_nearest(const ITensor *src, ITensor *dst, const ITensor *offsets,
45  float sampling_offset, bool align_corners, const Window &window)
46 {
47  const size_t in_stride_c = src->info()->dimension(0) + src->info()->padding().left + src->info()->padding().right;
48  const size_t in_stride_w = src->info()->dimension(1) + src->info()->padding().top + src->info()->padding().bottom;
49  const size_t in_stride_wc = in_stride_w * in_stride_c;
50  const size_t in_dim_h = src->info()->dimension(2);
51 
52  // Compute the ratio between source height and destination height
53  const auto hr = scale_utils::calculate_resize_ratio(in_dim_h, dst->info()->dimension(2), align_corners);
54  const auto window_start_x = static_cast<int32_t>(window.x().start());
55  const auto window_end_x = static_cast<int32_t>(window.x().end());
56 
57  Window win(window);
58  win.set(Window::DimX, Window::Dimension(0, 1, 1));
59  Iterator out(dst, win);
60 
61  const uint8_t *in_ptr_start = src->buffer() + src->info()->offset_first_element_in_bytes();
62  const unsigned int in_stride_bytes_hwc = src->info()->strides_in_bytes()[3];
63 
64  execute_window_loop(win, [&](const Coordinates & id)
65  {
66  const int32_t offset = *reinterpret_cast<const int32_t *>(offsets->ptr_to_element(Coordinates(id.y(), id.z()))) * in_stride_c;
67  const auto in_hi = static_cast<int>(align_corners ? utils::rounding::round_half_away_from_zero((id.z() + sampling_offset) * hr) : std::floor((id.z() + sampling_offset) * hr));
68  const int offset_row = in_hi * in_stride_wc;
69  const auto in_ptr = reinterpret_cast<const int8_t *>(in_ptr_start + in_stride_bytes_hwc * id[3]);
70  const auto out_ptr = reinterpret_cast<int8_t *>(out.ptr());
71 
72  // Compute S elements per iteration
73  int x = window_start_x;
74  svbool_t pg = svwhilelt_b8(x, window_end_x);
75  do
76  {
77  // Store results
78  svst1_s8(pg, out_ptr + x, svld1_s8(pg, in_ptr + offset + offset_row + x));
79 
80  x += svcntw();
81  pg = svwhilelt_b8(x, window_end_x);
82  }
83  while(svptest_any(svptrue_b8(), pg));
84  },
85  out);
86 }

88 void qasymm8_signed_sve_scale_bilinear(const ITensor *src, ITensor *dst, const ITensor *offsets, const ITensor *dx, const ITensor *dy,
89  BorderMode border_mode, PixelValue constant_border_value, float sampling_offset,
90  bool align_corners, const Window &window)
91 {
92  // Data layout is NHWC
93  const int idx_width = 1;
94  const int idx_height = 2;
95 
96  // Compute the ratio between source height and destination height
97  const auto hr = scale_utils::calculate_resize_ratio(src->info()->dimension(idx_height), dst->info()->dimension(idx_height), align_corners);
98  Window win_off;
99  win_off.set(Window::DimX, Window::Dimension(0, 0, 0));
100  win_off.set(Window::DimY, Window::Dimension(0, 0, 0));
101 
102  // Don't increment in X and Y direction for the input tensor
103  // A pointer to the start of this plane is needed as base for the precomputed offsets
104  Window win_in(window);
105  win_in.set(idx_width, Window::Dimension(0, 0, 0));
106  win_in.set(idx_height, Window::Dimension(0, 0, 0));
107 
108  for(size_t d = Window::DimZ; d < offsets->info()->num_dimensions(); ++d)
109  {
110  win_off.set(d, Window::Dimension(0, 0, 0));
111  }
112 
113  Iterator in(src, win_in);
114  Iterator out(dst, window);
115 
116  const int32_t in_dim_w = src->info()->dimension(idx_width);
117  const int32_t in_dim_h = src->info()->dimension(idx_height);
118  const int32_t stride_w = src->info()->strides_in_bytes()[idx_width];
119  const int32_t stride_h = src->info()->strides_in_bytes()[idx_height];
120 
121  const UniformQuantizationInfo iq_info = src->info()->quantization_info().uniform();
122  const UniformQuantizationInfo oq_info = dst->info()->quantization_info().uniform();
123 
124  if(border_mode == BorderMode::CONSTANT)
125  {
126  const int8_t const_border_value = static_cast<int8_t>(constant_border_value.get<int8_t>());
127  execute_window_loop(window, [&](const Coordinates & id)
128  {
129  const int32_t index_h = std::floor((id[idx_height] + sampling_offset) * hr - sampling_offset);
130  const int32_t index_w = *(reinterpret_cast<const int32_t *>(offsets->ptr_to_element(Coordinates(id[idx_width], id[idx_height]))));
131  const auto dx_val = *(reinterpret_cast<const float *>(dx->ptr_to_element(Coordinates(id[idx_width], id[idx_height]))));
132  const auto dy_val = *(reinterpret_cast<const float *>(dy->ptr_to_element(Coordinates(id[idx_width], id[idx_height]))));
133  const auto pixel_row_ptr = reinterpret_cast<const int8_t *>(in.ptr());
134 
135  const auto a00 = (0 <= index_w && index_w < in_dim_w && 0 <= index_h && index_h < in_dim_h) ?
136  (*(pixel_row_ptr + index_w * stride_w + index_h * stride_h)) :
137  const_border_value;
138  const auto a01 = (-1 <= index_w && index_w < in_dim_w - 1 && 0 <= index_h && index_h < in_dim_h) ?
139  (*(pixel_row_ptr + (index_w + 1) * stride_w + index_h * stride_h)) :
140  const_border_value;
141  const auto a10 = (0 <= index_w && index_w < in_dim_w && -1 <= index_h && index_h < in_dim_h - 1) ?
142  (*(pixel_row_ptr + index_w * stride_w + (index_h + 1) * stride_h)) :
143  const_border_value;
144  const auto a11 = (-1 <= index_w && index_w < in_dim_w - 1 && -1 <= index_h && index_h < in_dim_h - 1) ?
145  (*(pixel_row_ptr + (index_w + 1) * stride_w + (index_h + 1) * stride_h)) :
146  const_border_value;
147 
148  const float inp00 = Qasymm8QuantizationHelper<int8_t>::dequantize(a00, iq_info);
149  const float inp01 = Qasymm8QuantizationHelper<int8_t>::dequantize(a01, iq_info);
150  const float inp10 = Qasymm8QuantizationHelper<int8_t>::dequantize(a10, iq_info);
151  const float inp11 = Qasymm8QuantizationHelper<int8_t>::dequantize(a11, iq_info);
152  *reinterpret_cast<int8_t *>(out.ptr()) = Qasymm8QuantizationHelper<int8_t>::quantize(scale_helpers::delta_bilinear(inp00, inp01, inp10, inp11, dx_val, dy_val), oq_info);
153  },
154  in, out);
155  }
156  else if(border_mode == BorderMode::REPLICATE)
157  {
158  execute_window_loop(window, [&](const Coordinates & id)
159  {
160  const int index_h = std::floor((id[idx_height] + sampling_offset) * hr - sampling_offset);
161  const int32_t index_w = *(reinterpret_cast<const int32_t *>(offsets->ptr_to_element(Coordinates(id[idx_width], id[idx_height]))));
162  const auto dx_val = *(reinterpret_cast<const float *>(dx->ptr_to_element(Coordinates(id[idx_width], id[idx_height]))));
163  const auto dy_val = *(reinterpret_cast<const float *>(dy->ptr_to_element(Coordinates(id[idx_width], id[idx_height]))));
164  const auto pixel_row_ptr = reinterpret_cast<const int8_t *>(in.ptr());
165 
166  auto clamped_w = utility::clamp<int>(index_w, 0, in_dim_w - 1);
167  auto clamped_w1 = utility::clamp<int>(index_w + 1, 0, in_dim_w - 1);
168  auto clamped_h = utility::clamp<int>(index_h, 0, in_dim_h - 1);
169  auto clamped_h1 = utility::clamp<int>(index_h + 1, 0, in_dim_h - 1);
170 
171  const auto a00 = *(pixel_row_ptr + clamped_w * stride_w + clamped_h * stride_h);
172  const auto a01 = *(pixel_row_ptr + clamped_w1 * stride_w + clamped_h * stride_h);
173  const auto a10 = *(pixel_row_ptr + clamped_w * stride_w + clamped_h1 * stride_h);
174  const auto a11 = *(pixel_row_ptr + clamped_w1 * stride_w + clamped_h1 * stride_h);
175 
176  const float inp00 = Qasymm8QuantizationHelper<int8_t>::dequantize(a00, iq_info);
177  const float inp01 = Qasymm8QuantizationHelper<int8_t>::dequantize(a01, iq_info);
178  const float inp10 = Qasymm8QuantizationHelper<int8_t>::dequantize(a10, iq_info);
179  const float inp11 = Qasymm8QuantizationHelper<int8_t>::dequantize(a11, iq_info);
180  *reinterpret_cast<int8_t *>(out.ptr()) = Qasymm8QuantizationHelper<int8_t>::quantize(scale_helpers::delta_bilinear(inp00, inp01, inp10, inp11, dx_val, dy_val), oq_info);
181  },
182  in, out);
183  }
184  else
185  {
186  ARM_COMPUTE_ERROR("Not implemented");
187  }
188 }

192 void qasymm8_signed_sve_scale(const ITensor *src, ITensor *dst, const ITensor *offsets, const ITensor *dx, const ITensor *dy,
193  InterpolationPolicy policy, BorderMode border_mode, PixelValue constant_border_value, float sampling_offset,
194  bool align_corners, const Window &window)
