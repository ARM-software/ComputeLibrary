50 Status ClMatMul::validate(const ITensorInfo *lhs, const ITensorInfo *rhs, const ITensorInfo *dst, const MatMulInfo &matmul_info)
51 {
52  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(lhs, rhs, dst);
53  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(lhs, 1, DataType::QASYMM8, DataType::QASYMM8_SIGNED, DataType::F16, DataType::F32);
54  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(rhs, 1, DataType::QASYMM8, DataType::QASYMM8_SIGNED, DataType::F16, DataType::F32);
55 
56  const GPUTarget gpu_target = CLScheduler::get().target();
57 
58  std::unique_ptr<IClMatMulNativeKernelConfig> t = ClMatMulNativeKernelConfigurationFactory::create(gpu_target);
59 
60  MatMulKernelInfo kernel_info = t->configure(lhs, rhs, matmul_info);
61 
62  bool is_quantized = is_data_type_quantized_asymmetric(lhs->data_type());
63 
64  return is_quantized ? ClMatMulLowpNativeKernel::validate(lhs, rhs, dst, kernel_info) :
65  ClMatMulNativeKernel::validate(lhs, rhs, dst, kernel_info);
66 }

68 void ClMatMul::configure(const CLCompileContext &compile_context, ITensorInfo *lhs, ITensorInfo *rhs, ITensorInfo *dst, const MatMulInfo &matmul_info)
69 {
70  ARM_COMPUTE_ERROR_ON_NULLPTR(lhs, rhs, dst);
71  ARM_COMPUTE_LOG_PARAMS(lhs, rhs, dst, matmul_info);
72 
73  // Perform validation step
74  ARM_COMPUTE_ERROR_THROW_ON(validate(lhs, rhs, dst, matmul_info));
75 
76  _is_quantized = is_data_type_quantized_asymmetric(lhs->data_type());
77 
78  const GPUTarget gpu_target = CLScheduler::get().target();
79 
80  std::unique_ptr<IClMatMulNativeKernelConfig> t = ClMatMulNativeKernelConfigurationFactory::create(gpu_target);
81 
82  MatMulKernelInfo kernel_info = t->configure(lhs, rhs, matmul_info);
83 
84  if(_is_quantized)
85  {
86  _matmul_lowp_native_kernel->set_target(gpu_target);
87 
88  // Configure the low-precision native matrix multiply kernel
89  _matmul_lowp_native_kernel->configure(compile_context, lhs, rhs, dst, kernel_info);
90  }
91  else
92  {
93  _matmul_native_kernel->set_target(gpu_target);
94 
95  // Configure the native matrix multiply kernel
96  _matmul_native_kernel->configure(compile_context, lhs, rhs, dst, kernel_info);
97  }
98 }
