37 Status ClComponentCast::validate(
38  const Properties &properties,
39  const ArgumentPack<ITensorInfo> &tensors,
40  const Attributes &attributes,
41  const Settings &settings)
42 {
43  ARM_COMPUTE_UNUSED(properties, attributes, settings);
44 
45  const ITensorInfo *src = tensors.get_const_tensor(TensorType::ACL_SRC_0);
46  const ITensorInfo *dst = tensors.get_const_tensor(TensorType::ACL_DST_0);
47 
48  ARM_COMPUTE_RETURN_ERROR_ON_F16_UNSUPPORTED(src);
49  ARM_COMPUTE_RETURN_ERROR_ON_F16_UNSUPPORTED(dst);
50  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(src, dst);
51  ARM_COMPUTE_RETURN_ERROR_ON(src == dst);
52  ARM_COMPUTE_RETURN_ERROR_ON_MSG(src->data_type() == attributes.data_type(), "input and target data types should be different");
53 
54  // Validate in case of configured dst
55  if(dst->total_size() > 0)
56  {
57  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_SHAPES(src, dst);
58  ARM_COMPUTE_RETURN_ERROR_ON_MSG(dst->data_type() != attributes.data_type(), "dst and target data types should be same");
59  }
60 
61  return Status{};
62 }
