52 /** Creates a lifetime of three objects where the two of them can share the same underlying within the given scope
53  *
54  * @warning Contents and pointers of the objects registered will be invalid at the end of the function thus do not dereference.
55  *
56  * @param[in] mg The memory group under which the manageable objects will be registered
57  */
58 void generate_lifetime_info(MemoryGroup &mg)
59 {
60  MockMemoryManageable a{}, b{}, c{};
61  Memory m_a{}, m_b{}, m_c{};
62 
63  // Generate a custom lifetime for the objects
64  mg.manage(&a);
65  mg.manage(&b);
66  mg.finalize_memory(&a, m_a, 12U /* size */, 8U /* alignment */);
67  mg.manage(&c);
68  mg.finalize_memory(&b, m_b, 128U /* size */, 16U /* alignment */);
69  mg.finalize_memory(&c, m_c, 32U /* size */, 0U /* alignment */);
70 }
