58 /** Test case for memory injection in @ref arm_compute::experimental::ops::CpuGemm.
59  *
60  * Configure the operator once and inject memory at run-time in multiple executions.
61  *
62  * Checks performed in order:
63  * - Both runs compute the same output
64  */
65 TEST_CASE(OpsCpuGemmMemoryInjection, framework::DatasetMode::ALL)
66 {
67  auto gemm = std::make_unique<arm_compute::experimental::ops::CpuGemm>();
68  const auto lhs_info = TensorInfo(TensorShape(3U, 3U), 1, DataType::F32);
69  const auto rhs_info = TensorInfo(TensorShape(4U, 3U), 1, DataType::F32);
70  const auto c_info = TensorInfo(TensorShape(4U, 3U), 1, DataType::F32);
71  auto dst_info = TensorInfo(TensorShape(4U, 3U), 1, DataType::F32);
72  const auto gemm_info = GEMMInfo{};
73  gemm->configure(&lhs_info, &rhs_info, &c_info, &dst_info, 1.f, 1.f, gemm_info);
74 
75  // telhs are newly created every call of this lambda function
76  auto lhs = create_tensor<Tensor>(lhs_info);
77  auto rhs = create_tensor<Tensor>(rhs_info);
78  auto c = create_tensor<Tensor>(c_info);
79  lhs.allocator()->allocate();
80  rhs.allocator()->allocate();
81  c.allocator()->allocate();
82 
83  ITensorPack run_pack{ { TensorType::ACL_SRC_0, &lhs }, { TensorType::ACL_SRC_1, &rhs }, { TensorType::ACL_SRC_2, &c } };
84  ITensorPack prep_pack{ { TensorType::ACL_SRC_1, &rhs }, { TensorType::ACL_SRC_2, &c } };
85 
86  auto mg = MemoryGroup{};
87  auto ws = manage_workspace<Tensor>(gemm->workspace(), mg, run_pack, prep_pack);
88 
89  auto run_conv = [&]() -> Tensor
90  {
91  auto dst = create_tensor<Tensor>(dst_info);
92  dst.allocator()->allocate();
93  run_pack.add_tensor(TensorType::ACL_DST, &dst);
94 
95  library->fill_tensor_value(Accessor(lhs), 1.f);
96  library->fill_tensor_value(Accessor(rhs), 2.f);
97  library->fill_tensor_value(Accessor(c), 3.f);
98  // This operator is configured once and captured by this lambda.
99  gemm->prepare(prep_pack);
100  gemm->run(run_pack);
101  return dst;
102  };
103  auto result_0 = run_conv();
104  auto result_1 = run_conv();
105  for(size_t i = 0; i < result_0.info()->tensor_shape().total_size(); ++i)
106  {
107  ARM_COMPUTE_EXPECT(((float *)result_0.buffer())[i] == ((float *)result_1.buffer())[i], framework::LogLevel::ERRORS);
108  }
109 }
