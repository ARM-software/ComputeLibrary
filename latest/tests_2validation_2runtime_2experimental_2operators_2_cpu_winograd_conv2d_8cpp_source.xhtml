61 /** Test case for memory injection in @ref arm_compute::experimental::op::CpuWinogradConv2d.
62  *
63  * Configure the operator once and inject memory at run-time in multiple executions.
64  *
65  * Checks performed in order:
66  * - Both runs compute the same output
67  */
68 TEST_CASE(OpCpuWinogradConv2dMemoryInjection, framework::DatasetMode::ALL)
69 {
70  auto winograd = std::make_unique<cpu::CpuWinogradConv2d>();
71  const auto src_info = TensorInfo(TensorShape(8U, 8U, 32U), 1, DataType::F32);
72  const auto w_info = TensorInfo(TensorShape(1U), 1, DataType::F32);
73  const auto b_info = TensorInfo(TensorShape(1U, 3U, 32U, 1U), 1, DataType::F32);
74  auto dst_info = TensorInfo(TensorShape(8U, 6U, 1U), 1, DataType::F32);
75  const PadStrideInfo pad_info{};
76 
77  winograd->configure(&src_info, &b_info, &w_info, &dst_info, pad_info);
78 
79  // telhs are newly created every call of this lambda function
80  auto a = create_tensor<Tensor>(src_info);
81  auto b = create_tensor<Tensor>(b_info);
82  auto c = create_tensor<Tensor>(w_info);
83  a.allocator()->allocate();
84  b.allocator()->allocate();
85  c.allocator()->allocate();
86 
87  ITensorPack run_pack{{TensorType::ACL_SRC_0, &a}, {TensorType::ACL_SRC_1, &b}, {TensorType::ACL_SRC_2, &c}};
88  ITensorPack prep_pack{{TensorType::ACL_SRC_1, &b}, {TensorType::ACL_SRC_2, &c}};
89 
90  auto mg = MemoryGroup{};
91  auto ws = manage_workspace<Tensor>(winograd->workspace(), mg, run_pack, prep_pack);
92  auto run_conv = [&]() -> Tensor
93  {
94  auto dst = create_tensor<Tensor>(dst_info);
95  dst.allocator()->allocate();
96 
97  run_pack.add_tensor(TensorType::ACL_DST, &dst);
98  library->fill_tensor_value(Accessor(a), 1.f);
99  library->fill_tensor_value(Accessor(b), 2.f);
100  library->fill_tensor_value(Accessor(c), 3.f);
101 
102  // This operator is configured once and captured by this lambda.
103  winograd->prepare(prep_pack);
104  winograd->run(run_pack);
105  return dst;
106  };
107 
108  auto result_0 = run_conv();
109  auto result_1 = run_conv();
110 
111  for (size_t i = 0; i < result_0.info()->tensor_shape().total_size(); ++i)
112  {
113  ARM_COMPUTE_EXPECT(((float *)result_0.buffer())[i] == ((float *)result_1.buffer())[i],
114  framework::LogLevel::ERRORS);
115  }
116 }
