73 inline void validate_configuration(const TensorShape &shape, Format src_format, Format dst_format)
74 {
75  const unsigned int src_num_planes = num_planes_from_format(src_format);
76  const unsigned int dst_num_planes = num_planes_from_format(dst_format);
77 
78  TensorShape input = adjust_odd_shape(shape, src_format);
79  input = adjust_odd_shape(input, dst_format);
80 
81  // Create tensors
82  MultiImage ref_src = create_multi_image<MultiImage>(input, src_format);
83  MultiImage ref_dst = create_multi_image<MultiImage>(input, dst_format);
84 
85  // Create and Configure function
86  NEColorConvert color_convert;
87 
88  if(1U == src_num_planes)
89  {
90  const Tensor *src_plane = ref_src.plane(0);
91 
92  if(1U == dst_num_planes)
93  {
94  Tensor *dst_plane = ref_dst.plane(0);
95  color_convert.configure(src_plane, dst_plane);
96  }
97  else
98  {
99  color_convert.configure(src_plane, &ref_dst);
100  }
101  }
102  else
103  {
104  if(1U == dst_num_planes)
105  {
106  Tensor *dst_plane = ref_dst.plane(0);
107  color_convert.configure(&ref_src, dst_plane);
108  }
109  else
110  {
111  color_convert.configure(&ref_src, &ref_dst);
112  }
113  }
114 
115  for(unsigned int plane_idx = 0; plane_idx < src_num_planes; ++plane_idx)
116  {
117  const Tensor *src_plane = ref_src.plane(plane_idx);
118 
119  ARM_COMPUTE_EXPECT(src_plane->info()->is_resizable(), framework::LogLevel::ERRORS);
120  }
121  for(unsigned int plane_idx = 0; plane_idx < dst_num_planes; ++plane_idx)
122  {
123  const Tensor *dst_plane = ref_dst.plane(plane_idx);
124 
125  ARM_COMPUTE_EXPECT(dst_plane->info()->is_resizable(), framework::LogLevel::ERRORS);
126  }
127 }
