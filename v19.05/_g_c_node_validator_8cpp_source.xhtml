71 Status validate_convolution_layer(ConvolutionLayerNode &node)
72 {
73  ARM_COMPUTE_LOG_GRAPH_VERBOSE("Validating ConvolutionLayer node with ID : " << node.id() << " and Name: " << node.name() << std::endl);
74  ARM_COMPUTE_RETURN_ERROR_ON(node.num_inputs() != 3);
75  ARM_COMPUTE_RETURN_ERROR_ON(node.num_outputs() != 1);
76 
77  // Extract IO and info
78  arm_compute::ITensorInfo *weights = detail::get_backing_tensor_info(node.input(1));
79  const PadStrideInfo conv_info = node.convolution_info();
80  const ConvolutionMethod conv_algorithm = node.convolution_method();
81 
82  // Validate function
83  ARM_COMPUTE_RETURN_ERROR_ON_MSG(node.num_groups() != 1, "Grouping is not supported by ConvolutionLayer!");
84  if(conv_algorithm == ConvolutionMethod::Direct)
85  {
86  bool is_square = weights->tensor_shape().x() == weights->tensor_shape().y();
87  bool is_direct = (weights->tensor_shape().x() == 1) || (weights->tensor_shape().x() == 3) || (weights->tensor_shape().x() == 5);
88  bool is_correct_stride = (conv_info.stride().first) <= 2 && (conv_info.stride().second <= 2);
89  ARM_COMPUTE_RETURN_ERROR_ON_MSG(!(is_square && is_direct && is_correct_stride), "Direct convolution is not supported for given configuration");
90  }
91 
92  return Status{};
93 }

96 Status GCNodeValidator::validate(INode *node)
97 {
98  if(node == nullptr)
99  {
100  return Status{};
101  }
102 
103  NodeType type = node->type();
104  switch(type)
105  {
106  case NodeType::BoundingBoxTransformLayer:
107  return ARM_COMPUTE_CREATE_ERROR(arm_compute::ErrorCode::RUNTIME_ERROR, "Unsupported operation : BoundingBoxTransformLayer");
108  case NodeType::ChannelShuffleLayer:
109  return ARM_COMPUTE_CREATE_ERROR(arm_compute::ErrorCode::RUNTIME_ERROR, "Unsupported operation : ChannelShuffleLayer");
110  case NodeType::ConvolutionLayer:
111  return validate_convolution_layer(*polymorphic_downcast<ConvolutionLayerNode *>(node));
112  case NodeType::DepthwiseConvolutionLayer:
113  return validate_depthwise_convolution_layer(*polymorphic_downcast<DepthwiseConvolutionLayerNode *>(node));
114  case NodeType::DetectionOutputLayer:
115  return ARM_COMPUTE_CREATE_ERROR(arm_compute::ErrorCode::RUNTIME_ERROR, "Unsupported operation : DetectionOutputLayer");
116  case NodeType::FlattenLayer:
117  return ARM_COMPUTE_CREATE_ERROR(arm_compute::ErrorCode::RUNTIME_ERROR, "Unsupported operation : FlattenLayer");
118  case NodeType::GenerateProposalsLayer:
119  return ARM_COMPUTE_CREATE_ERROR(arm_compute::ErrorCode::RUNTIME_ERROR, "Unsupported operation : GenerateProposalsLayer");
120  case NodeType::NormalizePlanarYUVLayer:
121  return detail::validate_normalize_planar_yuv_layer<GCNormalizePlanarYUVLayer>(*polymorphic_downcast<NormalizePlanarYUVLayerNode *>(node));
122  case NodeType::PadLayer:
123  return ARM_COMPUTE_CREATE_ERROR(arm_compute::ErrorCode::RUNTIME_ERROR, "Unsupported operation : PadLayer");
124  case NodeType::PermuteLayer:
125  return ARM_COMPUTE_CREATE_ERROR(arm_compute::ErrorCode::RUNTIME_ERROR, "Unsupported operation : PermuteLayer");
126  case NodeType::PriorBoxLayer:
127  return ARM_COMPUTE_CREATE_ERROR(arm_compute::ErrorCode::RUNTIME_ERROR, "Unsupported operation : PriorBoxLayer");
128  case NodeType::ReorgLayer:
129  return ARM_COMPUTE_CREATE_ERROR(arm_compute::ErrorCode::RUNTIME_ERROR, "Unsupported operation : ReorgLayer");
130  case NodeType::ReshapeLayer:
131  return ARM_COMPUTE_CREATE_ERROR(arm_compute::ErrorCode::RUNTIME_ERROR, "Unsupported operation : ReshapeLayer");
132  case NodeType::ROIAlignLayer:
133  return ARM_COMPUTE_CREATE_ERROR(arm_compute::ErrorCode::RUNTIME_ERROR, "Unsupported operation : ROIAlignLayer");
134  case NodeType::SliceLayer:
135  return ARM_COMPUTE_CREATE_ERROR(arm_compute::ErrorCode::RUNTIME_ERROR, "Unsupported operation : SliceLayer");
136  case NodeType::UpsampleLayer:
137  return ARM_COMPUTE_CREATE_ERROR(arm_compute::ErrorCode::RUNTIME_ERROR, "Unsupported operation : UpsampleLayer");
138  case NodeType::YOLOLayer:
139  return ARM_COMPUTE_CREATE_ERROR(arm_compute::ErrorCode::RUNTIME_ERROR, "Unsupported operation : YOLOLayer");
140  default:
141  return Status{};
142  }
