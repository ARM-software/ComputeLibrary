144 void DotGraphPrinter::print_nodes(const Graph &g, std::ostream &os)
145 {
146  for(const auto &n : g.nodes())
147  {
148  if(n)
149  {
150  // Output node id
151  std::string node_id = std::string("n") + support::cpp11::to_string(n->id());
152  os << node_id << " ";
153 
154  // Output label
155  n->accept(_dot_node_visitor);
156 
157  std::string name = n->name().empty() ? node_id : n->name();
158  auto node_description = _dot_node_visitor.info();
159 
160  os << R"([label = ")" << name << R"( \n )" << n->assigned_target() << R"( \n )" << node_description << R"("])";
161  os << ";\n";
162  }
163  }
164 }
165 
166 void DotGraphPrinter::print_edges(const Graph &g, std::ostream &os)
167 {
168  for(const auto &e : g.edges())
169  {
170  if(e)
171  {
172  std::string source_node_id = std::string("n") + support::cpp11::to_string(e->producer_id());
173  std::string sink_node_id = std::string("n") + support::cpp11::to_string(e->consumer_id());
174  os << source_node_id << " -> " << sink_node_id << " ";
175  const Tensor *t = e->tensor();
176  ARM_COMPUTE_ERROR_ON(t == nullptr);
177  os << R"([label = ")" << t->desc().shape << R"( \n )" << t->desc().data_type << R"( \n )" << t->desc().layout << R"("])";
178  os << ";\n";
179  }
180  }
