50 void NELaplacianPyramid::run()
51 {
52  ARM_COMPUTE_ERROR_ON_MSG(0 == _num_levels, "Unconfigured function");
53 
54  // Compute Gaussian Pyramid
55  _gaussian_pyr_function.run();
56 
57  for(unsigned int i = 0; i < _num_levels; ++i)
58  {
59  // Apply Gaussian filter to gaussian pyramid image
60  _convf[i].run();
61  }
62 
63  for(unsigned int i = 0; i < _num_levels; ++i)
64  {
65  // Compute laplacian image
66  _subf[i].run();
67  }
68 
69  _depth_function.run();
70 }

72 void NELaplacianPyramid::configure(const ITensor *input, IPyramid *pyramid, ITensor *output, BorderMode border_mode, uint8_t constant_border_value)
73 {
74  ARM_COMPUTE_ERROR_ON(nullptr == pyramid);
75  ARM_COMPUTE_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(input, 1, DataType::U8);
76  ARM_COMPUTE_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(output, 1, DataType::S16);
77  ARM_COMPUTE_ERROR_ON(0 == pyramid->info()->num_levels());
78  ARM_COMPUTE_ERROR_ON(input->info()->dimension(0) != pyramid->info()->width());
79  ARM_COMPUTE_ERROR_ON(input->info()->dimension(1) != pyramid->info()->height());
80  ARM_COMPUTE_ERROR_ON(output->info()->dimension(0) != pyramid->get_pyramid_level(pyramid->info()->num_levels() - 1)->info()->dimension(0));
81  ARM_COMPUTE_ERROR_ON(output->info()->dimension(1) != pyramid->get_pyramid_level(pyramid->info()->num_levels() - 1)->info()->dimension(1));
82 
83  _num_levels = pyramid->info()->num_levels();
84 
85  // Create and initialize the gaussian pyramid and the convoluted pyramid
86  PyramidInfo pyramid_info;
87  pyramid_info.init(_num_levels, 0.5f, pyramid->info()->tensor_shape(), arm_compute::Format::U8);
88 
89  _gauss_pyr.init(pyramid_info);
90  _conv_pyr.init(pyramid_info);
91 
92  // Create Gaussian Pyramid function
93  _gaussian_pyr_function.configure(input, &_gauss_pyr, border_mode, constant_border_value);
94 
95  _convf.resize(_num_levels);
96  _subf.resize(_num_levels);
97 
98  for(unsigned int i = 0; i < _num_levels; ++i)
99  {
100  _convf[i].configure(_gauss_pyr.get_pyramid_level(i), _conv_pyr.get_pyramid_level(i), border_mode, constant_border_value);
101  _subf[i].configure(_gauss_pyr.get_pyramid_level(i), _conv_pyr.get_pyramid_level(i), pyramid->get_pyramid_level(i), ConvertPolicy::WRAP);
102  }
103 
104  _depth_function.configure(_conv_pyr.get_pyramid_level(_num_levels - 1), output, ConvertPolicy::WRAP, 0);
105 
106  _gauss_pyr.allocate();
107  _conv_pyr.allocate();
108 }
