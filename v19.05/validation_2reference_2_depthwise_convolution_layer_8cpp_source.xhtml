51 template <typename T, typename TB>
52 SimpleTensor<T> depthwise_convolution(const SimpleTensor<T> &src, const SimpleTensor<T> &weights, const SimpleTensor<TB> &biases, const TensorShape &dst_shape, const PadStrideInfo &conv_info,
53  unsigned int depth_multiplier, const Size2D &dilation, QuantizationInfo out_quant_info)
54 {
55  ARM_COMPUTE_UNUSED(out_quant_info);
56 
57  SimpleTensor<T> dst{ dst_shape, src.data_type(), 1 };
58 
59  // Compute reference
60  const int filter_width = weights.shape().x();
61  const int filter_height = weights.shape().y();
62  const int filter_plane = filter_width * filter_height;
63  const int input_width = src.shape().x();
64  const int input_height = src.shape().y();
65  const int input_depth = src.shape().z();
66  const int num_batches = src.shape().total_size() / (input_width * input_height * input_depth);
67 
68  const int pad_left = conv_info.pad_left();
69  const int pad_top = conv_info.pad_top();
70  const int pad_right = conv_info.pad_right();
71  const int pad_bottom = conv_info.pad_bottom();
72 
73  const float patch_width = (filter_width + (dilation.x() - 1) * (filter_width - 1));
74  const float patch_height = (filter_height + (dilation.y() - 1) * (filter_height - 1));
75 
76  const int patch_half_width_floor = patch_width / 2;
77  const int patch_half_height_floor = patch_height / 2;
78 
79  const auto patch_half_width_ceil = static_cast<int>(std::ceil(patch_width / 2));
80  const auto patch_half_height_ceil = static_cast<int>(std::ceil(patch_height / 2));
81 
82  const int minimum_x = -pad_left + patch_half_width_floor;
83  const int minimum_y = -pad_top + patch_half_height_floor;
84  const int maximum_x = input_width + pad_left + pad_right - static_cast<int>(patch_width);
85  const int maximum_y = input_height + pad_top + pad_bottom - static_cast<int>(patch_height);
86 
87  const T border_value(0);
88 
89  int out_pos = 0;
90  for(int r = 0; r < num_batches; ++r)
91  {
92  for(int z = 0; z < input_depth; ++z)
93  {
94  for(unsigned int m = 0; m < depth_multiplier; ++m)
95  {
96  const int out_z = z * depth_multiplier + m;
97 
98  for(int y = minimum_y; y <= minimum_y + maximum_y; y += conv_info.stride().second)
99  {
100  for(int x = minimum_x; x <= minimum_x + maximum_x; x += conv_info.stride().first)
101  {
102  Coordinates coords(static_cast<int>(x), static_cast<int>(y), static_cast<int>(z), static_cast<int>(r));
103  size_t filter_offset = filter_plane * out_z;
104 
105  T val(0);
106  for(int j = y - patch_half_height_floor; j < y + patch_half_height_ceil; j += dilation.y())
107  {
108  for(int i = x - patch_half_width_floor; i < x + patch_half_width_ceil; i += dilation.x())
109  {
110  coords.set(0, i);
111  coords.set(1, j);
112  val += *(weights.data() + filter_offset) * tensor_elem_at(src, coords, BorderMode::CONSTANT, border_value);
113  ++filter_offset;
114  }
115  }
116 
117  dst[out_pos++] = saturate_cast<T>(val + *static_cast<const TB *>(biases(Coordinates(out_z))));
118  }
119  }
120  }
121  }
122  }
123 
124  return dst;
125 }

127 template <>
128 SimpleTensor<uint8_t> depthwise_convolution(const SimpleTensor<uint8_t> &src, const SimpleTensor<uint8_t> &weights, const SimpleTensor<int32_t> &biases, const TensorShape &dst_shape,
129  const PadStrideInfo &conv_info, unsigned int depth_multiplier, const Size2D &dilation, QuantizationInfo out_quant_info)
130 {
131  // if no explicit quantization has been set you the same as src
132  if(out_quant_info == QuantizationInfo(0.0f, 0))
133  {
134  out_quant_info = src.quantization_info();
135  }
136  SimpleTensor<uint8_t> dst{ dst_shape, src.data_type(), 1, out_quant_info };
137 
138  // Create reference
139  const int input_offset = -src.quantization_info().offset;
140  const float input_scale = src.quantization_info().scale;
141  const int weights_offset = -weights.quantization_info().offset;
142  const float weights_scale = weights.quantization_info().scale;
143  const int output_offset = dst.quantization_info().offset;
144  const float output_scale = dst.quantization_info().scale;
145 
146  int output_multiplier;
147  int output_shift;
148  const float multiplier = input_scale * weights_scale / output_scale;
149  arm_compute::quantization::calculate_quantized_multiplier_less_than_one(multiplier, &output_multiplier, &output_shift);
150 
151  // Compute reference
152  const int filter_width = weights.shape().x();
153  const int filter_height = weights.shape().y();
154  const int filter_plane = filter_width * filter_height;
155  const int input_width = src.shape().x();
156  const int input_height = src.shape().y();
157  const int input_depth = src.shape().z();
158  const int num_batches = src.shape().total_size() / (input_width * input_height * input_depth);
159 
160  const int pad_left = conv_info.pad_left();
161  const int pad_top = conv_info.pad_top();
162  const int pad_right = conv_info.pad_right();
163  const int pad_bottom = conv_info.pad_bottom();
164 
165  const float patch_width = (filter_width + (dilation.x() - 1) * (filter_width - 1));
166  const float patch_height = (filter_height + (dilation.y() - 1) * (filter_height - 1));
167 
168  const int patch_half_width_floor = patch_width / 2;
169  const int patch_half_height_floor = patch_height / 2;
170 
171  const auto patch_half_width_ceil = static_cast<int>(std::ceil(patch_width / 2));
172  const auto patch_half_height_ceil = static_cast<int>(std::ceil(patch_height / 2));
173 
174  const int minimum_x = -pad_left + patch_half_width_floor;
175  const int minimum_y = -pad_top + patch_half_height_floor;
176  const int maximum_x = input_width + pad_left + pad_right - static_cast<int>(patch_width);
177  const int maximum_y = input_height + pad_top + pad_bottom - static_cast<int>(patch_height);
178 
179  int out_pos = 0;
180  for(int r = 0; r < num_batches; ++r)
181  {
182  for(int z = 0; z < input_depth; ++z)
183  {
184  for(unsigned int m = 0; m < depth_multiplier; ++m)
185  {
186  const int out_z = z * depth_multiplier + m;
187  const int32_t bias_val = *static_cast<const int32_t *>(biases(Coordinates(out_z)));
188 
189  for(int y = minimum_y; y <= minimum_y + maximum_y; y += conv_info.stride().second)
190  {
191  for(int x = minimum_x; x <= minimum_x + maximum_x; x += conv_info.stride().first)
192  {
193  Coordinates coords(x, y, z, r);
194  int filter_offset = filter_plane * out_z;
195 
196  int32_t val = 0;
197  for(int j = y - patch_half_height_floor; j < y + patch_half_height_ceil; j += dilation.y())
198  {
199  for(int i = x - patch_half_width_floor; i < x + patch_half_width_ceil; i += dilation.x())
200  {
201  coords.set(0, i);
202  coords.set(1, j);
203  const auto in_val = tensor_elem_at<uint8_t>(src, coords, BorderMode::CONSTANT, -input_offset);
204  const uint8_t w_val = *(weights.data() + filter_offset);
205  val += (in_val + input_offset) * (w_val + weights_offset);
206  ++filter_offset;
207  }
208  }
209  val += bias_val;
210  val = asymm_rounding_divide_by_pow2(asymm_int_mult(val, output_multiplier), output_shift);
211  val += output_offset;
212  val = std::max<int32_t>(val, 0);
213  val = std::min<int32_t>(val, 255);
214 
215  // Store the result
216  dst[out_pos++] = val;
217  }
218  }
219  }
220  }
221  }
222 
223  return dst;
224 }
