43 template <typename T>
44 inline T roi_align_1x1(const T *input, TensorShape input_shape,
45  float region_start_x,
46  float bin_size_x,
47  int grid_size_x,
48  float region_end_x,
49  float region_start_y,
50  float bin_size_y,
51  int grid_size_y,
52  float region_end_y,
53  int pz)
54 {
55  if((region_end_x <= region_start_x) || (region_end_y <= region_start_y))
56  {
57  return T(0);
58  }
59  else
60  {
61  float avg = 0;
62  // Iterate through the aligned pooling region
63  for(int iy = 0; iy < grid_size_y; ++iy)
64  {
65  for(int ix = 0; ix < grid_size_x; ++ix)
66  {
67  // Align the window in the middle of every bin
68  float y = region_start_y + (iy + 0.5) * bin_size_y / float(grid_size_y);
69  float x = region_start_x + (ix + 0.5) * bin_size_x / float(grid_size_x);
70 
71  // Interpolation in the [0,0] [0,1] [1,0] [1,1] square
72  const int y_low = y;
73  const int x_low = x;
74  const int y_high = y_low + 1;
75  const int x_high = x_low + 1;
76 
77  const float ly = y - y_low;
78  const float lx = x - x_low;
79  const float hy = 1. - ly;
80  const float hx = 1. - lx;
81 
82  const float w1 = hy * hx;
83  const float w2 = hy * lx;
84  const float w3 = ly * hx;
85  const float w4 = ly * lx;
86 
87  const size_t idx1 = coord2index(input_shape, Coordinates(x_low, y_low, pz));
88  T data1 = input[idx1];
89 
90  const size_t idx2 = coord2index(input_shape, Coordinates(x_high, y_low, pz));
91  T data2 = input[idx2];
92 
93  const size_t idx3 = coord2index(input_shape, Coordinates(x_low, y_high, pz));
94  T data3 = input[idx3];
95 
96  const size_t idx4 = coord2index(input_shape, Coordinates(x_high, y_high, pz));
97  T data4 = input[idx4];
98 
99  avg += w1 * data1 + w2 * data2 + w3 * data3 + w4 * data4;
100  }
101  }
102 
103  avg /= grid_size_x * grid_size_y;
104 
105  return T(avg);
106  }
107 }

116 template <typename T>
117 SimpleTensor<T> roi_align_layer(const SimpleTensor<T> &src, const SimpleTensor<T> &rois, const ROIPoolingLayerInfo &pool_info)
118 {
119  const size_t values_per_roi = rois.shape()[0];
120  const size_t num_rois = rois.shape()[1];
121  DataType dst_data_type = src.data_type();
122 
123  const auto *rois_ptr = static_cast<const T *>(rois.data());
124 
125  TensorShape input_shape = src.shape();
126  TensorShape output_shape(pool_info.pooled_width(), pool_info.pooled_height(), src.shape()[2], num_rois);
127  SimpleTensor<T> dst(output_shape, dst_data_type);
128 
129  // Iterate over every pixel of the input image
130  for(size_t px = 0; px < pool_info.pooled_width(); ++px)
131  {
132  for(size_t py = 0; py < pool_info.pooled_height(); ++py)
133  {
134  for(size_t pw = 0; pw < num_rois; ++pw)
135  {
136  const unsigned int roi_batch = rois_ptr[values_per_roi * pw];
137  const auto x1 = float(rois_ptr[values_per_roi * pw + 1]);
138  const auto y1 = float(rois_ptr[values_per_roi * pw + 2]);
139  const auto x2 = float(rois_ptr[values_per_roi * pw + 3]);
140  const auto y2 = float(rois_ptr[values_per_roi * pw + 4]);
141 
142  const float roi_anchor_x = x1 * pool_info.spatial_scale();
143  const float roi_anchor_y = y1 * pool_info.spatial_scale();
144  const float roi_dims_x = std::max((x2 - x1) * pool_info.spatial_scale(), 1.0f);
145  const float roi_dims_y = std::max((y2 - y1) * pool_info.spatial_scale(), 1.0f);
146 
147  float bin_size_x = roi_dims_x / pool_info.pooled_width();
148  float bin_size_y = roi_dims_y / pool_info.pooled_height();
149  float region_start_x = px * bin_size_x + roi_anchor_x;
150  float region_start_y = py * bin_size_y + roi_anchor_y;
151  float region_end_x = (px + 1) * bin_size_x + roi_anchor_x;
152  float region_end_y = (py + 1) * bin_size_y + roi_anchor_y;
153 
154  region_start_x = clamp(region_start_x, 0.0f, float(input_shape[0]));
155  region_start_y = clamp(region_start_y, 0.0f, float(input_shape[1]));
156  region_end_x = clamp(region_end_x, 0.0f, float(input_shape[0]));
157  region_end_y = clamp(region_end_y, 0.0f, float(input_shape[1]));
158 
159  const int roi_bin_grid_x = (pool_info.sampling_ratio() > 0) ? pool_info.sampling_ratio() : int(ceil(bin_size_x));
160  const int roi_bin_grid_y = (pool_info.sampling_ratio() > 0) ? pool_info.sampling_ratio() : int(ceil(bin_size_y));
161 
162  // Move input and output pointer across the fourth dimension
163  const size_t input_stride_w = input_shape[0] * input_shape[1] * input_shape[2];
164  const size_t output_stride_w = output_shape[0] * output_shape[1] * output_shape[2];
165  const T *input_ptr = src.data() + roi_batch * input_stride_w;
166  T *output_ptr = dst.data() + px + py * output_shape[0] + pw * output_stride_w;
167 
168  for(int pz = 0; pz < int(input_shape[2]); ++pz)
169  {
170  // For every pixel pool over an aligned region
171  *(output_ptr + pz * output_shape[0] * output_shape[1]) = roi_align_1x1(input_ptr, input_shape,
172  region_start_x,
173  bin_size_x,
174  roi_bin_grid_x,
175  region_end_x,
176  region_start_y,
177  bin_size_y,
178  roi_bin_grid_y,
179  region_end_y, pz);
180  }
181  }
182  }
183  }
184  return dst;
185 }
