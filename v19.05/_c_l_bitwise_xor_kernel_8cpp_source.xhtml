73 void CLBitwiseXorKernel::run(const Window &window, cl::CommandQueue &queue)
74 {
75  ARM_COMPUTE_ERROR_ON_UNCONFIGURED_KERNEL(this);
76  ARM_COMPUTE_ERROR_ON_INVALID_SUBWINDOW(ICLKernel::window(), window);
