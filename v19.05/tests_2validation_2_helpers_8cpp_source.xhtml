35 void fill_mask_from_pattern(uint8_t *mask, int cols, int rows, MatrixPattern pattern)
36 {
37  unsigned int v = 0;
38  std::mt19937 gen(library->seed());
39  std::bernoulli_distribution dist(0.5);
40 
41  for(int r = 0; r < rows; ++r)
42  {
43  for(int c = 0; c < cols; ++c, ++v)
44  {
45  uint8_t val = 0;
46 
47  switch(pattern)
48  {
49  case MatrixPattern::BOX:
50  val = 255;
51  break;
52  case MatrixPattern::CROSS:
53  val = ((r == (rows / 2)) || (c == (cols / 2))) ? 255 : 0;
54  break;
55  case MatrixPattern::DISK:
56  val = (((r - rows / 2.0f + 0.5f) * (r - rows / 2.0f + 0.5f)) / ((rows / 2.0f) * (rows / 2.0f)) + ((c - cols / 2.0f + 0.5f) * (c - cols / 2.0f + 0.5f)) / ((cols / 2.0f) *
57  (cols / 2.0f))) <= 1.0f ? 255 : 0;
58  break;
59  case MatrixPattern::OTHER:
60  val = (dist(gen) ? 0 : 255);
61  break;
62  default:
63  return;
64  }
65 
66  mask[v] = val;
67  }
68  }
69 
70  if(pattern == MatrixPattern::OTHER)
71  {
72  std::uniform_int_distribution<uint8_t> distribution_u8(0, ((cols * rows) - 1));
73  mask[distribution_u8(gen)] = 255;
74  }
75 }

77 HarrisCornersParameters harris_corners_parameters()
78 {
79  HarrisCornersParameters params;
80 
81  std::mt19937 gen(library->seed());
82  std::uniform_real_distribution<float> threshold_dist(0.f, 0.001f);
83  std::uniform_real_distribution<float> sensitivity(0.04f, 0.15f);
84  std::uniform_real_distribution<float> euclidean_distance(0.f, 30.f);
85  std::uniform_int_distribution<uint8_t> int_dist(0, 255);
86 
87  params.threshold = threshold_dist(gen);
88  params.sensitivity = sensitivity(gen);
89  params.min_dist = euclidean_distance(gen);
90  params.constant_border_value = int_dist(gen);
91 
92  return params;
93 }

133 template <typename T>
134 void matrix_multiply(const SimpleTensor<T> &a, const SimpleTensor<T> &b, SimpleTensor<T> &out)
135 {
136  ARM_COMPUTE_ERROR_ON(a.shape()[0] != b.shape()[1]);
137  ARM_COMPUTE_ERROR_ON(a.shape()[1] != out.shape()[1]);
138  ARM_COMPUTE_ERROR_ON(b.shape()[0] != out.shape()[0]);
139 
140  const int M = a.shape()[1]; // Rows
141  const int N = b.shape()[0]; // Cols
142  const int K = b.shape()[1];
143 
144  for(int y = 0; y < M; ++y)
145  {
146  for(int x = 0; x < N; ++x)
147  {
148  float acc = 0.0f;
149  for(int k = 0; k < K; ++k)
150  {
151  acc += a[y * K + k] * b[x + k * N];
152  }
153 
154  out[x + y * N] = acc;
155  }
156  }
157 }

160 void transpose_matrix(const SimpleTensor<T> &in, SimpleTensor<T> &out)
161 {
162  ARM_COMPUTE_ERROR_ON((in.shape()[0] != out.shape()[1]) || (in.shape()[1] != out.shape()[0]));
163 
164  const int width = in.shape()[0];
165  const int height = in.shape()[1];
166 
167  for(int y = 0; y < height; ++y)
168  {
169  for(int x = 0; x < width; ++x)
170  {
171  const T val = in[x + y * width];
172 
173  out[x * height + y] = val;
174  }
175  }

178 template <typename T>
179 void get_tile(const SimpleTensor<T> &in, SimpleTensor<T> &tile, const Coordinates &coord)
180 {
181  ARM_COMPUTE_ERROR_ON(tile.shape().num_dimensions() > 2);
182 
183  const int w_tile = tile.shape()[0];
184  const int h_tile = tile.shape()[1];
185 
186  // Fill the tile with zeros
187  std::fill(tile.data() + 0, (tile.data() + (w_tile * h_tile)), static_cast<T>(0));
188 
189  // Check if with the dimensions greater than 2 we could have out-of-bound reads
190  for(size_t d = 2; d < Coordinates::num_max_dimensions; ++d)
191  {
192  if(coord[d] < 0 || coord[d] >= static_cast<int>(in.shape()[d]))
193  {
194  ARM_COMPUTE_ERROR("coord[d] < 0 || coord[d] >= in.shape()[d] with d >= 2");
195  }
196  }
197 
198  // Since we could have out-of-bound reads along the X and Y dimensions,
199  // we start calculating the input address with x = 0 and y = 0
200  Coordinates start_coord = coord;
201  start_coord[0] = 0;
202  start_coord[1] = 0;
203 
204  // Get input and roi pointers
205  auto in_ptr = static_cast<const T *>(in(start_coord));
206  auto roi_ptr = static_cast<T *>(tile.data());
207 
208  const int x_in_start = std::max(0, coord[0]);
209  const int y_in_start = std::max(0, coord[1]);
210  const int x_in_end = std::min(static_cast<int>(in.shape()[0]), coord[0] + w_tile);
211  const int y_in_end = std::min(static_cast<int>(in.shape()[1]), coord[1] + h_tile);
212 
213  // Number of elements to copy per row
214  const int n = x_in_end - x_in_start;
215 
216  // Starting coordinates for the ROI
217  const int x_tile_start = coord[0] > 0 ? 0 : std::abs(coord[0]);
218  const int y_tile_start = coord[1] > 0 ? 0 : std::abs(coord[1]);
219 
220  // Update input pointer
221  in_ptr += x_in_start;
222  in_ptr += (y_in_start * in.shape()[0]);
223 
224  // Update ROI pointer
225  roi_ptr += x_tile_start;
226  roi_ptr += (y_tile_start * tile.shape()[0]);
227 
228  for(int y = y_in_start; y < y_in_end; ++y)
229  {
230  // Copy per row
231  std::copy(in_ptr, in_ptr + n, roi_ptr);
232 
233  in_ptr += in.shape()[0];
234  roi_ptr += tile.shape()[0];
235  }
236 }

238 template <typename T>
239 void zeros(SimpleTensor<T> &in, const Coordinates &anchor, const TensorShape &shape)
240 {
241  ARM_COMPUTE_ERROR_ON(anchor.num_dimensions() != shape.num_dimensions());
242  ARM_COMPUTE_ERROR_ON(in.shape().num_dimensions() > 2);
243  ARM_COMPUTE_ERROR_ON(shape.num_dimensions() > 2);
244 
245  // Check if with the dimensions greater than 2 we could have out-of-bound reads
246  for(size_t d = 0; d < Coordinates::num_max_dimensions; ++d)
247  {
248  if(anchor[d] < 0 || ((anchor[d] + shape[d]) > in.shape()[d]))
249  {
250  ARM_COMPUTE_ERROR("anchor[d] < 0 || (anchor[d] + shape[d]) > in.shape()[d]");
251  }
252  }
253 
254  // Get input pointer
255  auto in_ptr = static_cast<T *>(in(anchor[0] + anchor[1] * in.shape()[0]));
256 
257  const unsigned int n = in.shape()[0];
258 
259  for(unsigned int y = 0; y < shape[1]; ++y)
260  {
261  std::fill(in_ptr, in_ptr + shape[0], 0);
262  in_ptr += n;
263  }
264 }
