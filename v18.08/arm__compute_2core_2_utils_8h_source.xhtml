566 inline bool separate_matrix(const int16_t *conv, int16_t *conv_col, int16_t *conv_row, uint8_t size)
567 {
568  int32_t min_col = -1;
569  int16_t min_col_val = -1;
570 
571  for(int32_t i = 0; i < size; ++i)
572  {
573  if(conv[i] != 0 && (min_col < 0 || abs(min_col_val) > abs(conv[i])))
574  {
575  min_col = i;
576  min_col_val = conv[i];
577  }
578  }
579 
580  if(min_col < 0)
581  {
582  return false;
583  }
584 
585  for(uint32_t j = 0; j < size; ++j)
586  {
587  conv_col[j] = conv[min_col + j * size];
588  }
589 
590  for(uint32_t i = 0; i < size; i++)
591  {
592  if(static_cast<int>(i) == min_col)
593  {
594  conv_row[i] = 1;
595  }
596  else
597  {
598  int16_t coeff = conv[i] / conv[min_col];
599 
600  for(uint32_t j = 1; j < size; ++j)
601  {
602  if(conv[i + j * size] != (conv_col[j] * coeff))
603  {
604  return false;
605  }
606  }
607 
608  conv_row[i] = coeff;
609  }
610  }
611 
612  return true;
613 }

705 inline std::pair<DataType, DataType> data_type_for_convolution(const int16_t *conv_col, const int16_t *conv_row, size_t size)
706 {
707  DataType first_stage = DataType::UNKNOWN;
708  DataType second_stage = DataType::UNKNOWN;
709 
710  auto gez = [](const int16_t &v)
711  {
712  return v >= 0;
713  };
714 
715  auto accu_neg = [](const int &first, const int &second)
716  {
717  return first + (second < 0 ? second : 0);
718  };
719 
720  auto accu_pos = [](const int &first, const int &second)
721  {
722  return first + (second > 0 ? second : 0);
723  };
724 
725  const bool only_positive_coefficients = std::all_of(conv_row, conv_row + size, gez) && std::all_of(conv_col, conv_col + size, gez);
726 
727  if(only_positive_coefficients)
728  {
729  const int max_row_value = std::accumulate(conv_row, conv_row + size, 0) * UINT8_MAX;
730  const int max_value = std::accumulate(conv_col, conv_col + size, 0) * max_row_value;
731 
732  first_stage = (max_row_value <= UINT16_MAX) ? DataType::U16 : DataType::S32;
733 
734  second_stage = (max_value <= UINT16_MAX) ? DataType::U16 : DataType::S32;
735  }
736  else
737  {
738  const int min_row_value = std::accumulate(conv_row, conv_row + size, 0, accu_neg) * UINT8_MAX;
739  const int max_row_value = std::accumulate(conv_row, conv_row + size, 0, accu_pos) * UINT8_MAX;
740  const int neg_coeffs_sum = std::accumulate(conv_col, conv_col + size, 0, accu_neg);
741  const int pos_coeffs_sum = std::accumulate(conv_col, conv_col + size, 0, accu_pos);
742  const int min_value = neg_coeffs_sum * max_row_value + pos_coeffs_sum * min_row_value;
743  const int max_value = neg_coeffs_sum * min_row_value + pos_coeffs_sum * max_row_value;
744 
745  first_stage = ((INT16_MIN <= min_row_value) && (max_row_value <= INT16_MAX)) ? DataType::S16 : DataType::S32;
746 
747  second_stage = ((INT16_MIN <= min_value) && (max_value <= INT16_MAX)) ? DataType::S16 : DataType::S32;
748  }
749 
750  return std::make_pair(first_stage, second_stage);
751 }

762 {
