104 void SchedulerTimer::test_start()
105 {
106  // Start intercepting tasks:
107  ARM_COMPUTE_ERROR_ON(_real_graph_function != nullptr);
108  _real_graph_function = graph::TaskExecutor::get().execute_function;
109  auto task_interceptor = [this](graph::ExecutionTask & task)
110  {
111  Interceptor *scheduler = nullptr;
112  if(dynamic_cast<Interceptor *>(this->_interceptor.get()) != nullptr)
113  {
114  scheduler = arm_compute::utils::cast::polymorphic_downcast<Interceptor *>(_interceptor.get());
115  if(task.node != nullptr && !task.node->name().empty())
116  {
117  scheduler->set_prefix(task.node->name() + "/");
118  }
119  else
120  {
121  scheduler->set_prefix("");
122  }
123  }
124 
125  this->_real_graph_function(task);
126 
127  if(scheduler != nullptr)
128  {
129  scheduler->set_prefix("");
130  }
131  };
132 
133  ARM_COMPUTE_ERROR_ON(_real_scheduler != nullptr);
134  _real_scheduler_type = Scheduler::get_type();
135  //Note: We can't currently replace a custom scheduler
136  if(_real_scheduler_type != Scheduler::Type::CUSTOM)
137  {
138  _real_scheduler = &Scheduler::get();
139  _interceptor = std::make_shared<Interceptor>(_kernels, *_real_scheduler, _scale_factor);
140  Scheduler::set(std::static_pointer_cast<IScheduler>(_interceptor));
141  graph::TaskExecutor::get().execute_function = task_interceptor;
142  }
143 }
