111 /** This OpenCL kernel computes the matrix multiplication between 2 matrices plus 3 post ops:
112  * Post op 1: activation (optional)
113  * Post op 2: elementwise op
114  * Post op 3: activation (optional)
115  *
116  * @note (Optional) -DP1_ACTIVATION_TYPE, -DP1_ACTIVATION_A_VAL, -DP1_ACTIVATION_B_VAL: The activation type, alpha and beta values of the activation post op at slot 3
117  * @note (Required) -DP2_ELTWISE_OP: The (binary) elementwise post op to perform
118  * @note (Required) -DP2_ELTWISE_ARG1_HEIGHT: The height (Y dimension) of the eltwise operand matrix of the eltwise post op at slot 2
119  * @note (Required) -DP2_ELTWISE_ARG1_WIDTH: The width (X dimension) of the eltwise operand matrix of the eltwise post op at slot 2
120  * @note (Optional) -DP3_ACTIVATION_TYPE, -DP3_ACTIVATION_A_VAL, -DP3_ACTIVATION_B_VAL: The activation type, alpha and beta values of the activation post op at slot 3
121  *
122  * All parameters are similarly defined in kernel gemm_mm_native, with these additions:
123  *
124  * @param[in] eltwise_operand_ptr Pointer to the eltwise operand matrix. Supported data type: F16/F32
125  * @param[in] eltwise_operand_stride_x Stride of the eltwise operand matrix in X dimension (in bytes)
126  * @param[in] eltwise_operand_step_x eltwise_operand_stride_x * number of elements along X processed per workitem(in bytes)
127  * @param[in] eltwise_operand_stride_y Stride of the eltwise operand matrix in Y dimension (in bytes)
128  * @param[in] eltwise_operand_step_y eltwise_operand_stride_y * number of elements along Y processed per workitem(in bytes)
129  * @param[in] eltwise_operand_stride_z Stride of the eltwise operand tensor in Z dimension (in bytes)
130  */

181 
182 #if defined(MATRIX_B_DEPTH)

193  // The plane (zlhs) is calculated dividing M (y * M0) by HEIGHT_GEMM3D
194  CALCULATE_Z_OFFSET(M0, uint, zlhs, COMPUTE_M0_START_ROW(y, M0, PARTIAL_STORE_M0), HEIGHT_GEMM3D, DEPTH_GEMM3D, lhs_cross_plane_pad, lhs_stride_y);
195 
196  // Add offset for batched GEMM. The batches will be in the fourth dimension and for this reason we
197  // multiply lhs_stride_z by DEPTH_GEMM3D

199 
200 #else // defined(REINTERPRET_INPUT_AS_3D)
201

204 
205 #endif // defined(REINTERPRET_INPUT_AS_3D)

208  REPEAT_VAR_INIT_TO_CONST(M0, VEC_DATA_TYPE(DATA_TYPE, N0), c, 0); //VEC_DATA_TYPE(DATA_TYPE, N0) c0=0,c1=0,c2=0,... c(M0-1)=0;
209 
210  int i = 0;
211 #if K0 > 1
212  for(; i <= (K - K0); i += K0)
213  {
214  // Supported cases (M0, K0):
215  // 1,2 - 1,3 - 1,4 - 1,8 - 1,16
216  // 2,2 - 2,3 - 2,4 - 2,8 - 2,16
217  // 3,2 - 3,3 - 3,4 - 3,8 - 3,16
218  // 4,2 - 4,3 - 4,4 - 4,8 - 4,16

225

229  RHS_VFMA_M0xN0(0, a, b0, c);

242 #endif // K0 > 4

244  RHS_VFMA_M0xN0(8, a, b8, c);

256  }
257 #endif // K0 > 1
258  // Left-over accumulations
259  for(; i < K; ++i)

261  // Load values from LHS matrix

267 #endif // M0 > 1
268 #if M0 > 2

272 #if M0 > 3
273  VEC_DATA_TYPE(DATA_TYPE, 2)
274  a3 = *((__global DATA_TYPE *)(lhs_ptr + lhs_offset + 3 * lhs_stride_y + zlhs3));
275 #endif // M0 > 3
276 #if M0 > 4

283 #endif // M0 > 5

294  b = VLOAD(N0)(0, (__global DATA_TYPE *)(rhs_ptr + rhs_offset + 0 * rhs_stride_y));
295  RHS_VFMA_M0xN0(0, a, b, c);
296 
297  lhs_offset += sizeof(DATA_TYPE);

299  }

301  __global uchar *dst_addr = dst_ptr + dst_offset_first_element_in_bytes + (x * (uint)N0 * sizeof(DATA_TYPE)) + (COMPUTE_M0_START_ROW(y, M0, PARTIAL_STORE_M0) * dst_stride_y);
302 
303  REPEAT_VAR_INIT_TO_CONST(M0, uint, zout, 0);
304

341 
342  LOAD_BLOCK(M0, N0, DATA_TYPE, bias, bias_addr, 0, bias_stride_y, zero);

361  // c = act(c)
362  POST_OP3_ACTIVATION_OPTIONAL(M0, DATA_TYPE, DATA_TYPE_ACCUMULATOR, N0, c);
363 
364  // Store output block

366 }
