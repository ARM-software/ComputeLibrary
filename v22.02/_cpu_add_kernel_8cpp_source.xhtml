155 Status validate_arguments(const ITensorInfo &src0, const ITensorInfo &src1, const ITensorInfo &dst, ConvertPolicy policy)
156 {
157  ARM_COMPUTE_UNUSED(policy);
158 
159  ARM_COMPUTE_RETURN_ERROR_ON_CPU_F16_UNSUPPORTED(&src0);
160  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(&src0, 1, DataType::U8, DataType::QASYMM8, DataType::QASYMM8_SIGNED,
161  DataType::S16, DataType::QSYMM16, DataType::F16,
162  DataType::S32, DataType::F32);
163  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(&src0, &src1);
164 
165  const TensorShape out_shape = TensorShape::broadcast_shape(src0.tensor_shape(), src1.tensor_shape());
166 
167  ARM_COMPUTE_RETURN_ERROR_ON_MSG(out_shape.total_size() == 0, "Inputs are not broadcast compatible");
168  ARM_COMPUTE_RETURN_ERROR_ON_MSG((src0.tensor_shape().x() != src1.tensor_shape().x()) && ((src0.data_type() != src1.data_type()) || (src0.data_type() != dst.data_type())
169  || (src1.data_type() != dst.data_type())),
170  "Broadcasting across width is supported on configurations where all tensors have the same data type");
171 
172  // Validate in case of configured dst
173  if(dst.total_size() > 0)
174  {
175  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(&src0, &dst);
176  ARM_COMPUTE_RETURN_ERROR_ON_MSG(detail::have_different_dimensions(out_shape, dst.tensor_shape(), 0),
177  "Wrong shape for dst");
178  }
179 
180  const auto *uk = CpuAddKernel::get_implementation(DataTypeISASelectorData{ src0.data_type(), CPUInfo::get().get_isa() });
181  ARM_COMPUTE_RETURN_ERROR_ON(uk == nullptr || uk->ukernel == nullptr);
182 
183  return Status{};
184 }
185 
186 std::pair<Status, Window> validate_and_configure_window(const ITensorInfo &src0, const ITensorInfo &src1, ITensorInfo &dst)
187 {
188  const TensorShape &out_shape = TensorShape::broadcast_shape(src0.tensor_shape(), src1.tensor_shape());
189 
190  // Auto initialize dst if not initialized
191  set_shape_if_empty(dst, out_shape);
192  set_data_type_if_unknown(dst, src0.data_type());
193 
194  Window win = calculate_max_window(out_shape, Steps());
195 
196  // CpuAddKernel doesn't need padding so update_window_and_padding() can be skipped
197  return std::make_pair(Status{}, win);
198 }

201 void CpuAddKernel::configure(const ITensorInfo *src0, const ITensorInfo *src1, ITensorInfo *dst, ConvertPolicy policy)
202 {
203  ARM_COMPUTE_ERROR_ON_NULLPTR(src0, src1, dst);
204  ARM_COMPUTE_ERROR_THROW_ON(validate_arguments(*src0, *src1, *dst, policy));
205 
206  const auto uk = CpuAddKernel::get_implementation(DataTypeISASelectorData{ src0->data_type(), CPUInfo::get().get_isa() });
207 
208  ARM_COMPUTE_ERROR_ON_NULLPTR(uk);
209 
210  _policy = policy;
211  _run_method = uk->ukernel;
212  _name = std::string("CpuAddKernel").append("/").append(uk->name);
213 
214  // Configure kernel window
215  auto win_config = validate_and_configure_window(*src0, *src1, *dst);
216  ARM_COMPUTE_ERROR_THROW_ON(win_config.first);
217  ICpuKernel::configure(win_config.second);
218 }

226 
227  return Status{};

230 void CpuAddKernel::run_op(ITensorPack &tensors, const Window &window, const ThreadInfo &info)
231 {
232  ARM_COMPUTE_UNUSED(info);
233  ARM_COMPUTE_ERROR_ON_UNCONFIGURED_KERNEL(this);
234  ARM_COMPUTE_ERROR_ON_INVALID_SUBWINDOW(ICpuKernel::window(), window);
235 
236  ARM_COMPUTE_ERROR_ON(tensors.empty());
237  ARM_COMPUTE_ERROR_ON(_run_method == nullptr);
238 
239  const ITensor *src0 = tensors.get_const_tensor(TensorType::ACL_SRC_0);
240  const ITensor *src1 = tensors.get_const_tensor(TensorType::ACL_SRC_1);
241  ITensor *dst = tensors.get_tensor(TensorType::ACL_DST);
242 
243  _run_method(src0, src1, dst, _policy, window);
244 }
