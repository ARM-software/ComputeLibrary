102 template <typename InputVectorType, typename OutputVectorType, typename OperatorType,
103  typename InputScalarType = typename sve_scalar<InputVectorType>::type,
104  typename OutputScalarType = typename sve_scalar<OutputVectorType>::type>
105 void elementwise_op(const ITensor *in1, const ITensor *in2, ITensor *out, const Window &window,
106  OperatorType op,
107  LoopFuncType<InputScalarType, OutputScalarType, OperatorType> func,
108  BroadcastLoopFuncType<InputScalarType, OutputScalarType, OperatorType> broadcast_func)
109 {
110  const auto all_true_pg = svptrue<InputScalarType>();
111 
112  // Create input windows
113  Window input1_win = window.broadcast_if_dimension_le_one(in1->info()->tensor_shape());
114  Window input2_win = window.broadcast_if_dimension_le_one(in2->info()->tensor_shape());
115 
116  // Clear X Dimension on execution window as we handle manually
117  Window win = window;
118  win.set(Window::DimX, Window::Dimension(0, 1, 1));
119 
120  const auto window_start_x = static_cast<int>(window.x().start());
121  const auto window_end_x = static_cast<int>(window.x().end());
122  const bool is_broadcast_across_x = in1->info()->tensor_shape().x() != in2->info()->tensor_shape().x();
123 
124  if(is_broadcast_across_x)
125  {
126  const bool is_broadcast_input_2 = input2_win.x().step() == 0;
127  Window broadcast_win = is_broadcast_input_2 ? input2_win : input1_win;
128  Window non_broadcast_win = !is_broadcast_input_2 ? input2_win : input1_win;
129  const ITensor *broadcast_tensor = is_broadcast_input_2 ? in2 : in1;
130  const ITensor *non_broadcast_tensor = !is_broadcast_input_2 ? in2 : in1;
131 
132  // Clear X Dimension on execution window as we handle manually
133  non_broadcast_win.set(Window::DimX, Window::Dimension(0, 1, 1));
134 
135  Iterator broadcast_input(broadcast_tensor, broadcast_win);
136  Iterator non_broadcast_input(non_broadcast_tensor, non_broadcast_win);
137  Iterator output(out, win);
138 
139  execute_window_loop(win, [&](const Coordinates &)
140  {
141  auto output_ptr = reinterpret_cast<OutputScalarType *>(output.ptr());
142  const auto non_broadcast_input_ptr = reinterpret_cast<const InputScalarType *>(non_broadcast_input.ptr());
143  const InputScalarType broadcast_value = *reinterpret_cast<const InputScalarType *>(broadcast_input.ptr());
144 
145  int x = window_start_x;
146 
147  svbool_t pg = svwhilelt<InputScalarType>(x, window_end_x);
148  do
149  {
150  broadcast_func(pg,
151  {
152  op,
153  non_broadcast_input_ptr + x,
154  broadcast_value,
155  output_ptr + x,
156  !is_broadcast_input_2
157  });
158  x += svcnt<InputScalarType>();
159  pg = svwhilelt<InputScalarType>(x, window_end_x);
160  }
161  while(svptest_any(all_true_pg, pg));
162  },
163  broadcast_input, non_broadcast_input, output);
164  }
165  else
166  {
167  // Clear X Dimension on execution window as we handle manually
168  input1_win.set(Window::DimX, Window::Dimension(0, 1, 1));
169  input2_win.set(Window::DimX, Window::Dimension(0, 1, 1));
170 
171  Iterator input1(in1, input1_win);
172  Iterator input2(in2, input2_win);
173  Iterator output(out, win);
174 
175  execute_window_loop(win, [&](const Coordinates &)
176  {
177  auto output_ptr = reinterpret_cast<OutputScalarType *>(output.ptr());
178  const auto input1_ptr = reinterpret_cast<const InputScalarType *>(input1.ptr());
179  const auto input2_ptr = reinterpret_cast<const InputScalarType *>(input2.ptr());
180 
181  int x = window_start_x;
182 
183  svbool_t pg = svwhilelt<InputScalarType>(x, window_end_x);
184  do
185  {
186  func(pg,
187  {
188  op,
189  input1_ptr + x,
190  input2_ptr + x,
191  output_ptr + x
192  });
193  x += svcnt<InputScalarType>();
194  pg = svwhilelt<InputScalarType>(x, window_end_x);
195  }
196  while(svptest_any(all_true_pg, pg));
197  },
198  input1, input2, output);
199  }
200 }
