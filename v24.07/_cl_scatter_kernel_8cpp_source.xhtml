55 Status ClScatterKernel::validate(const ITensorInfo *updates,
56  const ITensorInfo *indices,
57  const ITensorInfo *dst,
58  const ScatterInfo &info)
59 {
60  ARM_COMPUTE_UNUSED(info);
61 
62  const TensorShape &ind_shape = indices->tensor_shape();
63  const TensorShape &upt_shape = updates->tensor_shape();
64  const TensorShape &dst_shape = dst->tensor_shape();
65 
66  const int32_t upt_dims = upt_shape.num_dimensions();
67  const int32_t dst_dims = dst_shape.num_dimensions();
68  const int32_t ind_dims = ind_shape.num_dimensions();
69  const int32_t data_dim = upt_dims - (ind_dims - 1); // Number of batch dims is the number of indices dims - 1
70 
71  const int32_t index_len = ind_shape[0];
72  bool unsupported_padding_config =
73  (dst_dims == index_len) && index_len > 1 && (dst->has_padding() || updates->has_padding());
74 
75  ARM_COMPUTE_RETURN_ERROR_ON_MSG(unsupported_padding_config, "Padding is not supported with these shapes.");
76  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(updates, dst);
77  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_NOT_IN(indices, DataType::S32);
78  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_NOT_IN(dst, DataType::F32, DataType::F16, DataType::S32, DataType::S16,
79  DataType::S8, DataType::U32, DataType::U16, DataType::U8);
80 
81  // Check data dims in update tensor and output tensor are equal
82  for (int32_t i = 0; i < data_dim; i++)
83  {
84  ARM_COMPUTE_RETURN_ERROR_ON_MSG(upt_shape[i] != dst_shape[i],
85  "Data dims should be same size in both updates and ouput tensor.");
86  }
87 
88  // Check if batch dims in indices and updates tensor are equal.
89  for (int32_t i = 0; i < ind_dims - 1; i++)
90  {
91  ARM_COMPUTE_RETURN_ERROR_ON_MSG(upt_shape[data_dim + i] != ind_shape[i + 1],
92  "Batch dimensions should be the same in updates and indices tensor.");
93  }
94 
95  ARM_COMPUTE_RETURN_ERROR_ON_MSG(ind_shape[1] != upt_shape[data_dim],
96  "Height of indices tensor should match size of highest dimension in updates tensor "
97  "(Excluding batch dimension)");
98 
99  ARM_COMPUTE_RETURN_ERROR_ON_MSG(
100  data_dim >= dst_dims, "Update tensor cannot have more dims than output tensor. (Excluding batch dimensions)");
101  ARM_COMPUTE_RETURN_ERROR_ON(index_len != dst_dims - data_dim);
102  ARM_COMPUTE_RETURN_ERROR_ON_MSG((ind_dims < 2), "Shape of Indices tensor must be at least 2D");
103 
104  ARM_COMPUTE_RETURN_ERROR_ON_MSG(index_len > max_index_length, "Maximum supported index length is 5!");
105  ARM_COMPUTE_RETURN_ERROR_ON_MSG(index_len > dst_dims && dst_dims != 1,
106  "Index length should be smaller than or equal to number of output dims");
107 
108  return Status{};
109 }

111 void ClScatterKernel::configure(const ClCompileContext &compile_context,
112  const ITensorInfo *updates,
113  const ITensorInfo *indices,
114  ITensorInfo *dst,
115  const ScatterInfo &info)
116 {
117  ARM_COMPUTE_ERROR_ON_NULLPTR(updates, dst, indices);
118  ARM_COMPUTE_LOG_PARAMS(updates, indices, dst, info);
119 
120  const TensorShape &dst_shape = dst->tensor_shape();
121  const int index_len = indices->dimension(0);
122 
123  // Check for single element data block
124  const bool is_scalar_block = (dst->num_dimensions() == static_cast<uint32_t>(index_len));
125 
126  const int n0 = adjust_vec_size(16 / updates->element_size(), is_scalar_block ? 1 : updates->dimension(0));
127  const int partial_n0 = updates->dimension(0) % n0;
128 
129  // The GWS will be 2D [x, y]
130  // x-dimension refers to the x coordinate of the dst tensor
131  // y-dimension refers to the collapsed y-coordinate of the data part of the dst tensor
132  Window win;
133 
134  if (!is_scalar_block)
135  {
136  win = calculate_max_window(dst_shape, Steps(n0));
137 
138  // Collapse the dimensions corresponding to indices in the execution window
139  for (int i = 0; i < index_len; ++i)
140  {
141  win.set(dst->num_dimensions() - (i + 1), Window::Dimension(0, 1, 1));
142  }
143 
144  win = win.collapse(win, 1);
145  }
146 
147  // Set build options
148  CLBuildOptions build_opts;
149  build_opts.add_option("-DDATA_TYPE=" + get_cl_type_from_data_type(dst->data_type()));
150  build_opts.add_option_if(is_data_type_float(dst->data_type()), "-DIS_FLOAT");
151 
152  const int num_dims = dst->num_dimensions();
153  TensorShape ind_collapsed = indices->tensor_shape().collapsed_from(1);
154  build_opts.add_option("-DNUM_INDICES=" + support::cpp11::to_string(ind_collapsed[1]));
155  build_opts.add_option("-DINDEX_LENGTH=" + support::cpp11::to_string(index_len));
156 
157  // We provide 5 variables to use in a constant array
158  for (int i = 1; i <= max_index_length; i++)
159  {
160  build_opts.add_option("-DOUT_SHAPE_N_MINUS_" + support::cpp11::to_string(i) + "=" +
161  support::cpp11::to_string(dst_shape[std::max(num_dims - i, 0)]));
162  }
163 
164  build_opts.add_option("-DN0=" + support::cpp11::to_string(n0));
165  build_opts.add_option("-DPARTIAL_N0=" + support::cpp11::to_string(partial_n0));
166 
167  switch (info.func)
168  {
169  case ScatterFunction::Update:
170  build_opts.add_option("-DSCATTER_FUNCTION=UPDATE_OP");
171  build_opts.add_option("-DSKIP_OUTPUT_READ");
172  break;
173  case ScatterFunction::Add:
174  build_opts.add_option("-DSCATTER_FUNCTION=ADD_OP");
175  break;
176  case ScatterFunction::Sub:
177  build_opts.add_option("-DSCATTER_FUNCTION=SUB_OP");
178  break;
179  case ScatterFunction::Max:
180  build_opts.add_option("-DSCATTER_FUNCTION=MAX_OP");
181  break;
182  case ScatterFunction::Min:
183  build_opts.add_option("-DSCATTER_FUNCTION=MIN_OP");
184  break;
185  default:
186  ARM_COMPUTE_ERROR("Not implemented");
187  }
188 
189  // Create kernel
190  std::string kernel_name = "scatter_mp1d_2d_mpnd";
191  build_opts.add_option("-D" + upper_string(kernel_name));
192 
193  ICLKernel::configure_internal(win);
194  _kernel = create_kernel(compile_context, kernel_name, build_opts.options());
195 
196  // Set config_id for enabling LWS tuning
197  _config_id = kernel_name;
198  _config_id += "_";
199  _config_id += lower_string(string_from_data_type(updates->data_type()));
200  _config_id += "_";
201  _config_id += support::cpp11::to_string(dst->dimension(1));
202  _config_id += "_";
203  _config_id += support::cpp11::to_string(dst->dimension(0));
204  _config_id += "_";
205  _config_id += support::cpp11::to_string(dst->dimension(2));
206  _config_id += "_";
207 }

209 void ClScatterKernel::run_op(ITensorPack &tensors, const Window &window, cl::CommandQueue &queue)
210 {
211  const auto updates =
212  utils::cast::polymorphic_downcast<const ICLTensor *>(tensors.get_const_tensor(TensorType::ACL_SRC_0));
213  const auto indices =
214  utils::cast::polymorphic_downcast<const ICLTensor *>(tensors.get_const_tensor(TensorType::ACL_SRC_1));
215  auto dst = utils::cast::polymorphic_downcast<ICLTensor *>(tensors.get_tensor(TensorType::ACL_DST));
216 
217  const ITensorInfo *dst_info = dst->info();
218  const ITensorInfo *upd_info = updates->info();
219  const int num_dims = dst_info->num_dimensions();
220  const int ind_dims = indices->info()->num_dimensions();
221  const int index_len = indices->info()->dimension(0);
222 
223  bool unsupported_padding_config =
224  num_dims == index_len && index_len > 1 && (dst_info->has_padding() || upd_info->has_padding());
225  if (unsupported_padding_config)
226  {
227  ARM_COMPUTE_ERROR("Unsupported Configuration! Padding not supported with these shapes.");
228  }
229 
230  // calculate m-dimensional data block strides in updates and destination tensors
231  const int upt_block_stride =
232  updates->info()->strides_in_bytes()[updates->info()->num_dimensions() - (ind_dims - 1)];
233 
234  const int out_block_stride = dst_info->strides_in_bytes()[num_dims - index_len];
235 
236  unsigned int idx = 0;
237 
238  add_2D_tensor_argument(idx, updates, window);
239  add_2D_tensor_argument(idx, indices, window);
240  add_2D_tensor_argument(idx, dst, window);
241 
242  _kernel.setArg<cl_int>(idx++, upt_block_stride);
243  _kernel.setArg<cl_int>(idx++, out_block_stride);
244 
245  enqueue(queue, *this, window, lws_hint());
246 }
