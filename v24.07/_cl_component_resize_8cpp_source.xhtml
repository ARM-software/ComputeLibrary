43 Status ClComponentResize::validate(const IGpuKernelComponent::Properties &properties,
44  const ArgumentPack<ITensorInfo> &tensors,
45  const ClComponentResize::Attributes &attributes)
46 {
47  ARM_COMPUTE_UNUSED(properties);
48 
49  const ITensorInfo *src = tensors.get_const_tensor(TensorType::ACL_SRC_0);
50  const ITensorInfo *dst = tensors.get_const_tensor(TensorType::ACL_DST_0);
51 
52  // Mismatching data types and quantization info
53  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(src, dst);
54  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_QUANTIZATION_INFO(src, dst);
55 
56  // Device requirements met
57  ARM_COMPUTE_RETURN_ERROR_ON_F16_UNSUPPORTED(src);
58 
59  // Align corners and sampling policy conformance
60  ARM_COMPUTE_RETURN_ERROR_ON(
61  attributes.align_corners() &&
62  !arm_compute::scale_utils::is_align_corners_allowed_sampling_policy(attributes.sampling_policy()));
63 
64  // All tensor infos are initialized
65  ARM_COMPUTE_RETURN_ERROR_ON(src->tensor_shape().total_size() == 0);
66  ARM_COMPUTE_RETURN_ERROR_ON(dst->tensor_shape().total_size() == 0);
67 
68  return Status();
69 }
