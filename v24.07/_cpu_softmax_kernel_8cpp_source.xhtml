106 void init_lut(std::vector<float> &lut, DataType type, float scale, float beta)
107 {
108  if (type == DataType::QASYMM8)
109  {
110  for (int i = 0; i < 256; ++i)
111  {
112  lut.push_back(std::exp(-scale * beta * i));
113  }
114  }
115  else if (type == DataType::QASYMM8_SIGNED)
116  {
117  for (int i = -128; i < 128; ++i)
118  {
119  lut.push_back(std::exp(-scale * beta * i));
120  }
121  }
122  else
123  {
124  ARM_COMPUTE_ERROR("Invalid datatype for QASYMM8/QASYMM8_SIGNED softmax");
125  }
126 }

128 Status validate_arguments_softmax(
129  const ITensorInfo &src, const ITensorInfo &dst, float beta, int axis, const ITensorInfo &tmp, bool is_log)
130 {
131  ARM_COMPUTE_UNUSED(beta);
132  // Check input
133  ARM_COMPUTE_RETURN_ERROR_ON_CPU_F16_UNSUPPORTED(&src);
134  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(&src, 1, DataType::QASYMM8, DataType::QASYMM8_SIGNED,
135  DataType::F16, DataType::F32);
136 
137  ARM_COMPUTE_RETURN_ERROR_ON(axis < 0 || axis > 3);
138 
139  const bool is_quantized_asymmetric = is_data_type_quantized_asymmetric(src.data_type());
140 
141  // Check output if configured
142  if (dst.total_size() != 0)
143  {
144  const QuantizationInfo output_quantization =
145  is_quantized_asymmetric ? arm_compute::get_softmax_output_quantization_info(src.data_type(), is_log)
146  : dst.quantization_info();
147  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(&src, &dst);
148  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_SHAPES(&src, &dst);
149  ARM_COMPUTE_RETURN_ERROR_ON(dst.quantization_info() != output_quantization);
150  }
151 
152  // Check tmp if configured
153  if (tmp.total_size() != 0)
154  {
155  // We have temporary storage only if src data type is quantized.
156  // Therefore, tmp data type must be F32
157  ARM_COMPUTE_RETURN_ERROR_ON(tmp.data_type() != DataType::F32);
158  ARM_COMPUTE_RETURN_ERROR_ON(!is_quantized_asymmetric);
159 
160  // We could potentially reduce tmp memory if we could predict or make an assumption
161  // on the maximum number of threads that will run in parallel.
162  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_SHAPES(&src, &tmp);
163  }
164 
165  return Status{};
166 }

174 void CpuSoftmaxKernel::configure(
175  const ITensorInfo *src, ITensorInfo *dst, float beta, bool is_log, int axis, ITensorInfo *tmp)
176 {
177  _axis = axis;
178 
179  ARM_COMPUTE_ERROR_ON_NULLPTR(src, dst, tmp);
180  ARM_COMPUTE_ERROR_THROW_ON(validate_arguments_softmax(*src, *dst, beta, axis, *tmp, is_log));
181 
182  // Configure kernel window
183  const bool is_quantized_asymmetric = is_data_type_quantized_asymmetric(src->data_type());
184 
185  // Output auto initialization if not yet initialized
186  const QuantizationInfo output_quantization =
187  is_quantized_asymmetric ? arm_compute::get_softmax_output_quantization_info(src->data_type(), is_log)
188  : dst->quantization_info();
189  auto_init_if_empty(*dst, TensorInfo(*src).set_quantization_info(output_quantization).reset_padding());
190 
191  // Tmp auto initialization if not yet initialized and src is quantized
192  if (is_quantized_asymmetric)
193  {
194  auto_init_if_empty(*tmp, TensorInfo(*src).set_data_type(DataType::F32).reset_padding());
195  }
196 
197  const auto *uk = CpuSoftmaxKernel::get_implementation(SoftmaxKernelDataTypeISASelectorData{
198  src->data_type(), CPUInfo::get().get_isa(), is_log, axis, CPUInfo::get().get_sme2_vector_length()});
199  ARM_COMPUTE_ERROR_ON(uk == nullptr || uk->ukernel == nullptr);
200 
201  std::string kernel_name = is_log ? std::string("CpuLogSoftmaxKernel") : std::string("CpuSoftmaxKernel");
202 
203  _beta = beta;
204  _run_method = uk->ukernel;
205  _name = kernel_name.append("/").append(uk->name);
206 
207  Window win;
208 
209  int vec_size = 16 / dst->element_size();
210 
211  if (_axis == 0)
212  {
213  win = calculate_max_window(*dst, Steps());
214 
215  /// TODO:Check dimensions > 0 for holes only. For this, we need
216  /// a utility function checking if there are holes after some dimension.
217  if (!has_holes(*dst, dst->num_dimensions() - 1))
218  {
219  win = win.collapse(win, Window::DimY);
220  }
221  }
222  else if (_axis > 0 && _axis <= 3)
223  {
224  win = calculate_max_window(*dst, Steps(vec_size));
225  }
226  else
227  {
228  ARM_COMPUTE_ERROR("Invalid axis");
229  }
230 
231  win.set(_axis, Window::Dimension(0, 1, 1));
232 
233  ICpuKernel<CpuSoftmaxKernel>::configure(win);
234 
235  const std::string uk_name = uk->name;
236  if (uk_name == "sme2_qu8_softmax_lut_512VL" || uk_name == "sme2_qs8_softmax_lut_512VL")
237  {
238  const float scale = src->quantization_info().uniform().scale;
239  init_lut(_lut, src->data_type(), scale, beta);
240  }
241 }

252 void CpuSoftmaxKernel::run_op(ITensorPack &tensors, const Window &window, const ThreadInfo &info)
253 {
254  ARM_COMPUTE_ERROR_ON_UNCONFIGURED_KERNEL(this);
255  ARM_COMPUTE_ERROR_ON_INVALID_SUBWINDOW(ICpuKernel<CpuSoftmaxKernel>::window(), window);
256  ARM_COMPUTE_ERROR_ON(_run_method == nullptr);
257 
258  const auto src = tensors.get_const_tensor(TensorType::ACL_SRC_0);
259  auto dst = tensors.get_tensor(TensorType::ACL_DST_0);
260 
261  if (is_data_type_quantized_asymmetric(src->info()->data_type()))
262  {
263  auto tmp = tensors.get_tensor(TensorType::ACL_DST_1);
264  unsigned int num_elems_processed_per_iteration;
265  if (_axis == 0)
266  {
267  num_elems_processed_per_iteration = src->info()->valid_region().shape[_axis];
268  }
269  else
270  {
271  //16 QASYMM8/QASYMM8_SIGNED elements can fit into the 16-byte vectors.
272  num_elems_processed_per_iteration = 16;
273  }
274  const unsigned int tmp_size_for_thread = tmp->info()->element_size() * num_elems_processed_per_iteration;
275 
276  void *tmp_for_thread = tmp->buffer() + (info.thread_id * tmp_size_for_thread);
277  _run_method(src, tmp_for_thread, dst, _beta, _axis, window, _lut.data());
278  }
279  else
280  {
281  _run_method(src, nullptr, dst, _beta, _axis, window, nullptr);
282  }
283 }
