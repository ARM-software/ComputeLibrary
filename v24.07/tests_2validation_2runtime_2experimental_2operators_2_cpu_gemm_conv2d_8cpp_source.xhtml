59 /** Test case for memory injection in @ref arm_compute::experimental::op::CpuGemmConv2d.
60  *
61  * Configure the operator once and inject memory at run-time in multiple executions.
62  *
63  * Checks performed in order:
64  * - Both runs compute the same output
65  */
66 TEST_CASE(OpCpuGemmConv2dMemoryInjection, framework::DatasetMode::ALL)
67 {
68  auto conv = std::make_unique<arm_compute::experimental::op::CpuGemmConv2d>();
69 
70  const auto src_info = TensorInfo(TensorShape(1U, 5U, 2U), 1, DataType::F32, DataLayout::NCHW);
71  const auto weights_info = TensorInfo(TensorShape(1U, 3U, 2U, 3U), 1, DataType::F32, DataLayout::NCHW);
72  const auto biases_info = TensorInfo(TensorShape(3U), 1, DataType::F32, DataLayout::NCHW);
73  auto dst_info = TensorInfo(TensorShape(1U, 7U, 3U), 1, DataType::F32, DataLayout::NCHW);
74  const auto pad_info = PadStrideInfo(1, 1, 0, 0, 2, 2, DimensionRoundingType::FLOOR);
75 
76  conv->configure(&src_info, &weights_info, &biases_info, &dst_info, pad_info);
77  auto const status = conv->validate(&src_info, &weights_info, &biases_info, &dst_info, pad_info);
78  ARM_COMPUTE_ASSERT(status);
79 
80  auto src = create_tensor<Tensor>(src_info);
81  auto weights = create_tensor<Tensor>(weights_info);
82  auto biases = create_tensor<Tensor>(biases_info);
83 
84  src.allocator()->allocate();
85  weights.allocator()->allocate();
86  biases.allocator()->allocate();
87 
88  ITensorPack run_pack{
89  {TensorType::ACL_SRC_0, &src}, {TensorType::ACL_SRC_1, &weights}, {TensorType::ACL_SRC_2, &biases}};
90  ITensorPack prep_pack{{TensorType::ACL_SRC_1, &weights}, {TensorType::ACL_SRC_2, &biases}};
91 
92  auto mg = MemoryGroup{};
93  auto ws = manage_workspace<Tensor>(conv->workspace(), mg, run_pack, prep_pack);
94 
95  auto run_conv = [&]() -> Tensor
96  {
97  auto dst = create_tensor<Tensor>(dst_info);
98  dst.allocator()->allocate();
99  run_pack.add_tensor(TensorType::ACL_DST, &dst);
100 
101  library->fill_tensor_value(Accessor(src), 1.f);
102  library->fill_tensor_value(Accessor(weights), 2.f);
103  library->fill_tensor_value(Accessor(biases), 3.f);
104  // This operator is configured once and captured by this lambda.
105  conv->prepare(prep_pack);
106  conv->run(run_pack);
107  return dst;
108  };
109  auto result_0 = run_conv();
110  auto result_1 = run_conv();
111  for (size_t i = 0; i < result_0.info()->tensor_shape().total_size(); ++i)
112  {
113  ARM_COMPUTE_EXPECT(((float *)result_0.buffer())[i] == ((float *)result_1.buffer())[i],
114  framework::LogLevel::ERRORS);
115  }
116 }
