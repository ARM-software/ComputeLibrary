45 Status validate_arguments(const ITensorInfo *input,
46  uint32_t axis,
47  uint32_t idx_input,
48  uint32_t num_tensors,
49  uint32_t rank,
50  const ITensorInfo *output)
51 {
52  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(input, output);
53  // Note: ARM_COMPUTE_RETURN_ERROR_ON_CPU_F16_UNSUPPORTED(input) is not needed here as this kernel doesn't use CPU FP16 instructions.
54  ARM_COMPUTE_RETURN_ERROR_ON(input->data_type() == DataType::UNKNOWN);
55  ARM_COMPUTE_RETURN_ERROR_ON(idx_input >= num_tensors);
56  ARM_COMPUTE_RETURN_ERROR_ON(axis > input->num_dimensions());
57  ARM_COMPUTE_RETURN_ERROR_ON(input->num_dimensions() > 4);
58  ARM_COMPUTE_RETURN_ERROR_ON(input->num_dimensions() != rank);
59 
60  if (output->total_size() != 0)
61  {
62  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DIMENSIONS(output->tensor_shape(),
63  compute_stack_shape(*input, axis, num_tensors));
64  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input, output);
65  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_QUANTIZATION_INFO(input, output);
66  }
67 
68  return Status{};
69 }

72 shift_from_axis_and_replace_coordinate(const Coordinates &id, uint32_t axis, uint32_t idx_input, uint32_t num_dims)
73 {
74  Coordinates id_out = id;
75  for (uint32_t i = num_dims; i > axis; --i)

77  id_out.set(i, id[i - 1]);
78  }
79  id_out.set(axis, idx_input);

83 void elementwise_stack(const std::vector<ITensor *> &input, ITensor *output, uint32_t axis, const Window &window)
84 {
85  Window window_out;
86  window_out.use_tensor_dimensions(output->info()->tensor_shape());
87 
88  const int32_t num_tensors = input.size();
89  const size_t element_size = input[0]->info()->element_size();
90  const uint32_t num_dims = static_cast<uint32_t>(input[0]->info()->num_dimensions());
91 
92  for (int32_t idx_input = 0; idx_input < num_tensors; ++idx_input)
93  {
94  Iterator input_it(input[idx_input], window);
95 
96  execute_window_loop(
97  window,
98  [&](const Coordinates &id)
99  {
100  Coordinates id_out = shift_from_axis_and_replace_coordinate(id, axis, idx_input, num_dims);
101  std::memcpy(output->ptr_to_element(id_out), input_it.ptr(), element_size);
102  },
103  input_it);
104  }
105 }

107 void memcpy_stack(const std::vector<ITensor *> &input, ITensor *output, uint32_t axis, const Window &window)
108 {
109  const int32_t element_size = input[0]->info()->element_size();
110  const int32_t chunk_size = input[0]->info()->tensor_shape().total_size_lower(axis) * element_size;
111  const int32_t num_tensors = input.size();
112  const int32_t out_chunk_step = chunk_size * num_tensors;
113 
114  const int32_t start_x = window.x().start();
115  const int32_t end_x = window.x().end();
116  const int32_t start_y = window.y().start();
117  const int32_t end_y = window.y().end();
118 
119  uint8_t *out_ptr_base = output->buffer() + output->info()->offset_first_element_in_bytes() + start_x * chunk_size;
120 
121  for (int32_t x = start_x; x < end_x; ++x)
122  {
123  const uint8_t *in_ptr =
124  input[x]->buffer() + input[x]->info()->offset_first_element_in_bytes() + start_y * chunk_size;
125  uint8_t *out_ptr = out_ptr_base + start_y * out_chunk_step;
126 
127  for (int32_t y = start_y; y < end_y; ++y)
128  {
129  std::memcpy(out_ptr, in_ptr, chunk_size);
130 
131  in_ptr += chunk_size;
132  out_ptr += out_chunk_step;
133  }
134 
135  out_ptr_base += chunk_size;
136  }
137 }

145 void NEStackLayerKernel::configure(const std::vector<ITensor *> &input, uint32_t axis, ITensor *output)
146 {
147  ARM_COMPUTE_ERROR_ON_NULLPTR(output);
148 
149  const int32_t num_tensors = input.size();
150  ARM_COMPUTE_ERROR_ON(num_tensors == 0);
151 
152  const uint32_t rank = input[0]->info()->num_dimensions();
153  ARM_COMPUTE_UNUSED(rank);
154 
155  for (int32_t i = 0; i < num_tensors; ++i)
156  {
157  ARM_COMPUTE_ERROR_ON_NULLPTR(input[i]);
158  ARM_COMPUTE_ERROR_THROW_ON(validate_arguments(input[i]->info(), axis, i, num_tensors, rank, output->info()));
159  }
160 
161  auto_init_if_empty(*output->info(), input[0]->info()->clone()->set_tensor_shape(
162  compute_stack_shape(*input[0]->info(), axis, num_tensors)));
163 
164  _input = input;
165  _output = output;
166  _axis = axis;
167 }

183 void NEStackLayerKernel::prepare()
184 {
185  // Prepare calculates the window at runtime, in case there is padding being added after configure()
186  const ITensorInfo *input_info = _input[0]->info();
187  const int32_t num_dims = input_info->num_dimensions();
188  const int32_t num_tensors = _input.size();
189 
190  // Check if there are any paddings in the input tensors
191  bool has_padding = false;
192  for (const ITensor *in : _input)
193  {
194  if (has_holes(*in->info(), num_dims - 1))
195  {
196  has_padding = true;
197  break;
198  }
199  }
200 
201  has_padding = has_padding || has_holes(*_output->info(), num_dims);
202 
203  Window win;
204  if (!has_padding)
205  {
206  _stack_fn = memcpy_stack;
207 
208  // 2D execution window (X,Y): [Num_tensors, Dimensions >= axis]
209  win.set(Window::DimX, Window::Dimension(0, num_tensors, 1));
210  win.set(Window::DimY, Window::Dimension(0, input_info->tensor_shape().total_size_upper(_axis), 1));
211  }
212  else
213  {
214  _stack_fn = elementwise_stack;
215  win = calculate_max_window(*input_info);
216  }
217 
218  INEKernel::configure(win);
219 }
