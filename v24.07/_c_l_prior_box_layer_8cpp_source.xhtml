59  info.min_sizes().size() * sizeof(float));
60  _aspect_ratios = cl::Buffer(CLScheduler::get().context(), CL_MEM_ALLOC_HOST_PTR | CL_MEM_READ_WRITE,

62  if (!info.max_sizes().empty())
63  {
64  _max = cl::Buffer(CLScheduler::get().context(), CL_MEM_ALLOC_HOST_PTR | CL_MEM_READ_WRITE,
65  info.max_sizes().size() * sizeof(float));
66  }
67 
68  auto k = std::make_unique<CLPriorBoxLayerKernel>();
69  k->configure(compile_context, input1, input2, output, info, &_min, &_max, &_aspect_ratios);
