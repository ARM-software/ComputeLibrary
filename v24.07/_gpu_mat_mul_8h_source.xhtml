76  /* Create an operator and fuse it into the workload sketch.
77  * @note If @ref validate_op() fails, the creation also fails and may throw an error.
78  * @note If @ref validate_op() fails, @p sketch remains unchanged and valid.
79  *
80  * Valid data type configurations:
81  * |LHS |RHS |dst |
82  * |:--------------|:--------------|:--------------|
83  * |F16 |F16 |F16 |
84  * |F32 |F32 |F32 |
85  *
86  *
87  * @param[in,out] sketch Workload sketch into which the operator will be fused
88  * @param[in] lhs Input tensor info for the LHS matrix. Data type supported: F32/F16. Dimensions above 2 are collapsed onto dimension 2 and represent the batch.
89  * @param[in] rhs Input tensor info for the RHS matrix. Data type supported: same as @p lhs. Dimensions above 2 are collapsed onto dimension 2 and represent the batch.
90  * @param[in] attributes Operator attributes
91  * @param[in] settings Operator settings
92  *
93  * @return Pointer for the destination tensor info
94  */
95  static ITensorInfo *create_op(GpuWorkloadSketch &sketch,
96  ITensorInfo *lhs,
97  ITensorInfo *rhs,
98  const Attributes &attributes,
99  const Settings &settings);
100 
101  /* Check if the operator configuration is supported, irrespective of fusion
102  *
103  * @param[in] context Workload context within which the operator is running
104  * @param[in] lhs Input tensor info for the LHS matrix.
105  * @param[in] rhs Input tensor info for the RHS matrix.
106  * @param[in] attributes Operator attributes
107  * @param[in] settings Operator settings
108  *
109  * @return Status
110  */
111  static Status is_supported_op(const GpuWorkloadContext &context,
112  const ITensorInfo *lhs,
113  const ITensorInfo *rhs,
114  const Attributes &attributes,

122  */
123  static Status validate_op(const GpuWorkloadSketch &sketch,
