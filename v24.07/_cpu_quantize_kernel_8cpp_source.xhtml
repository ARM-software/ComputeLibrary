51 Status validate_arguments(const ITensorInfo *src, const ITensorInfo *dst)
52 {
53  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(src, dst);
54  ARM_COMPUTE_RETURN_ERROR_ON_CPU_F16_UNSUPPORTED(src);

59  DataType::QASYMM8_SIGNED, DataType::QASYMM16);
60  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_SHAPES(src, dst);
61 
62  return Status{};
63 }

67 void CpuQuantizeKernel::configure(const ITensorInfo *src, ITensorInfo *dst)
68 {
69  ARM_COMPUTE_ERROR_ON_NULLPTR(src, dst);
70  ARM_COMPUTE_ERROR_THROW_ON(validate_arguments(src, dst));
71 
72  static const std::map<std::string, QuantizeFunctionExecutorPtr> quant_map = {
73  {"op_QASYMM8_QASYMM8", REGISTER_INTEGER_NEON(u8_u8_run_quantize_qasymm8)},
74  {"op_QASYMM8_QASYMM8_SIGNED", REGISTER_INTEGER_NEON(u8_i8_run_quantize_qasymm8)},
75  {"op_QASYMM8_QASYMM16", REGISTER_INTEGER_NEON(u8_run_quantize_qasymm16)},
76 
77  {"op_QASYMM8_SIGNED_QASYMM8", REGISTER_INTEGER_NEON(i8_u8_run_quantize_qasymm8)},
78  {"op_QASYMM8_SIGNED_QASYMM8_SIGNED", REGISTER_INTEGER_NEON(i8_i8_run_quantize_qasymm8)},
79  {"op_QASYMM8_SIGNED_QASYMM16", REGISTER_INTEGER_NEON(i8_run_quantize_qasymm16)},
80 
81  // Functions for offset only requantization
82  {"op_OFFSET_ONLY_QASYMM8_QASYMM8", REGISTER_INTEGER_NEON(u8_u8_run_requantize_offset_only)},
83  {"op_OFFSET_ONLY_QASYMM8_QASYMM8_SIGNED", REGISTER_INTEGER_NEON(u8_i8_run_requantize_offset_only)},
84  {"op_OFFSET_ONLY_QASYMM8_SIGNED_QASYMM8", REGISTER_INTEGER_NEON(i8_u8_run_requantize_offset_only)},
85  {"op_OFFSET_ONLY_QASYMM8_SIGNED_QASYMM8_SIGNED", REGISTER_INTEGER_NEON(i8_i8_run_requantize_offset_only)},
86 
87  // Functions for offset uint8 to int8 and vice versa quantization (no scale changes)
88  {"op_OFFSET_ONLY_CONVERT_QASYMM8_SIGNED_QASYMM8",
89  REGISTER_INTEGER_NEON(i8_u8_run_requantize_offset_only_convert)},
90  {"op_OFFSET_ONLY_CONVERT_QASYMM8_QASYMM8_SIGNED",
91  REGISTER_INTEGER_NEON(u8_i8_run_requantize_offset_only_convert)},
92 
93  {"op_F32_QSYMM8", REGISTER_FP32_NEON(fp32_i8_run_quantize_qsymm8)},
94  {"op_F32_QASYMM8", REGISTER_FP32_NEON(fp32_u8_run_quantize_qasymm8)},
95  {"op_F32_QASYMM8_SIGNED", REGISTER_FP32_NEON(fp32_i8_run_quantize_qasymm8)},
96  {"op_F32_QASYMM16", REGISTER_FP32_NEON(fp32_run_quantize_qasymm16)},
97 
98 #ifdef ARM_COMPUTE_ENABLE_FP16
99  {"op_F16_QASYMM8", REGISTER_FP16_NEON(fp16_u8_run_quantize_qasymm8)},
100  {"op_F16_QASYMM8_SIGNED", REGISTER_FP16_NEON(fp16_i8_run_quantize_qasymm8)},
101  {"op_F16_QASYMM16", REGISTER_FP16_NEON(fp16_run_quantize_qasymm16)},
102 #endif /* ARM_COMPUTE_ENABLE_FP16 */
103  };
104 
105  std::string function_to_call("op_");
106 
107  // For offset only functions - must be 8-bit and have identical scale values.
108  if (src->quantization_info().scale() == dst->quantization_info().scale() &&
109  (is_data_type_quantized_asymmetric_char(src->data_type()) &&
110  is_data_type_quantized_asymmetric_char(dst->data_type())))
111  {
112  function_to_call += "OFFSET_ONLY_";
113  // For optimized datatype conversion 8-bit re-quantization offset only functions.
114  // These must have an offset of exactly 128 to match requirements - has specific circumstances to match use case.
115  auto uqinfo =
116  compute_requantization_scale_offset(src->quantization_info().uniform(), dst->quantization_info().uniform());
117  const auto src_dt = src->data_type();
118  if (src->data_type() != dst->data_type() && ((src_dt == DataType::QASYMM8_SIGNED && uqinfo.offset == 128) ||
119  (src_dt == DataType::QASYMM8 && uqinfo.offset == -128)))
120  {
121  function_to_call += "CONVERT_";
122  }
123  }
124 
125  // Specify datatype for function
126  function_to_call += string_from_data_type(src->data_type()) + "_";
127  function_to_call += string_from_data_type(dst->data_type());
128 
129  auto it = quant_map.find(function_to_call);
130 
131  if (it == quant_map.end())
132  {
133  ARM_COMPUTE_ERROR("Unsupported combination of input and output data types");
134  }
135  _func = it->second;
136 
137  // Calculate window. Squash if possible.
138  Window win;
139  std::tie(win, _split_dimension) = calculate_squashed_or_max_window(*src);
140 
141  ICpuKernel::configure(win);
142 }

150 void CpuQuantizeKernel::run_op(ITensorPack &tensors, const Window &window, const ThreadInfo &info)
151 {
152  ARM_COMPUTE_UNUSED(info);
153  ARM_COMPUTE_ERROR_ON_UNCONFIGURED_KERNEL(this);
154  ARM_COMPUTE_ERROR_ON_INVALID_SUBWINDOW(ICpuKernel::window(), window);
155  ARM_COMPUTE_ERROR_ON(_func == nullptr);
156 
157  const auto src = tensors.get_const_tensor(TensorType::ACL_SRC);
158  auto dst = tensors.get_tensor(TensorType::ACL_DST);
159  (*_func)(src, dst, window);
