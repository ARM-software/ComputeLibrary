92 template <typename TIn, typename TOut>
93 void run_quantize_qsymm8(const ITensor *src, ITensor *dst, const Window &window)
94 {
95  const auto window_start_x = static_cast<int>(window.x().start());
96  const auto window_end_x = static_cast<int>(window.x().end());
97 
98  const UniformQuantizationInfo uqinfo_in = src->info()->quantization_info().uniform();
99  UniformQuantizationInfo uqinfo = dst->info()->quantization_info().uniform();
100  uqinfo = compute_requantization_scale_offset(uqinfo_in, uqinfo);
101 
102  // Collapse window and reset first dimension to handle tail calculations manually
103  Window win_collapsed = window.collapse_if_possible(window, Window::DimZ);
104  win_collapsed.set(Window::DimX, Window::Dimension(0, 1, 1));
105 
106  Iterator input(src, win_collapsed);
107  Iterator output(dst, win_collapsed);
108  execute_window_loop(
109  win_collapsed,
110  [&](const Coordinates &)
111  {
112  auto input_ptr = reinterpret_cast<const TIn *>(input.ptr());
113  auto output_ptr = reinterpret_cast<TOut *>(output.ptr());
114  int x = window_start_x;
115  for (; x <= (window_end_x - window_step); x += window_step)
116  {
117  wrapper::vstore(&output_ptr[x], vquantize_qasymm8<TOut>(load_value(&input_ptr[x]), uqinfo));
118  }
119  // Compute left-over elements
120  for (; x < window_end_x; ++x)
121  {
122  output_ptr[x] = quantize_qsymm8(input_ptr[x], dst->info()->quantization_info());
123  }
124  },
125  input, output);
126 }

128 template <typename TIn, typename TOut>
129 void run_requantize_offset_only_convert(const ITensor *src, ITensor *dst, const Window &window)
130 {
131  const auto window_start_x = static_cast<int>(window.x().start());
132  const auto window_end_x = static_cast<int>(window.x().end());
133 
134  // Calculate output offset difference.
135  const UniformQuantizationInfo uqinfo_in = src->info()->quantization_info().uniform();
136  UniformQuantizationInfo uqinfo = dst->info()->quantization_info().uniform();
137  uqinfo = compute_requantization_scale_offset(uqinfo_in, uqinfo);
138 
139  // Collapse window and reset first dimension to handle tail calculations manually
140  Window win_collapsed = window.collapse_if_possible(window, Window::DimZ);
141 
142  win_collapsed.set(Window::DimX, Window::Dimension(0, 1, 1));
143 
144  // Duplicate offset in signed vector format
145  const int8x16_t offset = wrapper::vdup_n(static_cast<int8_t>(uqinfo.offset), wrapper::traits::vector_128_tag{});
146 
147  Iterator input(src, win_collapsed);
148  Iterator output(dst, win_collapsed);
149  execute_window_loop(
150  win_collapsed,
151  [&](const Coordinates &)
152  {
153  auto input_ptr = reinterpret_cast<const TIn *>(input.ptr());
154  auto output_ptr = reinterpret_cast<TOut *>(output.ptr());
155  int x = window_start_x;
156  for (; x <= (window_end_x - window_step); x += window_step)
157  {
158  const wrapper::traits::neon_vector_t<TIn, window_step> qv =
159  wrapper::vloadq(input_ptr + x); // load 128 bit vector of 8 bit datatype
160 
161  // Signed addition.
162  auto res = vaddq_s8(reinterpret_cast<int8x16_t>(qv), offset);
163 
164  // Output is dependent on datatype.
165  wrapper::vstore(&output_ptr[x],
166  reinterpret_cast<wrapper::traits::neon_vector_t<TOut, window_step>>(res));
167  }
168  // Compute left-over elements
169  for (; x < window_end_x; ++x)
170  {
171  auto result = uqinfo.offset + static_cast<int32_t>(input_ptr[x]);
172  output_ptr[x] = static_cast<TOut>(result);
173  }
174  },
175  input, output);
176 }

178 template <typename TIn, typename TOut>
179 void run_requantize_offset_only(const ITensor *src, ITensor *dst, const Window &window)
180 {
181  const auto window_start_x = static_cast<int>(window.x().start());
182  const auto window_end_x = static_cast<int>(window.x().end());
183 
184  const UniformQuantizationInfo uqinfo_in = src->info()->quantization_info().uniform();
185  UniformQuantizationInfo uqinfo = dst->info()->quantization_info().uniform();
186  uqinfo = compute_requantization_scale_offset(uqinfo_in, uqinfo);
187 
188  // Collapse window and reset first dimension to handle tail calculations manually
189  Window win_collapsed = window.collapse_if_possible(window, Window::DimZ);
190  win_collapsed.set(Window::DimX, Window::Dimension(0, 1, 1));
191 
192  // Duplicate offset in signed vector format
193  const int16x8_t offset = wrapper::vdup_n(static_cast<int16_t>(uqinfo.offset), wrapper::traits::vector_128_tag{});
194 
195  const int32_t low_bound = (dst->info()->data_type() == DataType::QASYMM8) ? 0 : -128;
196  const int32_t upper_bound = (dst->info()->data_type() == DataType::QASYMM8) ? 255 : 127;
197 
198  Iterator input(src, win_collapsed);
199  Iterator output(dst, win_collapsed);
200  execute_window_loop(
201  win_collapsed,
202  [&](const Coordinates &)
203  {
204  auto input_ptr = reinterpret_cast<const TIn *>(input.ptr());
205  TOut *output_ptr = reinterpret_cast<TOut *>(output.ptr());
206 
207  int x = window_start_x;
208  for (; x <= (window_end_x - window_step); x += window_step)
209  {
210  const auto qv = wrapper::vloadq(input_ptr + x); // load 128 bit vector of 8 bit datatype
211  int16x8_t lower = reinterpret_cast<int16x8_t>(wrapper::vmovl(wrapper::vgetlow(qv)));
212  int16x8_t upper = reinterpret_cast<int16x8_t>(wrapper::vmovl(wrapper::vgethigh(qv)));
213 
214  // Signed addition.
215  lower = wrapper::vqadd(lower, offset);
216  upper = wrapper::vqadd(upper, offset);
217 
218  // Output is dependent on datatype.
219  auto res = recombine_8_16<TOut>(lower, upper);
220  wrapper::vstore(&output_ptr[x], res);
221  }
222  // Compute left-over elements
223  for (; x < window_end_x; ++x)
224  {
225  // Add offset and clamp result to within the range of the output datatype.
226  int32_t result = uqinfo.offset + static_cast<int32_t>(input_ptr[x]);
227  result = utility::clamp<int32_t>(result, low_bound, upper_bound);
228 
229  // Cast result to output datatype.
230  output_ptr[x] = static_cast<TOut>(result);
231  }
232  },
233  input, output);
234 }
